27 /**
28  * @brief Particle cache on the master.
29  *
30  * This class implements cached access to all particles in a
31  * particle range on the master node.
32  * This implementation fetches all particles to
33  * the master on first access. Updates of the particle data are
34  * triggered automatically on access. The data in the cache
35  * is invalidated automatically on_particle_change, and then
36  * updated on the next access.
37  *
38  */
39 class PartCfg {
40  /** The particle data */
41  std::vector<Particle> m_parts;
42  /** State */
43  bool m_valid;
44 
45 public:
46  using value_type = Particle;
47  PartCfg() : m_valid(false) {}
48 
49  /**
50  * @brief Iterator pointing to the particle with the lowest
51  * id.
52  *
53  * Returns a random access iterator that traverses the
54  * particles
55  * in order of ascending id. If the cache is not up-to-date,
56  * an update is triggered. This iterator stays valid as long
57  * as the cache is valid. Since the cache could be invalidated
58  * and updated elsewhere, iterators into the cache should not
59  * be stored.
60  */
61  auto begin() {
62  if (!m_valid)
63  update();
64 
65  return m_parts.begin();
66  }
67 
68  /**
69  * @brief Iterator pointing past the particle with the highest
70  * id.
71  *
72  * If the cache is not up-to-date,
73  * an update is triggered.
74  */
75  auto end() {
76  if (!m_valid)
77  update();
78 
79  return m_parts.end();
80  }
81 
82  /**
83  * @brief Returns true if the cache is up-to-date.
84  *
85  * If false, particle access will trigger an update.
86  */
87  bool valid() const { return m_valid; }
88 
89  /**
90  * @brief Invalidate the cache and free memory.
91  */
92  void invalidate() {
93  /* Release memory */
94  m_parts = std::vector<Particle>();
95  /* Adjust state */
96  m_valid = false;
97  }
98 
99  /**
100  * @brief Update particle information.
101  *
102  * This triggers a global update. All nodes
103  * sort their particle by id, and send them
104  * to the master.
105  */
106 private:
107  void update();
108 
109 public:
110  /** Number of particles in the config.
111  */
112  std::size_t size() {
113  if (!m_valid)
114  update();
115 
116  return m_parts.size();
117  }
118 
119  /**
120  * @brief size() == 0 ?
121  */
122  bool empty() {
123  if (!m_valid)
124  update();
125 
126  return m_parts.empty();
127  }
128 };
