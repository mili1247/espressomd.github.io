52 /**
53  * @brief Vector pointing from the real particle
54  * to the virtual site.
55  *
56  * @return Relative distance.
57  */
58 inline Utils::Vector3d connection_vector(
59 
60  Particle const *p_ref,
61  const ParticleProperties::VirtualSitesRelativeParameters &vs_rel) {
62  // Calculate the quaternion defining the orientation of the vector connecting
63  // the virtual site and the real particle
64  // This is obtained, by multiplying the quaternion representing the director
65  // of the real particle with the quaternion of the virtual particle, which
66  // specifies the relative orientation.
67  auto const director = Utils::convert_quaternion_to_director(
68  p_ref->r.quat * vs_rel.rel_orientation)
69  .normalize();
70 
71  return vs_rel.distance * director;
72 }

86 /**
87  * @brief Velocity of the virtual site
88  * @param p_ref Reference particle for the virtual site.
89  * @param vs_rel Parameters for the virtual site.
90  * @return Velocity of the virtual site.
91  */
92 Utils::Vector3d
93 velocity(const Particle *p_ref,
94  const ParticleProperties::VirtualSitesRelativeParameters &vs_rel) {
95  auto const d = connection_vector(p_ref, vs_rel);
96 
97  // Get omega of real particle in space-fixed frame
98  auto const omega_space_frame =
99  convert_vector_body_to_space(*p_ref, p_ref->m.omega);
100  // Obtain velocity from v=v_real particle + omega_real_particle \times
101  // director
102  return vector_product(omega_space_frame, d) + p_ref->m.v;
103 }

121 /**
122  * @brief Constraint force on the real particle.
123  *
124  * Calculates the force exerted by the constraint on the
125  * reference particle.
126  */
127 ParticleForce constraint_force(
128  const ParticleForce &f, const Particle *p_ref,
129  const ParticleProperties::VirtualSitesRelativeParameters &vs_rel) {
130  return {f.f,
131  vector_product(connection_vector(p_ref, vs_rel), f.f) + f.torque};
132 }
133 
134 /**
135  * @brief Constraint force to hold the particles at its prescribed position.
136  *
137  * @param f Force on the virtual site.
138  * @param p_ref Reference particle.
139  * @param vs_rel Parameters.
140  * @return Constraint force.
141  */
142 auto constraint_stress(
143  const Utils::Vector3d &f, const Particle *p_ref,
144  const ParticleProperties::VirtualSitesRelativeParameters &vs_rel) {
145  /* The constraint force is minus the force on the particle, make it force
146  * free. The counter force is translated by the connection vector to the
147  * real particle, hence the virial stress is */
148  return tensor_product(-f, connection_vector(p_ref, vs_rel));
149 }

152 void VirtualSitesRelative::update() const {
153  cell_structure.ghosts_update(Cells::DATA_PART_POSITION |
154  Cells::DATA_PART_MOMENTUM);
155 
156  for (auto &p : cell_structure.local_particles()) {
157  if (!p.p.is_virtual)
158  continue;
159 
160  const Particle *p_ref = get_reference_particle(p.p.vs_relative);
161 
162  auto const new_pos = position(p_ref, p.p.vs_relative);
163  /* The shift has to respect periodic boundaries: if the reference
164  * particles is not in the same image box, we potentially avoid to shift
165  * to the other side of the box. */
166  p.r.p += box_geo.get_mi_vector(new_pos, p.r.p);
167 
168  p.m.v = velocity(p_ref, p.p.vs_relative);
169 
170  if (get_have_quaternion())
171  p.r.quat = orientation(p_ref, p.p.vs_relative);
172 
173  if ((p.r.p - p.l.p_old).norm2() > Utils::sqr(0.5 * skin))
174  cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
175  } // namespace
176 }

178 // Distribute forces that have accumulated on virtual particles to the
179 // associated real particles
180 void VirtualSitesRelative::back_transfer_forces_and_torques() const {
181  cell_structure.ghosts_reduce_forces();
182 
183  init_forces_ghosts(cell_structure.ghost_particles());
184 
185  // Iterate over all the particles in the local cells
186  for (auto &p : cell_structure.local_particles()) {
187  // We only care about virtual particles
188  if (p.p.is_virtual) {
189  // First obtain the real particle responsible for this virtual particle:
190  Particle *p_ref = get_reference_particle(p.p.vs_relative);
191 
192  // Add forces and torques
193  p_ref->f += constraint_force(p.f, p_ref, p.p.vs_relative);
194  }
195  }
196 }
