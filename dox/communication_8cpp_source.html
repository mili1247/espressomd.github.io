67 /** Workaround for "Read -1, expected XXXXXXX, errno = 14" that sometimes
68  * appears when CUDA is used. This is a bug in OpenMPI 2.0-2.1.2 and 3.0.0
69  * according to
70  * https://www.mail-archive.com/users@lists.open-mpi.org/msg32357.html,
71  * so we set btl_vader_single_copy_mechanism = none.
72  */
73 void openmpi_fix_vader() {
74  if ((OMPI_MAJOR_VERSION == 2 && OMPI_MINOR_VERSION == 1 &&
75  OMPI_RELEASE_VERSION < 3) or
76  (OMPI_MAJOR_VERSION == 3 && OMPI_MINOR_VERSION == 0 &&
77  OMPI_RELEASE_VERSION == 0)) {
78  setenv("OMPI_MCA_btl_vader_single_copy_mechanism", "none", 0);
79  }
80 }

82 /**
83  * @brief Assure that openmpi is loaded to the global namespace.
84  *
85  * This was originally inspired by mpi4py
86  * (https://github.com/mpi4py/mpi4py/blob/4e3f47b6691c8f5a038e73f84b8d43b03f16627f/src/lib-mpi/compat/openmpi.h).
87  * It's needed because OpenMPI dlopens its submodules. These are unable to find
88  * the top-level OpenMPI library if that was dlopened itself, i.e. when the
89  * Python interpreter dlopens a module that is linked against OpenMPI. It's
90  * about some weird two-level symbol namespace thing.
91  */
92 void openmpi_global_namespace() {
93  if (OMPI_MAJOR_VERSION >= 3)
94  return;
95 #ifdef RTLD_NOLOAD
96  const int mode = RTLD_NOW | RTLD_GLOBAL | RTLD_NOLOAD;
97 #else
98  const int mode = RTLD_NOW | RTLD_GLOBAL;
99 #endif
100 
101  const void *_openmpi_symbol = dlsym(RTLD_DEFAULT, "MPI_Init");
102  if (!_openmpi_symbol) {
103  fprintf(stderr, "Aborting because unable to find OpenMPI symbol.\n");
104  errexit();
105  }
106 
107  Dl_info _openmpi_info;
108  dladdr(_openmpi_symbol, &_openmpi_info);
109 
110  const void *handle = dlopen(_openmpi_info.dli_fname, mode);
111 
112  if (!handle) {
113  fprintf(stderr, "Aborting because unable to load libmpi into the "
114  "global symbol space.\n");
115  errexit();
116  }
117 }
118 } // namespace
119 #endif
