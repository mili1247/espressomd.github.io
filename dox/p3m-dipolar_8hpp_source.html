123 /** Tune dipolar P3M parameters to desired accuracy.
124  *
125  * The parameters
126  * @ref P3MParameters::mesh "mesh",
127  * @ref P3MParameters::cao "cao",
128  * @ref P3MParameters::r_cut_iL "r_cut_iL" and
129  * @ref P3MParameters::alpha_L "alpha_L"
130  * are tuned to obtain the target accuracy (initially stored in
131  * @ref P3MParameters::accuracy "accuracy") in optimal time.
132  * These parameters are stored in the @ref dp3m object.
133  *
134  * The function utilizes the analytic expression of the error estimate
135  * for the dipolar P3M method in the paper of @cite cerda08d in
136  * order to obtain the rms error in the force for a system of N randomly
137  * distributed particles in a cubic box. For the real space error, the
138  * estimate in @cite kolafa92a is used.
139  *
140  * Parameter ranges if not given explicit values via dp3m_set_tune_params():
141  * - @p mesh is set up such that the number of mesh points is equal to the
142  * number of magnetic dipolar particles
143  * - @p cao explores all possible values
144  * - @p alpha_L is tuned for each tuple (@p r_cut_iL, @p mesh, @p cao) and
145  * calculated assuming that the error contributions of real and reciprocal
146  * space should be equal
147  *
148  * After checking if the total error lies below the target accuracy, the
149  * time needed for one force calculation (including Verlet list update)
150  * is measured via time_force_calc().
151  *
152  * The function generates a log of the performed tuning.
153  *
154  * The function is based on routines of the program HE_Q.cpp for charges
155  * written by M. Deserno.
156  *
157  * @param verbose printf output
158  * @param timings Number of test force calculations
159  * @retval ES_OK
160  * @retval ES_ERROR
161  */
162 int dp3m_adaptive_tune(int timings, bool verbose);

180 /** Calculate real space contribution of p3m dipolar pair forces and torques.
181  * If NPT is compiled in, update the NpT virial.
182  */
183 inline ParticleForce dp3m_pair_force(Particle const &p1, Particle const &p2,
184  Utils::Vector3d const &d, double dist2,
185  double dist) {
186  if ((p1.p.dipm == 0.) || (p2.p.dipm == 0.) || dist >= dp3m.params.r_cut ||
187  dist <= 0.)
188  return {};
189 
190  auto const dip1 = p1.calc_dip();
191  auto const dip2 = p2.calc_dip();
192  auto const alpsq = dp3m.params.alpha * dp3m.params.alpha;
193  auto const adist = dp3m.params.alpha * dist;
194 #if USE_ERFC_APPROXIMATION
195  auto const erfc_part_ri = Utils::AS_erfc_part(adist) / dist;
196 #else
197  auto const erfc_part_ri = erfc(adist) / dist;
198 #endif
199 
200  // Calculate scalar multiplications for vectors mi, mj, rij
201  auto const mimj = dip1 * dip2;
202 
203  auto const mir = dip1 * d;
204  auto const mjr = dip2 * d;
205 
206  auto const coeff = 2.0 * dp3m.params.alpha * Utils::sqrt_pi_i();
207  auto const dist2i = 1 / dist2;
208  auto const exp_adist2 = exp(-adist * adist);
209 
210  double B_r, C_r, D_r;
211  if (dp3m.params.accuracy > 5e-06)
212  B_r = (erfc_part_ri + coeff) * exp_adist2 * dist2i;
213  else
214  B_r = (erfc(adist) / dist + coeff * exp_adist2) * dist2i;
215 
216  C_r = (3 * B_r + 2 * alpsq * coeff * exp_adist2) * dist2i;
217  D_r = (5 * C_r + 4 * coeff * alpsq * alpsq * exp_adist2) * dist2i;
218 
219  // Calculate real-space forces
220  auto const force =
221  dipole.prefactor *
222  ((mimj * d + dip1 * mjr + dip2 * mir) * C_r - mir * mjr * D_r * d);
223 
224  // Calculate vector multiplications for vectors mi, mj, rij
225  auto const mixmj = vector_product(dip1, dip2);
226  auto const mixr = vector_product(dip1, d);
227 
228  // Calculate real-space torques
229  auto const torque = dipole.prefactor * (-mixmj * B_r + mixr * (mjr * C_r));
230 #ifdef NPT
231 #if USE_ERFC_APPROXIMATION
232  auto const fac = dipole.prefactor * p1.p.dipm * p2.p.dipm * exp_adist2;
233 #else
234  auto const fac = dipole.prefactor * p1.p.dipm * p2.p.dipm;
235 #endif
236  auto const energy = fac * (mimj * B_r - mir * mjr * C_r);
237  npt_add_virial_magnetic_contribution(energy);
238 #endif // NPT
239  return ParticleForce{force, torque};
240 }
