77 /**
78  * @brief Calculates the dihedral angle between particle quadruple p1, p2, p3
79  * and p4.
80  *
81  * The dihedral angle is the angle between the planes
82  * specified by the particle triples (p1,p2,p3) and (p2,p3,p4).
83  * Vectors a, b and c are the bond vectors between consecutive particles.
84  * If the a,b or b,c are parallel the dihedral angle is not defined in which
85  * case the routine returns phi=-1. Calling functions should check for that
86  *
87  * @param[in] r1 , r2 , r3 , r4 Positions of the particles forming the dihedral
88  * @param[out] a Vector from @p p1 to @p p2
89  * @param[out] b Vector from @p p2 to @p p3
90  * @param[out] c Vector from @p p3 to @p p4
91  * @param[out] aXb Vector product of a and b
92  * @param[out] l_aXb |aXB|
93  * @param[out] bXc Vector product of b and c
94  * @param[out] l_bXc |bXc|
95  * @param[out] cosphi Cosine of the dihedral angle
96  * @param[out] phi Dihedral angle
97  */
98 inline void
99 calc_dihedral_angle(Utils::Vector3d const &r1, Utils::Vector3d const &r2,
100  Utils::Vector3d const &r3, Utils::Vector3d const &r4,
101  Utils::Vector3d &a, Utils::Vector3d &b, Utils::Vector3d &c,
102  Utils::Vector3d &aXb, double *l_aXb, Utils::Vector3d &bXc,
103  double *l_bXc, double *cosphi, double *phi) {
104  a = box_geo.get_mi_vector(r2, r1);
105  b = box_geo.get_mi_vector(r3, r2);
106  c = box_geo.get_mi_vector(r4, r3);
107 
108  /* calculate vector product a X b and b X c */
109  aXb = vector_product(a, b);
110  bXc = vector_product(b, c);
111 
112  /* calculate the unit vectors */
113  *l_aXb = aXb.norm();
114  *l_bXc = bXc.norm();
115 
116  /* catch case of undefined dihedral angle */
117  if (*l_aXb <= TINY_LENGTH_VALUE || *l_bXc <= TINY_LENGTH_VALUE) {
118  *phi = -1.0;
119  *cosphi = 0;
120  return;
121  }
122 
123  aXb /= *l_aXb;
124  bXc /= *l_bXc;
125 
126  *cosphi = aXb * bXc;
127 
128  if (fabs(fabs(*cosphi) - 1) < TINY_SIN_VALUE)
129  *cosphi = std::round(*cosphi);
130 
131  /* Calculate dihedral angle */
132  *phi = acos(*cosphi);
133  if ((aXb * c) < 0.0)
134  *phi = (2.0 * Utils::pi()) - *phi;
135 }

137 /** Compute the four-body dihedral interaction force.
138  *
139  * @param[in] r1 Position of the first particle.
140  * @param[in] r2 Position of the second particle.
141  * @param[in] r3 Position of the third particle.
142  * @param[in] r4 Position of the fourth particle.
143  * @return the forces on @p p2, @p p1, @p p3
144  */
145 inline boost::optional<std::tuple<Utils::Vector3d, Utils::Vector3d,
146  Utils::Vector3d, Utils::Vector3d>>
147 DihedralBond::forces(Utils::Vector3d const &r1, Utils::Vector3d const &r2,
148  Utils::Vector3d const &r3,
149  Utils::Vector3d const &r4) const {
150  /* vectors for dihedral angle calculation */
151  Utils::Vector3d v12, v23, v34, v12Xv23, v23Xv34;
152  double l_v12Xv23, l_v23Xv34;
153  /* dihedral angle, cosine of the dihedral angle */
154  double phi, cosphi, sinmphi_sinphi;
155  /* force factors */
156  double fac;
157 
158  /* dihedral angle */
159  calc_dihedral_angle(r1, r2, r3, r4, v12, v23, v34, v12Xv23, &l_v12Xv23,
160  v23Xv34, &l_v23Xv34, &cosphi, &phi);
161  /* dihedral angle not defined - force zero */
162  if (phi == -1.0) {
163  return {};
164  }
165 
166  auto const f1 = (v23Xv34 - cosphi * v12Xv23) / l_v12Xv23;
167  auto const f4 = (v12Xv23 - cosphi * v23Xv34) / l_v23Xv34;
168 
169  auto const v23Xf1 = vector_product(v23, f1);
170  auto const v23Xf4 = vector_product(v23, f4);
171  auto const v34Xf4 = vector_product(v34, f4);
172  auto const v12Xf1 = vector_product(v12, f1);
173 
174  /* calculate force magnitude */
175  fac = -bend * mult;
176 
177  if (fabs(sin(phi)) < TINY_SIN_VALUE) {
178  /*(comes from taking the first term of the MacLaurin expansion of
179  sin(n*phi - phi0) and sin(phi) and then making the division).
180  The original code had a 2PI term in the cosine (cos(2PI - nPhi))
181  but I removed it because it wasn't doing anything. AnaVV*/
182  sinmphi_sinphi = mult * cos(mult * phi - phase) / cosphi;
183  } else {
184  sinmphi_sinphi = sin(mult * phi - phase) / sin(phi);
185  }
186 
187  fac *= sinmphi_sinphi;
188 
189  /* store dihedral forces */
190  auto const force1 = fac * v23Xf1;
191  auto const force2 = fac * (v34Xf4 - v12Xf1 - v23Xf1);
192  auto const force3 = fac * (v12Xf1 - v23Xf4 - v34Xf4);
193 
194  return std::make_tuple(force2, force1, force3, -(force1 + force2 + force3));
195 }

197 /** Compute the four-body dihedral interaction energy.
198  *
199  * @param[in] r1 Position of the first particle.
200  * @param[in] r2 Position of the second particle.
201  * @param[in] r3 Position of the third particle.
202  * @param[in] r4 Position of the fourth particle.
203  */
204 inline boost::optional<double>
205 DihedralBond::energy(Utils::Vector3d const &r1, Utils::Vector3d const &r2,
206  Utils::Vector3d const &r3,
207  Utils::Vector3d const &r4) const {
208  /* vectors for dihedral calculations. */
209  Utils::Vector3d v12, v23, v34, v12Xv23, v23Xv34;
210  double l_v12Xv23, l_v23Xv34;
211  /* dihedral angle, cosine of the dihedral angle */
212  double phi, cosphi;
213 
214  calc_dihedral_angle(r1, r2, r3, r4, v12, v23, v34, v12Xv23, &l_v12Xv23,
215  v23Xv34, &l_v23Xv34, &cosphi, &phi);
216  /* dihedral angle not defined - force zero */
217  if (phi == -1.0) {
218  return {};
219  }
220 
221  return bend * (1. - cos(mult * phi - phase));
222 }
