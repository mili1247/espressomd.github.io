32 /** Type codes for the type of %Coulomb interaction.
33  * Enumeration of implemented methods for the electrostatic interaction.
34  */
35 enum CoulombMethod {
36  COULOMB_NONE, ///< %Coulomb interaction switched off
37  COULOMB_DH, ///< %Coulomb method is Debye-Hueckel
38  COULOMB_P3M, ///< %Coulomb method is P3M
39  COULOMB_P3M_GPU, ///< %Coulomb method is P3M with GPU-based long-range part
40  COULOMB_ELC_P3M, ///< %Coulomb method is P3M plus ELC
41  COULOMB_MMM1D, ///< %Coulomb method is one-dimensional MMM
42  COULOMB_RF, ///< %Coulomb method is Reaction-Field
43  COULOMB_MMM1D_GPU, ///< %Coulomb method is one-dimensional MMM running on GPU
44  COULOMB_SCAFACOS, ///< %Coulomb method is ScaFaCoS
45 };
