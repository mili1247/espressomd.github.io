48 /** Calculate Thole force */
49 inline Utils::Vector3d thole_pair_force(Particle const &p1, Particle const &p2,
50  IA_parameters const &ia_params,
51  Utils::Vector3d const &d, double dist) {
52  auto const thole_q1q2 = ia_params.thole.q1q2;
53  auto const thole_s = ia_params.thole.scaling_coeff;
54 
55  if (thole_s != 0 && thole_q1q2 != 0 &&
56  !(pair_bond_enum_exists_between<ThermalizedBond>(p1, p2))) {
57  // Calc damping function (see @cite thole81a)
58  // S(r) = 1.0 - (1.0 + thole_s*r/2.0) * exp(-thole_s*r);
59  // Calc F = - d/dr ( S(r)*q1q2/r) =
60  // -(1/2)*(-2+(r^2*s^2+2*r*s+2)*exp(-s*r))*q1q2/r^2 Everything before
61  // q1q2/r^2 can be used as a factor for the Coulomb::central_force method
62  auto const sr = thole_s * dist;
63  auto const dS_r = 0.5 * (2.0 - (exp(-sr) * (sr * (sr + 2.0) + 2.0)));
64  return Coulomb::central_force(thole_q1q2 * (-1. + dS_r), d, dist);
65  }
66  return {};
67 }

69 /** Calculate Thole energy */
70 inline double thole_pair_energy(Particle const &p1, Particle const &p2,
71  IA_parameters const &ia_params,
72  Utils::Vector3d const &d, double dist) {
73 
74  auto const thole_s = ia_params.thole.scaling_coeff;
75  auto const thole_q1q2 = ia_params.thole.q1q2;
76 
77  if (thole_s != 0 && thole_q1q2 != 0 &&
78  dist < Coulomb::cutoff(box_geo.length()) &&
79  !(pair_bond_enum_exists_between<ThermalizedBond>(p1, p2))) {
80  auto const sd = thole_s * dist;
81  auto const S_r = 1.0 - (1.0 + sd / 2.0) * exp(-sd);
82  // Subtract p3m shortrange energy and add thole energy
83  return Coulomb::pair_energy(p1, p2, thole_q1q2 * (-1.0 + S_r), d, dist,
84  dist * dist);
85  }
86  return 0.0;
87 }
