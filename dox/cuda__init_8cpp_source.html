42 struct CompareDevices {
43  bool operator()(const EspressoGpuDevice &a,
44  const EspressoGpuDevice &b) const {
45  const int name_comp = strncmp(a.proc_name, b.proc_name, 63);
46  /* if both devices are from the same node, order by id */
47  if (name_comp == 0)
48  return a.id < b.id;
49 
50  return name_comp < 0;
51  }
52 };

54 /** Gather list of CUDA devices on all nodes on the master node.
55  * It relies on <tt>MPI_Get_processor_name()</tt> to get a unique identifier
56  * of the physical node, as opposed to the logical rank of which there can
57  * be more than one per node.
58  */
59 static std::vector<EspressoGpuDevice> mpi_cuda_gather_gpus_local() {
60  /* List of local devices */
61  std::vector<EspressoGpuDevice> devices_local;
62  /* Global unique device list (only relevant on master) */
63  std::vector<EspressoGpuDevice> devices_global;
64 
65  int n_devices;
66  try {
67  n_devices = cuda_get_n_gpus();
68  } catch (cuda_runtime_error const &err) {
69  n_devices = 0;
70  }
71 
72  int proc_name_len;
73  char proc_name[MPI_MAX_PROCESSOR_NAME];
74  MPI_Get_processor_name(proc_name, &proc_name_len);
75  if (std::strlen(proc_name) > 63)
76  proc_name[63] = '\0';
77 
78  for (int i = 0; i < n_devices; ++i) {
79  try {
80  EspressoGpuDevice device = cuda_get_device_props(i);
81  std::strncpy(device.proc_name, proc_name, 64);
82  device.proc_name[63] = '\0';
83  device.node = this_node;
84  devices_local.push_back(device);
85  } catch (cuda_runtime_error const &err) {
86  // pass
87  }
88  }
89 
90  int const n_gpus = static_cast<int>(devices_local.size());
91 
92  if (this_node == 0) {
93  std::set<EspressoGpuDevice, CompareDevices> device_set;
94  int *n_gpu_array = new int[n_nodes];
95  MPI_Gather(&n_gpus, 1, MPI_INT, n_gpu_array, 1, MPI_INT, 0, MPI_COMM_WORLD);
96 
97  /* insert local devices */
98  std::copy(devices_local.begin(), devices_local.end(),
99  std::inserter(device_set, device_set.begin()));
100 
101  EspressoGpuDevice device;
102  MPI_Status s;
103  /* Get devices from other nodes */
104  for (int i = 1; i < n_nodes; ++i) {
105  for (int j = 0; j < n_gpu_array[i]; ++j) {
106  MPI_Recv(&device, sizeof(EspressoGpuDevice), MPI_BYTE, i, 0,
107  MPI_COMM_WORLD, &s);
108  device_set.insert(device);
109  }
110  }
111  /* Copy unique devices to result, if any */
112  std::copy(device_set.begin(), device_set.end(),
113  std::inserter(devices_global, devices_global.begin()));
114  delete[] n_gpu_array;
115  } else {
116  /* Send number of devices to master */
117  MPI_Gather(&n_gpus, 1, MPI_INT, nullptr, 1, MPI_INT, 0, MPI_COMM_WORLD);
118  /* Send devices to master */
119  for (auto const &device : devices_local) {
120  MPI_Send(&device, sizeof(EspressoGpuDevice), MPI_BYTE, 0, 0,
121  MPI_COMM_WORLD);
122  }
123  }
124  return devices_global;
125 }
