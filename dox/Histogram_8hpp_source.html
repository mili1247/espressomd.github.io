38 /**
39  * \brief Histogram in Cartesian coordinates.
40  * \tparam T Histogram data type.
41  * \tparam N Histogram data dimensionality.
42  * \tparam M Coordinates data dimensionality.
43  * \tparam U Coordinates data type.
44  */
45 template <typename T, std::size_t N, std::size_t M = 3, typename U = double>
46 class Histogram {
47  using array_type = boost::multi_array<T, M + 1>;
48  using count_type = boost::multi_array<std::size_t, M + 1>;
49  using array_index = typename array_type::index;
50 
51 public:
52  /**
53  * \brief Histogram constructor.
54  * \param n_bins the number of bins in each histogram dimension.
55  * \param limits the minimum/maximum data values to consider for the
56  * histogram.
57  */
58  explicit Histogram(std::array<std::size_t, M> n_bins,
59  std::array<std::pair<U, U>, M> limits)
60  : m_n_bins(std::move(n_bins)), m_limits(std::move(limits)),
61  m_bin_sizes(calc_bin_sizes()), m_array(m_array_dim()),
62  m_count(m_array_dim()) {
63  m_ones.fill(T{1});
64  }
65 
66  /** \brief Get the number of bins for each dimension. */
67  std::array<std::size_t, M> get_n_bins() const { return m_n_bins; }
68 
69  /** \brief Get the histogram data. */
70  std::vector<T> get_histogram() const {
71  return {m_array.data(), m_array.data() + m_array.num_elements()};
72  }
73 
74  /** \brief Get the histogram count data. */
75  std::vector<std::size_t> get_tot_count() const {
76  return {m_count.data(), m_count.data() + m_count.num_elements()};
77  }
78 
79  /** \brief Get the ranges (min, max) for each dimension. */
80  std::array<std::pair<U, U>, M> get_limits() const { return m_limits; }
81 
82  /** \brief Get the bin sizes. */
83  std::array<U, M> get_bin_sizes() const { return m_bin_sizes; }
84 
85  /**
86  * \brief Add data to the histogram.
87  * \param pos Position to update.
88  */
89  void update(Span<const U> pos) { update(pos, m_ones); }
90 
91  /**
92  * \brief Add data to the histogram.
93  * \param pos Position to update.
94  * \param value Value to add.
95  */
96  void update(Span<const U> pos, Span<const T> value) {
97  if (pos.size() != M) {
98  throw std::invalid_argument("Wrong dimensions for the coordinates");
99  }
100  if (value.size() != N) {
101  throw std::invalid_argument("Wrong dimensions for the value");
102  }
103  if (check_limits(pos)) {
104  boost::array<array_index, M + 1> index;
105  for (std::size_t i = 0; i < M; ++i) {
106  index[i] = calc_bin_index(pos[i], m_limits[i].first, m_bin_sizes[i]);
107  }
108  for (array_index i = 0; i < N; ++i) {
109  index.back() = i;
110  m_array(index) += value[i];
111  m_count(index)++;
112  }
113  }
114  }
115 
116  /** \brief Normalize histogram. */
117  virtual void normalize() {
118  auto const bin_volume = std::accumulate(
119  m_bin_sizes.begin(), m_bin_sizes.end(), U{1}, std::multiplies<U>());
120  std::transform(
121  m_array.data(), m_array.data() + m_array.num_elements(), m_array.data(),
122  [bin_volume](T v) { return static_cast<T>(v / bin_volume); });
123  }
124 
125 private:
126  /**
127  * \brief Calculate the bin index.
128  * \param value Position on that dimension.
129  * \param offset Bin offset on that dimension.
130  * \param size Bin size on that dimension.
131  */
132  array_index calc_bin_index(double value, double offset, double size) const {
133  return static_cast<array_index>(std::floor((value - offset) / size));
134  }
135 
136  /**
137  * \brief Calculate the bin sizes.
138  */
139  std::array<U, M> calc_bin_sizes() const {
140  std::array<U, M> bin_sizes;
141  for (std::size_t i = 0; i < M; ++i) {
142  bin_sizes[i] = (m_limits[i].second - m_limits[i].first) /
143  static_cast<U>(m_n_bins[i]);
144  }
145  return bin_sizes;
146  }
147 
148  /**
149  * \brief Check if the position lies within the histogram limits.
150  * \param pos Position to check.
151  */
152  bool check_limits(Span<const U> pos) const {
153  if (pos.size() != M) {
154  throw std::invalid_argument("Wrong dimensions for the coordinates");
155  }
156  bool within_range = true;
157  for (std::size_t i = 0; i < pos.size(); ++i) {
158  if (pos[i] < m_limits[i].first or pos[i] >= m_limits[i].second)
159  within_range = false;
160  }
161  return within_range;
162  }
163 
164  std::array<std::size_t, M + 1> m_array_dim() const {
165  std::array<std::size_t, M + 1> dimensions;
166  std::copy(m_n_bins.begin(), m_n_bins.end(), dimensions.begin());
167  dimensions.back() = N;
168  return dimensions;
169  }
170 
171 protected:
172  /// Number of bins for each dimension.
173  std::array<std::size_t, M> m_n_bins;
174  /// Min and max values for each dimension.
175  std::array<std::pair<U, U>, M> m_limits;
176  /// Bin sizes for each dimension.
177  std::array<U, M> m_bin_sizes;
178  /// Histogram data.
179  array_type m_array;
180  /// Track the number of total hits per bin entry.
181  count_type m_count;
182  std::array<T, N> m_ones;
183 };

185 /**
186  * \brief Histogram in cylindrical coordinates.
187  * \tparam T Histogram data type.
188  * \tparam N Histogram data dimensionality.
189  * \tparam M Coordinates data dimensionality.
190  * \tparam U Coordinates data type.
191  */
192 template <typename T, std::size_t N, std::size_t M = 3, typename U = double>
193 class CylindricalHistogram : public Histogram<T, N, M, U> {
194  using Histogram<T, N, M, U>::m_n_bins;
195  using Histogram<T, N, M, U>::m_limits;
196  using Histogram<T, N, M, U>::m_bin_sizes;
197  using Histogram<T, N, M, U>::m_array;
198 
199 public:
200  using Histogram<T, N, M, U>::Histogram;
201 
202  void normalize() override {
203  auto const min_r = m_limits[0].first;
204  auto const r_bin_size = m_bin_sizes[0];
205  auto const phi_bin_size = m_bin_sizes[1];
206  auto const z_bin_size = m_bin_sizes[2];
207  auto const n_bins_r = m_n_bins[0];
208  for (std::size_t i = 0; i < n_bins_r; i++) {
209  auto const r_left = min_r + static_cast<U>(i) * r_bin_size;
210  auto const r_right = r_left + r_bin_size;
211  auto const bin_volume =
212  (r_right * r_right - r_left * r_left) * z_bin_size * phi_bin_size / 2;
213  auto *begin = m_array[i].origin();
214  std::transform(
215  begin, begin + m_array[i].num_elements(), begin,
216  [bin_volume](T v) { return static_cast<T>(v / bin_volume); });
217  }
218  }
219 };
