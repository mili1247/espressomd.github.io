27  * Template specialization for `Particle`. The traits mechanism is used to get
28  * indirect access to particle properties. This helps making the implementation
29  * of observables independent of the particle type.
30  */
31 template <> struct traits<Particle> {
32  auto position(Particle const &p) const { return p.r.p; }
33  auto velocity(Particle const &p) const { return p.m.v; }
34  auto mass(Particle const &p) const {
35 #ifdef VIRTUAL_SITES
36  // we exclude virtual particles since their mass does not have a meaning
37  if (p.p.is_virtual)
38  return decltype(p.p.mass){};
39 #endif
40  return p.p.mass;
41  }
42  auto charge(Particle const &p) const { return p.p.q; }
43  auto dipole_moment(Particle const &p) const {
44 #if defined(ROTATION) && defined(DIPOLES)
45  return p.calc_dip();
46 #else

48 #endif
