46 bool steepest_descent_step(const ParticleRange &particles) {
47  // Maximal force encountered on node
48  auto f_max = -std::numeric_limits<double>::max();
49 
50  // Iteration over all local particles
51  for (auto &p : particles) {
52  auto f = 0.0;
53 
54  // For all Cartesian coordinates
55  for (int j = 0; j < 3; j++) {
56  // Skip, if coordinate is fixed
57  if (!(p.p.ext_flag & COORD_FIXED(j)))
58  // Skip positional increments of virtual particles
59  if (!p.p.is_virtual) {
60  // Square of force on particle
61  f += Utils::sqr(p.f.f[j]);
62 
63  // Positional increment, crop to maximum allowed by user
64  auto const dp = boost::algorithm::clamp(params.gamma * p.f.f[j],
65  -params.max_displacement,
66  params.max_displacement);
67 
68  // Move particle
69  p.r.p[j] += dp;
70  }
71  }
72 #ifdef ROTATION
73  {
74  // Rotational increment
75  auto const dq = params.gamma * p.f.torque; // Vector parallel to torque
76  auto const t = p.f.torque.norm2();
77 
78  // Normalize rotation axis and compute amount of rotation
79  auto const l = dq.norm();
80  if (l > 0.0) {
81  auto const axis = dq / l;
82  auto const angle = boost::algorithm::clamp(l, -params.max_displacement,
83  params.max_displacement);
84 
85  // Rotate the particle around axis dq by amount l
86  local_rotate_particle(p, axis, angle);
87  }
88 
89  f_max = std::max(f_max, t);
90  }
91 #endif
92  // Note maximum force/torque encountered
93  f_max = std::max(f_max, f);
94  }
95 
96  cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
97 
98  // Synchronize maximum force/torque encountered
99  namespace mpi = boost::mpi;
100  auto const f_max_global =
101  mpi::all_reduce(comm_cart, f_max, mpi::maximum<double>());
102 
103  return sqrt(f_max_global) < params.f_max;
104 }
