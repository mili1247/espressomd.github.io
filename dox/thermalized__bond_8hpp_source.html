84 /** Separately thermalizes the com and distance of a particle pair.
85  * @param[in] p1 First particle.
86  * @param[in] p2 Second particle.
87  * @param[in] dx %Distance between the particles.
88  * @return the forces on @p p1 and @p p2
89  */
90 inline boost::optional<std::tuple<Utils::Vector3d, Utils::Vector3d>>
91 ThermalizedBond::forces(Particle const &p1, Particle const &p2,
92  Utils::Vector3d const &dx) const {
93  // Bond broke?
94  if (r_cut > 0.0 && dx.norm() > r_cut) {
95  return {};
96  }
97 
98  auto const mass_tot = p1.p.mass + p2.p.mass;
99  auto const mass_tot_inv = 1.0 / mass_tot;
100  auto const sqrt_mass_tot = sqrt(mass_tot);
101  auto const sqrt_mass_red = sqrt(p1.p.mass * p2.p.mass / mass_tot);
102  auto const com_vel = mass_tot_inv * (p1.p.mass * p1.m.v + p2.p.mass * p2.m.v);
103  auto const dist_vel = p2.m.v - p1.m.v;
104 
105  extern ThermalizedBondThermostat thermalized_bond;
106  Utils::Vector3d force1{};
107  Utils::Vector3d force2{};
108  auto const noise = Random::noise_uniform<RNGSalt::THERMALIZED_BOND>(
109  thermalized_bond.rng_counter(), thermalized_bond.rng_seed(),
110  p1.p.identity, p2.p.identity);
111 
112  for (int i = 0; i < 3; i++) {
113  double force_lv_com, force_lv_dist;
114 
115  // Langevin thermostat for center of mass
116  if (pref2_com > 0.0) {
117  force_lv_com =
118  -pref1_com * com_vel[i] + sqrt_mass_tot * pref2_com * noise[i];
119  } else {
120  force_lv_com = -pref1_com * com_vel[i];
121  }
122 
123  // Langevin thermostat for distance p1->p2
124  if (pref2_dist > 0.0) {
125  force_lv_dist =
126  -pref1_dist * dist_vel[i] + sqrt_mass_red * pref2_dist * noise[i];
127  } else {
128  force_lv_dist = -pref1_dist * dist_vel[i];
129  }
130  // Add forces
131  force1[i] = p1.p.mass * mass_tot_inv * force_lv_com - force_lv_dist;
132  force2[i] = p2.p.mass * mass_tot_inv * force_lv_com + force_lv_dist;
133  }
134 
135  return std::make_tuple(force1, force2);
136 }
