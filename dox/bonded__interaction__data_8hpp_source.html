113 /** Calculate the maximal cutoff of bonded interactions, required to
114  * determine the cell size for communication.
115  *
116  * Bond angle and dihedral potentials do not contain a cutoff intrinsically.
117  * The cutoff for these potentials depends on the bond length potentials
118  * (it is assumed that particles participating in a bond angle or dihedral
119  * potential are bound to each other by some bond length potential). For bond
120  * angle potentials nothing has to be done. For dihedral potentials the cutoff
121  * is set to twice the maximal cutoff because the particle in which the bond
122  * is stored is only bonded to the first two partners, one of which has an
123  * additional bond to the third partner.
124  */
125 double maximal_cutoff_bonded();
126 
127 /** Return the number of bonded partners for the specified bond */
128 inline int number_of_partners(Bonded_IA_Parameters const &iaparams) {
129  return boost::apply_visitor(BondNumPartners(), iaparams);
