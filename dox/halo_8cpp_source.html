45 /** Set halo region to a given value
46  * @param[out] dest pointer to the halo buffer
47  * @param value integer value to write into the halo buffer
48  * @param type halo field layout description
49  */
50 void halo_dtset(char *dest, int value, std::shared_ptr<FieldType> type) {
51  auto const vblocks = type->vblocks;
52  auto const vstride = type->vstride;
53  auto const vskip = type->vskip;
54  auto const &lens = type->lengths;
55  auto const &disps = type->disps;
56  auto const extent = type->extent;
57  auto const block_size = static_cast<long>(vskip) * static_cast<long>(extent);
58 
59  for (int i = 0; i < vblocks; i++) {
60  for (int j = 0; j < vstride; j++) {
61  for (std::size_t k = 0; k < disps.size(); k++)
62  memset(dest + disps[k], value, lens[k]);
63  }
64  dest += block_size;
65  }
66 }

71 void halo_copy_vector(char *r_buffer, char *s_buffer, int count,
72  std::shared_ptr<FieldType> type, bool vflag) {
73 
74  auto const vblocks = type->vblocks;
75  auto const vstride = type->vstride;
76  auto const extent = type->extent;
77 
78  auto block_size = static_cast<long>(type->vskip);
79  if (vflag) {
80  block_size *= static_cast<long>(type->subtype->extent);
81  }
82 
83  for (int i = 0; i < count; i++, s_buffer += extent, r_buffer += extent) {
84  char *dest = r_buffer, *src = s_buffer;
85  for (int j = 0; j < vblocks; j++, dest += block_size, src += block_size) {
86  halo_dtcopy(dest, src, vstride, type->subtype);
87  }
88  }
89 }

91 /** Copy lattice data with layout described by @p type.
92  * @param r_buffer data destination
93  * @param s_buffer data source
94  * @param count amount of data to copy
95  * @param type field layout type
96  */
97 void halo_dtcopy(char *r_buffer, char *s_buffer, int count,
98  std::shared_ptr<FieldType> type) {
99 
100  if (type->subtype) {
101  halo_copy_vector(r_buffer, s_buffer, count, type, type->vflag);
102  } else {
103 
104  for (int i = 0; i < count;
105  i++, s_buffer += type->extent, r_buffer += type->extent) {
106  if (!type->count) {
107  memmove(r_buffer, s_buffer, type->extent);
108  } else {
109  for (int j = 0; j < type->count; j++) {
110  memmove(r_buffer + type->disps[j], s_buffer + type->disps[j],
111  type->lengths[j]);
112  }
113  }
114  }
115  }

118 void prepare_halo_communication(HaloCommunicator &hc, const Lattice &lattice,
119  MPI_Datatype datatype,
120  const Utils::Vector3i &local_node_grid) {
121 
122  const auto &grid = lattice.grid;
123  const auto &period = lattice.halo_grid;
124 
125  for (int n = 0; n < hc.num; n++) {
126  MPI_Type_free(&(hc.halo_info[n].datatype));
127  }
128 
129  int const num = 2 * 3; /* two communications in each space direction */
130  hc.num = num;
131  hc.halo_info.resize(num);
132 
133  auto const extent = static_cast<long>(fieldtype_double->extent);
134 
135  auto const node_neighbors = calc_node_neighbors(comm_cart);
136 
137  int cnt = 0;
138  for (int dir = 0; dir < 3; dir++) {
139  for (int lr = 0; lr < 2; lr++) {
140 
141  HaloInfo &hinfo = hc.halo_info[cnt];
142 
143  int nblocks = 1;
144  for (int k = dir + 1; k < 3; k++) {
145  nblocks *= period[k];
146  }
147  int stride = 1;
148  for (int k = 0; k < dir; k++) {
149  stride *= period[k];
150  }
151  int skip = 1;
152  for (int k = 0; k < dir + 1 && k < 2; k++) {
153  skip *= period[k];
154  }
155 
156  if (lr == 0) {
157  /* send to left, recv from right */
158  hinfo.s_offset = extent * static_cast<long>(stride * 1);
159  hinfo.r_offset = extent * static_cast<long>(stride * (grid[dir] + 1));
160  } else {
161  /* send to right, recv from left */
162  hinfo.s_offset = extent * static_cast<long>(stride * grid[dir]);
163  hinfo.r_offset = extent * static_cast<long>(stride * 0);
164  }
165 
166  hinfo.source_node = node_neighbors[2 * dir + 1 - lr];
167  hinfo.dest_node = node_neighbors[2 * dir + lr];
168 
169  hinfo.fieldtype = std::make_shared<FieldType>(nblocks, stride, skip, true,
170  fieldtype_double);
171 
172  MPI_Type_vector(nblocks, stride, skip, datatype, &hinfo.datatype);
173  MPI_Type_commit(&hinfo.datatype);
174 
175  if (!box_geo.periodic(dir) &&
176  (local_geo.boundary()[2 * dir + lr] != 0 ||
177  local_geo.boundary()[2 * dir + 1 - lr] != 0)) {
178  if (local_node_grid[dir] == 1) {
179  hinfo.type = HALO_OPEN;
180  } else if (lr == 0) {
181  if (local_geo.boundary()[2 * dir + lr] == 1) {
182  hinfo.type = HALO_RECV;
183  } else {
184  hinfo.type = HALO_SEND;
185  }
186  } else {
187  if (local_geo.boundary()[2 * dir + lr] == -1) {
188  hinfo.type = HALO_RECV;
189  } else {
190  hinfo.type = HALO_SEND;
191  }
192  }
193  } else {
194  if (local_node_grid[dir] == 1) {
195  hc.halo_info[cnt].type = HALO_LOCL;
196  } else {
197  hc.halo_info[cnt].type = HALO_SENDRECV;
198  }
199  }
200  cnt++;
201  }
202  }
203 }

211 void halo_communication(const HaloCommunicator &hc, char *const base) {
212 
213  std::shared_ptr<FieldType> fieldtype;
214  MPI_Datatype datatype;
215  MPI_Request request;
216  MPI_Status status;
217 
218  for (int n = 0; n < hc.num; n++) {
219  int s_node, r_node;
220  int comm_type = hc.halo_info[n].type;
221  char *s_buffer = (char *)base + hc.halo_info[n].s_offset;
222  char *r_buffer = (char *)base + hc.halo_info[n].r_offset;
223 
224  switch (comm_type) {
225 
226  case HALO_LOCL:
227  fieldtype = hc.halo_info[n].fieldtype;
228  halo_dtcopy(r_buffer, s_buffer, 1, fieldtype);
229  break;
230 
231  case HALO_SENDRECV:
232  datatype = hc.halo_info[n].datatype;
233  s_node = hc.halo_info[n].source_node;
234  r_node = hc.halo_info[n].dest_node;
235  MPI_Sendrecv(s_buffer, 1, datatype, r_node, REQ_HALO_SPREAD, r_buffer, 1,
236  datatype, s_node, REQ_HALO_SPREAD, comm_cart, &status);
237  break;
238 
239  case HALO_SEND:
240  datatype = hc.halo_info[n].datatype;
241  fieldtype = hc.halo_info[n].fieldtype;
242  r_node = hc.halo_info[n].dest_node;
243  MPI_Isend(s_buffer, 1, datatype, r_node, REQ_HALO_SPREAD, comm_cart,
244  &request);
245  halo_dtset(r_buffer, 0, fieldtype);
246  MPI_Wait(&request, &status);
247  break;
248 
249  case HALO_RECV:
250  datatype = hc.halo_info[n].datatype;
251  s_node = hc.halo_info[n].source_node;
252  MPI_Irecv(r_buffer, 1, datatype, s_node, REQ_HALO_SPREAD, comm_cart,
253  &request);
254  MPI_Wait(&request, &status);
255  break;
256 
257  case HALO_OPEN:
258  fieldtype = hc.halo_info[n].fieldtype;
259  /** \todo this does not work for the n_i - <n_i> */
260  halo_dtset(r_buffer, 0, fieldtype);
261  break;
262  }
263  }
