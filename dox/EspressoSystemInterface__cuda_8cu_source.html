37 __global__ void split_kernel_rq(CUDA_particle_data *particles, float *r,
38  float *q, unsigned int n) {
39  auto const idx = blockDim.x * blockIdx.x + threadIdx.x;
40  if (idx >= n)
41  return;
42 
43  CUDA_particle_data p = particles[idx];
44 
45  r[3 * idx + 0] = p.p[0];
46  r[3 * idx + 1] = p.p[1];
47  r[3 * idx + 2] = p.p[2];
48 #ifdef ELECTROSTATICS
49  q[idx] = p.q;
50 #endif
51 }

136 void EspressoSystemInterface::reallocDeviceMemory(std::size_t n) {

138  if (m_r_gpu_begin != nullptr)
139  cuda_safe_mem(cudaFree(m_r_gpu_begin));

176 void EspressoSystemInterface::split_particle_struct() {
177  auto const device_particles = gpu_get_particle_pointer();
178  auto const n = static_cast<unsigned int>(device_particles.size());
179  if (n == 0)
180  return;
181 
182  dim3 grid(n / 512 + 1, 1, 1);
183  dim3 block(512, 1, 1);
184 
185  if (m_needsQGpu && m_needsRGpu)
186  split_kernel_rq<<<dim3(grid), dim3(block), 0, nullptr>>>(
187  device_particles.data(), m_r_gpu_begin, m_q_gpu_begin, n);
188  if (m_needsQGpu && !m_needsRGpu)
189  split_kernel_q<<<dim3(grid), dim3(block), 0, nullptr>>>(
190  device_particles.data(), m_q_gpu_begin, n);
191  if (!m_needsQGpu && m_needsRGpu)
192  split_kernel_r<<<dim3(grid), dim3(block), 0, nullptr>>>(
193  device_particles.data(), m_r_gpu_begin, n);
194 #ifdef CUDA
195  if (m_needsVGpu)
196  split_kernel_v<<<dim3(grid), dim3(block), 0, nullptr>>>(
197  device_particles.data(), m_v_gpu_begin, n);
198 #endif
199 #ifdef DIPOLES
200  if (m_needsDipGpu)
201  split_kernel_dip<<<dim3(grid), dim3(block), 0, nullptr>>>(
202  device_particles.data(), m_dip_gpu_begin, n);
203 
204 #endif
205 
206  if (m_needsDirectorGpu)
207  split_kernel_director<<<dim3(grid), dim3(block), 0, nullptr>>>(
208  device_particles.data(), m_director_gpu_begin, n);
209 }
