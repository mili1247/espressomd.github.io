86 /**
87  * @brief Resize a @ref device_vector.
88  *
89  * Due to a bug in thrust (https://github.com/thrust/thrust/issues/939),
90  * resizing or appending to default constructed containers causes undefined
91  * behavior by dereferencing a null-pointer for certain types. This
92  * function is used instead of the resize member function to side-step
93  * the problem. This is done by replacing the existing vector by a new
94  * one constructed with the desired size if resizing from capacity zero.
95  * Behaves as-if vec.resize(n) was called.
96  *
97  * @tparam T Type contained in the vector.
98  * @param vec Vector To resize.
99  * @param n Desired new size of the element.
100  */
101 template <class T>
102 void resize_or_replace(device_vector<T> &vec, std::size_t n) {
103  if (vec.capacity() == 0) {
104  vec = device_vector<T>(n);
105  } else {
106  vec.resize(n);
107  }
108 }

123 /**
124  * @brief Setup and call particle reallocation from the host.
125  * Note that in addition to calling this function the parameters must be
126  * broadcast with either:
127  * 1. @ref cuda_bcast_global_part_params() (when just being executed on the
128  * head node) or
129  * 2. `MPI_Bcast(gpu_get_global_particle_vars_pointer_host(),
130  * sizeof(CUDA_global_part_vars), MPI_BYTE, 0, comm_cart)` (when executed
131  * on all nodes)
132  */
133 void gpu_init_particle_comm() {
134  if (this_node == 0 && global_part_vars_host.communication_enabled == 0) {
135  try {
136  if (cuda_get_n_gpus() == 0) {
137  fprintf(stderr, "ERROR: No GPU was found.\n");
138  errexit();
139  }
140  auto const devID = cuda_get_device();
141  auto const compute_capability = cuda_check_gpu_compute_capability(devID);
142  auto const communication_test = cuda_test_device_access();
143  if (compute_capability != ES_OK or communication_test != ES_OK) {
144  fprintf(stderr,
145  "ERROR: CUDA device %i is not capable of running ESPResSo.\n",
146  devID);
147  errexit();
148  }
149  } catch (cuda_runtime_error const &err) {
150  fprintf(stderr, "ERROR: %s\n", err.what());
151  errexit();
152  }
153  }
154  global_part_vars_host.communication_enabled = 1;
155 }

171 void copy_part_data_to_gpu(ParticleRange particles) {
172  if (global_part_vars_host.communication_enabled == 1) {
173  cuda_mpi_get_particles(particles, particle_data_host);
174 
175  resize_buffers(particle_data_host.size());
176 
177  /* get espressomd particle values */
178  if (this_node == 0) {
179  cudaMemsetAsync(raw_data_pointer(particle_forces_device), 0x0,
180  byte_size(particle_forces_device), stream[0]);
181 #ifdef ROTATION
182  cudaMemsetAsync(raw_data_pointer(particle_torques_device), 0x0,
183  byte_size(particle_torques_device), stream[0]);
184 #endif
185  cudaMemcpyAsync(raw_data_pointer(particle_data_device),
186  particle_data_host.data(), byte_size(particle_data_host),
187  cudaMemcpyHostToDevice, stream[0]);
188  }
189  }
190 }

192 /** setup and call kernel to copy particle forces to host
193  */
194 void copy_forces_from_GPU(ParticleRange &particles) {
195  if (global_part_vars_host.communication_enabled == 1) {
196  /* Copy result from device memory to host memory*/
197  if (this_node == 0 && (not particle_forces_device.empty())) {
198  thrust::copy(particle_forces_device.begin(), particle_forces_device.end(),
199  particle_forces_host.begin());
200 #ifdef ROTATION
201  thrust::copy(particle_torques_device.begin(),
202  particle_torques_device.end(),
203  particle_torques_host.begin());
204 #endif
205  }
206 
207  cuda_mpi_send_forces(
208  particles, {particle_forces_host.data(), particle_forces_host.size()},
209  {particle_torques_host.data(), particle_torques_host.size()});
210  }
211 }

221 CUDA_energy copy_energy_from_GPU() {

223  return {};

229 void cuda_safe_mem_exit(cudaError_t CU_err, const char *file,
230  unsigned int line) {
231  if (CU_err != cudaSuccess) {
232  fprintf(stderr, "CUDA Memory error at %s:%u.\n", file, line);
233  fprintf(stderr, "CUDA error: %s\n", cudaGetErrorString(CU_err));
234  if (CU_err == cudaErrorInvalidValue)
235  fprintf(stderr, "You may have tried to allocate zero memory at %s:%u.\n",
236  file, line);
237  errexit();
238  } else {
239  CU_err = cudaGetLastError();
240  if (CU_err != cudaSuccess) {
241  fprintf(stderr,
242  "Error found during memory operation. Possibly however "
243  "from a failed operation before. %s:%u.\n",
244  file, line);
245  printf("CUDA error: %s\n", cudaGetErrorString(CU_err));
246  if (CU_err == cudaErrorInvalidValue)
247  fprintf(stderr,
248  "You may have tried to allocate zero memory before %s:%u.\n",
249  file, line);
250  errexit();
251  }
252  }
