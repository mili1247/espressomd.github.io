39 void p3m_add_block(double const *in, double *out, int const start[3],
40  int const size[3], int const dim[3]) {
41  /* fast,mid and slow changing indices */
42  int f, m, s;
43  /* linear index of in grid, linear index of out grid */
44  int li_in = 0, li_out = 0;
45  /* offsets for indices in output grid */
46  int m_out_offset, s_out_offset;
47 
48  li_out = start[2] + (dim[2] * (start[1] + (dim[1] * start[0])));
49  m_out_offset = dim[2] - size[2];
50  s_out_offset = (dim[2] * (dim[1] - size[1]));
51 
52  for (s = 0; s < size[0]; s++) {
53  for (m = 0; m < size[1]; m++) {
54  for (f = 0; f < size[2]; f++) {
55  out[li_out++] += in[li_in++];
56  }
57  li_out += m_out_offset;
58  }
59  li_out += s_out_offset;
60  }
61 }

63 double p3m_analytic_cotangent_sum(int n, double mesh_i, int cao) {
64  double c, res = 0.0;
65  c = Utils::sqr(cos(Utils::pi() * mesh_i * (double)n));
66 
67  switch (cao) {
68  case 1: {
69  res = 1;
70  break;
71  }
72  case 2: {
73  res = (1.0 + c * 2.0) / 3.0;
74  break;
75  }
76  case 3: {
77  res = (2.0 + c * (11.0 + c * 2.0)) / 15.0;
78  break;
79  }
80  case 4: {
81  res = (17.0 + c * (180.0 + c * (114.0 + c * 4.0))) / 315.0;
82  break;
83  }
84  case 5: {
85  res = (62.0 + c * (1072.0 + c * (1452.0 + c * (247.0 + c * 2.0)))) / 2835.0;
86  break;
87  }
88  case 6: {
89  res = (1382.0 +
90  c * (35396.0 +
91  c * (83021.0 + c * (34096.0 + c * (2026.0 + c * 4.0))))) /
92  155925.0;
93  break;
94  }
95  case 7: {
96  res =
97  (21844.0 +
98  c * (776661.0 + c * (2801040.0 +
99  c * (2123860.0 +
100  c * (349500.0 + c * (8166.0 + c * 4.0)))))) /
101  6081075.0;
102  break;
103  }
104  default: {
105  fprintf(stderr,
106  "%d: INTERNAL_ERROR: The value %d for the interpolation order "
107  "should not occur!\n",
108  this_node, cao);
109  errexit();
110  }
111  }
112 
113  return res;
114 }

116 void p3m_calc_local_ca_mesh(P3MLocalMesh &local_mesh,
117  const P3MParameters &params,
118  const LocalBox<double> &local_geo, double skin,
119  double space_layer) {
120  int i;
121  int ind[3];
122  /* total skin size */
123  double full_skin[3];
124 
125  for (i = 0; i < 3; i++)
126  full_skin[i] = params.cao_cut[i] + skin;
127 
128  full_skin[2] += space_layer;
129 
130  /* inner left down grid point (global index) */
131  for (i = 0; i < 3; i++)
132  local_mesh.in_ld[i] =
133  (int)ceil(local_geo.my_left()[i] * params.ai[i] - params.mesh_off[i]);
134  /* inner up right grid point (global index) */
135  for (i = 0; i < 3; i++)
136  local_mesh.in_ur[i] =
137  (int)floor(local_geo.my_right()[i] * params.ai[i] - params.mesh_off[i]);
138 
139  /* correct roundoff errors at boundary */
140  for (i = 0; i < 3; i++) {
141  if ((local_geo.my_right()[i] * params.ai[i] - params.mesh_off[i]) -
142  local_mesh.in_ur[i] <
143  ROUND_ERROR_PREC)
144  local_mesh.in_ur[i]--;
145  if (1.0 + (local_geo.my_left()[i] * params.ai[i] - params.mesh_off[i]) -
146  local_mesh.in_ld[i] <
147  ROUND_ERROR_PREC)
148  local_mesh.in_ld[i]--;
149  }
150  /* inner grid dimensions */
151  for (i = 0; i < 3; i++)
152  local_mesh.inner[i] = local_mesh.in_ur[i] - local_mesh.in_ld[i] + 1;
153  /* index of left down grid point in global mesh */
154  for (i = 0; i < 3; i++)
155  local_mesh.ld_ind[i] =
156  (int)ceil((local_geo.my_left()[i] - full_skin[i]) * params.ai[i] -
157  params.mesh_off[i]);
158  /* left down margin */
159  for (i = 0; i < 3; i++)
160  local_mesh.margin[i * 2] = local_mesh.in_ld[i] - local_mesh.ld_ind[i];
161  /* up right grid point */
162  for (i = 0; i < 3; i++)
163  ind[i] =
164  (int)floor((local_geo.my_right()[i] + full_skin[i]) * params.ai[i] -
165  params.mesh_off[i]);
166  /* correct roundoff errors at up right boundary */
167  for (i = 0; i < 3; i++)
168  if (((local_geo.my_right()[i] + full_skin[i]) * params.ai[i] -
169  params.mesh_off[i]) -
170  ind[i] ==
171  0)
172  ind[i]--;
173  /* up right margin */
174  for (i = 0; i < 3; i++)
175  local_mesh.margin[(i * 2) + 1] = ind[i] - local_mesh.in_ur[i];
176 
177  /* grid dimension */
178  local_mesh.size = 1;
179  for (i = 0; i < 3; i++) {
180  local_mesh.dim[i] = ind[i] - local_mesh.ld_ind[i] + 1;
181  local_mesh.size *= local_mesh.dim[i];
182  }
183  /* reduce inner grid indices from global to local */
184  for (i = 0; i < 3; i++)
185  local_mesh.in_ld[i] = local_mesh.margin[i * 2];
186  for (i = 0; i < 3; i++)
187  local_mesh.in_ur[i] = local_mesh.margin[i * 2] + local_mesh.inner[i];
188 
189  local_mesh.q_2_off = local_mesh.dim[2] - params.cao;
190  local_mesh.q_21_off = local_mesh.dim[2] * (local_mesh.dim[1] - params.cao);
191 }
