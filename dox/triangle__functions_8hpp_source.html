30 /**
31  * @brief Computes the normal vector of a triangle.
32  *
33  * The sign convention is such that P1P2, P1P3 and
34  * the normal form a right-handed system.
35  * The normal vector is not normalized, i.e. its length
36  * is arbitrary.
37  */
38 inline Vector3d get_n_triangle(const Vector3d &P1, const Vector3d &P2,
39  const Vector3d &P3) {
40  auto const u = P2 - P1;
41  auto const v = P3 - P1;
42 
43  return vector_product(u, v);
44 }

46 /** Computes the area of triangle between vectors P1,P2,P3, by computing
47  * the cross product P1P2 x P1P3 and taking the half of its norm.
48  */
49 inline double area_triangle(const Vector3d &P1, const Vector3d &P2,
50  const Vector3d &P3) {
51  return 0.5 * get_n_triangle(P1, P2, P3).norm();
52 }

54 /**
55  * @brief Computes the angle between two triangles in 3D space
56  *
57  * Returns the angle between two triangles in 3D space given by points P1P2P3
58  * and P2P3P4. Note, that the common edge is given as the second and the third
59  * argument. Here, the angle can have values from 0 to 2 * PI, depending on the
60  * orientation of the two triangles. So the angle can be convex or concave. For
61  * each triangle, an inward direction has been defined as the direction of one
62  * of the two normal vectors. Particularly, for triangle P1P2P3 it is the vector
63  * N1 = P2P1 x P2P3 and for triangle P2P3P4 it is N2 = P2P3 x P2P4. The method
64  * first computes the angle between N1 and N2, which gives always value between
65  * 0 and PI and then it checks whether this value must be corrected to a value
66  * between PI and 2 * PI.
67  *
68  * As an example, consider 4 points A,B,C,D in space given by coordinates
69  * A = [1,1,1], B = [2,1,1], C = [1,2,1], D = [1,1,2]. We want to determine
70  * the angle between triangles ABC and ACD. In case the orientation of the
71  * triangle ABC is [0,0,1] and orientation of ACD is [1,0,0], the resulting
72  * angle must be PI/2.0. To get correct results, note that the common edge is
73  * AC, and one must call the method as <tt>angle_btw_triangles(B,A,C,D)</tt>.
74  * With this call we have ensured that N1 = AB x AC (which coincides with
75  * [0,0,1]) and N2 = AC x AD (which coincides with [1,0,0]). Alternatively,
76  * if the orientations of the two triangles were the opposite, the correct
77  * call would be <tt>angle_btw_triangles(B,C,A,D)</tt> so that N1 = CB x CA
78  * and N2 = CA x CD.
79  */
80 inline double angle_btw_triangles(const Vector3d &P1, const Vector3d &P2,
81  const Vector3d &P3, const Vector3d &P4) {
82  auto const normal1 = get_n_triangle(P2, P1, P3);
83  auto const normal2 = get_n_triangle(P2, P3, P4);
84  auto const cosine = boost::algorithm::clamp(
85  normal1 * normal2 / std::sqrt(normal1.norm2() * normal2.norm2()), -1.0,
86  1.0);
87  // The angle between the faces (not considering
88  // the orientation, always less or equal to Pi)
89  // is equal to Pi minus angle between the normals
90  auto const phi = Utils::pi() - std::acos(cosine);
91 
92  // Now we need to determine, if the angle between two triangles is less than
93  // Pi or greater than Pi. To do this, we check if the point P4 lies in the
94  // halfspace given by triangle P1P2P3 and the normal to this triangle. If yes,
95  // we have an angle less than Pi, otherwise we have an angle greater than Pi.
96  // General equation of the plane is n_x*x + n_y*y + n_z*z + d = 0 where
97  // (n_x,n_y,n_z) is the normal to the plane.
98  // Point P1 lies in the plane, therefore d = -(n_x*P1_x + n_y*P1_y + n_z*P1_z)
99  // Point P4 lies in the halfspace given by normal iff n_x*P4_x + n_y*P4_y +
100  // n_z*P4_z + d >= 0
101  if (normal1 * P4 - normal1 * P1 < 0)
102  return 2 * Utils::pi() - phi;
103 
104  return phi;
105 }
