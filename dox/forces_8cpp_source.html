56 void init_forces(const ParticleRange &particles, double time_step, double kT) {
57  ESPRESSO_PROFILER_CXX_MARK_FUNCTION;
58  /* The force initialization depends on the used thermostat and the
59  thermodynamic ensemble */
60 
61 #ifdef NPT
62  npt_reset_instantaneous_virials();
63 #endif
64 
65  /* initialize forces with Langevin thermostat forces
66  or zero depending on the thermostat
67  set torque to zero for all and rescale quaternions
68  */
69  for (auto &p : particles) {
70  p.f = init_local_particle_force(p, time_step, kT);
71  }
72 
73  /* initialize ghost forces with zero
74  set torque to zero for all and rescale quaternions
75  */
76  for (auto &p : cell_structure.ghost_particles()) {
77  p.f = init_ghost_force(p);
78  }
79 }

87 void force_calc(CellStructure &cell_structure, double time_step, double kT) {
88  ESPRESSO_PROFILER_CXX_MARK_FUNCTION;
89 
90  espressoSystemInterface.update();
91 
92 #ifdef COLLISION_DETECTION
93  prepare_local_collision_queue();
94 #endif
95 
96  auto particles = cell_structure.local_particles();
97  auto ghost_particles = cell_structure.ghost_particles();
98 #ifdef ELECTROSTATICS
99  icc_iteration(particles, cell_structure.ghost_particles());
100 #endif
101  init_forces(particles, time_step, kT);
102 
103  for (auto &forceActor : forceActors) {
104  forceActor->computeForces(espressoSystemInterface);
105 #ifdef ROTATION
106  forceActor->computeTorques(espressoSystemInterface);
107 #endif
108  }
109 
110  calc_long_range_forces(particles);
111 
112 #ifdef ELECTROSTATICS
113  auto const coulomb_cutoff = Coulomb::cutoff(box_geo.length());
114 #else
115  auto const coulomb_cutoff = INACTIVE_CUTOFF;
116 #endif
117 
118 #ifdef DIPOLES
119  auto const dipole_cutoff = Dipole::cutoff(box_geo.length());
120 #else
121  auto const dipole_cutoff = INACTIVE_CUTOFF;
122 #endif
123 
124  short_range_loop(
125  add_bonded_force,
126  [](Particle &p1, Particle &p2, Distance const &d) {
127  add_non_bonded_pair_force(p1, p2, d.vec21, sqrt(d.dist2), d.dist2);
128 #ifdef COLLISION_DETECTION
129  if (collision_params.mode != COLLISION_MODE_OFF)
130  detect_collision(p1, p2, d.dist2);
131 #endif
132  },
133  maximal_cutoff(), maximal_cutoff_bonded(),
134  VerletCriterion{skin, interaction_range(), coulomb_cutoff, dipole_cutoff,
135  collision_detection_cutoff()});
136 
137  Constraints::constraints.add_forces(particles, get_sim_time());
138 
139  if (max_oif_objects) {
140  // There are two global quantities that need to be evaluated:
141  // object's surface and object's volume. One can add another
142  // quantity.
143  Utils::Vector2d area_volume;
144  area_volume[0] = 0.0;
145  area_volume[1] = 0.0;
146  for (int i = 0; i < max_oif_objects; i++) {
147  calc_oif_global(area_volume, i, cell_structure);
148  if (fabs(area_volume[0]) < 1e-100 && fabs(area_volume[1]) < 1e-100)
149  break;
150  add_oif_global_forces(area_volume, i, cell_structure);
151  }
152  }
153 
154  // Must be done here. Forces need to be ghost-communicated
155  immersed_boundaries.volume_conservation(cell_structure);
156 
157  lb_lbcoupling_calc_particle_lattice_ia(thermo_virtual, particles,
158  ghost_particles, time_step);
159 
160 #ifdef CUDA
161  copy_forces_from_GPU(particles);
162 #endif
163 
164 // VIRTUAL_SITES distribute forces
165 #ifdef VIRTUAL_SITES
166  virtual_sites()->back_transfer_forces_and_torques();
167 #endif
168 
169  // Communication Step: ghost forces
170  cell_structure.ghosts_reduce_forces();
171 
172  // should be pretty late, since it needs to zero out the total force
173  comfixed.apply(comm_cart, particles);
174 
175  // Needs to be the last one to be effective
176  forcecap_cap(particles);
177 
178  // mark that forces are now up-to-date
179  recalc_forces = false;
180 }
