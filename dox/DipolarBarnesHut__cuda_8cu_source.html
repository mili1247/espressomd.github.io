53 // The "half-convolution" multi-thread reduction.
54 // The thread with a lower index will operate longer and
55 // final result (here: the sum) is flowing down towards zero thread.
56 __device__ void dds_sumReduction_BH(float *input, float *sum) {
57  auto tid = static_cast<int>(threadIdx.x);
58  for (auto i = static_cast<int>(blockDim.x); i > 1; i /= 2) {
59  __syncthreads();
60  if (tid < i / 2)
61  input[tid] += input[i / 2 + tid];
62  if ((i % 2 == 1) && (tid == 0))
63  input[tid] += input[i - 1];
64  }
65  __syncthreads();
66  if (threadIdx.x == 0) {
67  sum[0] = input[0];
68  }
69 }

90 __global__ __launch_bounds__(THREADS1, FACTOR1) void boundingBoxKernel() {
91  int i, j, k, l, t, inc, n_blocks;
92  float val;
93  // min/max positions per the thread:
94  float minp[3], maxp[3];
95  // min/max positions per block:
96  __shared__ float smin[3 * THREADS1], smax[3 * THREADS1];
97  for (l = 0; l < 3; l++) {
98  minp[l] = maxp[l] = bhpara->r[l];
99  }
100 
101  // Scan all bodies.
102  // In order to iterate over all bodies assigned to thread,
103  // it is necessary to step over all threads in the GPU:
104  // inc = [number of blocks: gridDim.x] * [THREADS1 per block within given
105  // kernel]. Hence, this approach could handle an infinite number of bodies
106  // (particles)
107  i = static_cast<int>(threadIdx.x);
108  inc = THREADS1 * gridDim.x;
109  // j is an absolute index of the particle.
110  // It is shifted over a count of the passed block threads behind: blockIdx.x *
111  // THREADS1. NOTE: this loop is extrema search among all particles of the
112  // given thread in the present block. However, one is not among all threads of
113  // this block.
114  for (j = i + static_cast<int>(blockIdx.x) * THREADS1; j < bhpara->nbodies;
115  j += inc)
116  for (l = 0; l < 3; l++) {
117  val = bhpara->r[3 * j + l];
118  minp[l] = min(minp[l], val);
119  maxp[l] = max(maxp[l], val);
120  }
121 
122  // For a start point of a reduction in the given block shared memory
123  // of the i-th thread extrema:
124  for (l = 0; l < 3; l++) {
125  smin[3 * i + l] = minp[l];
126  smax[3 * i + l] = maxp[l];
127  }
128 
129  // Now it's time to (min)maximize among all threads of the given block.
130  // Each mim/max operation will be applied between
131  // the shared memory smin/smax and the given thread.
132  // The "half-convolution" multi-thread reduction
133  // the thread with a lower index will operate longer and
134  // final result (here: the shared memory extrema)
135  // is flowing down towards zero thread.
136  for (t = THREADS1 / 2; t > 0; t /= 2) {
137  __syncthreads();
138  if (i < t) {
139  k = i + t;
140  for (l = 0; l < 3; l++) {
141  smin[3 * i + l] = minp[l] = min(minp[l], smin[3 * k + l]);
142  smax[3 * i + l] = maxp[l] = max(maxp[l], smax[3 * k + l]);
143  // very last minp/maxp assignment will be made by zero thread (i == 0)
144  }
145  }
146  }
147 
148  // Thread i == 0 is responsible for a writing
149  // of a given block result into the global memory
150  // and other per-block operations.
151  if (i == 0) {
152  // per k-th block
153  k = static_cast<int>(blockIdx.x);
154  for (l = 0; l < 3; l++) {
155  // global memory storage of the per-block extrema
156  bhpara->minp[3 * k + l] = minp[l];
157  bhpara->maxp[3 * k + l] = maxp[l];
158  // note, that we are in zero thread and its variables minp/maxp
159  // contain de facto already reduced (see above) shared extrema smin/smax
160  }
161 
162  n_blocks = static_cast<int>(gridDim.x) - 1;
163  // The block increment is performing by its zero thread.
164  if (n_blocks == atomicInc((unsigned int *)&blkcntd, n_blocks)) {
165  // I'm the (randomly) last block, so combine all other blocks' results
166  // over the index j:
167  for (j = 0; j <= n_blocks; j++)
168  for (l = 0; l < 3; l++) {
169  minp[l] = min(minp[l], bhpara->minp[3 * j + l]);
170  maxp[l] = max(maxp[l], bhpara->maxp[3 * j + l]);
171  }
172 
173  // Compute 'radius':
174  val = max(maxp[0] - minp[0], maxp[1] - minp[1]);
175  radiusd = max(val, maxp[2] - minp[2]) * 0.5f;
176 
177  // NOTE: now the extrema are global.
178  // Present code fragment will be executed once: in zero thread of the last
179  // block.
180 
181  k = bhpara->nnodes;
182  // Create the root node of the Barnes-Hut octree.
183  // Bottom node is defined with max possible index just to start
184  // It will be updated within further tree building in
185  // corresponding kernel.
186  bottomd = k;
187  // Weight of the root node init.
188  bhpara->mass[k] = -1.0f;
189  // Sorting init for the tree root.
190  bhpara->start[k] = 0;
191  // Position of the root node should be in the center of just defined BH
192  // box:
193  for (l = 0; l < 3; l++)
194  bhpara->r[3 * k + l] = (minp[l] + maxp[l]) * 0.5f;
195  // Init further tree building octo- meaning their absence at the
196  // beginning:
197  for (i = 0; i < 8; i++)
198  bhpara->child[8 * k + i] = -1;
199  }
200  }
201 }

207 __global__ __launch_bounds__(THREADS2, FACTOR2) void treeBuildingKernel() {
208  int i, j, k, l, depth, localmaxdepth, skip, inc, lps;
209  float r;
210  float pos[3];
211  float p[3];
212  int ch, n, cell, locked, patch;
213  float radius;
214  float root[3];
215 
216  // Radius is determined in boundingBoxKernel
217  radius = radiusd;
218  // The root node has been created at the end of the boundingBoxKernel.
219  // Cache the root data:
220  for (l = 0; l < 3; l++)
221  root[l] = bhpara->r[3 * bhpara->nnodes + l];
222  // Maximum tree depth within the given thread.
223  localmaxdepth = 1;
224  // Skip the branch following and start from the root.
225  skip = 1;
226  // Number of loops for the threads sync algorithm
227  lps = 0;
228  // Increment to move among the bodies assigned to the given thread.
229  // Hence, one should step over all other threads in GPU with
230  // a quantity of blockDim.x * gridDim.x.
231  inc = static_cast<int>(blockDim.x * gridDim.x);
232  // Just a regular 1D GPU index
233  i = static_cast<int>(threadIdx.x + blockIdx.x * blockDim.x);
234 
235  // Iterate over all bodies assigned to thread.
236  while (i < bhpara->nbodies) {
237  if (skip != 0) {
238  // New body, so start traversing at root. Skip it further.
239  skip = 0;
240  // Particle position corresponding to the given thread and block:
241  for (l = 0; l < 3; l++)
242  p[l] = bhpara->r[3 * i + l];
243  // Let's start a moving via the tree from the root node 8 * nbodiesd:
244  n = bhpara->nnodes;
245  depth = 1;
246  r = radius;
247  // Determine which child to follow.
248  // j=0..7 determines the octant in a binary representations
249  j = 0;
250  for (l = 0; l < 3; l++)
251  if (root[l] < p[l])
252  j += static_cast<int>(pow(2, l));
253  }
254 
255  // Follow path to leaf cell. Should not happen at the first iteration of
256  // this loop.
257  ch = bhpara->child[n * 8 + j];
258 
259  // Global memory writing related threads sync
260  if (lps++ > THREADS2) {
261  *bhpara->max_lps = lps;
262  }
263  //.. now wait for global memory updates. This impacts on race conditions and
264  // frameworks level optimizations. Further kernels contain a similar code
265  // fragment.
266  __threadfence();
267 
268  // The child with the index higher than nbodiesd (number of particles) means
269  // that it is an octant cell, not a body.
270  // Actually, we need nnodesd == 8 * nbodiesd nodes for the cells storage.
271  // Let's iterate this "while" loop before we will reach the particle
272  // or an absence of a child:
273  while (ch >= bhpara->nbodies) {
274  n = ch;
275  depth++;
276  // Going down through the octree depth: radius split corresponds to the
277  // cube split. Corresponding octants are cells, not bodies. Smaller radius
278  // will be used until the next skip == 1
279  r *= 0.5f;
280  j = 0;
281  // Determine which child octant to follow based on body coordinates.
282  // j=0..7 determines the octant in a binary representations.
283  for (l = 0; l < 3; l++)
284  if (bhpara->r[3 * n + l] < p[l])
285  j += static_cast<int>(pow(2, l));
286  ch = bhpara->child[n * 8 + j];
287  }
288  // Now we are deep enough in the tree, passed all levels of cells and
289  // reached the body (particle).
290  if (ch != -2) { // Skip if child pointer is locked (-2) and try again later.
291  locked = n * 8 + j;
292  // Try to lock and iterate towards next body:
293  if (ch == atomicCAS((int *)&bhpara->child[locked], ch, -2)) {
294  // If we are here then child[locked] was equal to "ch" and now is
295  // assigned to -2, it is locked. We will not came here in a case if this
296  // child will be already locked by other threads because other particle
297  // could already do it in other thread. Several particles simultaneously
298  // could force a new octants' split. In this case we will not came here
299  // and "i += inc" will be not executed below. Hence, the present loop
300  // "while (i < nbodiesd)" will mandatory repeat in the given thread for
301  // the given i-th particle until other threads will unlock this cell for
302  // either a body insertion and/or new octant level cells creation.
303  if (ch == -2) {
304  // Cannot be here..
305  *bhpara->err = 1;
306  break;
307  }
308  if (ch == -1) {
309  // If -1 (i.e. no child index) then just insert a new body
310  bhpara->child[locked] = i;
311  } else {
312  patch = -1;
313  // There already is a body and/or cell(s) in this position.
314  // We should start from a loop of the deepest child cell
315  // determination. Then, we need to create new cells and to distribute
316  // existing and new bodies between these new cells.
317  do {
318  // If we are here then the tree bottom level is moving further:
319  // Note, that the bottomd is not a global tree depth.
320  // It is rather a tree size in 1D representation of nnodesd == 8 *
321  // nbodiesd nodes. These lists will be correctly handled by the
322  // sortKernel later.
323  depth++;
324  cell = atomicSub((int *)&bottomd, 1) - 1;
325  if (cell <= bhpara->nbodies) {
326  // This should not happen. A cell cannot have such index. Error.
327  *bhpara->err = 1;
328  bottomd = bhpara->nnodes;
329  }
330  // The "patch" is saving the information about a first cell created
331  // in the current thread before it continues to dive into the tree
332  // depth. Hence, the "patch" defines the new branch inception index.
333  patch = max(patch, cell);
334 
335  // Note: "j" is defined by the body position against the parent cell
336  // within above "while (ch >= nbodiesd)" loop.
337  // The new cell will be placed below relatively the center of
338  // corresponding j-th octant:
339  for (l = 0; l < 3; l++)
340  pos[l] = static_cast<float>((j >> l) & 1) * r;
341  // Note, that negative octants correspond to pos[l] == 0 and
342  // positive octants correspond to pos[l] == r.
343 
344  // Going down through the octree depth: radius split corresponds to
345  // the cube split. Corresponding octants are cells, not bodies.
346  // Smaller radius will be used until the next skip == 1
347  r *= 0.5f;
348 
349  // Init the node weight coefficients.
350  // Note: particles has mass=1.0 is defined in allocBHmemCopy().
351  bhpara->mass[cell] = -1.0f;
352 
353  // The startd array is crucial for the sortKernel.
354  // The original root node already has startd = 0.
355  // Here, we assign -1 to the cells in contrast to bodies.
356  // Bodies do not need this array. They need only array sortd,
357  // which will be defined in the sortKernel for a usage by the force
358  // and energy calculation kernels.
359  bhpara->start[cell] = -1;
360 
361  // Now, let's save the cell coordinates locally (pos[l]) and
362  // globally (xd[3 * cell + l]). This location should be shifted from
363  // the octant center defined above (pos[l] before this assignment).
364  // Parent cell coordinates bhpara->r[3 * n + l] will be added.
365  // Parent radius now is equal to 2 * r, where "r" is already updated
366  // above: r *= 0.5f. Hence, the negative octant is defined above by
367  // pos[l] == 0 and positive - by pos[l] == 2 * r. In order to
368  // transform these coordinates into relative octant positions, we
369  // need to add -r to obtain -r and r for negative and positive
370  // octants. Now, the child (cell) octants centers are deriving from
371  // the parent (n) octant center:
372  for (l = 0; l < 3; l++)
373  pos[l] = bhpara->r[3 * cell + l] =
374  bhpara->r[3 * n + l] - r + pos[l];
375 
376  // By default, the new cell has no children in all k-th octants:
377  for (k = 0; k < 8; k++)
378  bhpara->child[cell * 8 + k] = -1;
379 
380  // This condition should always be true cause "patch" is -1 at the
381  // beginning and the bottomd/cell reduces further.
382  if (patch != cell) {
383  // New cell is assigned as a child of previous "n" parent:
384  bhpara->child[n * 8 + j] = cell;
385  }
386 
387  // pos[l] already contains the child cell coordinates.
388  // Let's assign "child" then. First the octant should be selected:
389  j = 0;
390  for (l = 0; l < 3; l++)
391  if (pos[l] < bhpara->r[3 * ch + l])
392  j += static_cast<int>(pow(2, l));
393  // New element just appeared in the chain of cells. Hence, that what
394  // supposed to be a child ("ch") before entering the present
395  // iteration, now will be a child of the new cell (after this
396  // smallest octant split into new octants):
397  bhpara->child[cell * 8 + j] = ch;
398 
399  // Now cell is claimed to be a parent of further iteration of the
400  // present loop.
401  n = cell;
402  j = 0;
403  __threadfence();
404  // Let's handle the particle position (p[l]) corresponding to the
405  // given thread and block against new octant cell (pos[l]):
406  for (l = 0; l < 3; l++)
407  if (pos[l] < p[l])
408  j += static_cast<int>(pow(2, l));
409 
410  // Now the current cell's child should be considering in the new
411  // particle new octant:
412  ch = bhpara->child[n * 8 + j];
413  // Repeat until the two bodies will be different children.
414  // Hence, the current "child" should have no children.
415  // It is equivalent to an absence of other particles
416  // in the i-th particle new smallest octant, otherwise we should
417  // split octants further until these two particles will come to
418  // different octants:
419  } while (ch >= 0);
420 
421  // i-th particle assignment as a child to the last created cell:
422  bhpara->child[n * 8 + j] = i;
423  // Push out the subtree among the whole grid.
424  // Data setting must be completed after this point.
425  __threadfence();
426  // The final locked child index is defined by a maximal cell index,
427  // i.e. by a beginning of the new tree of cells created within
428  // the loop "while (ch >= 0)".
429  // The "patch" defines the new just created branch inception index:
430  bhpara->child[locked] = patch;
431  }
432 
433  localmaxdepth = max(depth, localmaxdepth);
434  // Each thread started from the skip=1 and made the above
435  // tree building loop procedure (while (ch >= 0)).
436  // They should do the same for remaining (each) particles.
437  // Note, that bottomd is a global variable and it is already updated.
438  // This is already taken into account in further sortKernel logic.
439  // Hence, move on to the next body assigned to the given thread:
440  i += inc;
441  skip = 1;
442  lps = 0;
443  }
444  }
445  }
446  // Record maximum tree depth:
447  atomicMax((int *)&maxdepthd, localmaxdepth);
448 }

454 __global__ __launch_bounds__(THREADS3, FACTOR3) void summarizationKernel() {
455  int i, j, k, l, im, ch, inc, missing, missing_max, cnt, bottom, lps;
456  int iteration, repeat_flag;
457  // the node "mass" and its count respectively:
458  float m, cm;
459  // position of equivalent total dipole and its magnitude:
460  // (like a mass and the center of mass)
461  float p[3], u[3];
462  // Per-block BH tree caching:
463  __shared__ int child[THREADS3 * 8];
464 
465  // no children by default:
466  for (i = 0; i < 8; i++)
467  child[i * THREADS3 + threadIdx.x] = -1;
468  bottom = bottomd;
469  // Increment towards other particles assigned to the given thread:
470  inc = static_cast<int>(blockDim.x * gridDim.x);
471  // Nodes iteration "k" should start from the "bottomd" level of the cells,
472  // which is a minimal index of the last created cell.
473  // Starting "k" value should be aligned using the warp size
474  // according to the designed threads performance.
475  // k = (bottom & (-WARPSIZE)) + threadIdx.x + blockIdx.x * blockDim.x;
476  k = bottom + static_cast<int>(threadIdx.x + blockIdx.x * blockDim.x);
477  // Threads below the bottom line could proceed to their next cells.
478  // if (k < bottom) k += inc;
479 
480  // Assume no missing children:
481  missing = 0;
482  iteration = 0;
483  repeat_flag = 0;
484  __syncthreads(); // throttle
485  // threads sync related
486  lps = 0;
487  // Iterate over all cells (not particles) assigned to the thread:
488  while (k <= bhpara->nnodes) {
489  if (lps++ > THREADS3) {
490  *bhpara->max_lps = lps;
491  __threadfence();
492  }
493  if (bhpara->mass[k] < 0.) {
494  iteration++;
495  if (missing == 0) {
496  // New cell, so initialize:
497  cm = 0.0f;
498  for (l = 0; l < 3; l++) {
499  p[l] = 0.0f;
500  u[l] = 0.0f;
501  }
502  cnt = 0;
503  j = 0;
504  for (i = 0; i < 8; i++) {
505  ch = bhpara->child[k * 8 + i];
506  if (ch >= 0) {
507  if (i != j) {
508  // Move child to front (needed later for a speed only).
509  // The child's octant change is incorrect from
510  // a tree organization perspective. However, the sum
511  // will be the same.
512  bhpara->child[k * 8 + i] = -1;
513  bhpara->child[k * 8 + j] = ch;
514  }
515  // Cache a missing child in the block shared memory:
516  child[missing * THREADS3 + threadIdx.x] = ch;
517  m = bhpara->mass[ch];
518  // Is a child the particle? Only particles have non-negative mass
519  // initialized originally. Another option: a cell which already
520  // aggregated masses of other cells and particles. "missing" means
521  // that a non-zero contribution of such kind is missing:
522  missing++;
523  if (m >= 0.0f) {
524  // child is ready
525  missing--;
526  // The child is a cell, not a body (ch >= nbodiesd).
527  // Also, the previous condition (m >= 0.0f) reveals
528  // that its' children total mass is already calculated.
529  // Hence, below command "countd[k] = cnt" is already executed by
530  // other threads/blocks and we can add this count
531  if (ch >= bhpara->nbodies) { // count bodies (needed later)
532  // As far as a child is a cell, its "countd" was already
533  // calculated.
534  cnt += bhpara->count[ch] - 1;
535  }
536  // add child's contribution
537  cm += m;
538  for (l = 0; l < 3; l++) {
539  p[l] += bhpara->r[3 * ch + l] * m;
540  u[l] += bhpara->u[3 * ch + l];
541  }
542  }
543  j++;
544  } // if (ch >= 0)
545  }
546  missing_max = missing;
547  // Count of children:
548  cnt += j;
549  }
550 
551  //__syncthreads(); // throttle
552 
553  if (missing != 0) {
554  for (im = 0; im < missing_max; im++) {
555  // poll missing child
556  ch = child[im * THREADS3 + threadIdx.x];
557  if (ch >= 0) {
558  m = bhpara->mass[ch];
559  // Is a child the particle? Only particles have non-negative mass
560  // initialized originally. Another option: a cell which already
561  // aggregated masses of other cells and particles.
562  if (m >= 0.0f) {
563  // child is now ready
564  missing--;
565  child[im * THREADS3 + threadIdx.x] = -1;
566  // The child is a cell, not a body (ch >= nbodiesd).
567  if (ch >= bhpara->nbodies) {
568  // count bodies (needed later)
569  cnt += bhpara->count[ch] - 1;
570  }
571  // add child's contribution
572  cm += m;
573  for (l = 0; l < 3; l++) {
574  p[l] += bhpara->r[3 * ch + l] * m;
575  u[l] += bhpara->u[3 * ch + l];
576  }
577  } // m >= 0.0f
578  } // ch >= 0
579  } // missing_max
580  // repeat until we are done or child is not ready
581  }

585  // (missing == 0) could be true and threads will move to next particles (k
586  // += inc) only if previous conditions (m >= 0.0f) will be true. It can
587  // happen only if cell will obtain the mass (only here below: "massd[k] =
588  // cm") or they will find the very last child: particles. Before that:
589  // do/while loop will continue.
590  if (missing == 0) {
591  // all children are ready, so store computed information
592  bhpara->count[k] = cnt;
593  m = 1.0f / cm;
594  for (l = 0; l < 3; l++) {
595  bhpara->r[3 * k + l] = p[l] * m;
596  bhpara->u[3 * k + l] = u[l];
597  }
598  __threadfence(); // make sure data are visible before setting mass
599  bhpara->mass[k] = cm;
600  __threadfence();
601  k += inc;
602  iteration = 0;
603  lps = 0;
604  }
605  //__syncthreads(); // throttle
606  if (iteration > THREADS3 + 1) {
607  k += inc;
608  repeat_flag = 1;
609  iteration = 0;
610  missing = 0;
611  }
612  } else {
613  k += inc;
614  }
615  if ((k > bhpara->nnodes) && (repeat_flag)) {
616  repeat_flag = 0;
617  missing = 0;
618  k = bottom + static_cast<int>(threadIdx.x + blockIdx.x * blockDim.x);
619  }
620  } // while
621 }

623 /******************************************************************************/
624 /*** sort bodies **************************************************************/
625 /******************************************************************************/
626 
627 // This kernel concurrently places the bodies into an array such that the bodies
628 // appear in the same order in the array as they would during an in-order
629 // traversal of the octree. This sorting groups spatially close bodies (in the
630 // same octant cells) together, and these grouped bodies are crucial to speed up
631 // forceCalculationKernel and energyCalculationKernel
632 __global__ __launch_bounds__(THREADS4, FACTOR4) void sortKernel() {
633  int i, k, ch, dec, start, bottom, lps;
634 
635  bottom = bottomd;
636  dec = static_cast<int>(blockDim.x * gridDim.x);
637  // Start from the end of the nnodesd == 8 * nbodiesd.
638  // Reverse order is required now cause octant cells which are more close
639  // to the root have a larger count of entities inside (countd[k]).
640  // Particles should be sorted over all entities count in the tree array
641  // representation made by treeBuildingKernel.
642  k = bhpara->nnodes + 1 - dec +
643  static_cast<int>(threadIdx.x + blockIdx.x * blockDim.x);
644  // threads sync related
645  lps = 0;
646 
647  // iterate over all cells assigned to thread
648  while (k >= bottom) {
649  start = bhpara->start[k];
650  // Threads sync related
651  if (lps++ > THREADS4) {
652  *bhpara->max_lps = lps;
653  __threadfence();
654  }
655  // Let's start from the root which has only startd=0 defined
656  // in boundingBoxKernel. All other bodies and cells have -1.
657  if (start >= 0) {
658  for (i = 0; i < 8; i++) {
659  ch = bhpara->child[k * 8 + i];
660  if (ch >= bhpara->nbodies) {
661  // child is a cell
662  bhpara->start[ch] = start; // set start ID of child
663  start += bhpara->count[ch]; // add # of bodies in subtree
664  } else if (ch >= 0) {
665  // Child is a body.
666  // This particle should be saved with a stepping over
667  // a count of particles in the cells.
668  // treeBuildingKernel already has ordered cells in a
669  // linear array way. The sortKernel just order random particle
670  // indices in the same order. Hence, they will be much faster accessed
671  // by forceCalculationKernel and energyCalculationKernel.
672  bhpara->sort[start] = ch; // record body in 'sorted' array
673  start++;
674  }
675  }
676  k -= dec; // move on to next cell
677  // Threads sync related
678  lps = 0;
679  }
680  //__syncthreads(); // throttle
681  }

684 /******************************************************************************/
685 /*** compute dipole-dipole force and torque ***********************************/
686 /******************************************************************************/
687 
688 __global__ __launch_bounds__(THREADS5, FACTOR5) void forceCalculationKernel(
689  float pf, float *force, float *torque) {
690  int i, j, k, l, n, depth, base, sbase, diff, t;
691  float tmp;
692  // dr is a distance between particles.
693  // f,h, and N are a target force, field, and torque respectively.
694  // u and uc are dipole moments of two particles.
695  float dr[3], f[3], h[3], u[3], uc[3], N[3];
696  // Shared memory aggregates of each warp-specific stacks
697  // with the MAXDEPTH size per each warp:
698  // "node" is the BH octant sub-cell in the stack.
699  // "pos"=0..7 - which octant we are examining now in the stack.
700  // dq is an array used to determine that the given BH cell is far enough.
701  __shared__ int pos[MAXDEPTH * THREADS5 / WARPSIZE],
702  node[MAXDEPTH * THREADS5 / WARPSIZE];
703  __shared__ float dq[MAXDEPTH * THREADS5 / WARPSIZE];
704  float b, b2, d1, dd5;
705  float bb2d7, umd5;
706 
707  // Zero thread of the block initialize shared data for all warps.
708  if (0 == threadIdx.x) {
709  // Precompute values that depend only on tree level.
710  // The method's parameters (a trade-off accuracy/performance)
711  // which determine that the
712  // cell is far enough are "itolsqd" and
713  // "epssqd" which define a fraction of the octant cell and
714  // an additive distance respectively.
715  // Their joint contribution for the given tree depth are
716  // calculated within the array dq[i], which will
717  // be compared later with squared distance between the particle
718  // and the cell depending on a cell level.
719  // Original tree box edge (2*radiusd) should be divided *0.5
720  // as much as the tree depth takes place.
721  tmp = radiusd;
722  dq[0] = tmp * tmp * *itolsqd;
723  for (i = 1; i < maxdepthd; i++) {
724  dq[i] = dq[i - 1] * 0.25f;
725  dq[i - 1] += *epssqd;
726  }
727  dq[i - 1] += *epssqd;
728 
729  // Only maximal Barnes-Hut tree depth is allowed.
730  // This error is technically possible, however, most applications
731  // are far from the 1/2^32 particles' convergence.
732  if (maxdepthd > MAXDEPTH) {
733  *bhpara->err = maxdepthd;
734  }
735  }
736  __syncthreads();
737 
738  // Only maximal Barnes-Hut tree depth is allowed.
739  if (maxdepthd <= MAXDEPTH) {
740  // How many warps are behind the current thread (?):
741  base = static_cast<int>(threadIdx.x) / WARPSIZE;
742  // Figure out first thread in each warp (lane 0):
743  sbase = base * WARPSIZE;
744  // Initial stack index is its MAXDEPTH portion start for the given warp
745  // count base:
746  j = base * MAXDEPTH;
747 
748  // How far the thread is from the warp beginning (?):
749  diff = static_cast<int>(threadIdx.x) - sbase;
750  // Make multiple copies to avoid index calculations later:
751  if (diff < MAXDEPTH) {
752  // Each thread copies its own dq[] element to a part of
753  // dq array dedicated to the given warp:
754  dq[diff + j] = dq[diff];
755  }
756  __syncthreads();
757 
758  // Iterate over all bodies assigned to thread:
759  for (k = static_cast<int>(threadIdx.x + blockIdx.x * blockDim.x);
760  k < bhpara->nbodies; k += static_cast<int>(blockDim.x * gridDim.x)) {
761  // Sorted body indexes assigned to me:
762  i = bhpara->sort[k]; // get permuted/sorted index
763  // Cache the particle position info:
764  for (l = 0; l < 3; l++) {
765  u[l] = bhpara->u[3 * i + l];
766  h[l] = 0.0f;
767  f[l] = 0.0f;
768  }
769 
770  // Force will be calculated for i-th particle.
771  // All other space is of interest, hence all cells will be considered.
772  // Hence, we should start from the root node (whole Barnes-Hut cube).
773  // Initialize iteration stack, i.e., push root node onto stack.
774  // Let's start from zero octant.
775  depth = j;
776  if (sbase == threadIdx.x) {
777  node[j] = bhpara->nnodes;
778  pos[j] = 0;
779  }
780 
781  while (depth >= j) {
782  // Stack is not empty (depth is still higher than j-level).
783  // Hence, there are still some children to consider.
784  while ((t = pos[depth]) < 8) {
785  // Node on top of stack has more children to process:
786  n = bhpara->child[node[depth] * 8 + t]; // load child pointer
787  if (sbase == threadIdx.x) {
788  // I'm the first thread in the warp.
789  // Let me check for the current depth level of the tree
790  // whether we have more children among 8 octant cells.
791  // Hence, let's go to the next octant in the next iteration only:
792  pos[depth] = t + 1;
793  }
794  __threadfence_block();
795  // There is a child (octant cell) with a dipole moment uxd[3 * n + l]
796  // and the center position bhpara->r[3 * n + l]:
797  if (n >= 0) {
798  tmp = 0.0f; // compute distance squared
799  for (l = 0; l < 3; l++) {
800  dr[l] = -bhpara->r[3 * n + l] + bhpara->r[3 * i + l];
801  tmp += dr[l] * dr[l];
802  }
803 
804  // NOTE: i-th particle is specific for the given thread.
805  // However, the above-mentioned index "i" is already sorted by the
806  // sortKernel. Hence, below stack-diving operations will be
807  // performed by different threads of the same warp almost
808  // synchronously because adjacent threads have particles located in
809  // the space not far from each other (=adjacent octants of
810  // corresponding local maxdepth). Hence, such particles will have a
811  // similar stack of the below execution from a perspective of other
812  // Barnes-Hut tree branches distancing against them. Same global
813  // memory fragments (child[]) will be loaded to the same warp
814  // access and no corresponding threads' divergence will take place.
815  // Only in this way, zero thread could control others: see "if
816  // (sbase == threadIdx.x)" above and below.. and this control will
817  // be correct. It will be even with a single stack arrays per the
818  // single warp: The pos, node and dq array fragments are shared
819  // between all threads of the whole warp.
820 
821  // Check if all threads agree that cell is far enough away (or is a
822  // body, i.e. n < nbodiesd).
823 #if defined(__CUDACC__) && CUDA_VERSION >= 9000
824  if ((n < bhpara->nbodies) ||
825  __all_sync(__activemask(), tmp >= dq[depth])) {
826 #else
827  if ((n < bhpara->nbodies) || __all(tmp >= dq[depth])) {
828 #endif
829  if (n != i) {
830 
831  d1 = sqrtf(tmp /*, 0.5f*/);
832  dd5 = __fdividef(1.0f, tmp * tmp * d1);
833  b = 0.0f;
834  b2 = 0.0f;
835  umd5 = 0.0f;
836  for (l = 0; l < 3; l++) {
837  uc[l] = bhpara->u[3 * n + l];
838  b += uc[l] * dr[l];
839  b2 += u[l] * dr[l];
840  umd5 += u[l] * uc[l];
841  }
842  umd5 *= -3.0f * dd5;
843 
844  bb2d7 = 15.0f * b * b2 * __fdividef(dd5, tmp);
845 
846  for (l = 0; l < 3; l++) {
847  h[l] += (b * 3.0f * dr[l] - tmp * uc[l]) * dd5;
848  f[l] += -dr[l] * (umd5 + bb2d7) +
849  3.0f * (b * u[l] + b2 * uc[l]) * dd5;
850  }
851  }
852  } else {
853  // If it is not then let's split octants further (more depth).
854  // Push the cell onto the stack:
855  depth++;
856  if (sbase == threadIdx.x) {
857  // The given warp should start from this child as a root
858  // further:
859  node[depth] = n;
860  // Let's start from it zero octant:
861  pos[depth] = 0;
862  }
863  __threadfence_block();
864  }
865  } else {
866  // Early out because all remaining children are also zero.
867  // We should move to the next octant or to the next depth if other
868  // threads already checked other octants:
869  depth = max(j, depth - 1);
870  }
871  }
872  depth--; // Done with this level
873  }
874 
875  // Torque:
876  N[0] = u[1] * h[2] - u[2] * h[1];
877  N[1] = u[2] * h[0] - u[0] * h[2];
878  N[2] = u[0] * h[1] - u[1] * h[0];
879 
880  for (l = 0; l < 3; l++) {
881  if (f[l] != f[l] || h[l] != h[l]) { // nan
882  printf("Force Kernel: NAN in particle[%d]\n", i);
883  printf("x = %f, y = %f, z = %f,\n", bhpara->u[3 * i + 0],
884  bhpara->u[3 * i + 1], bhpara->u[3 * i + 2]);
885  printf("fx = %f, fy = %f, fz = %f,\n", f[0], f[1], f[2]);
886  printf("hx = %f, hy = %f, hz = %f,\n", h[0], h[1], h[2]);
887  }
888  atomicAdd(force + 3 * i + l, f[l] * pf);
889  atomicAdd(torque + 3 * i + l, N[l] * pf);
890  }
891  }
892  }
893 }

898 /******************************************************************************/
899 
900 __global__ __launch_bounds__(THREADS5, FACTOR5) void energyCalculationKernel(
901  float pf, float *energySum) {
902  // NOTE: the algorithm of this kernel is almost identical to
903  // forceCalculationKernel. See comments there.
904 
905  int i, j, k, l, n, depth, base, sbase, diff, t;
906  float tmp;
907  float dr[3], h[3], u[3], uc[3];
908  __shared__ int pos[MAXDEPTH * THREADS5 / WARPSIZE],
909  node[MAXDEPTH * THREADS5 / WARPSIZE];
910  __shared__ float dq[MAXDEPTH * THREADS5 / WARPSIZE];
911  float sum = 0.0;
912  extern __shared__ float res[];
913 
914  float b, d1, dd5;
915 
916  if (0 == threadIdx.x) {
917  tmp = radiusd;
918  // precompute values that depend only on tree level
919  dq[0] = tmp * tmp * *itolsqd;
920  for (i = 1; i < maxdepthd; i++) {
921  dq[i] = dq[i - 1] * 0.25f;
922  dq[i - 1] += *epssqd;
923  }
924  dq[i - 1] += *epssqd;
925 
926  if (maxdepthd > MAXDEPTH) {
927  *bhpara->err = maxdepthd;
928  }
929  }
930  __syncthreads();
931 
932  if (maxdepthd <= MAXDEPTH) {
933  // figure out first thread in each warp (lane 0)
934  base = static_cast<int>(threadIdx.x) / WARPSIZE;
935  sbase = base * WARPSIZE;
936  j = base * MAXDEPTH;
937 
938  diff = static_cast<int>(threadIdx.x) - sbase;
939  // make multiple copies to avoid index calculations later
940  if (diff < MAXDEPTH) {
941  dq[diff + j] = dq[diff];
942  }
943  __syncthreads();
944 
945  // iterate over all bodies assigned to thread
946  for (k = static_cast<int>(threadIdx.x + blockIdx.x * blockDim.x);
947  k < bhpara->nbodies; k += static_cast<int>(blockDim.x * gridDim.x)) {
948  i = bhpara->sort[k]; // get permuted/sorted index
949  // cache position info
950  for (l = 0; l < 3; l++) {
951  u[l] = bhpara->u[3 * i + l];
952  h[l] = 0.0f;
953  }
954 
955  // initialize iteration stack, i.e., push root node onto stack
956  depth = j;
957  if (sbase == threadIdx.x) {
958  node[j] = bhpara->nnodes;
959  pos[j] = 0;
960  }
961 
962  while (depth >= j) {
963  // stack is not empty
964  while ((t = pos[depth]) < 8) {
965  // node on top of stack has more children to process
966  n = bhpara->child[node[depth] * 8 + t]; // load child pointer
967  if (sbase == threadIdx.x) {
968  // I'm the first thread in the warp
969  pos[depth] = t + 1;
970  }
971  __threadfence_block();
972  if (n >= 0) {
973  tmp = 0.0f;
974  for (l = 0; l < 3; l++) {
975  dr[l] = -bhpara->r[3 * n + l] + bhpara->r[3 * i + l];
976  tmp += dr[l] * dr[l];
977  }
978 #if defined(__CUDACC__) && CUDA_VERSION >= 9000
979  if ((n < bhpara->nbodies) ||
980  __all_sync(
981  __activemask(),
982  tmp >= dq[depth])) { // check if all threads agree that cell
983  // is far enough away (or is a body)
984 #else
985  if ((n < bhpara->nbodies) ||
986  __all(tmp >=
987  dq[depth])) { // check if all threads agree that cell
988  // is far enough away (or is a body)
989 #endif
990  if (n != i) {
991  d1 = sqrtf(tmp /*, 0.5f*/);
992  dd5 = __fdividef(1.0f, tmp * tmp * d1);
993  b = 0.0f;
994  for (l = 0; l < 3; l++) {
995  uc[l] = bhpara->u[3 * n + l];
996  b += uc[l] * dr[l];
997  }
998 
999  for (l = 0; l < 3; l++)
1000  h[l] += (b * 3.0f * dr[l] - tmp * uc[l]) * dd5;
1001  }
1002  } else {
1003  // push cell onto stack
1004  depth++;
1005  if (sbase == threadIdx.x) {
1006  node[depth] = n;
1007  pos[depth] = 0;
1008  }
1009  __threadfence_block();
1010  }
1011  } else {
1012  depth = max(j, depth - 1); // early out because all remaining
1013  // children are also zero
1014  }
1015  }
1016  depth--; // done with this level
1017  }
1018 
1019  for (l = 0; l < 3; l++) {
1020  sum += -u[l] * h[l];
1021  if (h[l] != h[l]) { // nan
1022  printf("Energy Kernel: NAN in particle[%d]\n", i);
1023  printf("x = %f, y = %f, z = %f,\n", bhpara->u[3 * i + 0],
1024  bhpara->u[3 * i + 1], bhpara->u[3 * i + 2]);
1025  printf("hx = %f, hy = %f, hz = %f,\n", h[0], h[1], h[2]);
1026  }
1027  }
1028  }
1029  }
1030 
1031  sum *= pf;
1032  // the self-consistent field energy;
1033  // the Barnes-Hut algorithm, probably, does not allow to avoid this /2 cause
1034  // it is not symmetric:
1035  sum /= 2.0f;
1036  // Save per thread result into block shared mem
1037  res[threadIdx.x] = sum;
1038  // Sum results within a block
1039  __syncthreads(); // Wait til all threads in block are done
1040  dds_sumReduction_BH(res, &(energySum[blockIdx.x]));
1041 }
