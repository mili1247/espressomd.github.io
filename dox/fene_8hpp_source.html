72 /** Compute the FENE bond force.
73  * @param[in] dx %Distance between the particles.
74  */
75 inline boost::optional<Utils::Vector3d>
76 FeneBond::force(Utils::Vector3d const &dx) const {
77  auto const len = dx.norm();
78  auto const dr = len - r0;
79 
80  if (dr >= drmax) {
81  return {};
82  }
83 
84  auto fac = -k * dr / (1.0 - dr * dr * drmax2i);
85  if (len > ROUND_ERROR_PREC) {
86  fac /= len;
87  } else {
88  fac = 0.0;
89  }
90 
91  return fac * dx;
92 }

94 /** Compute the FENE bond energy.
95  * @param[in] dx %Distance between the particles.
96  */
97 inline boost::optional<double>
98 FeneBond::energy(Utils::Vector3d const &dx) const {
99  /* compute bond stretching (r-r0) */
100  double const dr = dx.norm() - r0;
101 
102  /* check bond stretching */
103  if (dr >= drmax) {
104  return {};
105  }
106 
107  return -0.5 * k * drmax2 * log(1.0 - dr * dr * drmax2i);
108 }
