21 /** \file
22  * This file contains implementations for some special functions which are
23  * needed by the MMM family of algorithms. This are the modified Hurwitz zeta
24  * function and the modified Bessel functions of first and second kind. The
25  * implementations are based on the GSL code (see \ref specfunc.cpp for the
26  * original GSL header).
27  *
28  * The Hurwitz zeta function is evaluated using the Euler-Maclaurin summation
29  * formula, the Bessel functions are evaluated using several different
30  * Chebychev expansions. Both achieve a precision of nearly machine precision,
31  * which is no problem for the Hurwitz zeta function, which is only used when
32  * determining the coefficients for the modified polygamma functions (see \ref
33  * mmm-common.hpp).
34  */

79 /** Evaluate the polynomial interpreted as a Taylor series via the
80  * Horner scheme.
81  */
82 inline double evaluateAsTaylorSeriesAt(Utils::Span<const double> series,
83  double x) {
84  assert(not series.empty());
85  auto cnt = static_cast<int>(series.size()) - 1;
86  const double *c = series.data();
87  double r = c[cnt];
88  while (--cnt >= 0)
89  r = r * x + c[cnt];
90  return r;
91 }

93 /** Evaluate the polynomial interpreted as a Chebychev series. Requires a
94  * series with at least three coefficients, i.e. no linear approximations!
95  */
96 inline double evaluateAsChebychevSeriesAt(Utils::Span<const double> series,
97  double x) {
98  assert(series.size() >= 3);
99 
100  const double *c = series.data();
101  double const x2 = 2.0 * x;
102  double dd = c[series.size() - 1];
103  double d = x2 * dd + c[series.size() - 2];
104  for (auto j = static_cast<int>(series.size()) - 3; j >= 1; j--) {
105  auto const tmp = d;
106  d = x2 * d - dd + c[j];
107  dd = tmp;
108  }
109  return x * d - dd + 0.5 * c[0];
110 }
