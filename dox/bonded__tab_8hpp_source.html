125 /** Compute a tabulated bond length force.
126  *
127  * The force acts in the direction of the connecting vector between the
128  * particles. For distances smaller than the tabulated range it uses a linear
129  * extrapolation based on the first two tabulated force values.
130  *
131  * @param[in] dx %Distance between the particles.
132  */
133 inline boost::optional<Utils::Vector3d>
134 TabulatedDistanceBond::force(Utils::Vector3d const &dx) const {
135  auto const dist = dx.norm();
136 
137  if (dist < pot->cutoff()) {
138  auto const fac = pot->force(dist) / dist;
139  return fac * dx;
140  }
141  return {};
142 }
143 
144 /** Compute a tabulated bond length energy.
145  *
146  * For distances smaller than the tabulated range it uses a quadratic
147  * extrapolation based on the first two tabulated force values and the first
148  * tabulated energy value.
149  *
150  * @param[in] dx %Distance between the particles.
151  */
152 inline boost::optional<double>
153 TabulatedDistanceBond::energy(Utils::Vector3d const &dx) const {
154  auto const dist = dx.norm();
155 
156  if (dist < pot->cutoff()) {
157  return pot->energy(dist);
158  }
159  return {};
160 }

162 /** Compute the three-body angle interaction force.
163  * @param[in] r_mid Position of second/middle particle.
164  * @param[in] r_left Position of first/left particle.
165  * @param[in] r_right Position of third/right particle.
166  * @return Forces on the second, first and third particles, in that order.
167  */
168 inline std::tuple<Utils::Vector3d, Utils::Vector3d, Utils::Vector3d>
169 TabulatedAngleBond::forces(Utils::Vector3d const &r_mid,
170  Utils::Vector3d const &r_left,
171  Utils::Vector3d const &r_right) const {
172 
173  auto forceFactor = [this](double const cos_phi) {
174  auto const sin_phi = sqrt(1 - Utils::sqr(cos_phi));
175 #ifdef TABANGLEMINUS
176  auto const phi = acos(-cos_phi);
177 #else
178  auto const phi = acos(cos_phi);
179 #endif
180  auto const tab_pot = pot;
181  auto const gradient = tab_pot->force(phi);
182  return -gradient / sin_phi;
183  };
184 
185  return angle_generic_force(r_mid, r_left, r_right, forceFactor, true);
186 }

188 /** Compute the three-body angle interaction energy.
189  * It is assumed that the potential is tabulated
190  * for all angles between 0 and Pi.
191  *
192  * @param[in] r_mid Position of second/middle particle.
193  * @param[in] r_left Position of first/left particle.
194  * @param[in] r_right Position of third/right particle.
195  */
196 inline double TabulatedAngleBond::energy(Utils::Vector3d const &r_mid,
197  Utils::Vector3d const &r_left,
198  Utils::Vector3d const &r_right) const {
199  auto const vectors = calc_vectors_and_cosine(r_mid, r_left, r_right, true);
200  auto const cos_phi = std::get<4>(vectors);
201  /* calculate phi */
202 #ifdef TABANGLEMINUS
203  auto const phi = acos(-cos_phi);
204 #else
205  auto const phi = acos(cos_phi);
206 #endif
207  return pot->energy(phi);
208 }

210 /** Compute the four-body dihedral interaction force.
211  * This function is not tested yet.
212  *
213  * @param[in] r1 Position of the first particle.
214  * @param[in] r2 Position of the second particle.
215  * @param[in] r3 Position of the third particle.
216  * @param[in] r4 Position of the fourth particle.
217  * @return the forces on @p p2, @p p1, @p p3
218  */
219 inline boost::optional<std::tuple<Utils::Vector3d, Utils::Vector3d,
220  Utils::Vector3d, Utils::Vector3d>>
221 TabulatedDihedralBond::forces(Utils::Vector3d const &r1,
222  Utils::Vector3d const &r2,
223  Utils::Vector3d const &r3,
224  Utils::Vector3d const &r4) const {
225  /* vectors for dihedral angle calculation */
226  Utils::Vector3d v12, v23, v34, v12Xv23, v23Xv34;
227  double l_v12Xv23, l_v23Xv34;
228  /* dihedral angle, cosine of the dihedral angle, cosine of the bond angles */
229  double phi, cos_phi;
230 
231  /* dihedral angle */
232  calc_dihedral_angle(r1, r2, r3, r4, v12, v23, v34, v12Xv23, &l_v12Xv23,
233  v23Xv34, &l_v23Xv34, &cos_phi, &phi);
234  /* dihedral angle not defined - force zero */
235  if (phi == -1.0) {
236  return {};
237  }
238 
239  /* calculate force components (directions) */
240  auto const f1 = (v23Xv34 - cos_phi * v12Xv23) / l_v12Xv23;
241  auto const f4 = (v12Xv23 - cos_phi * v23Xv34) / l_v23Xv34;
242 
243  auto const v23Xf1 = vector_product(v23, f1);
244  auto const v23Xf4 = vector_product(v23, f4);
245  auto const v34Xf4 = vector_product(v34, f4);
246  auto const v12Xf1 = vector_product(v12, f1);
247 
248  /* table lookup */
249  auto const fac = pot->force(phi);
250 
251  /* store dihedral forces */
252  auto const force1 = fac * v23Xf1;
253  auto const force2 = fac * (v34Xf4 - v12Xf1 - v23Xf1);
254  auto const force3 = fac * (v12Xf1 - v23Xf4 - v34Xf4);
255 
256  return std::make_tuple(force2, force1, force3, -(force2 + force1 + force3));
257 }
