62 /** Compute the bonded Coulomb pair force.
63  * @param[in] q1q2 Product of the particle charges.
64  * @param[in] dx %Distance between the particles.
65  */
66 inline boost::optional<Utils::Vector3d>
67 BondedCoulomb::force(double const q1q2, Utils::Vector3d const &dx) const {
68 #ifdef ELECTROSTATICS
69  auto const dist2 = dx.norm2();
70  auto const dist3 = dist2 * std::sqrt(dist2);
71  auto const fac = prefactor * q1q2 / dist3;
72  return fac * dx;
73 #else
74  return Utils::Vector3d{};
75 #endif
76 }
