41 double maximal_cutoff_bonded() {
42  auto const max_cut_bonded = boost::accumulate(
43  bonded_ia_params, BONDED_INACTIVE_CUTOFF,
44  [](auto max_cut, Bonded_IA_Parameters const &bond) {
45  return std::max(max_cut, boost::apply_visitor(BondCutoff(), bond));
46  });

48  /* Check if there are dihedrals */
49  auto const any_dihedrals = std::any_of(
50  bonded_ia_params.begin(), bonded_ia_params.end(), [](auto const &bond) {
51  return (boost::get<DihedralBond>(&bond) ||
52  boost::get<TabulatedDihedralBond>(&bond));
53  });
54 
55  /* dihedrals: the central particle is indirectly connected to the fourth
56  * particle via the third particle, so we have to double the cutoff */
57  return (any_dihedrals) ? 2 * max_cut_bonded : max_cut_bonded;
58 }
