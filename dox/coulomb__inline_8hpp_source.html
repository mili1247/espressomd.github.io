41 inline Utils::Vector3d central_force(double const q1q2,
42  Utils::Vector3d const &d, double dist) {
43  Utils::Vector3d f{};
44 
45  switch (coulomb.method) {
46 #ifdef P3M
47  case COULOMB_P3M_GPU:
48  case COULOMB_P3M:
49  case COULOMB_ELC_P3M:
50  p3m_add_pair_force(q1q2, d, dist, f);
51  break;
52 #endif
53  case COULOMB_MMM1D:
54  add_mmm1d_coulomb_pair_force(q1q2, d, dist, f);
55  break;
56  case COULOMB_DH:
57  add_dh_coulomb_pair_force(q1q2, d, dist, f);
58  break;
59  case COULOMB_RF:
60  add_rf_coulomb_pair_force(q1q2, d, dist, f);
61  break;
62 #ifdef SCAFACOS
63  case COULOMB_SCAFACOS:
64  Scafacos::fcs_coulomb()->add_pair_force(q1q2, d, dist, f);
65  break;
66 #endif
67  default:
68  break;
69  }
70 
71  return coulomb.prefactor * f;
72 }

74 inline std::tuple<Utils::Vector3d, Utils::Vector3d, Utils::Vector3d>
75 pair_force(Particle const &p1, Particle const &p2, Utils::Vector3d const &d,
76  double dist) {
77  auto const q1q2 = p1.p.q * p2.p.q;
78 
79  if (q1q2 == 0) {
80  return {};
81  }
82 
83  auto const force = central_force(q1q2, d, dist);
84 
85 #ifdef P3M
86  if ((coulomb.method == COULOMB_ELC_P3M) &&
87  (elc_params.dielectric_contrast_on)) {
88 
89  auto const f1 =
90  coulomb.prefactor *
91  ELC_P3M_dielectric_layers_force_contribution(p2.r.p, p1.r.p, q1q2);
92  auto const f2 =
93  coulomb.prefactor *
94  ELC_P3M_dielectric_layers_force_contribution(p1.r.p, p2.r.p, q1q2);
95 
96  return {force, f1, f2};
97  }
98 #endif
99 
100  return {force, {}, {}};
101 }

103 /**
104  * @brief Pair contribution to the pressure tensor.
105  *
106  * If supported by the method, this returns the virial
107  * contribution to the pressure tensor for this pair.
108  *
109  * @param p1 particle
110  * @param p2 particle
111  * @param d distance vector
112  * @param dist distance norm
113  * @return Contribution to the pressure tensor.
114  */
115 inline Utils::Matrix<double, 3, 3> pair_pressure(Particle const &p1,
116  Particle const &p2,
117  Utils::Vector3d const &d,
118  double dist) {
119  switch (coulomb.method) {
120  case COULOMB_NONE:
121  break;
122 #ifdef P3M
123  case COULOMB_P3M_GPU:
124  case COULOMB_P3M:
125 #endif
126  case COULOMB_MMM1D:
127  case COULOMB_DH:
128  case COULOMB_RF: {
129  auto const force = central_force(p1.p.q * p2.p.q, d, dist);
130 
131  return Utils::tensor_product(force, d);
132  }
133  default:
134  fprintf(stderr, "calculating pressure for electrostatics method that "
135  "doesn't have it implemented\n");
136  break;
137  }
138 
139  return {};
140 }

142 // energy_inline
143 inline double pair_energy(Particle const &p1, Particle const &p2,
144  double const q1q2, Utils::Vector3d const &d,
145  double dist, double dist2) {
146  /* real space Coulomb */
147  auto E = [&]() {
148  switch (coulomb.method) {
149 #ifdef P3M
150  case COULOMB_P3M_GPU:
151  case COULOMB_P3M:
152  return p3m_pair_energy(q1q2, dist);
153  case COULOMB_ELC_P3M:
154  if (elc_params.dielectric_contrast_on) {
155  return 0.5 * ELC_P3M_dielectric_layers_energy_contribution(p1, p2) +
156  p3m_pair_energy(q1q2, dist);
157  } else {
158  return p3m_pair_energy(q1q2, dist);
159  }
160 #endif
161 #ifdef SCAFACOS
162  case COULOMB_SCAFACOS:
163  return Scafacos::fcs_coulomb()->pair_energy(q1q2, dist);
164 #endif
165  case COULOMB_DH:
166  return dh_coulomb_pair_energy(q1q2, dist);
167  case COULOMB_RF:
168  return rf_coulomb_pair_energy(q1q2, dist);
169  case COULOMB_MMM1D:
170  return mmm1d_coulomb_pair_energy(q1q2, d, dist2, dist);
171  default:
172  return 0.;
173  }
174  }();
175  return coulomb.prefactor * E;
176 }
