31  * @brief Calculate the distance function in the coordinates of the cylinder.
32  *
33  * @param r Distance from the cylinder axis.
34  * @param z Distance from the center along the axis.
35  *
36  * @returns The distance vector from the surface in the cylinder system.
37  */
38 std::pair<double, double> SimplePore::dist_half_pore(double r, double z) const {
39  assert(z >= 0.0);
40  assert(r >= 0.0);
41 
42  /*
43  * We have to find the line that splits area 1 (r determines distance) from
44  * area 2 (z determines distance) inside pore. In area 3 we have to consider
45  * z and r to determine the distance.
46  *
47  * | x
48  * | 2 x
49  * | x
50  * _|_ x 1
51  * \| ^ r
52  * 3 |-------|
53  * | z <-
54  */
55 
56  if ((z <= c_z) && (r <= (c_z + c_r - z))) {
57  /* Cylinder section, inner */
58  return {m_rad - r, 0};
59  }
60  if (((z >= c_z) && (r >= c_r)) || ((z <= c_z) && (r > (c_z + c_r - z)))) {
61  /* Wall section and outer cylinder */
62  return {0, m_half_length - z};
63  }
64  /* Smoothing area */
65  /* Vector to center of torus segment */
66  auto const dr = c_r - r;
67  auto const dz = c_z - z;
68 
69  /* Rescale to surface */
70  auto const d = std::sqrt(dr * dr + dz * dz);
71  auto const fac = (d - m_smoothing_rad) / d;
72 
73  return {fac * dr, fac * dz};
74 }

76 void SimplePore::calculate_dist(const Utils::Vector3d &pos, double &dist,
77  Utils::Vector3d &vec) const {
78  /* Coordinate transform to cylinder coords
79  with origin at m_center. */
80  Utils::Vector3d const c_dist = pos - m_center;
81  auto const z = e_z * c_dist;
82  auto const r_vec = c_dist - z * e_z;
83  auto const r = r_vec.norm();
84 
85  /* If exactly on the axis, chose e_r orthogonal
86  to e_z. */
87  auto const e_r = (r == 0) ? e_r_axis : r_vec / r;
88 
89  /* The pore has mirror symmetry in z with regard to
90  the center in the {r,z} system. We calculate always
91  for the z > 0 case, and flip the result if appropriate. */
92  double dr, dz;
93  std::tie(dr, dz) = dist_half_pore(r, std::abs(z));
94 
95  double side = -1;
96  if (((dz == 0) && (r <= m_rad)) || // cylinder section
97  ((dr == 0) && (std::abs(z) > m_half_length))) { // ||
98  side = 1;
99  } else {
100  // smoothing area
101  if (std::abs(z) >= c_z) {
102  auto const angle = std::asin((std::abs(z) - c_z) / m_smoothing_rad);
103  auto const dist_offset =
104  m_smoothing_rad - (std::cos(angle) * m_smoothing_rad);
105  if (m_half_length < std::abs(z) || r <= (m_rad + dist_offset)) {
106  side = 1;
107  }
108  }
109  }
110 
111  if (z <= 0.0) {
112  dz *= -1;
113  }
114 
115  dist = std::sqrt(dr * dr + dz * dz) * side;
116  vec = -dr * e_r - dz * e_z;
117 }
