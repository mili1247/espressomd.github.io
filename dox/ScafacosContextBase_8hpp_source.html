42  * @brief Public interface to the ScaFaCoS context.

49  /** @brief Collect particle data in continuous arrays. */

52  virtual void update_particle_forces() const = 0;
53  /** @brief Calculate long-range part of the energy. */
54  virtual double long_range_energy() = 0;
55  /** @brief Add long-range part of the forces to particles. */
56  virtual void add_long_range_force() = 0;
57  /** @brief Add near-field pair force. */
58  inline void add_pair_force(double q1q2, Utils::Vector3d const &d, double dist,
59  Utils::Vector3d &force) {
60  if (dist > get_r_cut())
61  return;
62 
63  auto const field = get_pair_force(dist);
64  auto const fak = q1q2 * field / dist;
65  force -= fak * d;
66  }
67  /** @brief Calculate near-field pair energy. */
68  inline double pair_energy(double q1q2, double dist) {
69  if (dist > get_r_cut())
70  return 0.;
71 
72  return q1q2 * get_pair_energy(dist);
73  }
74  /** @brief Reinitialize number of particles, box shape and periodicity. */
75  virtual void update_system_params() = 0;
76  virtual double get_r_cut() const = 0;
77  virtual double get_pair_force(double dist) const = 0;
78  virtual double get_pair_energy(double dist) const = 0;
79  virtual std::string get_method_and_parameters() = 0;
80 };
