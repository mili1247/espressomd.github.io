54 /**
55  * @brief Get pairs of particles that are closer than a distance and fulfill a
56  * filter criterion.
57  *
58  * It uses link_cell to get pairs out of the cellsystem
59  * by a simple distance criterion and
60  *
61  * Pairs are sorted so that first.id < second.id
62  */
63 template <class Filter>
64 std::vector<std::pair<int, int>> get_pairs_filtered(double const distance,
65  Filter filter) {
66  std::vector<std::pair<int, int>> ret;
67  on_observable_calc();
68  auto const cutoff2 = distance * distance;
69  auto pair_kernel = [&ret, &cutoff2, &filter](Particle const &p1,
70  Particle const &p2,
71  Distance const &d) {
72  if (d.dist2 < cutoff2 and filter(p1) and filter(p2))
73  ret.emplace_back(p1.p.identity, p2.p.identity);
74  };
75 
76  cell_structure.non_bonded_loop(pair_kernel);
77 
78  /* Sort pairs */
79  for (auto &pair : ret) {
80  if (pair.first > pair.second)
81  std::swap(pair.first, pair.second);
82  }
83 
84  return ret;
85 }

141 namespace detail {
142 void search_distance_sanity_check(double const distance) {
143  /* get_pairs_filtered() finds pairs via the non_bonded_loop. The maximum
144  * finding range is therefore limited by the decomposition that is used.
145  */
146  auto range = *boost::min_element(cell_structure.max_range());
147  if (distance > range) {
148  runtimeErrorMsg() << "pair search distance " << distance
149  << " bigger than the decomposition range " << range
150  << ".\n";
151  }
152 }

243 void cells_update_ghosts(unsigned data_parts) {
244  /* data parts that are only updated on resort */
245  auto constexpr resort_only_parts =
246  Cells::DATA_PART_PROPERTIES | Cells::DATA_PART_BONDS;
247 
248  auto const global_resort =
249  boost::mpi::all_reduce(comm_cart, cell_structure.get_resort_particles(),
250  std::bit_or<unsigned>());
251 
252  if (global_resort != Cells::RESORT_NONE) {
253  int global = (global_resort & Cells::RESORT_GLOBAL)
254  ? CELL_GLOBAL_EXCHANGE
255  : CELL_NEIGHBOR_EXCHANGE;
256 
257  /* Resort cell system */
258  cell_structure.resort_particles(global);
259  cell_structure.ghosts_count();
260  cell_structure.ghosts_update(data_parts);
261 
262  /* Add the ghost particles to the index if we don't already
263  * have them. */
264  for (auto &part : cell_structure.ghost_particles()) {
265  if (cell_structure.get_local_particle(part.p.identity) == nullptr) {
266  cell_structure.update_particle_index(part.identity(), &part);
267  }
268  }
269 
270  /* Particles are now sorted */
271  cell_structure.clear_resort_particles();
272  } else {
273  /* Communication step: ghost information */
274  cell_structure.ghosts_update(data_parts & ~resort_only_parts);
275  }
276 }
