50 bool in_local_domain(Utils::Vector3d const &pos) {
51  auto const lblattice = lb_lbfluid_get_lattice();
52  auto const my_left = local_geo.my_left();
53  auto const my_right = local_geo.my_right();
54 
55  return (pos[0] >= my_left[0] - 0.5 * lblattice.agrid &&
56  pos[0] < my_right[0] + 0.5 * lblattice.agrid &&
57  pos[1] >= my_left[1] - 0.5 * lblattice.agrid &&
58  pos[1] < my_right[1] + 0.5 * lblattice.agrid &&
59  pos[2] >= my_left[2] - 0.5 * lblattice.agrid &&
60  pos[2] < my_right[2] + 0.5 * lblattice.agrid);
61 }

64 /** Put the calculated force stored on the ibm particles into the fluid by
65  * updating the @ref lbfields structure.
66  * Called from the integration loop right after the forces have been
67  * calculated.
68  */
69 void IBM_ForcesIntoFluid_CPU() {
70  // Update the forces on the ghost particles
71  cell_structure.ghosts_update(Cells::DATA_PART_FORCE);
72 
73  // Loop over local cells
74  for (auto &p : cell_structure.local_particles()) {
75  if (p.p.is_virtual)
76  CoupleIBMParticleToFluid(p);
77  }
78 
79  for (auto &p : cell_structure.ghost_particles()) {
80  // for ghost particles we have to check if they lie
81  // in the range of the local lattice nodes
82  if (in_local_domain(p.r.p)) {
83  if (p.p.is_virtual)
84  CoupleIBMParticleToFluid(p);
85  }
86  }
87 }

90  * particles.
91  * Called from the integration loop right after the LB update.
92  */
93 void IBM_UpdateParticlePositions(ParticleRange const &particles,
94  double time_step) {
95  // Get velocities
96  if (lattice_switch == ActiveLB::CPU)
97  ParticleVelocitiesFromLB_CPU();
98 #ifdef CUDA
99  if (lattice_switch == ActiveLB::GPU)
100  ParticleVelocitiesFromLB_GPU(particles);
101 #endif
102 
103  // Do update: Euler
104  const double skin2 = Utils::sqr(0.5 * skin);
105  // Loop over particles in local cells
106  for (auto &p : particles) {
107  if (p.p.is_virtual) {
108 #ifdef EXTERNAL_FORCES
109  if (!(p.p.ext_flag & 2))
110 #endif
111  p.r.p[0] += p.m.v[0] * time_step;
112 #ifdef EXTERNAL_FORCES
113  if (!(p.p.ext_flag & 4))
114 #endif
115  p.r.p[1] += p.m.v[1] * time_step;
116 #ifdef EXTERNAL_FORCES
117  if (!(p.p.ext_flag & 8))
118 #endif
119  p.r.p[2] += p.m.v[2] * time_step;
120 
121  // Check if the particle might have crossed a box border
122  const double dist2 = (p.r.p - p.l.p_old).norm2();
123  if (dist2 > skin2) {
124  cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
125  }
126  }
127  }

130 /** Put the momentum of a given particle into the LB fluid. */
131 void CoupleIBMParticleToFluid(Particle &p) {
132  // Convert units from MD to LB
133  auto const delta_j = p.f.f * Utils::sqr(lbpar.tau * lbpar.tau) / lbpar.agrid;
134 
135  // Get indices and weights of affected nodes using discrete delta function
136  Utils::Vector<std::size_t, 8> node_index{};
137  Utils::Vector6d delta{};
138  lblattice.map_position_to_lattice(p.r.p, node_index, delta);
139 
140  // Loop over all affected nodes
141  for (int z = 0; z < 2; z++) {
142  for (int y = 0; y < 2; y++) {
143  for (int x = 0; x < 2; x++) {
144  // Do not put force into a halo node
145  if (!IsHalo(node_index[(z * 2 + y) * 2 + x])) {
146  // Add force into the lbfields structure
147  auto &local_f =
148  lbfields[node_index[(z * 2 + y) * 2 + x]].force_density;
149 
150  local_f +=
151  delta[3 * x + 0] * delta[3 * y + 1] * delta[3 * z + 2] * delta_j;
152  }
153  }
154  }
155  }
156 }

158 /** Calculate the LB fluid velocity at a particle position.
159  * Very similar to the velocity interpolation done in standard ESPResSo,
160  * except that we add the f/2 contribution, cf. @cite guo02a.
161  * The fluid velocity is obtained by linear interpolation,
162  * cf. eq. (11) in @cite ahlrichs99a.
163  */
164 void GetIBMInterpolatedVelocity(const Utils::Vector3d &pos, Utils::Vector3d &v,
165  Utils::Vector3d &forceAdded) {
166  /* determine elementary lattice cell surrounding the particle
167  and the relative position of the particle in this cell */
168  Utils::Vector<std::size_t, 8> node_index{};
169  Utils::Vector6d delta{};
170  lblattice.map_position_to_lattice(pos, node_index, delta);
171 
172  Utils::Vector3d interpolated_u = {};
173  // This for the f/2 contribution to the velocity
174  forceAdded = {};
175 
176  for (int z = 0; z < 2; z++) {
177  for (int y = 0; y < 2; y++) {
178  for (int x = 0; x < 2; x++) {
179  auto const index = node_index[(z * 2 + y) * 2 + x];
180  auto const local_delta =
181  delta[3 * x + 0] * delta[3 * y + 1] * delta[3 * z + 2];
182  const auto &f = lbfields[index].force_density_buf;
183 
184  double local_density;
185  Utils::Vector3d local_j;
186 
187  // This can be done more easily without copying the code twice.
188  // We probably can even set the boundary velocity directly.
189 #ifdef LB_BOUNDARIES
190  if (lbfields[index].boundary) {
191  local_density = lbpar.density;
192  local_j = lbpar.density *
193  (*LBBoundaries::lbboundaries[lbfields[index].boundary - 1])
194  .velocity();
195  } else
196 #endif
197  {
198  auto const modes = lb_calc_modes(index, lbfluid);
199  local_density = lbpar.density + modes[0];
200 
201  // Add the +f/2 contribution!!
202  local_j[0] = modes[1] + f[0] / 2.;
203  local_j[1] = modes[2] + f[1] / 2.;
204  local_j[2] = modes[3] + f[2] / 2.;
205 
206  // Keep track of the forces that we added to the fluid.
207  // This is necessary for communication because this part is executed
208  // for real and ghost particles.
209  // Later on we sum the real and ghost contributions.
210  auto const f_ext =
211  lbpar.ext_force_density * Utils::sqr(lbpar.agrid * lbpar.tau);
212  forceAdded += local_delta * (f - f_ext) / (2. * local_density);
213  }
214 
215  // Interpolate velocity
216  interpolated_u += local_delta * local_j / local_density;
217  }
218  }
219  }
220 
221  v = interpolated_u * lbpar.agrid / lbpar.tau;
222 }

224 /** Build a cache structure which contains a flag for each LB node whether
225  * that node is a halo node or not.
226  */
227 bool IsHalo(const int indexCheck) {
228  // First call --> build cache
229  if (isHaloCache == nullptr) {
230  isHaloCache = new bool[lblattice.halo_grid_volume];
231  // Assume everything is a halo and correct in the next step
232  for (int i = 0; i < lblattice.halo_grid_volume; i++)
233  isHaloCache[i] = true;
234  // Loop through and check where indexCheck occurs
235  int index = lblattice.halo_offset;
236  for (int z = 1; z <= lblattice.grid[2]; z++) {
237  for (int y = 1; y <= lblattice.grid[1]; y++) {
238  for (int x = 1; x <= lblattice.grid[0]; x++) {
239  isHaloCache[index] = false;
240  ++index;
241  }
242  index += 2; /* skip halo region */
243  }
244  index += 2 * lblattice.halo_grid[0]; /* skip halo region */
245  }
246  }
247 
248  // Return
249  return isHaloCache[indexCheck];
250 }

252 /** Get particle velocities from LB and set the velocity field in the
253  * particles data structure.
254  */
255 void ParticleVelocitiesFromLB_CPU() {
256  // Loop over particles in local cells.
257  // Here all contributions are included: velocity, external force and
258  // particle force.
259  for (auto &p : cell_structure.local_particles()) {
260  if (p.p.is_virtual) {
261  // Get interpolated velocity and store in the force (!) field
262  // for later communication (see below)
263  Utils::Vector3d dummy;
264  GetIBMInterpolatedVelocity(p.r.p, p.f.f, dummy);
265  }
266  }
267 
268  // Loop over particles in ghost cells
269  // Here we only add the particle forces stemming from the ghosts
270  for (auto &p : cell_structure.ghost_particles()) {
271  // This criterion include the halo on the left, but excludes the halo on
272  // the right
273  // Try if we have to use *1.5 on the right
274  if (in_local_domain(p.r.p)) {
275  if (p.p.is_virtual) {
276  Utils::Vector3d force{}; // The force stemming from the ghost particle
277  Utils::Vector3d dummy;
278  GetIBMInterpolatedVelocity(p.r.p, dummy, force);
279 
280  // Rescale and store in the force field of the particle (for
281  // communication, see below)
282  p.f.f = force * lbpar.agrid / lbpar.tau;
283  } else {
284  p.f.f = {};
285  } // Reset, necessary because we add all forces below. Also needs to
286  // be done for the real particles!
287  } else {
288  p.f.f = {};
289  } // Reset, necessary because we add all forces below
290  }
291 
292  // Now the local particles contain a velocity (stored in the force field)
293  // and the ghosts contain the rest of the velocity in their respective force
294  // fields.
295  // We need to add these. Since we have stored them in the force, not the
296  // velocity fields, we can use the standard force communicator and then
297  // transfer to the velocity afterwards.
298  // Note that this overwrites the actual force which would be a problem for
299  // real particles.
300  // This could be solved by keeping a backup of the local forces before this
301  // operation is attempted.
302  cell_structure.ghosts_reduce_forces();
303 
304  // Transfer to velocity field
305  for (auto &p : cell_structure.local_particles()) {
306  if (p.p.is_virtual) {
307  p.m.v = p.f.f;
308  }
309  }
310 }
