81 /** External particle forces */
82 inline ParticleForce external_force(Particle const &p) {
83  ParticleForce f = {};
84 
85 #ifdef EXTERNAL_FORCES
86  f.f += p.p.ext_force;
87 #ifdef ROTATION
88  f.torque += p.p.ext_torque;
89 #endif
90 #endif
91 
92 #ifdef ENGINE
93  // apply a swimming force in the direction of
94  // the particle's orientation axis
95  if (p.p.swim.swimming) {
96  f.f += p.p.swim.f_swim * p.r.calc_director();
97  }
98 #endif
99 
100  return f;
101 }

127 inline ParticleForce calc_non_bonded_pair_force(Particle const &p1,
128  Particle const &p2,
129  IA_parameters const &ia_params,
130  Utils::Vector3d const &d,
131  double const dist) {
132 
133  ParticleForce pf{};
134  double force_factor = 0;
135 /* Lennard-Jones */
136 #ifdef LENNARD_JONES
137  force_factor += lj_pair_force_factor(ia_params, dist);
138 #endif
139 /* WCA */
140 #ifdef WCA
141  force_factor += wca_pair_force_factor(ia_params, dist);
142 #endif
143 /* Lennard-Jones generic */
144 #ifdef LENNARD_JONES_GENERIC
145  force_factor += ljgen_pair_force_factor(ia_params, dist);
146 #endif
147 /* smooth step */
148 #ifdef SMOOTH_STEP
149  force_factor += SmSt_pair_force_factor(ia_params, dist);
150 #endif
151 /* Hertzian force */
152 #ifdef HERTZIAN
153  force_factor += hertzian_pair_force_factor(ia_params, dist);
154 #endif
155 /* Gaussian force */
156 #ifdef GAUSSIAN
157  force_factor += gaussian_pair_force_factor(ia_params, dist);
158 #endif
159 /* BMHTF NaCl */
160 #ifdef BMHTF_NACL
161  force_factor += BMHTF_pair_force_factor(ia_params, dist);
162 #endif
163 /* Buckingham*/
164 #ifdef BUCKINGHAM
165  force_factor += buck_pair_force_factor(ia_params, dist);
166 #endif
167 /* Morse*/
168 #ifdef MORSE
169  force_factor += morse_pair_force_factor(ia_params, dist);
170 #endif
171 /*soft-sphere potential*/
172 #ifdef SOFT_SPHERE
173  force_factor += soft_pair_force_factor(ia_params, dist);
174 #endif
175 /*hat potential*/
176 #ifdef HAT
177  force_factor += hat_pair_force_factor(ia_params, dist);
178 #endif
179 /* Lennard-Jones cosine */
180 #ifdef LJCOS
181  force_factor += ljcos_pair_force_factor(ia_params, dist);
182 #endif
183 /* Lennard-Jones cosine */
184 #ifdef LJCOS2
185  force_factor += ljcos2_pair_force_factor(ia_params, dist);
186 #endif
187 /* Thole damping */
188 #ifdef THOLE
189  pf.f += thole_pair_force(p1, p2, ia_params, d, dist);
190 #endif
191 /* tabulated */
192 #ifdef TABULATED
193  force_factor += tabulated_pair_force_factor(ia_params, dist);
194 #endif
195 /* Gay-Berne */
196 #ifdef GAY_BERNE
197  // The gb force function isn't inlined, probably due to its size
198  if (dist < ia_params.gay_berne.cut) {
199  pf += gb_pair_force(p1.r.calc_director(), p2.r.calc_director(), ia_params,
200  d, dist);
201  }
202 #endif
203  pf.f += force_factor * d;
204  return pf;
205 }

207 inline ParticleForce calc_opposing_force(ParticleForce const &pf,
208  Utils::Vector3d const &d) {
209  ParticleForce out{-pf.f};
210 #ifdef ROTATION
211  // if torque is a null vector, the opposing torque is a null vector too
212  // (this check guards from returning a small yet non-null opposing
213  // torque due to numerical imprecision)
214  if (pf.torque[0] != 0. || pf.torque[1] != 0. || pf.torque[2] != 0.) {
215  out.torque = -(pf.torque + vector_product(d, pf.f));
216  }
217 #endif
218  return out;
219 }

221 /** Calculate non-bonded forces between a pair of particles and update their
222  * forces and torques.
223  * @param[in,out] p1 particle 1.
224  * @param[in,out] p2 particle 2.
225  * @param[in] d vector between @p p1 and @p p2.
226  * @param dist distance between @p p1 and @p p2.
227  * @param dist2 distance squared between @p p1 and @p p2.
228  */
229 inline void add_non_bonded_pair_force(Particle &p1, Particle &p2,
230  Utils::Vector3d const &d, double dist,
231  double dist2) {
232  IA_parameters const &ia_params = *get_ia_param(p1.p.type, p2.p.type);
233  ParticleForce pf{};
234 
235  /***********************************************/
236  /* non-bonded pair potentials */
237  /***********************************************/
238 
239  if (dist < ia_params.max_cut) {
240 #ifdef EXCLUSIONS
241  if (do_nonbonded(p1, p2))
242 #endif
243  pf += calc_non_bonded_pair_force(p1, p2, ia_params, d, dist);
244  }
245 
246  /***********************************************/
247  /* short-range electrostatics */
248  /***********************************************/
249 
250 #ifdef ELECTROSTATICS
251  {
252  auto const forces = Coulomb::pair_force(p1, p2, d, dist);
253  pf.f += std::get<0>(forces);
254 #ifdef P3M
255  // forces from the virtual charges
256  p1.f.f += std::get<1>(forces);
257  p2.f.f += std::get<2>(forces);
258 #endif
259  }
260 #endif
261 
262  /*********************************************************************/
263  /* everything before this contributes to the virial pressure in NpT, */
264  /* but nothing afterwards */
265  /*********************************************************************/
266 #ifdef NPT
267  npt_add_virial_force_contribution(pf.f, d);
268 #endif
269 
270  /***********************************************/
271  /* thermostat */
272  /***********************************************/
273 
274  /* The inter dpd force should not be part of the virial */
275 #ifdef DPD
276  if (thermo_switch & THERMO_DPD) {
277  auto const force = dpd_pair_force(p1, p2, ia_params, d, dist, dist2);
278  p1.f.f += force;
279  p2.f.f -= force;
280  }
281 #endif
282 
283  /***********************************************/
284  /* long-range magnetostatics */
285  /***********************************************/
286 
287 #ifdef DIPOLES
288  /* real space magnetic dipole-dipole */
289  pf += Dipole::pair_force(p1, p2, d, dist, dist2);
290 #endif
291 
292  /***********************************************/
293  /* add total non-bonded forces to particles */
294  /***********************************************/
295 
296  p1.f += pf;
297  p2.f += calc_opposing_force(pf, d);
298 }

300 /** Compute the bonded interaction force between particle pairs.
301  *
302  * @param[in] p1 First particle.
303  * @param[in] p2 Second particle.
304  * @param[in] iaparams Bonded parameters for the interaction.
305  * @param[in] dx Vector between @p p1 and @p p2.
306  */
307 inline boost::optional<Utils::Vector3d>
308 calc_bond_pair_force(Particle const &p1, Particle const &p2,
309  Bonded_IA_Parameters const &iaparams,
310  Utils::Vector3d const &dx) {
311  if (auto const *iap = boost::get<FeneBond>(&iaparams)) {
312  return iap->force(dx);
313  }
314  if (auto const *iap = boost::get<HarmonicBond>(&iaparams)) {
315  return iap->force(dx);
316  }
317  if (auto const *iap = boost::get<QuarticBond>(&iaparams)) {
318  return iap->force(dx);
319  }
320 #ifdef ELECTROSTATICS
321  if (auto const *iap = boost::get<BondedCoulomb>(&iaparams)) {
322  return iap->force(p1.p.q * p2.p.q, dx);
323  }
324  if (auto const *iap = boost::get<BondedCoulombSR>(&iaparams)) {
325  return iap->force(dx);
326  }
327 #endif
328 #ifdef TABULATED
329  if (auto const *iap = boost::get<TabulatedDistanceBond>(&iaparams)) {
330  return iap->force(dx);
331  }
332 #endif
333  if (boost::get<VirtualBond>(&iaparams) || boost::get<RigidBond>(&iaparams)) {
334  return Utils::Vector3d{};
335  }
336  throw BondUnknownTypeError();
337 }

339 inline bool add_bonded_two_body_force(Bonded_IA_Parameters const &iaparams,
340  Particle &p1, Particle &p2) {
341  auto const dx = box_geo.get_mi_vector(p1.r.p, p2.r.p);
342 
343  if (auto const *iap = boost::get<ThermalizedBond>(&iaparams)) {
344  auto result = iap->forces(p1, p2, dx);
345  if (result) {
346  using std::get;
347  p1.f.f += get<0>(result.get());
348  p2.f.f += get<1>(result.get());
349 
350  return false;
351  }
352  } else {
353  auto result = calc_bond_pair_force(p1, p2, iaparams, dx);
354  if (result) {
355  p1.f.f += result.get();
356  p2.f.f -= result.get();
357 
358 #ifdef NPT
359  npt_add_virial_force_contribution(result.get(), dx);
360 #endif
361  return false;
362  }
363  }
364  return true;
365 }

403  p1.f.f += get<0>(forces);

406 
407  return false;

453 inline bool add_bonded_force(Particle &p1, int bond_id,
454  Utils::Span<Particle *> partners) {
455  auto const &iaparams = bonded_ia_params[bond_id];
456 
457  switch (number_of_partners(iaparams)) {
458  case 0:
459  return false;
460  case 1:
461  return add_bonded_two_body_force(iaparams, p1, *partners[0]);
462  case 2:
463  return add_bonded_three_body_force(iaparams, p1, *partners[0],
464  *partners[1]);
465  case 3:
466  return add_bonded_four_body_force(iaparams, p1, *partners[0], *partners[1],
467  *partners[2]);
468  default:
469  throw BondInvalidSizeError{number_of_partners(iaparams)};
470  }
471 }
