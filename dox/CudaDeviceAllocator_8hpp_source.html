35 /**
36  * @brief Allocator that uses CUDA to allocate GPU memory.
37  *
38  * This allocator uses thrust::malloc and thrust::free to
39  * manage device memory. To be able to have static containers
40  * for device memory, exceptions in deallocations are suppressed.
41  * This is needed because static containers may be destroyed after
42  * the CUDA system was teared down, leading to device_free to fail.
43  *
44  * @tparam T Type to allocate memory for.
45  */
46 template <class T> struct CudaDeviceAllocator {
47  using value_type = T;
48  using pointer = thrust::device_ptr<T>;
49  using const_pointer = thrust::device_ptr<const T>;
50  using reference = thrust::device_reference<T>;
51  using const_reference = thrust::device_reference<const T>;
52 
53  CudaDeviceAllocator() noexcept = default;
54  template <class U>
55  explicit CudaDeviceAllocator(const CudaDeviceAllocator<U> &) {}
56  template <class U> bool operator==(const CudaDeviceAllocator<U> &) const {
57  return true;
58  }
59  template <class U> bool operator!=(const CudaDeviceAllocator<U> &) const {
60  return false;
61  }
62 
63  pointer allocate(const std::size_t n) const {
64  return thrust::device_malloc<T>(n);
65  }
66  void deallocate(pointer p, std::size_t) const noexcept {
67  try {
68  thrust::device_free(p);
69  } catch (thrust::system::system_error const &) {
70  ;
71  }
72  }
73 };
