41 void calc_oif_global(Utils::Vector2d &area_volume, int molType,
42  CellStructure &cs) {
43  // first-fold-then-the-same approach
44  double partArea = 0.0;
45  // z volume
46  double VOL_partVol = 0.;
47 
48  Utils::Vector2d part_area_volume; // added
49 
50  cs.bond_loop(
51  [&partArea, &VOL_partVol, molType](Particle &p1, int bond_id,
52  Utils::Span<Particle *> partners) {
53  if (p1.p.mol_id != molType)
54  return false;
55 
56  if (boost::get<OifGlobalForcesBond>(&bonded_ia_params[bond_id]) !=
57  nullptr) {
58  // remaining neighbors fetched
59  auto const p11 = unfolded_position(p1.r.p, p1.l.i, box_geo.length());
60  auto const p22 = p11 + box_geo.get_mi_vector(partners[0]->r.p, p11);
61  auto const p33 = p11 + box_geo.get_mi_vector(partners[1]->r.p, p11);
62 
63  // unfolded positions correct
64  auto const VOL_A = area_triangle(p11, p22, p33);
65  partArea += VOL_A;
66 
67  auto const VOL_norm = get_n_triangle(p11, p22, p33);
68  auto const VOL_dn = VOL_norm.norm();
69  auto const VOL_hz = 1.0 / 3.0 * (p11[2] + p22[2] + p33[2]);
70  VOL_partVol += VOL_A * -1 * VOL_norm[2] / VOL_dn * VOL_hz;
71  }
72 
73  return false;
74  });
75 
76  part_area_volume[0] = partArea;
77  part_area_volume[1] = VOL_partVol;
78 
79  MPI_Allreduce(part_area_volume.data(), area_volume.data(), 2, MPI_DOUBLE,
80  MPI_SUM, MPI_COMM_WORLD);
81 }

83 void add_oif_global_forces(Utils::Vector2d const &area_volume, int molType,
84  CellStructure &cs) {
85  // first-fold-then-the-same approach
86  double area = area_volume[0];
87  double VOL_volume = area_volume[1];
88 
89  cs.bond_loop([area, VOL_volume, molType](Particle &p1, int bond_id,
90  Utils::Span<Particle *> partners) {
91  if (p1.p.mol_id != molType)
92  return false;
93 
94  if (auto const *iaparams =
95  boost::get<OifGlobalForcesBond>(&bonded_ia_params[bond_id])) {
96  auto const p11 = unfolded_position(p1.r.p, p1.l.i, box_geo.length());
97  auto const p22 = p11 + box_geo.get_mi_vector(partners[0]->r.p, p11);
98  auto const p33 = p11 + box_geo.get_mi_vector(partners[1]->r.p, p11);
99 
100  // unfolded positions correct
101  // starting code from volume force
102  auto const VOL_norm = get_n_triangle(p11, p22, p33).normalize();
103  auto const VOL_A = area_triangle(p11, p22, p33);
104  auto const VOL_vv = (VOL_volume - iaparams->V0) / iaparams->V0;
105 
106  auto const VOL_force =
107  (1.0 / 3.0) * iaparams->kv * VOL_vv * VOL_A * VOL_norm;
108 
109  auto const h = (1. / 3.) * (p11 + p22 + p33);
110 
111  auto const deltaA = (area - iaparams->A0_g) / iaparams->A0_g;
112 
113  auto const m1 = h - p11;
114  auto const m2 = h - p22;
115  auto const m3 = h - p33;
116 
117  auto const m1_length = m1.norm();
118  auto const m2_length = m2.norm();
119  auto const m3_length = m3.norm();
120 
121  auto const fac = iaparams->ka_g * VOL_A * deltaA /
122  (m1_length * m1_length + m2_length * m2_length +
123  m3_length * m3_length);
124 
125  p1.f.f += fac * m1 + VOL_force;
126  partners[0]->f.f += fac * m2 + VOL_force;
127  partners[1]->f.f += fac * m3 + VOL_force;
128  }
129 
130  return false;
131  });
132 }
