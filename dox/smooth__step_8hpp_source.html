44 /** Calculate smooth step force factor */
45 inline double SmSt_pair_force_factor(IA_parameters const &ia_params,
46  double dist) {
47  if (dist >= ia_params.smooth_step.cut) {
48  return 0.0;
49  }
50 
51  auto const frac = ia_params.smooth_step.d / dist;
52  auto const fracP = pow(frac, ia_params.smooth_step.n);
53  auto const er =
54  exp(2. * ia_params.smooth_step.k0 * (dist - ia_params.smooth_step.sig));
55  return (ia_params.smooth_step.n * fracP +
56  2. * ia_params.smooth_step.eps * ia_params.smooth_step.k0 * dist *
57  er / Utils::sqr(1.0 + er)) /
58  Utils::sqr(dist);
59 }
60 
61 /** Calculate smooth step energy */
62 inline double SmSt_pair_energy(IA_parameters const &ia_params, double dist) {
63  if (dist >= ia_params.smooth_step.cut) {
64  return 0.0;
65  }
66 
67  auto const frac = ia_params.smooth_step.d / dist;
68  auto const fracP = pow(frac, ia_params.smooth_step.n);
69  auto const er =
70  exp(2. * ia_params.smooth_step.k0 * (dist - ia_params.smooth_step.sig));
71  return fracP + ia_params.smooth_step.eps / (1.0 + er);
72 }
