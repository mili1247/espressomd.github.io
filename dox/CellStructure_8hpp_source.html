392  /**
393  * @brief Resolve ids to particles.
394  *
395  * @throws BondResolutionError if one of the ids
396  * was not found.
397  *
398  * @param partner_ids Ids to resolve.
399  * @return Vector of Particle pointers.
400  */
401  auto resolve_bond_partners(Utils::Span<const int> partner_ids) {
402  boost::container::static_vector<Particle *, 4> partners;
403  get_local_particles(partner_ids, std::back_inserter(partners));
404 
405  /* Check if id resolution failed for any partner */
406  if (boost::algorithm::any_of(
407  partners, [](Particle *partner) { return partner == nullptr; })) {
408  throw BondResolutionError{};
409  }
410 
411  return partners;
412  }

414  /**
415  * @brief Execute kernel for every bond on particle.
416  * @tparam Handler Callable, which can be invoked with
417  * (Particle, int, Utils::Span<Particle *>),
418  * returning a bool.
419  * @param p Particles for whom the bonds are evaluated.
420  * @param handler is called for every bond, and handed
421  * p, the bond id and a span with the bond
422  * partners as arguments. Its return value
423  * should indicate if the bond was broken.
424  */
425  template <class Handler>
426  void execute_bond_handler(Particle &p, Handler handler) {
427  for (const BondView bond : p.bonds()) {
428  auto const partner_ids = bond.partner_ids();
429 
430  try {
431  auto partners = resolve_bond_partners(partner_ids);
432 
433  auto const bond_broken =
434  handler(p, bond.bond_id(), Utils::make_span(partners));
435 
436  if (bond_broken) {
437  bond_broken_error(p.identity(), partner_ids);
438  }
439  } catch (const BondResolutionError &) {
440  bond_broken_error(p.identity(), partner_ids);
441  }
442  }
443  }

508  /**
509  * @brief Run link_cell algorithm for local cells.
510  *
511  * @tparam Kernel Needs to be callable with (Particle, Particle, Distance).
512  * @param kernel Pair kernel functor.
513  */
514  template <class Kernel> void link_cell(Kernel kernel) {
515  auto const maybe_box = decomposition().minimum_image_distance();
516  auto const first = boost::make_indirect_iterator(local_cells().begin());
517  auto const last = boost::make_indirect_iterator(local_cells().end());
518 
519  if (maybe_box) {
520  Algorithm::link_cell(
521  first, last,
522  [&kernel, df = detail::MinimalImageDistance{*maybe_box}](
523  Particle &p1, Particle &p2) { kernel(p1, p2, df(p1, p2)); });
524  } else {
525  Algorithm::link_cell(
526  first, last,
527  [&kernel, df = detail::EuclidianDistance{}](
528  Particle &p1, Particle &p2) { kernel(p1, p2, df(p1, p2)); });
529  }
530  }

532  /** Non-bonded pair loop with verlet lists.
533  *
534  * @param pair_kernel Kernel to apply
535  * @param verlet_criterion Filter for verlet lists.
536  */
537  template <class PairKernel, class VerletCriterion>
538  void verlet_list_loop(PairKernel pair_kernel,
539  const VerletCriterion &verlet_criterion) {
540  /* In this case the verlet list update is attached to
541  * the pair kernel, and the verlet list is rebuilt as
542  * we go. */
543  if (m_rebuild_verlet_list) {
544  m_verlet_list.clear();
545 
546  link_cell([&](Particle &p1, Particle &p2, Distance const &d) {
547  if (verlet_criterion(p1, p2, d)) {
548  m_verlet_list.emplace_back(&p1, &p2);
549  pair_kernel(p1, p2, d);
550  }
551  });
552 
553  m_rebuild_verlet_list = false;
554  } else {
555  auto const maybe_box = decomposition().minimum_image_distance();
556  /* In this case the pair kernel is just run over the verlet list. */
557  if (maybe_box) {
558  auto const distance_function = detail::MinimalImageDistance{*maybe_box};
559  for (auto &pair : m_verlet_list) {
560  pair_kernel(*pair.first, *pair.second,
561  distance_function(*pair.first, *pair.second));
562  }
563  } else {
564  auto const distance_function = detail::EuclidianDistance{};
565  for (auto &pair : m_verlet_list) {
566  pair_kernel(*pair.first, *pair.second,
567  distance_function(*pair.first, *pair.second));
568  }
569  }
570  }
571  }
