36 inline void brownian_dynamics_propagator(BrownianThermostat const &brownian,
37  const ParticleRange &particles,
38  double time_step, double kT) {
39  for (auto &p : particles) {
40  // Don't propagate translational degrees of freedom of vs
41  if (!(p.p.is_virtual) or thermo_virtual) {
42  p.r.p += bd_drag(brownian.gamma, p, time_step);
43  p.m.v = bd_drag_vel(brownian.gamma, p);
44  p.r.p += bd_random_walk(brownian, p, time_step, kT);
45  p.m.v += bd_random_walk_vel(brownian, p);
46  /* Verlet criterion check */
47  if ((p.r.p - p.l.p_old).norm2() > Utils::sqr(0.5 * skin))
48  cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
49  }
50 #ifdef ROTATION
51  if (!p.p.rotation)
52  continue;
53  convert_torque_to_body_frame_apply_fix(p);
54  p.r.quat = bd_drag_rot(brownian.gamma_rotation, p, time_step);
55  p.m.omega = bd_drag_vel_rot(brownian.gamma_rotation, p);
56  p.r.quat = bd_random_walk_rot(brownian, p, time_step, kT);
57  p.m.omega += bd_random_walk_vel_rot(brownian, p);
58 #endif // ROTATION
59  }
60  increment_sim_time(time_step);
61 }
