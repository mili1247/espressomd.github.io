52 double ShapeBasedConstraint::min_dist(const ParticleRange &particles) {
53  double global_mindist = std::numeric_limits<double>::infinity();
54 
55  auto const local_mindist = std::accumulate(
56  particles.begin(), particles.end(),
57  std::numeric_limits<double>::infinity(),
58  [this](double min, Particle const &p) {
59  IA_parameters const &ia_params =
60  *get_ia_param(p.p.type, part_rep.p.type);
61  if (checkIfInteraction(ia_params)) {
62  double dist;
63  Utils::Vector3d vec;
64  m_shape->calculate_dist(folded_position(p.r.p, box_geo), dist, vec);
65  return std::min(min, dist);
66  }
67  return min;
68  });
69  boost::mpi::reduce(comm_cart, local_mindist, global_mindist,
70  boost::mpi::minimum<double>(), 0);
71  return global_mindist;
72 }

74 ParticleForce ShapeBasedConstraint::force(Particle const &p,
75  Utils::Vector3d const &folded_pos,
76  double t) {
77  ParticleForce pf{};
78  IA_parameters const &ia_params = *get_ia_param(p.p.type, part_rep.p.type);
79 
80  if (checkIfInteraction(ia_params)) {
81  double dist = 0.;
82  Utils::Vector3d dist_vec;
83  m_shape->calculate_dist(folded_pos, dist, dist_vec);
84 
85 #ifdef DPD
86  Utils::Vector3d dpd_force{};
87 #endif
88  Utils::Vector3d outer_normal_vec{};
89 
90  if (dist > 0) {
91  outer_normal_vec = -dist_vec / dist;
92  pf = calc_non_bonded_pair_force(p, part_rep, ia_params, dist_vec, dist);
93 #ifdef DPD
94  if (thermo_switch & THERMO_DPD) {
95  dpd_force =
96  dpd_pair_force(p, part_rep, ia_params, dist_vec, dist, dist * dist);
97  // Additional use of DPD here requires counter increase
98  dpd.rng_increment();
99  }
100 #endif
101  } else if (m_penetrable && (dist <= 0)) {
102  if ((!m_only_positive) && (dist < 0)) {
103  pf =
104  calc_non_bonded_pair_force(p, part_rep, ia_params, dist_vec, -dist);
105 #ifdef DPD
106  if (thermo_switch & THERMO_DPD) {
107  dpd_force = dpd_pair_force(p, part_rep, ia_params, dist_vec, dist,
108  dist * dist);
109  // Additional use of DPD here requires counter increase
110  dpd.rng_increment();
111  }
112 #endif
113  }
114  } else {
115  runtimeErrorMsg() << "Constraint"
116  << " violated by particle " << p.p.identity << " dist "
117  << dist;
118  }
119 
120 #ifdef ROTATION
121  part_rep.f.torque += calc_opposing_force(pf, dist_vec).torque;
122 #endif
123 #ifdef DPD
124  pf.f += dpd_force;
125 #endif
126  m_local_force -= pf.f;
127  m_outer_normal_force -= outer_normal_vec * pf.f;
128  }
129  return pf;
130 }

132 void ShapeBasedConstraint::add_energy(const Particle &p,
133  const Utils::Vector3d &folded_pos,
134  double t,
135  Observable_stat &obs_energy) const {
136  double energy = 0.0;
137 
138  IA_parameters const &ia_params = *get_ia_param(p.p.type, part_rep.p.type);
139 
140  if (checkIfInteraction(ia_params)) {
141  double dist = 0.0;
142  Utils::Vector3d vec;
143  m_shape->calculate_dist(folded_pos, dist, vec);
144  if (dist > 0) {
145  energy = calc_non_bonded_pair_energy(p, part_rep, ia_params, vec, dist);
146  } else if ((dist <= 0) && m_penetrable) {
147  if (!m_only_positive && (dist < 0)) {
148  energy = calc_non_bonded_pair_energy(p, part_rep, ia_params, vec,
149  -1.0 * dist);
150  }
151  } else {
152  runtimeErrorMsg() << "Constraint violated by particle " << p.p.identity;
153  }
154  }
155  if (part_rep.p.type >= 0)
156  obs_energy.add_non_bonded_contribution(p.p.type, part_rep.p.type, energy);
157 }
