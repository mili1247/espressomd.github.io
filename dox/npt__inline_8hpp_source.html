36 /** Add velocity-dependent noise and friction for NpT-sims to the particle's
37  * velocity
38  * @tparam step Which half time step to integrate (1 or 2)
39  * @param npt_iso Parameters
40  * @param vel particle velocity
41  * @param p_identity particle identity
42  * @return noise added to the velocity, already rescaled by
43  * dt/2 (contained in prefactors)
44  */
45 template <std::size_t step>
46 inline Utils::Vector3d
47 friction_therm0_nptiso(IsotropicNptThermostat const &npt_iso,
48  Utils::Vector3d const &vel, int p_identity) {
49  static_assert(step == 1 or step == 2, "NPT only has 2 integration steps");
50  constexpr auto const salt =
51  (step == 1) ? RNGSalt::NPTISO0_HALF_STEP1 : RNGSalt::NPTISO0_HALF_STEP2;
52  if (thermo_switch & THERMO_NPT_ISO) {
53  if (npt_iso.pref_noise_0 > 0.0) {
54  return npt_iso.pref_rescale_0 * vel +
55  npt_iso.pref_noise_0 *
56  Random::noise_uniform<salt>(npt_iso.rng_counter(),
57  npt_iso.rng_seed(), p_identity);
58  }
59  return npt_iso.pref_rescale_0 * vel;
60  }
61  return {};
62 }

64 /** Add p_diff-dependent noise and friction for NpT-sims to \ref
65  * NptIsoParameters::p_diff
66  */
67 inline double friction_thermV_nptiso(IsotropicNptThermostat const &npt_iso,
68  double p_diff) {
69  if (thermo_switch & THERMO_NPT_ISO) {
70  if (npt_iso.pref_noise_V > 0.0) {
71  return npt_iso.pref_rescale_V * p_diff +
72  npt_iso.pref_noise_V *
73  Random::noise_uniform<RNGSalt::NPTISOV, 1>(
74  npt_iso.rng_counter(), npt_iso.rng_seed(), 0);
75  }
76  return npt_iso.pref_rescale_V * p_diff;
77  }
78  return 0.0;
79 }
