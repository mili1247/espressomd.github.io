72 /** Compute the three-body angle interaction force.
73  * @param[in] r_mid Position of second/middle particle.
74  * @param[in] r_left Position of first/left particle.
75  * @param[in] r_right Position of third/right particle.
76  * @return Forces on the second, first and third particles, in that order.
77  */
78 inline std::tuple<Utils::Vector3d, Utils::Vector3d, Utils::Vector3d>
79 AngleCosineBond::forces(Utils::Vector3d const &r_mid,
80  Utils::Vector3d const &r_left,
81  Utils::Vector3d const &r_right) const {
82 
83  auto forceFactor = [this](double const cos_phi) {
84  auto const sin_phi = sqrt(1 - Utils::sqr(cos_phi));
85  // angle force term: K(phi) = -k * sin(phi - phi0) / sin(phi)
86  // trig identity: sin(a - b) = sin(a)cos(b) - cos(a)sin(b)
87  return -bend * (sin_phi * cos_phi0 - cos_phi * sin_phi0) / sin_phi;
88  };
89 
90  return angle_generic_force(r_mid, r_left, r_right, forceFactor, false);
91 }

93 /** Computes the three-body angle interaction energy.
94  * @param[in] r_mid Position of second/middle particle.
95  * @param[in] r_left Position of first/left particle.
96  * @param[in] r_right Position of third/right particle.
97  */
98 inline double AngleCosineBond::energy(Utils::Vector3d const &r_mid,
99  Utils::Vector3d const &r_left,
100  Utils::Vector3d const &r_right) const {
101  auto const vectors = calc_vectors_and_cosine(r_mid, r_left, r_right, true);
102  auto const cos_phi = std::get<4>(vectors);
103  auto const sin_phi = sqrt(1 - Utils::sqr(cos_phi));
104  // potential: U(phi) = k * [1 - cos(phi - phi0)]
105  // trig identity: cos(phi - phi0) = cos(phi)cos(phi0) + sin(phi)sin(phi0)
106  return bend * (1 - (cos_phi * cos_phi0 + sin_phi * sin_phi0));
107 }
