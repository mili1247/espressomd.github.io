34 /** Compute the cosine of the angle between three particles.
35  *
36  * Also return all intermediate quantities: normalized vectors
37  * @f$ \vec{r_{ij}} @f$ (from particle @f$ j @f$ to particle @f$ i @f$)
38  * and @f$ \vec{r_{kj}} @f$, and their normalization constants.
39  *
40  * @param[in] r_mid Position of second/middle particle.
41  * @param[in] r_left Position of first/left particle.
42  * @param[in] r_right Position of third/right particle.
43  * @param[in] sanitize_cosine Sanitize the cosine of the angle.
44  * @return @f$ \vec{r_{ij}} @f$, @f$ \vec{r_{kj}} @f$,
45  * @f$ \left\|\vec{r_{ij}}\right\|^{-1} @f$,
46  * @f$ \left\|\vec{r_{kj}}\right\|^{-1} @f$,
47  * @f$ \cos(\theta_{ijk}) @f$
48  */
49 inline std::tuple<Utils::Vector3d, Utils::Vector3d, double, double, double>
50 calc_vectors_and_cosine(Utils::Vector3d const &r_mid,
51  Utils::Vector3d const &r_left,
52  Utils::Vector3d const &r_right,
53  bool sanitize_cosine = false) {
54  /* normalized vector from p_mid to p_left */
55  auto vec1 = box_geo.get_mi_vector(r_left, r_mid);
56  auto const d1i = 1.0 / vec1.norm();
57  vec1 *= d1i;
58  /* normalized vector from p_mid to p_right */
59  auto vec2 = box_geo.get_mi_vector(r_right, r_mid);
60  auto const d2i = 1.0 / vec2.norm();
61  vec2 *= d2i;
62  /* cosine of the angle between vec1 and vec2 */
63  auto cosine = vec1 * vec2;
64  if (sanitize_cosine) {
65  if (cosine > TINY_COS_VALUE)
66  cosine = TINY_COS_VALUE;
67  if (cosine < -TINY_COS_VALUE)
68  cosine = -TINY_COS_VALUE;
69  }
70  return std::make_tuple(vec1, vec2, d1i, d2i, cosine);
71 }

73 /** Compute a three-body angle interaction force.
74  *
75  * See the details in @ref bondedIA_angle_force. The @f$ K(\theta_{ijk}) @f$
76  * term is provided as a lambda function in @p forceFactor.
77  *
78  * @param[in] r_mid Position of second/middle particle.
79  * @param[in] r_left Position of first/left particle.
80  * @param[in] r_right Position of third/right particle.
81  * @param[in] forceFactor Angle force term.
82  * @param[in] sanitize_cosine Sanitize the cosine of the angle.
83  * @tparam ForceFactor Function evaluating the angle force term
84  * for a given angle.
85  * @return Forces on the second, first and third particles, in that order.
86  */
87 template <typename ForceFactor>
88 std::tuple<Utils::Vector3d, Utils::Vector3d, Utils::Vector3d>
89 angle_generic_force(Utils::Vector3d const &r_mid, Utils::Vector3d const &r_left,
90  Utils::Vector3d const &r_right, ForceFactor forceFactor,
91  bool sanitize_cosine) {
92  Utils::Vector3d vec1, vec2;
93  double d1i, d2i, cosine;
94  std::tie(vec1, vec2, d1i, d2i, cosine) =
95  calc_vectors_and_cosine(r_mid, r_left, r_right, sanitize_cosine);
96  /* force factor */
97  auto const fac = forceFactor(cosine);
98  /* distribute forces */
99  auto f_left = (fac * d1i) * (vec1 * cosine - vec2);
100  auto f_right = (fac * d2i) * (vec2 * cosine - vec1);
101  auto f_mid = -(f_left + f_right);
102  return std::make_tuple(f_mid, f_left, f_right);
103 }
