36 std::array<double, 4> calc_re(int chain_start, int chain_n_chains,
37  int chain_length) {
38  double dist = 0.0, dist2 = 0.0, dist4 = 0.0;
39  std::array<double, 4> re;
40 
41  for (int i = 0; i < chain_n_chains; i++) {
42  auto const &p1 =
43  get_particle_data(chain_start + i * chain_length + chain_length - 1);
44  auto const &p2 = get_particle_data(chain_start + i * chain_length);
45 
46  auto const d = unfolded_position(p1.r.p, p1.l.i, box_geo.length()) -
47  unfolded_position(p2.r.p, p2.l.i, box_geo.length());
48  auto const norm2 = d.norm2();
49  dist += sqrt(norm2);
50  dist2 += norm2;
51  dist4 += norm2 * norm2;
52  }
53  auto const tmp = static_cast<double>(chain_n_chains);
54  re[0] = dist / tmp;
55  re[2] = dist2 / tmp;
56  re[1] = sqrt(re[2] - re[0] * re[0]);
57  re[3] = sqrt(dist4 / tmp - re[2] * re[2]);
58  return re;
59 }

61 std::array<double, 4> calc_rg(int chain_start, int chain_n_chains,
62  int chain_length) {
63  double r_G = 0.0, r_G2 = 0.0, r_G4 = 0.0;
64  double tmp;
65  std::array<double, 4> rg;
66 
67  for (int i = 0; i < chain_n_chains; i++) {
68  double M = 0.0;
69  Utils::Vector3d r_CM{};
70  for (int j = 0; j < chain_length; j++) {
71  auto const &p = get_particle_data(chain_start + i * chain_length + j);
72 
73  if (p.p.is_virtual) {
74  throw std::runtime_error(
75  "Gyration tensor is not well-defined for chains including virtual "
76  "sites. Virtual sites do not have a meaningful mass.");
77  }
78  r_CM += unfolded_position(p.r.p, p.l.i, box_geo.length()) * p.p.mass;
79  M += p.p.mass;
80  }
81  r_CM /= M;
82  tmp = 0.0;
83  for (int j = 0; j < chain_length; ++j) {
84  auto const &p = get_particle_data(chain_start + i * chain_length + j);
85  Utils::Vector3d const d =
86  unfolded_position(p.r.p, p.l.i, box_geo.length()) - r_CM;
87  tmp += d.norm2();
88  }
89  tmp /= (double)chain_length;
90  r_G += sqrt(tmp);
91  r_G2 += tmp;
92  r_G4 += tmp * tmp;
93  }
94  tmp = static_cast<double>(chain_n_chains);
95  rg[0] = r_G / tmp;
96  rg[2] = r_G2 / tmp;
97  rg[1] = sqrt(rg[2] - rg[0] * rg[0]);
98  rg[3] = sqrt(r_G4 / tmp - rg[2] * rg[2]);
99  return rg;
100 }

102 std::array<double, 2> calc_rh(int chain_start, int chain_n_chains,
103  int chain_length) {
104  double r_H = 0.0, r_H2 = 0.0, ri = 0.0, prefac, tmp;
105  std::array<double, 2> rh;
106 
107  prefac = 0.5 * chain_length * (chain_length - 1);
108  for (int p = 0; p < chain_n_chains; p++) {
109  ri = 0.0;
110  for (int i = chain_start + chain_length * p;
111  i < chain_start + chain_length * (p + 1); i++) {
112  auto const &p1 = get_particle_data(i);
113  for (int j = i + 1; j < chain_start + chain_length * (p + 1); j++) {
114  auto const &p2 = get_particle_data(j);
115  auto const d = unfolded_position(p1.r.p, p1.l.i, box_geo.length()) -
116  unfolded_position(p2.r.p, p2.l.i, box_geo.length());
117  ri += 1.0 / d.norm();
118  }
119  }
120  tmp = prefac / ri;
121  r_H += tmp;
122  r_H2 += tmp * tmp;
123  }
124  tmp = static_cast<double>(chain_n_chains);
125  rh[0] = r_H / tmp;
126  rh[1] = sqrt(r_H2 / tmp - rh[0] * rh[0]);
127  return rh;
