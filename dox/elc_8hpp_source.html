41 /** @brief Parameters for the ELC method */
42 struct ELCParameters {
43  /** Maximal allowed pairwise error for the potential and force.
44  * Used at least by the near formula, since this does the error control at
45  * runtime.
46  */
47  double maxPWerror;
48  /** Cutoff of the exponential sum. Since in all other MMM methods this is
49  * the far formula, we call it here the same, although in the ELC context
50  * it does not make much sense.
51  */
52  double far_cut;
53  /** Squared value of #far_cut. */
54  double far_cut2;
55  /** Size of the empty gap. Note that ELC relies on the user to make sure
56  * that this condition is fulfilled.
57  */
58  double gap_size;
59  /** Flag whether #far_cut was set by the user, or calculated by ESPResSo.
60  * In the latter case, the cutoff will be adapted if important parameters,
61  * such as the box dimensions, change.
62  */
63  bool far_calculated;
64  /** Flag whether the box is neutralized by a homogeneous background.
65  * If true, use a homogeneous neutralizing background for nonneutral
66  * systems. Unlike the 3D case, this background adds an additional
67  * force pointing towards the system center, so be careful with this.
68  */
69  bool neutralize;
70 
71  /// flag whether there is any dielectric contrast in the system.
72  bool dielectric_contrast_on;
73 
74  /// dielectric contrast in the upper part of the simulation cell.
75  double delta_mid_top;
76  /// dielectric contrast in the lower part of the simulation cell.
77  double delta_mid_bot;
78  /// @brief Flag whether a const. potential is applied.
79  bool const_pot;
80  /// @brief Const. potential.
81  double pot_diff;
82 
83  /** Minimal distance of two charges for which the far formula is used.
84  * For plain ELC, this equals #gap_size, but for dielectric ELC it is
85  * only 1/3 of that.
86  */
87  double minimal_dist;
88  /** Layer around the dielectric contrast in which we trick around. */
89  double space_layer;
90  /** The space that is finally left. */
91  double space_box;
92  /** Up to where particles can be found. */
93  double h;
94 };
