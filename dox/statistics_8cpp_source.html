52 double mindist(PartCfg &partCfg, const std::vector<int> &set1,
53  const std::vector<int> &set2) {
54  using Utils::contains;
55 
56  auto mindist2 = std::numeric_limits<double>::infinity();
57 
58  for (auto jt = partCfg.begin(); jt != partCfg.end(); ++jt) {
59  /* check which sets particle j belongs to (bit 0: set1, bit1: set2) */
60  auto in_set = 0u;
61  if (set1.empty() || contains(set1, jt->p.type))
62  in_set = 1u;
63  if (set2.empty() || contains(set2, jt->p.type))
64  in_set |= 2u;
65  if (in_set == 0)
66  continue;
67 
68  for (auto it = std::next(jt); it != partCfg.end(); ++it)
69  /* accept a pair if particle j is in set1 and particle i in set2 or vice
70  * versa. */
71  if (((in_set & 1u) && (set2.empty() || contains(set2, it->p.type))) ||
72  ((in_set & 2u) && (set1.empty() || contains(set1, it->p.type))))
73  mindist2 =
74  std::min(mindist2, box_geo.get_mi_vector(jt->r.p, it->r.p).norm2());
75  }
76 
77  return std::sqrt(mindist2);
78 }

108 Utils::Vector3d centerofmass(PartCfg &partCfg, int type) {
109  Utils::Vector3d com{};
110  double mass = 0.0;
111 
112  for (auto const &p : partCfg) {
113  if ((p.p.type == type) || (type == -1))
114  if (not p.p.is_virtual) {
115  com += p.r.p * p.p.mass;
116  mass += p.p.mass;
117  }
118  }
119  com /= mass;
120  return com;
121 }

135 void momentofinertiamatrix(PartCfg &partCfg, int type, double *MofImatrix) {
136  int i, count;
137  double massi;
138  Utils::Vector3d p1{};
139  count = 0;
140 
141  for (i = 0; i < 9; i++)
142  MofImatrix[i] = 0.;
143 
144  auto const com = centerofmass(partCfg, type);
145  for (auto const &p : partCfg) {
146  if (type == p.p.type and (not p.p.is_virtual)) {
147  count++;
148  p1 = p.r.p - com;
149  massi = p.p.mass;
150  MofImatrix[0] += massi * (p1[1] * p1[1] + p1[2] * p1[2]);
151  MofImatrix[4] += massi * (p1[0] * p1[0] + p1[2] * p1[2]);
152  MofImatrix[8] += massi * (p1[0] * p1[0] + p1[1] * p1[1]);
153  MofImatrix[1] -= massi * (p1[0] * p1[1]);
154  MofImatrix[2] -= massi * (p1[0] * p1[2]);
155  MofImatrix[5] -= massi * (p1[1] * p1[2]);
156  }
157  }
158  /* use symmetry */
159  MofImatrix[3] = MofImatrix[1];
160  MofImatrix[6] = MofImatrix[2];
161  MofImatrix[7] = MofImatrix[5];
162 }

164 std::vector<int> nbhood(PartCfg &partCfg, const Utils::Vector3d &pos,
165  double r_catch, const Utils::Vector3i &planedims) {
166  std::vector<int> ids;
167 
168  auto const r2 = r_catch * r_catch;
169  auto const pt = Utils::Vector3d{pos[0], pos[1], pos[2]};
170 
171  Utils::Vector3d d;
172 
173  for (auto const &p : partCfg) {
174  if ((planedims[0] + planedims[1] + planedims[2]) == 3) {
175  d = box_geo.get_mi_vector(pt, p.r.p);
176  } else {
177  /* Calculate the in plane distance */
178  for (int j = 0; j < 3; j++) {
179  d[j] = planedims[j] * (p.r.p[j] - pt[j]);
180  }
181  }
182 
183  if (d.norm2() < r2) {
184  ids.push_back(p.p.identity);
185  }
186  }
187 
188  return ids;
189 }

191 double distto(PartCfg &partCfg, const Utils::Vector3d &pos, int pid) {
192  auto mindist = std::numeric_limits<double>::infinity();
193 
194  for (auto const &part : partCfg) {
195  if (pid != part.p.identity) {
196  auto const d = box_geo.get_mi_vector({pos[0], pos[1], pos[2]}, part.r.p);
197  mindist = std::min(mindist, d.norm2());
198  }
199  }
200  return std::sqrt(mindist);
201 }

203 void calc_part_distribution(PartCfg &partCfg, std::vector<int> const &p1_types,
204  std::vector<int> const &p2_types, double r_min,
205  double r_max, int r_bins, bool log_flag,
206  double *low, double *dist) {
207  int ind, cnt = 0;
208  double inv_bin_width = 0.0;
209  double min_dist, min_dist2 = 0.0, start_dist2;
210 
211  start_dist2 = Utils::sqr(box_geo.length()[0] + box_geo.length()[1] +
212  box_geo.length()[2]);
213  /* bin preparation */
214  *low = 0.0;
215  for (int i = 0; i < r_bins; i++)
216  dist[i] = 0.0;
217  if (log_flag)
218  inv_bin_width = (double)r_bins / (log(r_max) - log(r_min));
219  else
220  inv_bin_width = (double)r_bins / (r_max - r_min);
221 
222  /* particle loop: p1_types */
223  for (auto const &p1 : partCfg) {
224  for (int t1 : p1_types) {
225  if (p1.p.type == t1) {
226  min_dist2 = start_dist2;
227  /* particle loop: p2_types */
228  for (auto const &p2 : partCfg) {
229  if (p1 != p2) {
230  for (int t2 : p2_types) {
231  if (p2.p.type == t2) {
232  auto const act_dist2 =
233  box_geo.get_mi_vector(p1.r.p, p2.r.p).norm2();
234  if (act_dist2 < min_dist2) {
235  min_dist2 = act_dist2;
236  }
237  }
238  }
239  }
240  }
241  min_dist = sqrt(min_dist2);
242  if (min_dist <= r_max) {
243  if (min_dist >= r_min) {
244  /* calculate bin index */
245  if (log_flag)
246  ind = (int)((log(min_dist) - log(r_min)) * inv_bin_width);
247  else
248  ind = (int)((min_dist - r_min) * inv_bin_width);
249  if (ind >= 0 && ind < r_bins) {
250  dist[ind] += 1.0;
251  }
252  } else {
253  *low += 1.0;
254  }
255  }
256  cnt++;
257  }
258  }
259  }
260  if (cnt == 0)
261  return;
262 
263  /* normalization */
264  *low /= (double)cnt;
265  for (int i = 0; i < r_bins; i++)
266  dist[i] /= (double)cnt;
267 }

269 void calc_structurefactor(PartCfg &partCfg, std::vector<int> const &p_types,
270  int order, std::vector<double> &wavevectors,
271  std::vector<double> &intensities) {
272 
273  if (order < 1)
274  throw std::domain_error("order has to be a strictly positive number");
275 
276  auto const order_sq = order * order;
277  std::vector<double> ff(2 * order_sq + 1);
278  auto const twoPI_L = 2 * Utils::pi() * box_geo.length_inv()[0];
279 
280  for (int i = 0; i <= order; i++) {
281  for (int j = -order; j <= order; j++) {
282  for (int k = -order; k <= order; k++) {
283  auto const n = i * i + j * j + k * k;
284  if ((n <= order_sq) && (n >= 1)) {
285  double C_sum = 0.0, S_sum = 0.0;
286  for (auto const &p : partCfg) {
287  for (int t : p_types) {
288  if (p.p.type == t) {
289  auto const qr = twoPI_L * (Utils::Vector3i{{i, j, k}} * p.r.p);
290  C_sum += cos(qr);
291  S_sum += sin(qr);
292  }
293  }
294  }
295  ff[2 * n - 2] += C_sum * C_sum + S_sum * S_sum;
296  ff[2 * n - 1]++;
297  }
298  }
299  }
300  }
301 
302  long n_particles = 0l;
303  for (auto const &p : partCfg) {
304  for (int t : p_types) {
305  if (p.p.type == t)
306  n_particles++;
307  }
308  }
309 
310  int length = 0;
311  for (int qi = 0; qi < order_sq; qi++) {
312  if (ff[2 * qi + 1] != 0) {
313  ff[2 * qi] /= static_cast<double>(n_particles) * ff[2 * qi + 1];
314  length++;
315  }
316  }
317 
318  wavevectors.resize(length);
319  intensities.resize(length);
320 
321  int cnt = 0;
322  for (int i = 0; i < order_sq; i++) {
323  if (ff[2 * i + 1] != 0) {
324  wavevectors[cnt] = twoPI_L * sqrt(i + 1);
325  intensities[cnt] = ff[2 * i];
326  cnt++;
327  }
328  }
329 }
