33 std::vector<double> CylindricalLBVelocityProfileAtParticlePositions::evaluate(
34  Utils::Span<std::reference_wrapper<const Particle>> particles,
35  const ParticleObservables::traits<Particle> &traits) const {
36  Utils::CylindricalHistogram<double, 3> histogram(n_bins(), limits());
37 
38  for (auto p : particles) {
39  auto const pos = folded_position(traits.position(p), box_geo);
40  auto const v = lb_lbfluid_get_interpolated_velocity(pos) *
41  lb_lbfluid_get_lattice_speed();
42 
43  histogram.update(
44  Utils::transform_coordinate_cartesian_to_cylinder(
45  pos - transform_params->center(), transform_params->axis(),
46  transform_params->orientation()),
47  Utils::transform_vector_cartesian_to_cylinder(
48  v, transform_params->axis(), pos - transform_params->center()));
49  }
50 
51  // normalize by number of hits per bin
52  auto hist_tmp = histogram.get_histogram();
53  auto tot_count = histogram.get_tot_count();
54  std::transform(hist_tmp.begin(), hist_tmp.end(), tot_count.begin(),
55  hist_tmp.begin(), [](auto hi, auto ci) {
56  return ci > 0 ? hi / static_cast<double>(ci) : 0.;
57  });
58  return hist_tmp;
59 }
