21 /** \file
22  *
23  * %Lattice Boltzmann algorithm for hydrodynamic degrees of freedom.
24  *
25  * For performance reasons it is clever to do streaming and collision at the
26  * same time because every fluid node has to be read and written only once.
27  * This increases mainly cache efficiency.
28  *
29  * The hydrodynamic fields, corresponding to density, velocity and pressure,
30  * are stored in @ref LB_FluidNode in the array @ref lbfields, the populations
31  * in @ref LB_Fluid in the array @ref lbfluid which is constructed as
32  * 2 x (Nx x Ny x Nz) x 19 array.
33  *
34  * Implementation in lb.cpp.
35  */

230 inline Utils::Vector19d lb_get_population(Lattice::index_t index) {
231  Utils::Vector19d pop{};
232  for (int i = 0; i < D3Q19::n_vel; ++i) {
233  pop[i] = lbfluid[i][index] + D3Q19::coefficients[i][0] * lbpar.density;
234  }
235  return pop;
236 }
237 
238 inline void lb_set_population(Lattice::index_t index,
239  const Utils::Vector19d &pop) {
240  for (int i = 0; i < D3Q19::n_vel; ++i) {
241  lbfluid[i][index] = pop[i] - D3Q19::coefficients[i][0] * lbpar.density;
242  }
243 }
