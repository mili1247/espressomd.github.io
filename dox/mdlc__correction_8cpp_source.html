54 /** Checks if a magnetic particle is in the forbidden gap region
55  */
56 inline void check_gap_mdlc(const Particle &p) {
57  if (p.p.dipm != 0.0) {
58  if (p.r.p[2] < 0.0)
59  runtimeErrorMsg() << "Particle " << p.p.identity << " entered MDLC gap "
60  << "region by " << (p.r.p[2]);
61  else if (p.r.p[2] > dlc_params.h) {
62  runtimeErrorMsg() << "Particle " << p.p.identity << " entered MDLC gap "
63  << "region by " << (p.r.p[2] - dlc_params.h);
64  }
65  }
66 }

68 /** Calculate the maximal dipole moment in the system */
69 double calc_mu_max() {
70  auto const local_particles = cell_structure.local_particles();
71  auto const mu_max_local = std::accumulate(
72  local_particles.begin(), local_particles.end(), 0.0,
73  [](double mu, Particle const &p) { return std::max(mu, p.p.dipm); });
74 
75  double mu_max;
76  boost::mpi::reduce(comm_cart, mu_max_local, mu_max,
77  boost::mpi::maximum<double>(), 0);
78  return mu_max;

83 inline double g1_DLC_dip(double g, double x) {
84  auto const c = g / x;
85  auto const cc2 = c * c;
86  auto const x3 = x * x * x;
87  auto const a = g * g * g / x + 1.5 * cc2 + 1.5 * g / x3 + 0.75 / (x3 * x);
88  return a;
89 }

91 inline double g2_DLC_dip(double g, double x) {
92  auto const x2 = x * x;
93  auto const a = g * g / x + 2.0 * g / x2 + 2.0 / (x2 * x);
94  return a;
95 }

110 /** Compute the dipolar DLC corrections for forces and torques.
111  * %Algorithm implemented accordingly to @cite brodka04a.
112  */
113 double get_DLC_dipolar(int kcut, std::vector<Utils::Vector3d> &fs,
114  std::vector<Utils::Vector3d> &ts,
115  const ParticleRange &particles) {
116  auto const n_local_particles = particles.size();
117 
118  std::vector<double> ReSjp(n_local_particles), ReSjm(n_local_particles);
119  std::vector<double> ImSjp(n_local_particles), ImSjm(n_local_particles);
120  std::vector<double> ReGrad_Mup(n_local_particles),
121  ImGrad_Mup(n_local_particles);
122  std::vector<double> ReGrad_Mum(n_local_particles),
123  ImGrad_Mum(n_local_particles);
124  double s1, s2, s3, s4;
125  double s1z, s2z, s3z, s4z;
126  double ss;
127 
128  auto const facux = 2.0 * Utils::pi() * box_geo.length_inv()[0];
129  auto const facuy = 2.0 * Utils::pi() * box_geo.length_inv()[1];
130  double energy = 0.0;
131 
132  for (int ix = -kcut; ix <= +kcut; ix++) {
133  for (int iy = -kcut; iy <= +kcut; iy++) {
134  if (!(ix == 0 && iy == 0)) {
135  auto const gx = static_cast<double>(ix) * facux;
136  auto const gy = static_cast<double>(iy) * facuy;
137 
138  auto const gr = sqrt(gx * gx + gy * gy);
139 
140  // We assume short slab direction is z direction
141  auto const fa1 = 1. / (gr * (exp(gr * box_geo.length()[2]) - 1.0));
142 
143  // ... Compute S+,(S+)*,S-,(S-)*, and Spj,Smj for the current g
144 
145  double S[4] = {0.0, 0.0, 0.0, 0.0}; // S of Brodka method, or is S[4] =
146  // {Re(S+), Im(S+), Re(S-), Im(S-)}
147  int ip = 0;
148 
149  for (auto const &p : particles) {
150  if (p.p.dipm > 0) {
151  Utils::Vector3d const dip = p.calc_dip();
152 
153  auto const a = gx * dip[0] + gy * dip[1];
154  auto const b = gr * dip[2];
155  auto const er = gx * p.r.p[0] + gy * p.r.p[1];
156  auto const ez = gr * p.r.p[2];
157  auto const c = cos(er);
158  auto const d = sin(er);
159  auto const f = exp(ez);
160 
161  ReSjp[ip] = (b * c - a * d) * f;
162  ImSjp[ip] = (c * a + b * d) * f;
163  ReSjm[ip] = (-b * c - a * d) / f;
164  ImSjm[ip] = (c * a - b * d) / f;
165  ReGrad_Mup[ip] = c * f;
166  ReGrad_Mum[ip] = c / f;
167  ImGrad_Mup[ip] = d * f;
168  ImGrad_Mum[ip] = d / f;
169 
170  S[0] += ReSjp[ip];
171  S[1] += ImSjp[ip];
172  S[2] += ReSjm[ip];
173  S[3] += ImSjm[ip];
174  }
175  ip++;
176  }
177 
178  MPI_Allreduce(MPI_IN_PLACE, S, 4, MPI_DOUBLE, MPI_SUM, comm_cart);
179 
180  // We compute the contribution to the energy ............
181 
182  // s2=(ReSm*ReSp+ImSm*ImSp); s2=s1!!!
183 
184  energy += fa1 * ((S[0] * S[2] + S[1] * S[3]) * 2.0);
185 
186  // ... Now we can compute the contributions to E,Fj,Ej for the current
187  // g-value
188  ip = 0;
189  for (auto &p : particles) {
190  if (p.p.dipm > 0) {
191  // We compute the contributions to the forces ............
192 
193  s1 = -(-ReSjp[ip] * S[3] + ImSjp[ip] * S[2]);
194  s2 = +(ReSjm[ip] * S[1] - ImSjm[ip] * S[0]);
195  s3 = -(-ReSjm[ip] * S[1] + ImSjm[ip] * S[0]);
196  s4 = +(ReSjp[ip] * S[3] - ImSjp[ip] * S[2]);
197 
198  s1z = +(ReSjp[ip] * S[2] + ImSjp[ip] * S[3]);
199  s2z = -(ReSjm[ip] * S[0] + ImSjm[ip] * S[1]);
200  s3z = -(ReSjm[ip] * S[0] + ImSjm[ip] * S[1]);
201  s4z = +(ReSjp[ip] * S[2] + ImSjp[ip] * S[3]);
202 
203  ss = s1 + s2 + s3 + s4;
204  fs[ip][0] += fa1 * gx * ss;
205  fs[ip][1] += fa1 * gy * ss;
206  fs[ip][2] += fa1 * gr * (s1z + s2z + s3z + s4z);
207 
208  // We compute the contributions to the electrical field
209  // ............
210 
211  s1 = -(-ReGrad_Mup[ip] * S[3] + ImGrad_Mup[ip] * S[2]);
212  s2 = +(ReGrad_Mum[ip] * S[1] - ImGrad_Mum[ip] * S[0]);
213  s3 = -(-ReGrad_Mum[ip] * S[1] + ImGrad_Mum[ip] * S[0]);
214  s4 = +(ReGrad_Mup[ip] * S[3] - ImGrad_Mup[ip] * S[2]);
215 
216  s1z = +(ReGrad_Mup[ip] * S[2] + ImGrad_Mup[ip] * S[3]);
217  s2z = -(ReGrad_Mum[ip] * S[0] + ImGrad_Mum[ip] * S[1]);
218  s3z = -(ReGrad_Mum[ip] * S[0] + ImGrad_Mum[ip] * S[1]);
219  s4z = +(ReGrad_Mup[ip] * S[2] + ImGrad_Mup[ip] * S[3]);
220 
221  ss = s1 + s2 + s3 + s4;
222  ts[ip][0] += fa1 * gx * ss;
223  ts[ip][1] += fa1 * gy * ss;
224  ts[ip][2] += fa1 * gr * (s1z + s2z + s3z + s4z);
225  } // if dipm>0 ....
226  ip++;
227  } // loop j
228  } // end of if(ii> ...
229  }
230  } // end of loops for gx,gy
231 
232  // Convert from the corrections to the Electrical field to the corrections
233  // for the torques ....
234 
235  int ip = 0;
236  for (auto const &p : particles) {
237  if (p.p.dipm > 0) {
238  ts[ip] = vector_product(p.calc_dip(), ts[ip]);
239  }
240  ip++;
241  }
242 
243  // Multiply by the factors we have left during the loops
244 
245  auto const piarea =
246  Utils::pi() * box_geo.length_inv()[0] * box_geo.length_inv()[1];
247 
248  for (int j = 0; j < n_local_particles; j++) {
249  fs[j] *= piarea;
250  ts[j] *= piarea;
251  }
252 
253  energy *= (-piarea);
254 
255  return energy;
256 }

258 /** Compute the dipolar DLC corrections
259  * %Algorithm implemented accordingly to @cite brodka04a.
260  */
261 double get_DLC_energy_dipolar(int kcut, const ParticleRange &particles) {
262  auto const facux = 2.0 * Utils::pi() * box_geo.length_inv()[0];
263  auto const facuy = 2.0 * Utils::pi() * box_geo.length_inv()[1];
264 
265  double energy = 0.0;
266  for (int ix = -kcut; ix <= +kcut; ix++) {
267  for (int iy = -kcut; iy <= +kcut; iy++) {
268 
269  if (!(ix == 0 && iy == 0)) {
270  auto const gx = static_cast<double>(ix) * facux;
271  auto const gy = static_cast<double>(iy) * facuy;
272  auto const gr = sqrt(gx * gx + gy * gy);
273  // We assume short slab direction is z direction
274  auto const fa1 = 1. / (gr * (exp(gr * box_geo.length()[2]) - 1.0));
275 
276  // ... Compute S+,(S+)*,S-,(S-)*, and Spj,Smj for the current g
277 
278  double S[4] = {0.0, 0.0, 0.0, 0.0};
279  int ip = 0;
280  for (auto const &p : particles) {
281  if (p.p.dipm > 0) {
282  const Utils::Vector3d dip = p.calc_dip();
283 
284  auto const a = gx * dip[0] + gy * dip[1];
285  {
286  auto const b = gr * dip[2];
287  auto const er = gx * p.r.p[0] + gy * p.r.p[1];
288  auto const ez = gr * p.r.p[2];
289  auto const c = cos(er);
290  auto const d = sin(er);
291  auto const f = exp(ez);
292 
293  S[0] += (b * c - a * d) * f;
294  S[1] += (c * a + b * d) * f;
295  S[2] += (-b * c - a * d) / f;
296  S[3] += (c * a - b * d) / f;
297  }
298  }
299  ip++;
300  }
301 
302  double global_S[4];
303  MPI_Reduce(S, global_S, 4, MPI_DOUBLE, MPI_SUM, 0, comm_cart);
304 
305  // We compute the contribution to the energy ............
306  auto const s1 = global_S[0] * global_S[2] + global_S[1] * global_S[3];
307  // s2=(ReSm*ReSp+ImSm*ImSp); s2=s1!!!
308 
309  energy += fa1 * (s1 * 2.0);
310 
311  } // end of if(...
312  }
313  } // end of loops for gx,gy
314 
315  // Multiply by the factors we have left during the loops
316 
317  auto const piarea =
318  Utils::pi() * box_geo.length_inv()[0] * box_geo.length_inv()[1];
319  energy *= (-piarea);
320  return (this_node == 0) ? energy : 0.0;
321 }

323 /** Compute and add the terms needed to correct the 3D dipolar
324  * methods when we have a slab geometry
325  */
326 void add_mdlc_force_corrections(const ParticleRange &particles) {
327  auto const volume = box_geo.volume();
328  auto const correc = 4. * Utils::pi() / volume;
329 
330  // --- Create arrays that should contain the corrections to
331  // the forces and torques, and set them to zero.
332  std::vector<Utils::Vector3d> dip_DLC_f(particles.size());
333  std::vector<Utils::Vector3d> dip_DLC_t(particles.size());
334 
335  //---- Compute the corrections ----------------------------------
336 
337  // First the DLC correction
338  get_DLC_dipolar(static_cast<int>(std::round(dlc_params.far_cut)), dip_DLC_f,
339  dip_DLC_t, particles);
340 
341  // Now we compute the correction like Yeh and Klapp to take into account
342  // the fact that you are using a 3D PBC method which uses spherical
343  // summation instead of slab-wise summation.
344  // Slab-wise summation is the one required to apply DLC correction.
345  // This correction is often called SDC = Shape Dependent Correction.
346  // See @cite brodka04a.
347 
348  auto const box_dip = calc_slab_dipole(particles);
349 
350  // --- Transfer the computed corrections to the Forces, Energy and torques
351  // of the particles
352 
353  int ip = 0;
354  for (auto &p : particles) {
355  check_gap_mdlc(p);
356 
357  if (p.p.dipm != 0.0) {
358  // SDC correction term is zero for the forces
359  p.f.f += dipole.prefactor * dip_DLC_f[ip];
360 
361  auto const dip = p.calc_dip();
362  // SDC correction for the torques
363  Utils::Vector3d d = {0.0, 0.0, -correc * box_dip[2]};
364 #ifdef DP3M
365  if (dipole.method == DIPOLAR_MDLC_P3M and
366  dp3m.params.epsilon != P3M_EPSILON_METALLIC) {
367  auto const correps = correc / (2.0 * dp3m.params.epsilon + 1.0);
368  d += correps * box_dip;
369  }
370 #endif
371  p.f.torque += dipole.prefactor * (dip_DLC_t[ip] + vector_product(dip, d));
372  }
373  ip++;
374  }
375 }

377 /** Compute and add the terms needed to correct the energy of
378  * 3D dipolar methods when we have a slab geometry
379  */
380 double add_mdlc_energy_corrections(const ParticleRange &particles) {
381 
382  auto const volume = box_geo.volume();
383  auto const prefactor = dipole.prefactor * 2. * Utils::pi() / volume;
384 
385  // Check if particles aren't in the forbidden gap region
386  // This loop is needed, because there is no other guaranteed
387  // single pass over all particles in this function.
388  for (auto const &p : particles) {
389  check_gap_mdlc(p);
390  }
391 
392  //---- Compute the corrections ----------------------------------
393 
394  // First the DLC correction
395  auto const k_cut = static_cast<int>(std::round(dlc_params.far_cut));
396  double dip_DLC_energy =
397  dipole.prefactor * get_DLC_energy_dipolar(k_cut, particles);
398 
399  // Now we compute the correction like Yeh and Klapp to take into account
400  // the fact that you are using a 3D PBC method which uses spherical
401  // summation instead of slab-wise summation.
402  // Slab-wise summation is the one required to apply DLC correction.
403  // This correction is often called SDC = Shape Dependent Correction.
404  // See @cite brodka04a.
405 
406  auto const box_dip = calc_slab_dipole(particles);
407 
408  if (this_node == 0) {
409  dip_DLC_energy += prefactor * Utils::sqr(box_dip[2]);
410 #ifdef DP3M
411  if (dipole.method == DIPOLAR_MDLC_P3M and
412  dp3m.params.epsilon != P3M_EPSILON_METALLIC) {
413  auto const correps = 1.0 / (2.0 * dp3m.params.epsilon + 1.0);
414  dip_DLC_energy -= prefactor * box_dip.norm2() * correps;
415  }
416 #endif
417  return dip_DLC_energy;
418  }
419  return 0.0;
420 }

422 /** Compute the cut-off in the DLC dipolar part to get a certain accuracy.
423  * We assume particles to have all the same value of the dipolar momentum
424  * modulus, which is taken as the largest value of mu inside the system.
425  * If we assume the gap has a width @c gap_size (within
426  * which there is no particles): <tt>Lz = h + gap_size</tt>
427  *
428  * BE CAREFUL:
429  * 1. We assume the short distance for the slab to be in the *z*-direction
430  * 2. You must also tune the other 3D method to the same accuracy, otherwise
431  * it makes no sense to have an accurate result for DLC-dipolar.
432  */
433 double mdlc_tune_far_cut(DLC_struct const &params) {
434  /* we take the maximum dipole in the system, to be sure that the errors
435  * in the other case will be equal or less than for this one */
436  auto const mu_max = mpi_call(Communication::Result::master_rank, calc_mu_max);
437  auto const mu_max_sq = mu_max * mu_max;
438 
439  const double n = get_n_part();
440  auto const lx = box_geo.length()[0];
441  auto const ly = box_geo.length()[1];
442  auto const lz = box_geo.length()[2];
443  auto const h = params.h;
444 
445  if (std::abs(lx - ly) > 0.001) {
446  throw std::runtime_error("MDLC tuning: box size in x direction is "
447  "different from y direction. The tuning "
448  "formula requires both to be equal.");
449  }
450 
451  constexpr int limitkc = 200;
452  for (int kc = 1; kc < limitkc; kc++) {
453  auto const gc = kc * 2.0 * Utils::pi() / lx;
454  auto const fa0 = sqrt(9.0 * exp(+2.0 * gc * h) * g1_DLC_dip(gc, lz - h) +
455  9.0 * exp(-2.0 * gc * h) * g1_DLC_dip(gc, lz + h) +
456  22.0 * g1_DLC_dip(gc, lz));
457  auto const fa1 = 0.5 * sqrt(Utils::pi() / (2.0 * lx * ly)) * fa0;
458  auto const fa2 = g2_DLC_dip(gc, lz);
459  auto const de = n * mu_max_sq / (4.0 * (exp(gc * lz) - 1.0)) * (fa1 + fa2);
460  if (de < params.maxPWerror) {
461  return static_cast<double>(kc);
462  }
463  }
464 
465  throw std::runtime_error("MDLC tuning failed: unable to find a proper "
466  "cut-off for the given accuracy.");
467 }

475 void mdlc_set_params(double maxPWerror, double gap_size, double far_cut) {
476  auto const h = box_geo.length()[2] - gap_size;
477  if (maxPWerror <= 0.) {
478  throw std::domain_error("maxPWerror must be > 0");
479  }
480  if (gap_size <= 0.) {
481  throw std::domain_error("gap_size must be > 0");
482  }
483  if (h < 0.) {
484  throw std::domain_error("gap size too large");
485  }
486  mdlc_sanity_checks();
487 
488  DipolarInteraction new_method;
489  switch (dipole.method) {
490 #ifdef DP3M
491  case DIPOLAR_MDLC_P3M:
492  case DIPOLAR_P3M:
493  new_method = DIPOLAR_MDLC_P3M;
494  break;
495 #endif
496  case DIPOLAR_MDLC_DS:
497  case DIPOLAR_DS:
498  new_method = DIPOLAR_MDLC_DS;
499  fprintf(stderr, "You are not using the P3M method, therefore dp3m.params."
500  "epsilon unknown, I will assume metallic borders.\n");
501  break;
502  default:
503  throw std::runtime_error(
504  "MDLC cannot extend the currently active magnetostatics solver.");
505  }
506 
507  DLC_struct new_dlc_params{maxPWerror, far_cut, gap_size, far_cut == -1., h};
508 
509  if (new_dlc_params.far_calculated) {
510  new_dlc_params.far_cut = mdlc_tune_far_cut(new_dlc_params);
511  }
512 
513  dlc_params = new_dlc_params;
514 
515  Dipole::set_method_local(new_method);
516  mpi_bcast_coulomb_params();
517 }
