200 /** Calculate indices that shift @ref P3MParameters::mesh "mesh" by `mesh/2`.
201  * For each mesh size @f$ n @f$ in @c mesh_size, create a sequence of integer
202  * values @f$ \left( 0, \ldots, \lfloor n/2 \rfloor, -\lfloor n/2 \rfloor,
203  * \ldots, -1\right) @f$ if @c zero_out_midpoint is false, otherwise
204  * @f$ \left( 0, \ldots, \lfloor n/2 - 1 \rfloor, 0, -\lfloor n/2 \rfloor,
205  * \ldots, -1\right) @f$.
206  */
207 std::array<std::vector<int>, 3> inline calc_meshift(
208  std::array<int, 3> const &mesh_size, bool zero_out_midpoint = false) {
209  std::array<std::vector<int>, 3> ret{};
210 
211  for (std::size_t i = 0; i < 3; i++) {
212  ret[i] = std::vector<int>(mesh_size[i]);
213 
214  for (int j = 1; j <= mesh_size[i] / 2; j++) {
215  ret[i][j] = j;
216  ret[i][mesh_size[i] - j] = -j;
217  }
218  if (zero_out_midpoint)
219  ret[i][mesh_size[i] / 2] = 0;
220  }
221 
222  return ret;
223 }
224 } // namespace detail
