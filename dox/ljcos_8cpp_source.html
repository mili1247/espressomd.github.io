34 int ljcos_set_params(int part_type_a, int part_type_b, double eps, double sig,
35  double cut, double offset) {
36  IA_parameters *data = get_ia_param_safe(part_type_a, part_type_b);
37 
38  if (!data)
39  return ES_ERROR;
40 
41  data->ljcos.eps = eps;
42  data->ljcos.sig = sig;
43  data->ljcos.cut = cut;
44  data->ljcos.offset = offset;
45 
46  /* Calculate dependent parameters */
47  auto const facsq = Utils::cbrt_2() * Utils::sqr(sig);
48  data->ljcos.rmin = sqrt(Utils::cbrt_2()) * sig;
49  data->ljcos.alfa = Utils::pi() / (Utils::sqr(data->ljcos.cut) - facsq);
50  data->ljcos.beta =
51  Utils::pi() * (1. - (1. / (Utils::sqr(data->ljcos.cut) / facsq - 1.)));
52 
53  /* broadcast interaction parameters */
54  mpi_bcast_ia_params(part_type_a, part_type_b);
55 
56  return ES_OK;
57 }
