48 /**
49  * Performs a randomly selected reaction in the reaction ensemble
50  */
51 int ReactionAlgorithm::do_reaction(int reaction_steps) {
52  for (int i = 0; i < reaction_steps; i++) {
53  int reaction_id = i_random(static_cast<int>(reactions.size()));
54  generic_oneway_reaction(reactions[reaction_id]);
55  }
56  return 0;
57 }

62 void ReactionAlgorithm::add_reaction(

81 /**
82  * Checks whether all necessary variables for the reaction ensemble have been
83  * set.
84  */
85 void ReactionAlgorithm::check_reaction_method() const {
86  if (reactions.empty()) {
87  throw std::runtime_error("Reaction system not initialized");
88  }
89 
90  if (kT < 0) {
91  throw std::runtime_error("kT cannot be negative,"
92  "normally it should be 1.0. This will be used"
93  "directly to calculate beta:=1/(k_B T) which"
94  "occurs in the exp(-beta*E)\n");
95  }
96 
97 #ifdef ELECTROSTATICS
98  // check for the existence of default charges for all types that take part in
99  // the reactions
100 
101  for (const auto &current_reaction : reactions) {
102  // check for reactants
103  for (int reactant_type : current_reaction.reactant_types) {
104  auto it = charges_of_types.find(reactant_type);
105  if (it == charges_of_types.end()) {
106  std::string message = std::string("Forgot to assign charge to type ") +
107  std::to_string(reactant_type);
108  throw std::runtime_error(message);
109  }
110  }
111  // check for products
112  for (int product_type : current_reaction.product_types) {
113  auto it = charges_of_types.find(product_type);
114  if (it == charges_of_types.end()) {
115  std::string message = std::string("Forgot to assign charge to type ") +
116  std::to_string(product_type);
117  throw std::runtime_error(message);
118  }
119  }
120  }
121 #endif
122 }

135  */
136 bool ReactionAlgorithm::all_reactant_particles_exist(

138  bool enough_particles = true;

140  int current_number =

163 /**
164  *Performs a trial reaction move
165  */
166 std::tuple<std::vector<StoredParticleProperty>, std::vector<int>,
167  std::vector<StoredParticleProperty>>
168 ReactionAlgorithm::make_reaction_attempt(
169  SingleReaction const &current_reaction) {
170  // create or hide particles of types with corresponding types in reaction
171  std::vector<int> p_ids_created_particles;
172  std::vector<StoredParticleProperty> hidden_particles_properties;
173  std::vector<StoredParticleProperty> changed_particles_properties;
174 
175  for (int i = 0; i < std::min(current_reaction.product_types.size(),
176  current_reaction.reactant_types.size());
177  i++) {
178  // change std::min(reactant_coefficients(i),product_coefficients(i)) many
179  // particles of reactant_types(i) to product_types(i)
180  for (int j = 0; j < std::min(current_reaction.product_coefficients[i],
181  current_reaction.reactant_coefficients[i]);
182  j++) {
183  append_particle_property_of_random_particle(
184  current_reaction.reactant_types[i], changed_particles_properties);
185  replace_particle(changed_particles_properties.back().p_id,
186  current_reaction.product_types[i]);
187  }
188  // create product_coefficients(i)-reactant_coefficients(i) many product
189  // particles iff product_coefficients(i)-reactant_coefficients(i)>0,
190  // iff product_coefficients(i)-reactant_coefficients(i)<0, hide this number
191  // of reactant particles
192  if (current_reaction.product_coefficients[i] -
193  current_reaction.reactant_coefficients[i] >
194  0) {
195  for (int j = 0; j < current_reaction.product_coefficients[i] -
196  current_reaction.reactant_coefficients[i];
197  j++) {
198  int p_id = create_particle(current_reaction.product_types[i]);
199  check_exclusion_radius(p_id);
200  p_ids_created_particles.push_back(p_id);
201  }
202  } else if (current_reaction.reactant_coefficients[i] -
203  current_reaction.product_coefficients[i] >
204  0) {
205  for (int j = 0; j < current_reaction.reactant_coefficients[i] -
206  current_reaction.product_coefficients[i];
207  j++) {
208  append_particle_property_of_random_particle(
209  current_reaction.reactant_types[i], hidden_particles_properties);
210  check_exclusion_radius(hidden_particles_properties.back().p_id);
211  hide_particle(hidden_particles_properties.back().p_id);
212  }
213  }
214  }
215  // create or hide particles of types with noncorresponding replacement types
216  for (auto i = std::min(current_reaction.product_types.size(),
217  current_reaction.reactant_types.size());
218  i < std::max(current_reaction.product_types.size(),
219  current_reaction.reactant_types.size());
220  i++) {
221  if (current_reaction.product_types.size() <
222  current_reaction.reactant_types.size()) {
223  // hide superfluous reactant_types particles
224  for (int j = 0; j < current_reaction.reactant_coefficients[i]; j++) {
225  append_particle_property_of_random_particle(
226  current_reaction.reactant_types[i], hidden_particles_properties);
227  check_exclusion_radius(hidden_particles_properties.back().p_id);
228  hide_particle(hidden_particles_properties.back().p_id);
229  }
230  } else {
231  // create additional product_types particles
232  for (int j = 0; j < current_reaction.product_coefficients[i]; j++) {
233  int p_id = create_particle(current_reaction.product_types[i]);
234  check_exclusion_radius(p_id);
235  p_ids_created_particles.push_back(p_id);
236  }
237  }
238  }
239 
240  return {changed_particles_properties, p_ids_created_particles,
241  hidden_particles_properties};
242 }

244 /**
245  * Restores the previously stored particle properties. This function is invoked
246  * when a reaction attempt is rejected.
247  */
248 void ReactionAlgorithm::restore_properties(
249  std::vector<StoredParticleProperty> const &property_list,
250  const int number_of_saved_properties) {
251  // this function restores all properties of all particles provided in the
252  // property list, the format of the property list is (p_id,charge,type)
253  // repeated for each particle that occurs in that list
254  for (auto &i : property_list) {
255  int type = i.type;
256 #ifdef ELECTROSTATICS
257  // set charge
258  double charge = i.charge;
259  set_particle_q(i.p_id, charge);
260 #endif
261  // set type
262  set_particle_type(i.p_id, type);
263  }

267  SingleReaction const &current_reaction) const {
268  std::map<int, int> old_particle_numbers;
269  // reactants
270  for (int type : current_reaction.reactant_types) {

281 /**
282  * Generic one way reaction
283  * A+B+...+G +... --> K+...X + Z +...
284  * You need to use 2A --> B instead of A+A --> B since in the last case you
285  * assume distinctness of the particles, however both ways to describe the
286  * reaction are equivalent in the thermodynamic limit (large particle numbers).
287  * Furthermore, it is crucial for the function in which order you provide the
288  * reactant and product types since particles will be replaced correspondingly!
289  * If there are less reactants than products, new product particles are created
290  * randomly in the box. Matching particles simply change the types. If there
291  * are more reactants than products, old reactant particles are deleted.
292  */
293 void ReactionAlgorithm::generic_oneway_reaction(
294  SingleReaction &current_reaction) {
295 
296  current_reaction.tried_moves += 1;
297  particle_inside_exclusion_radius_touched = false;
298  if (!all_reactant_particles_exist(current_reaction)) {
299  // makes sure, no incomplete reaction is performed -> only need to consider
300  // rollback of complete reactions
301  return;
302  }
303 
304  // calculate potential energy
305  const double E_pot_old =
306  calculate_current_potential_energy_of_system(); // only consider potential
307  // energy since we assume
308  // that the kinetic part
309  // drops out in the
310  // process of calculating
311  // ensemble averages
312  // (kinetic part may be
313  // separated and crossed
314  // out)
315 
316  // find reacting molecules in reactants and save their properties for later
317  // recreation if step is not accepted
318  // do reaction
319  // save old particle_numbers
320  std::map<int, int> old_particle_numbers =
321  save_old_particle_numbers(current_reaction);
322 
323  std::vector<int> p_ids_created_particles;
324  std::vector<StoredParticleProperty> hidden_particles_properties;
325  std::vector<StoredParticleProperty> changed_particles_properties;
326  // save p_id, charge and type of the reactant particle, only thing we
327  // need to hide the particle and recover it
328  const int number_of_saved_properties = 3;
329 
330  std::tie(changed_particles_properties, p_ids_created_particles,
331  hidden_particles_properties) =
332  make_reaction_attempt(current_reaction);
333 
334  auto const E_pot_new = (particle_inside_exclusion_radius_touched)
335  ? std::numeric_limits<double>::max()
336  : calculate_current_potential_energy_of_system();
337 
338  auto const bf = calculate_acceptance_probability(
339  current_reaction, E_pot_old, E_pot_new, old_particle_numbers);
340 
341  std::vector<double> exponential = {exp(-1.0 / kT * (E_pot_new - E_pot_old))};
342  current_reaction.accumulator_potential_energy_difference_exponential(
343  exponential);
344 
345  if (m_uniform_real_distribution(m_generator) < bf) {
346  // accept
347  // delete hidden reactant_particles (remark: don't delete changed particles)
348  // extract ids of to be deleted particles
349  auto len_hidden_particles_properties =
350  static_cast<int>(hidden_particles_properties.size());
351  std::vector<int> to_be_deleted_hidden_ids(len_hidden_particles_properties);
352  std::vector<int> to_be_deleted_hidden_types(
353  len_hidden_particles_properties);
354  for (int i = 0; i < len_hidden_particles_properties; i++) {
355  auto p_id = hidden_particles_properties[i].p_id;
356  to_be_deleted_hidden_ids[i] = p_id;
357  to_be_deleted_hidden_types[i] = hidden_particles_properties[i].type;
358  // change back type otherwise the bookkeeping algorithm is not working
359  set_particle_type(p_id, hidden_particles_properties[i].type);
360  }
361 
362  for (int i = 0; i < len_hidden_particles_properties; i++) {
363  delete_particle(to_be_deleted_hidden_ids[i]); // delete particle
364  }
365  current_reaction.accepted_moves += 1;
366  } else {
367  // reject
368  // reverse reaction
369  // 1) delete created product particles
370  for (int p_ids_created_particle : p_ids_created_particles) {
371  delete_particle(p_ids_created_particle);
372  }
373  // 2) restore previously hidden reactant particles
374  restore_properties(hidden_particles_properties, number_of_saved_properties);
375  // 3) restore previously changed reactant particles
376  restore_properties(changed_particles_properties,
377  number_of_saved_properties);
378  }
379 }

392 /**
393  * Hides a particle from short ranged interactions and from the electrostatic
394  * interaction. Additional hiding from interactions would need to be implemented
395  * here.
396  *
397  * Removing the particle charge and changing its type to a non existing one
398  * deactivates all interactions with other particles, as if the particle was
399  * inexistent (currently only type-based interactions are switched off, as well
400  * as the electrostatic interaction).
401  * This function does not break bonds for simple reactions, as long as there
402  * are no reactions like 2A -->B where one of the reacting A particles occurs
403  * in the polymer (think of bond breakages if the monomer in the polymer gets
404  * deleted in the reaction). This constraint is not of fundamental reason, but
405  * there would be a need for a rule for such "collision" reactions (a reaction
406  * like the one above).
407  */
408 void ReactionAlgorithm::hide_particle(int p_id) const {
409  set_particle_type(p_id, non_interacting_type);
410 #ifdef ELECTROSTATICS
411  set_particle_q(p_id, 0.0);
412 #endif
413 }

428 /**
429  * Deletes the particle with the given p_id and stores the id if the deletion
430  * created a hole in the particle id range. This method is intended to only
431  * delete unbonded particles since bonds are coupled to ids. This is used to
432  * avoid the id range becoming excessively huge.
433  */
434 int ReactionAlgorithm::delete_particle(int p_id) {
435  auto const old_max_seen_id = get_maximal_particle_id();
436  if (p_id == old_max_seen_id) {
437  // last particle, just delete
438  remove_particle(p_id);
439  // remove all saved empty p_ids which are greater than the max_seen_particle
440  // this is needed in order to avoid the creation of holes
441  for (auto p_id_iter = m_empty_p_ids_smaller_than_max_seen_particle.begin();
442  p_id_iter != m_empty_p_ids_smaller_than_max_seen_particle.end();) {
443  if ((*p_id_iter) >= old_max_seen_id)
444  p_id_iter = m_empty_p_ids_smaller_than_max_seen_particle.erase(
445  p_id_iter); // update iterator after container was modified
446  else
447  ++p_id_iter;
448  }
449  } else if (p_id <= old_max_seen_id) {
450  remove_particle(p_id);
451  m_empty_p_ids_smaller_than_max_seen_particle.push_back(p_id);
452  } else {
453  throw std::runtime_error(
454  "Particle id is greater than the max seen particle id");
455  }
456  return 0;
457 }

486 /**
487  * Writes a random position inside the central box into the provided array.
488  */
489 Utils::Vector3d ReactionAlgorithm::get_random_position_in_box() {
490  Utils::Vector3d out_pos{};
491 
492  if (m_reaction_constraint == ReactionConstraint::CYL_Z) {
493  // see http://mathworld.wolfram.com/DiskPointPicking.html
494  // for uniform disk point picking in cylinder
495  auto const random_radius =
496  m_cyl_radius * std::sqrt(m_uniform_real_distribution(m_generator));
497  auto const random_phi =
498  2. * Utils::pi() * m_uniform_real_distribution(m_generator);
499  out_pos[0] = m_cyl_x + random_radius * cos(random_phi);
500  out_pos[1] = m_cyl_y + random_radius * sin(random_phi);
501  out_pos[2] = box_geo.length()[2] * m_uniform_real_distribution(m_generator);
502  } else if (m_reaction_constraint == ReactionConstraint::SLAB_Z) {
503  out_pos[0] = box_geo.length()[0] * m_uniform_real_distribution(m_generator);
504  out_pos[1] = box_geo.length()[1] * m_uniform_real_distribution(m_generator);
505  out_pos[2] = m_slab_start_z + (m_slab_end_z - m_slab_start_z) *
506  m_uniform_real_distribution(m_generator);
507  } else {
508  assert(m_reaction_constraint == ReactionConstraint::NONE);
509  out_pos[0] = box_geo.length()[0] * m_uniform_real_distribution(m_generator);
510  out_pos[1] = box_geo.length()[1] * m_uniform_real_distribution(m_generator);
511  out_pos[2] = box_geo.length()[2] * m_uniform_real_distribution(m_generator);
512  }
513  return out_pos;
514 }

516 /**
517  * Creates a particle at the end of the observed particle id range.
518  */
519 int ReactionAlgorithm::create_particle(int desired_type) {
520  int p_id;
521  if (!m_empty_p_ids_smaller_than_max_seen_particle.empty()) {
522  auto p_id_iter = std::min_element(
523  std::begin(m_empty_p_ids_smaller_than_max_seen_particle),
524  std::end(m_empty_p_ids_smaller_than_max_seen_particle));
525  p_id = *p_id_iter;
526  m_empty_p_ids_smaller_than_max_seen_particle.erase(p_id_iter);
527  } else {
528  p_id = get_maximal_particle_id() + 1;
529  }
530 
531  // we use mass=1 for all particles, think about adapting this
532  move_particle(p_id, get_random_position_in_box(), std::sqrt(kT));
533  set_particle_type(p_id, desired_type);
534 #ifdef ELECTROSTATICS
535  set_particle_q(p_id, charges_of_types[desired_type]);
536 #endif
537  return p_id;
538 }

540 void ReactionAlgorithm::move_particle(int p_id, Utils::Vector3d const &new_pos,
541  double velocity_prefactor) {
542  place_particle(p_id, new_pos);

544  Utils::Vector3d vel;

546  vel[1] = velocity_prefactor * m_normal_distribution(m_generator);
547  vel[2] = velocity_prefactor * m_normal_distribution(m_generator);
548  set_particle_v(p_id, vel);

551 std::vector<std::pair<int, Utils::Vector3d>>
552 ReactionAlgorithm::generate_new_particle_positions(int type, int n_particles) {
553 
554  std::vector<int> p_id_s_changed_particles;
555  std::vector<std::pair<int, Utils::Vector3d>> old_positions;
556  old_positions.reserve(n_particles);
557 
558  // draw particle ids at random without replacement
559  int p_id = -1;
560  std::vector<int> drawn_pids{p_id};
561  for (int i = 0; i < n_particles; i++) {
562  // draw a new particle id
563  while (Utils::contains(drawn_pids, p_id)) {
564  auto const random_index = i_random(number_of_particles_with_type(type));
565  p_id = get_random_p_id(type, random_index);
566  }
567  drawn_pids.emplace_back(p_id);
568  // store original position
569  auto const &p = get_particle_data(p_id);
570  old_positions.emplace_back(std::pair<int, Utils::Vector3d>{p_id, p.r.p});
571  // write new position
572  auto const prefactor = std::sqrt(kT / p.p.mass);
573  auto const new_pos = get_random_position_in_box();
574  move_particle(p_id, new_pos, prefactor);
575  check_exclusion_radius(p_id);
576  }
577 
578  return old_positions;
579 }
