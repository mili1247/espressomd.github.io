69 inline void Accumulator::operator()(const std::vector<double> &data) {
70  if (data.size() != m_acc_data.size())
71  throw std::runtime_error(
72  "The given data size does not fit the initialized size!");
73  ++m_n;
74  if (m_n == 1) {
75  std::transform(data.begin(), data.end(), m_acc_data.begin(),
76  [](double d) -> AccumulatorData<double> {
77  return {d, 0.0};
78  });
79  } else {
80  std::transform(m_acc_data.begin(), m_acc_data.end(), data.begin(),
81  m_acc_data.begin(),
82  [this](AccumulatorData<double> &a,
83  double d) -> AccumulatorData<double> {
84  auto const old_mean = a.mean;
85  auto const new_mean =
86  old_mean + (d - old_mean) / static_cast<double>(m_n);
87  auto const new_m = a.m + (d - old_mean) * (d - new_mean);
88  return {new_mean, new_m};
89  });
90  }
91 }

101 inline std::vector<double> Accumulator::variance() const {
102  std::vector<double> res;
103  if (m_n == 1) {
104  res = std::vector<double>(m_acc_data.size(),
105  std::numeric_limits<double>::max());
106  } else {
107  std::transform(m_acc_data.begin(), m_acc_data.end(),
108  std::back_inserter(res),
109  [this](const AccumulatorData<double> &acc_data) {
110  return acc_data.m / (static_cast<double>(m_n) - 1);
111  });
112  }
113  return res;
114 }

116 /**

119 inline std::vector<double> Accumulator::std_error() const {
120  auto const var = variance();
121  std::vector<double> err(var.size());
122  std::transform(var.begin(), var.end(), err.begin(), [this](double d) {
123  return std::sqrt(d / static_cast<double>(m_n));
