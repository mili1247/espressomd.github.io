85 /** Update translational and rotational velocities of all particles. */
86 void sd_update_locally(ParticleRange const &parts) {
87  std::size_t i = 0;
88 
89  // Even though on the master node, the v_sd vector is larger than
90  // the (local) parts vector, this should still work. Because the local
91  // particles correspond to the first 6*n entries in the master's v_sd
92  // (which holds the velocities of ALL particles).
93 
94  for (auto &p : parts) {
95  // skip virtual particles
96  if (p.p.is_virtual) {
97  continue;
98  }
99 
100  // Copy velocities
101  p.m.v[0] = v_sd[6 * i + 0];
102  p.m.v[1] = v_sd[6 * i + 1];
103  p.m.v[2] = v_sd[6 * i + 2];
104 
105  p.m.omega[0] = v_sd[6 * i + 3];
106  p.m.omega[1] = v_sd[6 * i + 4];
107  p.m.omega[2] = v_sd[6 * i + 5];
108 
109  i++;
110  }
111 }

147 void propagate_vel_pos_sd(const ParticleRange &particles,
148  const boost::mpi::communicator &comm,
149  const double time_step) {
150  static std::vector<SD_particle_data> parts_buffer{};
151 
152  parts_buffer.clear();
153  boost::transform(particles, std::back_inserter(parts_buffer),
154  [](auto const &p) { return SD_particle_data(p); });
155  Utils::Mpi::gather_buffer(parts_buffer, comm, 0);
156 
157  /* Buffer that holds local particle data, and all particles on the master
158  * node used for sending particle data to master node. */
159  if (comm.rank() == 0) {
160  std::size_t n_part = parts_buffer.size();
161 
162  static std::vector<double> x_host{};
163  static std::vector<double> f_host{};
164  static std::vector<double> a_host{};
165 
166  x_host.resize(6 * n_part);
167  f_host.resize(6 * n_part);
168  a_host.resize(n_part);
169 
170  std::size_t i = 0;
171  for (auto const &p : parts_buffer) {
172  x_host[6 * i + 0] = p.pos[0];
173  x_host[6 * i + 1] = p.pos[1];
174  x_host[6 * i + 2] = p.pos[2];
175  // Actual orientation is not needed, just need default.
176  x_host[6 * i + 3] = 1;
177  x_host[6 * i + 4] = 0;
178  x_host[6 * i + 5] = 0;
179 
180  f_host[6 * i + 0] = p.ext_force.f[0];
181  f_host[6 * i + 1] = p.ext_force.f[1];
182  f_host[6 * i + 2] = p.ext_force.f[2];
183 
184  f_host[6 * i + 3] = p.ext_force.torque[0];
185  f_host[6 * i + 4] = p.ext_force.torque[1];
186  f_host[6 * i + 5] = p.ext_force.torque[2];
187 
188  double radius = radius_dict[p.type];
189 
190  a_host[i] = radius;
191 
192  ++i;
193  }
194 
195  v_sd = sd_cpu(x_host, f_host, a_host, n_part, sd_viscosity,
196  std::sqrt(sd_kT / time_step),
197  static_cast<std::size_t>(stokesian.rng_counter()),
198  static_cast<std::size_t>(stokesian.rng_seed()), sd_flags);
199  } else { // if (this_node == 0)
200  v_sd.resize(particles.size() * 6);
201  } // if (this_node == 0) {...} else
202 
203  Utils::Mpi::scatter_buffer(v_sd.data(),
204  static_cast<int>(particles.size() * 6), comm, 0);
205  sd_update_locally(particles);
206 }
