38 namespace ScriptInterface {
39 /**
40  * @brief Context of an object handle.
41  *
42  * Each instance of @ref ObjectHandle can have an
43  * attached context, which can e.g. synchronize
44  * distributed copies of the instance. The context
45  * does also provide facilities for serializing
46  * Objects into a string representation.
47  */
48 class Context : public std::enable_shared_from_this<Context> {
49 public:
50  /**
51  * @brief Call method on remote instances
52  *
53  * @param self Internal identifier of the instance
54  * @param name Name of the method to call
55  * @param arguments Arguments to the call
56  */
57  virtual void notify_call_method(const ObjectHandle *self,
58  std::string const &name,
59  VariantMap const &arguments) = 0;
60 
61  /**
62  * @brief Set a parameter on remote instances
63  *
64  * @param self Internal identifier of the instance to be modified
65  * @param name Name of the parameter to change
66  * @param value Value to set it to
67  */
68  virtual void notify_set_parameter(const ObjectHandle *self,
69  std::string const &name,
70  Variant const &value) = 0;
71 
72  /**
73  * @brief Get a new reference counted instance of a script interface by
74  * name.
75  *
76  * Objects created thru a Context get shared ownership of that context,
77  * e.g. the lifetime of the context is at least as long as the objects
78  * created by it. Therefore the object can always assume that the context
79  * is present.
80  */
81  virtual std::shared_ptr<ObjectHandle>
82  make_shared(std::string const &name, const VariantMap &parameters) = 0;
83 
84 protected:
85  /**
86  * @brief Set the context of an object to this.
87  *
88  * @param o Object to set the context for.
89  */
90  void set_context(ObjectHandle *o) { o->m_context = this->shared_from_this(); }
91 
92 public:
93  /**
94  * @brief Get the class name for an ObjectHandle instance.
95  *
96  * This returns the name by which the object can be created.
97  */
98  virtual boost::string_ref name(const ObjectHandle *o) const = 0;
99 
100  virtual ~Context() = default;
101 };
102 } // namespace ScriptInterface
