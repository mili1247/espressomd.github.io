33 int BMHTF_set_params(int part_type_a, int part_type_b, double A, double B,
34  double C, double D, double sig, double cut) {
35  double shift, dist2, pw6;
36  IA_parameters *data = get_ia_param_safe(part_type_a, part_type_b);
37 
38  if (!data)
39  return ES_ERROR;
40 
41  dist2 = cut * cut;
42  pw6 = dist2 * dist2 * dist2;
43  shift = -(A * exp(B * (sig - cut)) - C / pw6 - D / pw6 / dist2);
44 
45  data->bmhtf.A = A;
46  data->bmhtf.B = B;
47  data->bmhtf.C = C;
48  data->bmhtf.D = D;
49  data->bmhtf.sig = sig;
50  data->bmhtf.cut = cut;
51  data->bmhtf.computed_shift = shift;
52 
53  /* broadcast interaction parameters */
54  mpi_bcast_ia_params(part_type_a, part_type_b);
55 
56  return ES_OK;
57 }
