58 __constant__ float far_switch_radius_2[1] = {0.05f * 0.05f};
59 __constant__ float boxz[1];
60 __constant__ float uz[1];
61 __constant__ float coulomb_prefactor[1] = {1.0f};
62 __constant__ int bessel_cutoff[1] = {5};
63 __constant__ float maxPWerror[1] = {1e-5f};
64 
65 // As the coefficients are stored in __constant__ memory, the array needs to be
66 // sized in advance. We don't know exactly how many coefficients per order, so
67 // we size plentiful.
68 constexpr int modpsi_order = 30;
69 constexpr int modpsi_constant_size = modpsi_order * modpsi_order * 2;
70 
71 // linearized array on device
72 __constant__ int device_n_modPsi[1] = {0};
73 __constant__ unsigned int device_linModPsi_offsets[2 * modpsi_order];
74 __constant__ unsigned int device_linModPsi_lengths[2 * modpsi_order];
75 __constant__ float device_linModPsi[modpsi_constant_size];
76 
77 __device__ float dev_mod_psi_even(int n, float x) {
78  return evaluateAsTaylorSeriesAt(
79  &device_linModPsi[device_linModPsi_offsets[2 * n]],
80  static_cast<int>(device_linModPsi_lengths[2 * n]), x * x);
81 }
82 
83 __device__ float dev_mod_psi_odd(int n, float x) {
84  return x * evaluateAsTaylorSeriesAt(
85  &device_linModPsi[device_linModPsi_offsets[2 * n + 1]],
86  static_cast<int>(device_linModPsi_lengths[2 * n + 1]), x * x);
87 }

89 int modpsi_init() {
90  create_mod_psi_up_to(modpsi_order);
91 
92  // linearized array on host
93  std::vector<unsigned int> linModPsi_offsets(modPsi.size());
94  std::vector<unsigned int> linModPsi_lengths(modPsi.size());
95  for (std::size_t i = 0; i < modPsi.size(); i++) {
96  if (i)
97  linModPsi_offsets[i] =
98  linModPsi_offsets[i - 1] + linModPsi_lengths[i - 1];
99  linModPsi_lengths[i] = static_cast<unsigned int>(modPsi[i].size());
100  }
101 
102  // linearize the coefficients array
103  std::vector<float> linModPsi(linModPsi_offsets[modPsi.size() - 1] +
104  linModPsi_lengths[modPsi.size() - 1]);
105  for (std::size_t i = 0; i < modPsi.size(); i++) {
106  for (std::size_t j = 0; j < modPsi[i].size(); j++) {
107  linModPsi[linModPsi_offsets[i] + j] = static_cast<float>(modPsi[i][j]);
108  }
109  }
110 
111  for (int d = 0; d < deviceCount; d++) {
112  cudaSetDevice(d);
113 
114  // copy to GPU
115  auto const linModPsiSize = linModPsi_offsets[modPsi.size() - 1] +
116  linModPsi_lengths[modPsi.size() - 1];
117  if (linModPsiSize > static_cast<unsigned int>(modpsi_constant_size)) {
118  throw std::runtime_error(
119  "__constant__ device_linModPsi[] is not large enough");
120  }
121  cuda_safe_mem(cudaMemcpyToSymbol(device_linModPsi_offsets,
122  linModPsi_offsets.data(),
123  modPsi.size() * sizeof(int)));
124  cuda_safe_mem(cudaMemcpyToSymbol(device_linModPsi_lengths,
125  linModPsi_lengths.data(),
126  modPsi.size() * sizeof(int)));
127  cuda_safe_mem(cudaMemcpyToSymbol(device_linModPsi, linModPsi.data(),
128  linModPsiSize * sizeof(float)));
129  auto const n_modPsi = static_cast<int>(modPsi.size() >> 1);
130  cuda_safe_mem(cudaMemcpyToSymbol(device_n_modPsi, &n_modPsi, sizeof(int)));
131  }
132 
133  return 0;
134 }

159 void Mmm1dgpuForce::setup(SystemInterface &s) {
160  auto const box_z = static_cast<float>(s.box()[2]);
161  if (need_tune && s.npart_gpu() > 0) {
162  set_params(box_z, static_cast<float>(coulomb.prefactor), maxPWerror,
163  far_switch_radius, bessel_cutoff);
164  tune(s, maxPWerror, far_switch_radius, bessel_cutoff);
165  }
166  if (box_z != host_boxz) {
167  set_params(box_z, 0, -1, -1, -1);
168  }
169  if (s.npart_gpu() == host_npart) { // unchanged
170  return;
171  }
172 
173  // For all but the largest systems, it is faster to store force pairs and then
174  // sum them up. Atomics are just so slow: so unless we're limited by memory,
175  // do the latter.
176  auto const part_mem_size = 3 * Utils::sqr(s.npart_gpu()) * sizeof(float);
177  pairs = 2;
178  for (int d = 0; d < deviceCount; d++) {
179  cudaSetDevice(d);
180 
181  std::size_t freeMem, totalMem;
182  cudaMemGetInfo(&freeMem, &totalMem);
183  if (freeMem / 2 < part_mem_size) {
184  // don't use more than half the device's memory
185  fprintf(stderr, "Switching to atomicAdd due to memory constraints.\n");
186  pairs = 0;
187  break;
188  }
189  }
190  if (dev_forcePairs)
191  cudaFree(dev_forcePairs);
192  if (pairs) {
193  // we need memory to store force pairs
194  cuda_safe_mem(cudaMalloc((void **)&dev_forcePairs, part_mem_size));
195  }
196  if (dev_energyBlocks)
197  cudaFree(dev_energyBlocks);
198  cuda_safe_mem(
199  cudaMalloc((void **)&dev_energyBlocks, numBlocks(s) * sizeof(float)));
200  host_npart = static_cast<unsigned int>(s.npart_gpu());
201 }

213 __forceinline__ __device__ float sqpow(float x) { return x * x; }

216 __device__ void sumReduction(float *input, float *sum) {
217  auto const tid = threadIdx.x;
218  for (auto i = blockDim.x / 2; i > 0; i /= 2) {
219  __syncthreads();
220  if (tid < i)
221  input[tid] += input[i + tid];
222  }
223  __syncthreads();
224  if (tid == 0)
225  sum[0] = input[0];
226 }

228 __global__ void sumKernel(float *data, std::size_t N) {
229  extern __shared__ float partialsums[];
230  if (blockIdx.x != 0)
231  return;
232  std::size_t const tid = threadIdx.x;
233  auto result = 0.f;
234 
235  for (std::size_t i = 0; i < N; i += blockDim.x) {
236  if (i + tid >= N)
237  partialsums[tid] = 0.f;
238  else
239  partialsums[tid] = data[i + tid];
240 
241  sumReduction(partialsums, &result);
242  if (tid == 0) {
243  if (i == 0)
244  data[0] = 0.f;
245  data[0] += result;
246  }
247  }

250 __global__ void besselTuneKernel(int *result, float far_switch_radius,
251  int maxCut) {
252  constexpr auto c_2pif = 2 * Utils::pi<float>();
253  auto const arg = c_2pif * *uz * far_switch_radius;
254  auto const pref = 4 * *uz * max(1.0f, c_2pif * *uz);
255  float err;
256  int P = 1;
257  do {
258  err = pref * dev_K1(arg * static_cast<float>(P)) * exp(arg) / arg *
259  (static_cast<float>(P) - 1 + 1 / arg);
260  P++;
261  } while (err > *maxPWerror && P <= maxCut);
262  P--;
263 
264  result[0] = P;
265 }

267 void Mmm1dgpuForce::tune(SystemInterface &s, float _maxPWerror,
268  float _far_switch_radius, int _bessel_cutoff) {
269  float far_switch_radius = _far_switch_radius;
270  int bessel_cutoff = _bessel_cutoff;
271  float maxrad = host_boxz;
272 
273  if (_far_switch_radius < 0 && _bessel_cutoff < 0) {
274  // autodetermine switching radius and Bessel cutoff
275  float bestrad = 0, besttime = INFINITY;
276 
277  // NOLINTNEXTLINE(clang-analyzer-security.FloatLoopCounter)
278  for (far_switch_radius = 0.05f * maxrad; far_switch_radius < maxrad;
279  far_switch_radius += 0.05f * maxrad) {
280  set_params(0, 0, _maxPWerror, far_switch_radius, bessel_cutoff);
281  tune(s, _maxPWerror, far_switch_radius, -2); // tune Bessel cutoff
282  auto const runtime = force_benchmark(s);
283  if (runtime < besttime) {
284  besttime = runtime;
285  bestrad = far_switch_radius;
286  }
287  }
288  far_switch_radius = bestrad;
289 
290  set_params(0, 0, _maxPWerror, far_switch_radius, bessel_cutoff);
291  tune(s, _maxPWerror, far_switch_radius, -2); // tune Bessel cutoff
292  } else if (_bessel_cutoff < 0) {
293  // autodetermine Bessel cutoff
294  int *dev_cutoff;
295  constexpr auto maxCut = 30;
296  cuda_safe_mem(cudaMalloc((void **)&dev_cutoff, sizeof(int)));
297  besselTuneKernel<<<dim3(1), dim3(1), 0, nullptr>>>(
298  dev_cutoff, far_switch_radius, maxCut);
299  cuda_safe_mem(cudaMemcpy(&bessel_cutoff, dev_cutoff, sizeof(int),
300  cudaMemcpyDeviceToHost));
301  cudaFree(dev_cutoff);
302  if (_bessel_cutoff != -2 && bessel_cutoff >= maxCut) {
303  // we already had our switching radius and only needed to
304  // determine the cutoff, i.e. this was the final tuning round
305  throw std::runtime_error(
306  "No reasonable Bessel cutoff could be determined.");
307  }
308 
309  set_params(0, 0, _maxPWerror, far_switch_radius, bessel_cutoff);
310  }
311 }

313 void Mmm1dgpuForce::set_params(float _boxz, float _coulomb_prefactor,
314  float _maxPWerror, float _far_switch_radius,
315  int _bessel_cutoff, bool manual) {
316  if (_boxz > 0 && _far_switch_radius > _boxz) {
317  throw std::runtime_error(
318  "switching radius must not be larger than box length");
319  }
320  auto const _far_switch_radius_2 = Utils::sqr(_far_switch_radius);
321  auto const _uz = 1.0f / _boxz;
322  for (int d = 0; d < deviceCount; d++) {
323  // double colons are needed to access the constant memory variables because
324  // they are file globals and we have identically named class variables
325  cudaSetDevice(d);
326  if (manual) // tuning needs to be performed again
327  {
328  far_switch_radius = _far_switch_radius;
329  bessel_cutoff = _bessel_cutoff;
330  }
331  if (_far_switch_radius >= 0) {
332  mmm1d_params.far_switch_radius_2 = _far_switch_radius_2;
333  cuda_safe_mem(cudaMemcpyToSymbol(::far_switch_radius_2,
334  &_far_switch_radius_2, sizeof(float)));
335  far_switch_radius = _far_switch_radius;
336  }
337  if (_boxz > 0) {
338  host_boxz = _boxz;
339  cuda_safe_mem(cudaMemcpyToSymbol(::boxz, &_boxz, sizeof(float)));
340  cuda_safe_mem(cudaMemcpyToSymbol(::uz, &_uz, sizeof(float)));
341  }
342  if (_coulomb_prefactor != 0) {
343  cuda_safe_mem(cudaMemcpyToSymbol(::coulomb_prefactor, &_coulomb_prefactor,
344  sizeof(float)));
345  coulomb_prefactor = _coulomb_prefactor;
346  }
347  if (_bessel_cutoff > 0) {
348  mmm1d_params.bessel_cutoff = _bessel_cutoff;
349  cuda_safe_mem(
350  cudaMemcpyToSymbol(::bessel_cutoff, &_bessel_cutoff, sizeof(int)));
351  bessel_cutoff = _bessel_cutoff;
352  }
353  if (_maxPWerror > 0) {
354  mmm1d_params.maxPWerror = _maxPWerror;
355  cuda_safe_mem(
356  cudaMemcpyToSymbol(::maxPWerror, &_maxPWerror, sizeof(float)));
357  maxPWerror = _maxPWerror;
358  }
359  }
360  need_tune = true;
361 
362  // The changed parameters in @ref mmm1d_params do not need to be broadcast:
363  // they are only accessed by the CUDA code which only runs on the head node.
364  // We couldn't broadcast from here anyway because @ref set_params() is
365  // called from the @ref computeForces() and @ref tune() functions.
366 }

368 __global__ void forcesKernel(const float *__restrict__ r,
369  const float *__restrict__ q,
370  float *__restrict__ force, std::size_t N,
371  int pairs, std::size_t tStart) {
372 
373  constexpr auto c_2pif = 2 * Utils::pi<float>();
374  auto const tStop = Utils::sqr(N);
375 
376  for (std::size_t tid = threadIdx.x + blockIdx.x * blockDim.x + tStart;
377  tid < tStop; tid += blockDim.x * gridDim.x) {
378  auto const p1 = tid % N, p2 = tid / N;
379  auto x = r[3 * p2 + 0] - r[3 * p1 + 0];
380  auto y = r[3 * p2 + 1] - r[3 * p1 + 1];
381  auto z = r[3 * p2 + 2] - r[3 * p1 + 2];
382  auto const rxy2 = sqpow(x) + sqpow(y);
383  auto rxy = sqrt(rxy2);
384  auto sum_r = 0.f;
385  auto sum_z = 0.f;
386 
387  // if (*boxz <= 0.0) return; // in case we are not initialized yet
388 
389  while (fabs(z) > *boxz / 2) // make sure we take the shortest distance
390  z -= (z > 0 ? 1.f : -1.f) * *boxz;
391 
392  if (p1 == p2) // particle exerts no force on itself
393  {
394  rxy = 1.f; // so the division at the end doesn't fail with NaN
395  // (sum_r is 0 anyway)
396  } else if (rxy2 <= *far_switch_radius_2) // near formula
397  {
398  auto const uzz = *uz * z;
399  auto const uzr = *uz * rxy;
400  sum_z = dev_mod_psi_odd(0, uzz);
401  auto uzrpow = uzr;
402  for (int n = 1; n < *device_n_modPsi; n++) {
403  auto const sum_r_old = sum_r;
404  auto const mpe = dev_mod_psi_even(n, uzz);
405  auto const mpo = dev_mod_psi_odd(n, uzz);
406 
407  sum_r += 2 * static_cast<float>(n) * mpe * uzrpow;
408  uzrpow *= uzr;
409  sum_z += mpo * uzrpow;
410  uzrpow *= uzr;
411 
412  if (fabs(sum_r_old - sum_r) < *maxPWerror)
413  break;
414  }
415 
416  sum_r *= sqpow(*uz);
417  sum_z *= sqpow(*uz);
418 
419  sum_r += rxy * cbpow(rsqrt(rxy2 + sqpow(z)));
420  sum_r += rxy * cbpow(rsqrt(rxy2 + sqpow(z + *boxz)));
421  sum_r += rxy * cbpow(rsqrt(rxy2 + sqpow(z - *boxz)));
422 
423  sum_z += z * cbpow(rsqrt(rxy2 + sqpow(z)));
424  sum_z += (z + *boxz) * cbpow(rsqrt(rxy2 + sqpow(z + *boxz)));
425  sum_z += (z - *boxz) * cbpow(rsqrt(rxy2 + sqpow(z - *boxz)));
426 
427  if (rxy == 0) // particles at the same radial position only exert a force
428  // in z direction
429  {
430  rxy = 1.f; // so the division at the end doesn't fail with NaN
431  // (sum_r is 0 anyway)
432  }
433  } else // far formula
434  {
435  for (int p = 1; p < *bessel_cutoff; p++) {
436  float arg = c_2pif * *uz * static_cast<float>(p);
437  sum_r += static_cast<float>(p) * dev_K1(arg * rxy) * cos(arg * z);
438  sum_z += static_cast<float>(p) * dev_K0(arg * rxy) * sin(arg * z);
439  }
440  sum_r *= sqpow(*uz) * 4 * c_2pif;
441  sum_z *= sqpow(*uz) * 4 * c_2pif;
442  sum_r += 2 * *uz / rxy;
443  }
444 
445  auto const pref = *coulomb_prefactor * q[p1] * q[p2];
446  if (pairs) {
447  force[3 * (p1 + p2 * N - tStart) + 0] = pref * sum_r / rxy * x;
448  force[3 * (p1 + p2 * N - tStart) + 1] = pref * sum_r / rxy * y;
449  force[3 * (p1 + p2 * N - tStart) + 2] = pref * sum_z;
450  } else {
451  atomicAdd(&force[3 * p2 + 0], pref * sum_r / rxy * x);
452  atomicAdd(&force[3 * p2 + 1], pref * sum_r / rxy * y);
453  atomicAdd(&force[3 * p2 + 2], pref * sum_z);
454  }
455  }
456 }

458 __global__ void energiesKernel(const float *__restrict__ r,
459  const float *__restrict__ q,
460  float *__restrict__ energy, std::size_t N,
461  int pairs, std::size_t tStart) {
462 
463  constexpr auto c_2pif = 2 * Utils::pi<float>();
464  constexpr auto c_gammaf = Utils::gamma<float>();
465  auto const tStop = Utils::sqr(N);
466 
467  extern __shared__ float partialsums[];
468  if (!pairs) {
469  partialsums[threadIdx.x] = 0;
470  __syncthreads();
471  }
472  for (std::size_t tid = threadIdx.x + blockIdx.x * blockDim.x + tStart;
473  tid < tStop; tid += blockDim.x * gridDim.x) {
474  auto const p1 = tid % N, p2 = tid / N;
475  auto z = r[3 * p2 + 2] - r[3 * p1 + 2];
476  auto const rxy2 = sqpow(r[3 * p2 + 0] - r[3 * p1 + 0]) +
477  sqpow(r[3 * p2 + 1] - r[3 * p1 + 1]);
478  auto rxy = sqrt(rxy2);
479  auto sum_e = 0.f;
480 
481  // if (*boxz <= 0.0) return; // in case we are not initialized yet
482 
483  while (fabs(z) > *boxz / 2) // make sure we take the shortest distance
484  z -= (z > 0 ? 1.f : -1.f) * *boxz;
485 
486  if (p1 == p2) // particle exerts no force on itself
487  {
488  } else if (rxy2 <= *far_switch_radius_2) // near formula
489  {
490  auto const uzz = *uz * z;
491  auto const uzr2 = sqpow(*uz * rxy);
492  auto uzrpow = uzr2;
493  sum_e = dev_mod_psi_even(0, uzz);
494  for (int n = 1; n < *device_n_modPsi; n++) {
495  auto const sum_e_old = sum_e;
496  auto const mpe = dev_mod_psi_even(n, uzz);
497  sum_e += mpe * uzrpow;
498  uzrpow *= uzr2;
499 
500  if (fabs(sum_e_old - sum_e) < *maxPWerror)
501  break;
502  }
503 
504  sum_e *= -1 * *uz;
505  sum_e -= 2 * *uz * c_gammaf;
506  sum_e += rsqrt(rxy2 + sqpow(z));
507  sum_e += rsqrt(rxy2 + sqpow(z + *boxz));
508  sum_e += rsqrt(rxy2 + sqpow(z - *boxz));
509  } else // far formula
510  {
511  sum_e = -(log(rxy * *uz / 2) + c_gammaf) / 2;
512  for (int p = 1; p < *bessel_cutoff; p++) {
513  auto const arg = c_2pif * *uz * static_cast<float>(p);
514  sum_e += dev_K0(arg * rxy) * cos(arg * z);
515  }
516  sum_e *= *uz * 4;
517  }
518 
519  if (pairs) {
520  energy[p1 + p2 * N - tStart] = *coulomb_prefactor * q[p1] * q[p2] * sum_e;
521  } else {
522  partialsums[threadIdx.x] += *coulomb_prefactor * q[p1] * q[p2] * sum_e;
523  }
524  }
525  if (!pairs) {
526  sumReduction(partialsums, &energy[blockIdx.x]);
527  }
528 }

530 __global__ void vectorReductionKernel(float const *src, float *dst,
531  std::size_t N, std::size_t tStart) {
532 
533  auto const tStop = Utils::sqr(N);
534 
535  for (std::size_t tid = threadIdx.x + blockIdx.x * blockDim.x; tid < N;
536  tid += blockDim.x * gridDim.x) {
537  auto const offset = (tid + (tStart % N)) % N;
538  for (std::size_t i = 0; tid + i * N < (tStop - tStart); i++) {
539 #pragma unroll 3
540  for (std::size_t d = 0; d < 3; d++) {
541  dst[3 * offset + d] -= src[3 * (tid + i * N) + d];
542  }
