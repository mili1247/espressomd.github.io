28 namespace PairCriteria {
29 /** @brief Criterion which provides a true/false for a pair of particles */
30 class PairCriterion {
31 public:
32  /** @brief Make a decision based on two Particle objects */
33  virtual bool decide(const Particle &p1, const Particle &p2) const = 0;
34  /** @brief Make a decision based on particle ids.
35  * This can only run on the master node outside the integration loop */
36  bool decide(int id1, int id2) const {
37  // Retrieve particle data
38  auto const &p1 = get_particle_data(id1);
39  auto const &p2 = get_particle_data(id2);
40  const bool res = decide(p1, p2);
41  return res;
42  }
43  virtual ~PairCriterion() = default;
44 };

60 /** True if the short range energy is larger than a cut_off */
61 class EnergyCriterion : public PairCriterion {
62 public:
63  bool decide(const Particle &p1, const Particle &p2) const override {
64  // Distance between particles
65  auto const vec21 = box_geo.get_mi_vector(p1.r.p, p2.r.p);
66  const double dist_betw_part = vec21.norm();
67 
68  // Interaction parameters for particle types
69  IA_parameters const &ia_params = *get_ia_param(p1.p.type, p2.p.type);
70 
71  return (calc_non_bonded_pair_energy(p1, p2, ia_params, vec21,
72  dist_betw_part)) >= m_cut_off;
73  };
74  double get_cut_off() { return m_cut_off; }
75  void set_cut_off(double c) { m_cut_off = c; }
76 
77 private:
78  double m_cut_off;
79 };
