134 __device__ float evaluateAsChebychevSeriesAt(float const *c, int n, float x) {
135  auto const x2 = 2.0f * x;
136  auto dd = c[n - 1];
137  auto d = x2 * dd + c[n - 2];
138  for (int j = n - 3; j >= 1; j--) {
139  auto const tmp = d;
140  d = x2 * d - dd + c[j];
141  dd = tmp;
142  }
143  return x * d - dd + c[0] / 2.0f;
144 }

146 __device__ float evaluateAsTaylorSeriesAt(float const *c, int n, float x) {
147  int cnt = n - 1;
148  auto r = c[cnt];
149  while (--cnt >= 0)
150  r = r * x + c[cnt];
151  return r;
152 }

154 __device__ float dev_K0(float x) {
155  auto const c = evaluateAsChebychevSeriesAt(
156  x <= 2.0f ? bk0_data : x <= 8.0f ? ak0_data : ak02_data,
157  x <= 2.0f ? bk0_size : x <= 8.0f ? ak0_size : ak02_size,
158  x <= 2.0f ? x * x / 2.0f - 1.0f
159  : x <= 8.0f ? (16.0f / x - 5.0f) / 3.0f : (16.0f / x - 1.0f));
160  if (x <= 2.0f) {
161  auto const I0 =
162  evaluateAsChebychevSeriesAt(bi0_data, bi0_size, x * x / 4.5f - 1.0f);
163  return (-log(x) + Utils::ln_2<float>()) * I0 + c;
164  }
165  return exp(-x) * c * rsqrt(x);
166 }
