25 /**
26  * @file
27  *
28  * @ref Communication::MpiCallbacks manages MPI communication using a
29  * visitor pattern. The program runs on the head node and is responsible
30  * for calling callback functions on the worker nodes when necessary,
31  * e.g. to broadcast global variables or run an algorithm in parallel.
32  *
33  * Callbacks are registered on the head node as function pointers via
34  * the @ref callback_macros "callback macros". The visitor pattern
35  * allows using arbitrary function signatures. For non-void returning
36  * callbacks, several return value policies are available: ignore return
37  * value, return only one value (this is achieved using a boost optional
38  * that is empty on all but one node), return the value of the head node,
39  * or return a reduced value (by specifying the reduction operation).
40  */

87 /**
88  * @brief Check if a type can be used as a callback argument.
89  *
90  * This checks is a type can be a parameter type for a MPI callback.
91  * Not allowed are pointers and non-const references, as output
92  * parameters can not work across ranks.
93  */
94 template <class T>
95 using is_allowed_argument = std::integral_constant<
96  bool, not(std::is_pointer<T>::value ||
97  (!std::is_const<std::remove_reference_t<T>>::value &&
98  std::is_lvalue_reference<T>::value))>;
99 
100 template <class... Args>
101 using are_allowed_arguments =
102  typename Utils::conjunction<is_allowed_argument<Args>...>::type;
103 
104 /**
105  * @brief Invoke a callable with arguments from an mpi buffer.
106  *
107  * @tparam F A Callable that can be called with Args as parameters.
108  * @tparam Args Pack of arguments for @p F
109  *
110  * @param f Functor to be called
111  * @param ia Buffer to extract the parameters from
112  *
113  * @return Return value of calling @p f.
114  */
115 template <class F, class... Args>
116 auto invoke(F f, boost::mpi::packed_iarchive &ia) {
117  static_assert(are_allowed_arguments<Args...>::value,
118  "Pointers and non-const references are not allowed as "
119  "arguments for callbacks.");
120 
121  /* This is the local receive buffer for the parameters. We have to strip
122  away const so we can actually deserialize into it. */
123  std::tuple<std::remove_const_t<std::remove_reference_t<Args>>...> params;
124  Utils::for_each([&ia](auto &e) { ia >> e; }, params);
125 
126  /* We add const here, so that parameters can only be by value
127  or const reference. Output parameters on callbacks are not
128  sensible because the changes are not propagated back, so
129  we make sure this does not compile. */
130  return Utils::apply(f, Utils::as_const(params));
131 }

133 /**
134  * @brief Type-erased interface for callbacks.
135  *
136  * This encapsulates the signature of the callback
137  * and the parameter transfer, so that it can be
138  * called without any type information on the parameters.
139  */
140 struct callback_concept_t {
141  /**
142  * @brief Execute the callback.
143  *
144  * Unpack parameters for this callback, and then call it.
145  */
146  virtual void operator()(boost::mpi::communicator const &,
147  boost::mpi::packed_iarchive &) const = 0;
148  virtual ~callback_concept_t() = default;
149 };

151 /**
152  * @brief Callback without a return value.
153  *
154  * This is an implementation of a callback for a specific callable
155  * @p F and a set of arguments to call it with, where the there
156  * is no return value.
157  */
158 template <class F, class... Args>
159 struct callback_void_t final : public callback_concept_t {
160  F m_f;
161 
162  callback_void_t(callback_void_t const &) = delete;
163  callback_void_t(callback_void_t &&) = delete;
164 
165  template <class FRef>
166  explicit callback_void_t(FRef &&f) : m_f(std::forward<FRef>(f)) {}
167  void operator()(boost::mpi::communicator const &,
168  boost::mpi::packed_iarchive &ia) const override {
169  detail::invoke<F, Args...>(m_f, ia);
170  }
171 };

195 /**
196  * @brief Callback with a return value from one rank.
197  *
198  * This is an implementation of a callback for a specific callable
199  * @p F and a set of arguments to call it with, where the value from
200  * one rank is returned. Only one node is allowed to return a value.
201  */
202 template <class F, class... Args>
203 struct callback_one_rank_t final : public callback_concept_t {
204  F m_f;
205 
206  callback_one_rank_t(callback_one_rank_t const &) = delete;
207  callback_one_rank_t(callback_one_rank_t &&) = delete;
208 
209  template <class FRef>
210  explicit callback_one_rank_t(FRef &&f) : m_f(std::forward<FRef>(f)) {}
211  void operator()(boost::mpi::communicator const &comm,
212  boost::mpi::packed_iarchive &ia) const override {
213  auto const result = detail::invoke<F, Args...>(m_f, ia);
214 
215  assert(1 == boost::mpi::all_reduce(comm, static_cast<int>(!!result),
216  std::plus<>()) &&
217  "Incorrect number of return values");
218 
219  /* If this rank returned a result, send it to the head node. */
220  if (!!result) {
221  comm.send(0, 42, *result);
222  }
223  }
224 };

248 /**
249  * @brief Callback with return value reduction.
250  *
251  * This is an implementation of a callback for a specific callable
252  * @p F and a set of arguments to call it with, where the return
253  * value is reduced over the communicator.
254  */
255 template <class Op, class F, class... Args>
256 struct callback_reduce_t final : public callback_concept_t {
257  Op m_op;
258  F m_f;
259 
260  template <class OpRef, class FRef>
261  explicit callback_reduce_t(OpRef &&op, FRef &&f)
262  : m_op(std::forward<OpRef>(op)), m_f(std::forward<FRef>(f)) {}
263 
264  /**
265  * @brief Execute the callback.
266  *
267  * Receive parameters for this callback, and then call it.
268  *
269  * @param comm The communicator to receive the parameters on.
270  */
271  void operator()(boost::mpi::communicator const &comm,
272  boost::mpi::packed_iarchive &ia) const override {
273  /* Call the callback function, and reduce over the results with
274  * the stored reduction operation. */
275  boost::mpi::reduce(comm, detail::invoke<F, Args...>(m_f, ia), m_op, 0);
276  }
277 };

299 /**

301  *
302  * The signature is deduced from F::operator() const, which has
303  * to exist and can not be overloaded.

306  return make_model_impl(Result::Ignore{}, std::forward<F>(f),

346 /**
347  * @brief The interface of the MPI callback mechanism.
348  */
349 class MpiCallbacks {
350 public:
351  /**
352  * @brief RAII handle for a callback.
353  *
354  * This is what the client gets for registering a
355  * dynamic (= not function pointer) callback.
356  * It manages the lifetime of the callback handle
357  * needed to call it. The handle has a type derived
358  * from the signature of the callback, which makes
359  * it possible to do static type checking on the
360  * arguments.
361  */
362  template <class... Args> class CallbackHandle {
363  public:
364  template <typename F, class = std::enable_if_t<std::is_same<
365  typename detail::functor_types<F>::argument_types,
366  std::tuple<Args...>>::value>>
367  CallbackHandle(MpiCallbacks *cb, F &&f)
368  : m_id(cb->add(std::forward<F>(f))), m_cb(cb) {}
369 
370  CallbackHandle(CallbackHandle const &) = delete;
371  CallbackHandle(CallbackHandle &&rhs) noexcept = default;
372  CallbackHandle &operator=(CallbackHandle const &) = delete;
373  CallbackHandle &operator=(CallbackHandle &&rhs) noexcept = default;
374 
375  private:
376  int m_id;
377  MpiCallbacks *m_cb;
378 
379  public:
380  /**
381  * @brief Call the callback managed by this handle.
382  *
383  * The arguments are passed to the remote callees, it
384  * must be possible to call the function with the provided
385  * arguments, otherwise this will not compile.
386  */
387  template <class... ArgRef>
388  auto operator()(ArgRef &&... args) const
389  /* Enable if a hypothetical function with signature void(Args..)
390  * could be called with the provided arguments. */
391  -> std::enable_if_t<
392  std::is_void<decltype(std::declval<void (*)(Args...)>()(
393  std::forward<ArgRef>(args)...))>::value> {
394  if (m_cb)
395  m_cb->call(m_id, std::forward<ArgRef>(args)...);
396  }
397 
398  ~CallbackHandle() {
399  if (m_cb)
400  m_cb->remove(m_id);
401  }
402 
403  MpiCallbacks *cb() const { return m_cb; }
404  int id() const { return m_id; }
405  };
406 
407  /* Avoid accidental copy, leads to mpi deadlock or split brain */
408  MpiCallbacks(MpiCallbacks const &) = delete;
409  MpiCallbacks &operator=(MpiCallbacks const &) = delete;
410 
411 private:
412  static auto &static_callbacks() {
413  static std::vector<
414  std::pair<void (*)(), std::unique_ptr<detail::callback_concept_t>>>
415  m_callbacks;
416 
417  return m_callbacks;
418  }
419 
420 public:
421  explicit MpiCallbacks(boost::mpi::communicator comm,
422  bool abort_on_exit = true)
423  : m_abort_on_exit(abort_on_exit), m_comm(std::move(comm)) {
424  /* Add a dummy at id 0 for loop abort. */
425  m_callback_map.add(nullptr);
426 
427  for (auto &kv : static_callbacks()) {
428  m_func_ptr_to_id[kv.first] = m_callback_map.add(kv.second.get());
429  }
430  }
431 
432  ~MpiCallbacks() {
433  /* Release the clients on exit */
434  if (m_abort_on_exit && (m_comm.rank() == 0)) {
435  try {
436  abort_loop();
437  } catch (...) {
438  }
439  }
440  }
441 
442 private:
443  /**
444  * @brief Add a new callback.
445  *
446  * Add a new callback to the system. This is a collective
447  * function that must be run on all nodes.
448  *
449  * @tparam F An object with a const call operator.
450  *
451  * @param f The callback function to add.
452  * @return A handle with which the callback can be called.
453  */
454  template <typename F> auto add(F &&f) {
455  m_callbacks.emplace_back(detail::make_model(std::forward<F>(f)));
456  return m_callback_map.add(m_callbacks.back().get());
457  }
458 
459 public:
460  /**
461  * @brief Add a new callback.
462  *
463  * Add a new callback to the system. This is a collective
464  * function that must be run on all nodes.
465  *
466  * @param fp Pointer to the static callback function to add.
467  */
468  template <class... Args> void add(void (*fp)(Args...)) {
469  m_callbacks.emplace_back(detail::make_model(fp));
470  const int id = m_callback_map.add(m_callbacks.back().get());
471  m_func_ptr_to_id[reinterpret_cast<void (*)()>(fp)] = id;
472  }
473 
474  /**
475  * @brief Add a new callback.
476  *
477  * Add a new callback to the system. This is a collective
478  * function that must be run on all nodes.
479  *
480  * @param fp Pointer to the static callback function to add.
481  */
482  template <class... Args> static void add_static(void (*fp)(Args...)) {
483  static_callbacks().emplace_back(reinterpret_cast<void (*)()>(fp),
484  detail::make_model(fp));
485  }
486 
487  /**
488  * @brief Add a new callback with a return value.
489  *
490  * Add a new callback to the system. This is a collective
491  * function that must be run on all nodes.
492  * @p tag is one of the tag types from @ref Communication::Result,
493  * which indicates what to do with the return values.
494  *
495  * @param tag Tag type indicating return operation
496  * @param tag_args Argument for the return operation, if any.
497  * @param fp Pointer to the static callback function to add.
498  */
499  template <class Tag, class R, class... Args, class... TagArgs>
500  static void add_static(Tag tag, R (*fp)(Args...), TagArgs &&... tag_args) {
501  static_callbacks().emplace_back(
502  reinterpret_cast<void (*)()>(fp),
503  detail::make_model(tag, fp, std::forward<TagArgs>(tag_args)...));
504  }
505 
506 private:
507  /**
508  * @brief Remove callback.
509  *
510  * Remove the callback id from the callback list.
511  * This is a collective call that must be run on all node.
512  *
513  * @param id Identifier of the callback to remove.
514  */
515  void remove(int id) {
516  m_callbacks.erase(
517  boost::remove_if(m_callbacks,
518  [ptr = m_callback_map[id]](auto const &e) {
519  return e.get() == ptr;
520  }),
521  m_callbacks.end());
522  m_callback_map.remove(id);
523  }
524 
525 private:
526  /**
527  * @brief call a callback.
528  *
529  * Call the callback id.
530  * The method can only be called on the head node
531  * and has the prerequisite that the other nodes are
532  * in the MPI loop.
533  *
534  * @param id The callback to call.
535  * @param args Arguments for the callback.
536  */
537  template <class... Args> void call(int id, Args &&... args) const {
538  if (m_comm.rank() != 0) {
539  throw std::logic_error("Callbacks can only be invoked on rank 0.");
540  }
541 
542  /* Check if callback exists */
543  if (m_callback_map.find(id) == m_callback_map.end()) {
544  throw std::out_of_range("Callback does not exists.");
545  }
546 
547  /* Send request to worker nodes */
548  boost::mpi::packed_oarchive oa(m_comm);
549  oa << id;
550 
551  /* Pack the arguments into a packed mpi buffer. */
552  Utils::for_each([&oa](auto &&e) { oa << e; },
553  std::forward_as_tuple(std::forward<Args>(args)...));
554 
555  boost::mpi::broadcast(m_comm, oa, 0);
556  }
557 
558 public:
559  /**
560  * @brief call a callback.
561  *
562  * Call a static callback by pointer.
563  * The method can only be called on the head node
564  * and has the prerequisite that the other nodes are
565  * in the MPI loop. Also the function has to be previously
566  * registered e.g. with the @ref REGISTER_CALLBACK macro.
567  * The callback is **not** called on the head node.
568  *
569  * @param fp Pointer to the function to call.
570  * @param args Arguments for the callback.
571  */
572  template <class... Args, class... ArgRef>
573  auto call(void (*fp)(Args...), ArgRef &&... args) const ->
574  /* Enable only if fp can be called with the provided arguments,
575  * e.g. if fp(args...) is well-formed. */
576  std::enable_if_t<std::is_void<decltype(fp(args...))>::value> {
577  const int id = m_func_ptr_to_id.at(reinterpret_cast<void (*)()>(fp));
578 
579  call(id, std::forward<ArgRef>(args)...);
580  }
581 
582  /**
583  * @brief call a callback.
584  *
585  * Call a static callback by pointer.
586  * The method can only be called on the head node
587  * and has the prerequisite that the other nodes are
588  * in the MPI loop. Also the function has to be previously
589  * registered e.g. with the @ref REGISTER_CALLBACK macro.
590  * The callback is called on the head node.
591  *
592  * @param fp Pointer to the function to call.
593  * @param args Arguments for the callback.
594  */
595  template <class... Args, class... ArgRef>
596  void call_all(void (*fp)(Args...), ArgRef &&... args) const {
597  call(fp, args...);
598  fp(args...);
599  }

601  /**
602  * @brief Call a callback and reduce the result over all nodes.
603  *
604  * This calls a callback on all nodes, including the head node,
605  * and does a mpi reduction with the registered operation.
606  *
607  * This method can only be called on the head node.
608  */
609  template <class Op, class R, class... Args>
610  auto call(Result::Reduction, Op op, R (*fp)(Args...), Args... args) const
611  -> std::remove_reference_t<decltype(op(std::declval<R>(),
612  std::declval<R>()))> {
613  const int id = m_func_ptr_to_id.at(reinterpret_cast<void (*)()>(fp));
614 
615  call(id, args...);
616 
617  std::remove_cv_t<std::remove_reference_t<decltype(
618  op(std::declval<R>(), std::declval<R>()))>>
619  result{};
620  boost::mpi::reduce(m_comm, fp(args...), result, op, 0);
621 
622  return result;
623  }

625  /**
626  * @brief Call a callback and ignore the result over all nodes.
627  *
628  * This calls a callback on all nodes, including the head node,
629  * and ignore all return values.
630  *
631  * This method can only be called on the head node.
632  */
633  template <class R, class... Args, class... ArgRef>
634  auto call(Result::Ignore, R (*fp)(Args...), ArgRef... args) const
635  -> std::remove_reference_t<R> {
636 
637  const int id = m_func_ptr_to_id.at(reinterpret_cast<void (*)()>(fp));
638  call(id, args...);
639 
640  fp(std::forward<Args>(args)...);
641 
642  return {};
643  }

645  /**
646  * @brief Call a callback and reduce the result over all nodes.
647  *
648  * This calls a callback on all nodes, including the head node,
649  * and returns the value from the node which returned an engaged
650  * optional. Only one node is allowed to return a value.
651  *
652  * This method can only be called on the head node.
653  */
654  template <class R, class... Args, class... ArgRef>
655  auto call(Result::OneRank, boost::optional<R> (*fp)(Args...),
656  ArgRef... args) const -> std::remove_reference_t<R> {
657 
658  const int id = m_func_ptr_to_id.at(reinterpret_cast<void (*)()>(fp));
659  call(id, args...);
660 
661  auto const local_result = fp(std::forward<Args>(args)...);
662 
663  assert(1 == boost::mpi::all_reduce(m_comm, static_cast<int>(!!local_result),
664  std::plus<>()) &&
665  "Incorrect number of return values");
666 
667  if (!!local_result) {
668  return *local_result;
669  }
670 
671  std::remove_cv_t<std::remove_reference_t<R>> result;
672  m_comm.recv(boost::mpi::any_source, boost::mpi::any_tag, result);
673  return result;
674  }

677  * @brief Call a callback and return the result of the head node.
678  *
679  * This calls a callback on all nodes, including the head node,
680  * and returns the value from the head node.
681  *
682  * This method can only be called on the head node.
683  */
684  template <class R, class... Args, class... ArgRef>
685  auto call(Result::MasterRank, R (*fp)(Args...), ArgRef... args) const
686  -> std::remove_reference_t<R> {
687

689  call(id, args...);
690

694  /**
695  * @brief Mpi slave loop.
696  *
697  * This is the callback loop for the slaves. They block
698  * on the MPI call and wait for a new callback request
699  * coming from the master.
700  * This should be run on the slaves and must be running
701  * so that the master can issue call().
702  */
703  void loop() const {
704  for (;;) {
705  /* Communicate callback id and parameters */
706  boost::mpi::packed_iarchive ia(m_comm);
707  boost::mpi::broadcast(m_comm, ia, 0);
708 
709  int request;
710  ia >> request;
711 
712  if (request == LOOP_ABORT) {
713  break;
714  }
715  /* Call the callback */
716  m_callback_map[request]->operator()(m_comm, ia);
717  }
718  }
719 
720  /**
721  * @brief Abort mpi loop.
722  *
723  * Make the slaves exit the MPI loop.
724  */
725  void abort_loop() { call(LOOP_ABORT); }
726 
727  /**

729  */
730  boost::mpi::communicator const &comm() const { return m_comm; }
