143 __global__ void createGreensfcn() {
144  unsigned int index = fde_getThreadIndex();
145  unsigned int tmp;
146  unsigned int coord[3];
147 
148  coord[0] = index % (fde_parameters_gpu->dim_x / 2 + 1);
149  tmp = index / (fde_parameters_gpu->dim_x / 2 + 1);
150  coord[1] = tmp % fde_parameters_gpu->dim_y;
151  coord[2] = tmp / fde_parameters_gpu->dim_y;
152 
153  if (index < fde_parameters_gpu->dim_z * fde_parameters_gpu->dim_y *
154  (fde_parameters_gpu->dim_x / 2 + 1)) {
155 
156  if (index == 0) {
157  // setting 0th Fourier mode to 0 enforces charge neutrality
158  fde_parameters_gpu->greensfcn[index] = 0.0f;
159  } else {
160  constexpr cufftReal two_pi = 2.0f * Utils::pi<cufftReal>();
161  fde_parameters_gpu->greensfcn[index] =
162  -2.0f * two_pi * fde_parameters_gpu->prefactor *
163  fde_parameters_gpu->agrid * fde_parameters_gpu->agrid * 0.5f /
164  (cos(two_pi * static_cast<cufftReal>(coord[0]) /
165  static_cast<cufftReal>(fde_parameters_gpu->dim_x)) +
166  cos(two_pi * static_cast<cufftReal>(coord[1]) /
167  static_cast<cufftReal>(fde_parameters_gpu->dim_y)) +
168  cos(two_pi * static_cast<cufftReal>(coord[2]) /
169  static_cast<cufftReal>(fde_parameters_gpu->dim_z)) -
170  3.0f) /
171  static_cast<cufftReal>(fde_parameters_gpu->dim_x *
172  fde_parameters_gpu->dim_y *
173  fde_parameters_gpu->dim_z);
174  }
175 
176  // fde_parameters_gpu->greensfcn[index] = 0.0f; //TODO delete
177  }
178 }

195 void FdElectrostatics::calculatePotential(cufftComplex *charge_potential) {
196 
197  if (cufftExecR2C(plan_fft, (cufftReal *)charge_potential, charge_potential) !=
198  CUFFT_SUCCESS) {
199 
200  fprintf(stderr, "ERROR: Unable to execute FFT plan\n");
201  }
202 
203  dim3 dim_grid = calculate_dim_grid(
204  static_cast<unsigned>(parameters.dim_z * parameters.dim_y *
205  (parameters.dim_x / 2 + 1)),
206  4, threads_per_block);
207 
208  KERNELCALL(multiplyGreensfcn, dim_grid, threads_per_block, charge_potential);
209 
210  if (cufftExecC2R(plan_ifft, charge_potential,
211  (cufftReal *)charge_potential) != CUFFT_SUCCESS) {
212 
213  fprintf(stderr, "ERROR: Unable to execute iFFT plan\n");
214  }
