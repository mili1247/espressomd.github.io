30 /** Base class for observables.
31  *
32  * An observable extracts raw data from a system or compute a statistic based
33  * on the state of a system, and returns an array of doubles.
34  *
35  * %Observables typically don't have setters or getters to access and modify
36  * their member variables, and usually only have a default constructor with no
37  * argument. Each observable class has a corresponding interface in
38  * @ref ScriptInterface::Observables, where setters and getters are defined.
39  */
40 class Observable {
41 public:
42  Observable() = default;
43  virtual ~Observable() = default;
44  /** Calculate the set of values measured by the observable */
45  virtual std::vector<double> operator()() const = 0;
46 
47  /** Size of the flat array returned by the observable */
48  std::size_t n_values() const {
49  auto const v = shape();
50  return std::accumulate(v.begin(), v.end(), 1u, std::multiplies<>());
51  }
52 
53  /** Dimensions needed to reshape the flat array returned by the observable */
54  virtual std::vector<std::size_t> shape() const = 0;
55 };
