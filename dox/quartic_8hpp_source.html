62 /** Compute the quartic bond force.
63  * @param[in] dx %Distance between the particles.
64  */
65 inline boost::optional<Utils::Vector3d>
66 QuarticBond::force(Utils::Vector3d const &dx) const {
67  auto const dist = dx.norm();
68 
69  if ((r_cut > 0.0) && (dist > r_cut)) {
70  return {};
71  }
72 
73  auto const dr = dist - r;
74  auto const fac = (k0 * dr + k1 * Utils::int_pow<3>(dr)) / dist;
75  return -fac * dx;
76 }
77 
78 /** Compute the quartic bond energy.
79  * @param[in] dx %Distance between the particles.
80  */
81 inline boost::optional<double>
82 QuarticBond::energy(Utils::Vector3d const &dx) const {
83  auto const dist = dx.norm();
84 
85  if ((r_cut > 0.0) && (dist > r_cut)) {
86  return {};
87  }
88 
89  double dr2 = Utils::sqr(dist - r);
90 
91  return 0.5 * k0 * dr2 + 0.25 * k1 * Utils::sqr(dr2);
92 }
