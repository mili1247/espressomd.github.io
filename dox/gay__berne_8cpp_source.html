33 int gay_berne_set_params(int part_type_a, int part_type_b, double eps,
34  double sig, double cut, double k1, double k2,
35  double mu, double nu) {
36  IA_parameters *data = get_ia_param_safe(part_type_a, part_type_b);
37 
38  if (!data)
39  return ES_ERROR;
40 
41  data->gay_berne.eps = eps;
42  data->gay_berne.sig = sig;
43  data->gay_berne.cut = cut;
44  data->gay_berne.k1 = k1;
45  data->gay_berne.k2 = k2;
46  data->gay_berne.mu = mu;
47  data->gay_berne.nu = nu;
48 
49  /* Calculate dependent parameters */
50 
51  data->gay_berne.chi1 = ((data->gay_berne.k1 * data->gay_berne.k1) - 1) /
52  ((data->gay_berne.k1 * data->gay_berne.k1) + 1);
53  data->gay_berne.chi2 =
54  (pow(data->gay_berne.k2, (1 / data->gay_berne.mu)) - 1) /
55  (pow(data->gay_berne.k2, (1 / data->gay_berne.mu)) + 1);
56 
57  /* broadcast interaction parameters */
58  mpi_bcast_ia_params(part_type_a, part_type_b);
59 
60  return ES_OK;
61 }
