33 /** Propagate the velocities and positions. Integration steps before force
34  * calculation of the Velocity Verlet integrator: <br> \f[ v(t+0.5 \Delta t) =
35  * v(t) + 0.5 \Delta t f(t)/m \f] <br> \f[ p(t+\Delta t) = p(t) + \Delta t
36  * v(t+0.5 \Delta t) \f]
37  */
38 inline void velocity_verlet_propagate_vel_pos(const ParticleRange &particles,
39  double time_step) {
40 
41  auto const skin2 = Utils::sqr(0.5 * skin);
42  for (auto &p : particles) {
43 #ifdef ROTATION
44  propagate_omega_quat_particle(p, time_step);
45 #endif
46 
47  // Don't propagate translational degrees of freedom of vs
48  if (p.p.is_virtual)
49  continue;
50  for (int j = 0; j < 3; j++) {
51  if (!(p.p.ext_flag & COORD_FIXED(j))) {
52  /* Propagate velocities: v(t+0.5*dt) = v(t) + 0.5 * dt * a(t) */
53  p.m.v[j] += 0.5 * time_step * p.f.f[j] / p.p.mass;
54 
55  /* Propagate positions (only NVT): p(t + dt) = p(t) + dt *
56  * v(t+0.5*dt) */
57  p.r.p[j] += time_step * p.m.v[j];
58  }
59  }
60 
61  /* Verlet criterion check*/
62  if ((p.r.p - p.l.p_old).norm2() > skin2)
63  cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
64  }
65 }

67 /** Final integration step of the Velocity Verlet integrator
68  * \f[ v(t+\Delta t) = v(t+0.5 \Delta t) + 0.5 \Delta t f(t+\Delta t)/m \f]
69  */
70 inline void velocity_verlet_propagate_vel_final(const ParticleRange &particles,
71  double time_step) {
72 
73  for (auto &p : particles) {
74  // Virtual sites are not propagated during integration
75  if (p.p.is_virtual)
76  continue;
77 
78  for (int j = 0; j < 3; j++) {
79  if (!(p.p.ext_flag & COORD_FIXED(j))) {
80  /* Propagate velocity: v(t+dt) = v(t+0.5*dt) + 0.5*dt * a(t+dt) */
81  p.m.v[j] += 0.5 * time_step * p.f.f[j] / p.p.mass;
82  }
83  }
84  }
85 }
