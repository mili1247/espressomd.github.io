40 static void pack_particles(ParticleRange particles,
41  CUDA_particle_data *buffer) {
42  using Utils::Vector3f;
43 
44  int i = 0;
45  for (auto const &part : particles) {
46  buffer[i].p = static_cast<Vector3f>(folded_position(part.r.p, box_geo));
47 
48 #ifdef CUDA
49  buffer[i].identity = part.p.identity;
50  buffer[i].v = static_cast<Vector3f>(part.m.v);
51 #ifdef VIRTUAL_SITES
52  buffer[i].is_virtual = part.p.is_virtual;
53 #endif
54 #endif
55 
56 #ifdef DIPOLES
57  buffer[i].dip = static_cast<Vector3f>(part.calc_dip());
58 #endif
59 
60 #if defined(LB_ELECTROHYDRODYNAMICS) && defined(CUDA)
61  buffer[i].mu_E = static_cast<Vector3f>(part.p.mu_E);
62 #endif
63 
64 #ifdef ELECTROSTATICS
65  buffer[i].q = static_cast<float>(part.p.q);
66 #endif
67 
68 #ifdef MASS
69  buffer[i].mass = static_cast<float>(part.p.mass);
70 #endif
71 
72 #ifdef ROTATION
73  buffer[i].director = static_cast<Vector3f>(part.r.calc_director());
74 #endif
75 
76 #ifdef ENGINE
77  buffer[i].swim.v_swim = static_cast<float>(part.p.swim.v_swim);
78  buffer[i].swim.f_swim = static_cast<float>(part.p.swim.f_swim);
79  buffer[i].swim.director = buffer[i].director;
80 
81  buffer[i].swim.push_pull = part.p.swim.push_pull;
82  buffer[i].swim.dipole_length =
83  static_cast<float>(part.p.swim.dipole_length);
84  buffer[i].swim.swimming = part.p.swim.swimming;
85 #endif
86  i++;
87  }
88 }

90 void cuda_mpi_get_particles(
91  const ParticleRange &particles,
92  pinned_vector<CUDA_particle_data> &particle_data_host) {
93  auto const n_part = particles.size();
94 
95  if (this_node > 0) {
96  static std::vector<CUDA_particle_data> buffer;
97  buffer.resize(n_part);
98  /* pack local parts into buffer */
99  pack_particles(particles, buffer.data());
100 
101  Utils::Mpi::gather_buffer(buffer, comm_cart);
102  } else {
103  particle_data_host.resize(n_part);
104 
105  /* Pack own particles */
106  pack_particles(particles, particle_data_host.data());
107 
108  Utils::Mpi::gather_buffer(particle_data_host, comm_cart);
109  }
110 }

112 /**
113  * @brief Add a flat force (and torque) array to a range of particles.
114  *
115  * @param particles The particles the forces (and torques should be added to)
116  * @param forces The forces as flat array of size 3 * particles.size()
117  * @param torques The torques as flat array of size 3 * particles.size(),
118  * this is only touched if ROTATION is active.
119  */
120 static void add_forces_and_torques(ParticleRange particles,
121  Utils::Span<const float> forces,
122  Utils::Span<const float> torques) {
123  int i = 0;
124  for (auto &part : particles) {
125  for (int j = 0; j < 3; j++) {
126  part.f.f[j] += forces[3 * i + j];
127 #ifdef ROTATION
128  part.f.torque[j] += torques[3 * i + j];
129 #endif
130  }
131  i++;
132  }
133 }

135 void cuda_mpi_send_forces(const ParticleRange &particles,
136  Utils::Span<float> host_forces,
137  Utils::Span<float> host_torques) {
138  auto const n_elements = 3 * particles.size();
139 
140  if (this_node > 0) {
141  static std::vector<float> buffer_forces;
142  static std::vector<float> buffer_torques;
143  /* Alloc buffer */
144  buffer_forces.resize(n_elements);
145 
146  Utils::Mpi::scatter_buffer(buffer_forces.data(), n_elements, comm_cart);
147 #ifdef ROTATION
148  /* Alloc buffer */
149  buffer_torques.resize(n_elements);
150 
151  Utils::Mpi::scatter_buffer(buffer_torques.data(), n_elements, comm_cart);
152 #endif
153 
154  add_forces_and_torques(particles, buffer_forces, buffer_torques);
155  } else {
156  /* Scatter forces to slaves */
157  Utils::Mpi::scatter_buffer(host_forces.data(), n_elements, comm_cart);
158 #ifdef ROTATION
159  Utils::Mpi::scatter_buffer(host_torques.data(), n_elements, comm_cart);
160 #endif
161 
162  add_forces_and_torques(particles, host_forces, host_torques);
163  }
164 }
