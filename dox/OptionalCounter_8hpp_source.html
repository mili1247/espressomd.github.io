28  *

31  */
32 class OptionalCounter {

34  Utils::Counter<uint64_t> m_counter;

36 
37 public:
38  OptionalCounter() : m_counter{}, m_initialized(false) {}
39  OptionalCounter(Utils::Counter<uint64_t> const &counter)
40  : m_counter(counter), m_initialized(true) {}
41  OptionalCounter &operator=(Utils::Counter<uint64_t> counter) {
42  m_counter = std::move(counter);
43  m_initialized = true;
44  return *this;
45  }
46  template <class Archive>
