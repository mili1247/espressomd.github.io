69 /** Variant of @ref add_non_bonded_pair_force where only %Coulomb
70  * contributions are calculated
71  */
72 inline void add_non_bonded_pair_force_icc(Particle &p1, Particle &p2,
73  Utils::Vector3d const &d, double dist,
74  double dist2) {
75  auto forces = Coulomb::pair_force(p1, p2, d, dist);
76 
77  p1.f.f += std::get<0>(forces);
78  p2.f.f -= std::get<0>(forces);
79 #ifdef P3M
80  p1.f.f += std::get<1>(forces);
81  p2.f.f += std::get<2>(forces);
82 #endif
83 }
84 void icc_iteration(const ParticleRange &particles,
85  const ParticleRange &ghost_particles) {
86  if (icc_cfg.n_icc == 0)
87  return;
88 
89  Coulomb::icc_sanity_check();
90 
91  auto const pref = 1.0 / (coulomb.prefactor * 2 * Utils::pi());
92  icc_cfg.citeration = 0;
93 
94  double globalmax = 0.;
95 
96  for (int j = 0; j < icc_cfg.num_iteration; j++) {
97  double charge_density_max = 0.;
98 
99  force_calc_icc(particles, ghost_particles); /* Calculate electrostatic
100  forces (SR+LR) excluding source source interaction*/
101  cell_structure.ghosts_reduce_forces();
102 
103  double diff = 0;
104 
105  for (auto &p : particles) {
106  if (p.p.identity < icc_cfg.n_icc + icc_cfg.first_id &&
107  p.p.identity >= icc_cfg.first_id) {
108  auto const id = p.p.identity - icc_cfg.first_id;
109  /* the dielectric-related prefactor: */
110  auto const del_eps =
111  (icc_cfg.ein[id] - icc_cfg.eout) / (icc_cfg.ein[id] + icc_cfg.eout);
112  /* calculate the electric field at the certain position */
113  auto const local_e_field = p.f.f / p.p.q + icc_cfg.ext_field;
114 
115  if (local_e_field.norm2() == 0) {
116  runtimeErrorMsg()
117  << "ICC found zero electric field on a charge. This must "
118  "never happen";
119  }
120 
121  auto const charge_density_old = p.p.q / icc_cfg.areas[id];
122 
123  charge_density_max =
124  std::max(charge_density_max, std::abs(charge_density_old));
125 
126  auto const charge_density_update =
127  del_eps * pref * (local_e_field * icc_cfg.normals[id]) +
128  2 * icc_cfg.eout / (icc_cfg.eout + icc_cfg.ein[id]) *
129  icc_cfg.sigma[id];
130  /* relative variation: never use an estimator which can be negative
131  * here */
132  auto const charge_density_new =
133  (1. - icc_cfg.relax) * charge_density_old +
134  (icc_cfg.relax) * charge_density_update;
135 
136  /* Take the largest error to check for convergence */
137  auto const relative_difference =
138  std::abs((charge_density_new - charge_density_old) /
139  (charge_density_max +
140  std::abs(charge_density_new + charge_density_old)));
141 
142  diff = std::max(diff, relative_difference);
143 
144  p.p.q = charge_density_new * icc_cfg.areas[id];
145 
146  /* check if the charge now is more than 1e6, to determine if ICC still
147  * leads to reasonable results. This is kind of an arbitrary measure
148  * but does a good job spotting divergence! */
149  if (std::abs(p.p.q) > 1e6) {
150  runtimeErrorMsg()
151  << "too big charge assignment in icc! q >1e6 , assigned "
152  "charge= "
153  << p.p.q;
154 
155  diff = 1e90; /* A very high value is used as error code */
156  break;
157  }
158  }
159  } /* cell particles */
160  /* Update charges on ghosts. */
161  cell_structure.ghosts_update(Cells::DATA_PART_PROPERTIES);
162 
163  icc_cfg.citeration++;
164 
165  boost::mpi::all_reduce(comm_cart, diff, globalmax,
166  boost::mpi::maximum<double>());
167 
168  if (globalmax < icc_cfg.convergence)
169  break;
170  } /* iteration */
171 
172  if (globalmax > icc_cfg.convergence) {
173  runtimeErrorMsg()
174  << "ICC failed to converge in the given number of maximal steps.";
175  }
176 
177  on_particle_charge_change();
178 }

180 void force_calc_icc(const ParticleRange &particles,
181  const ParticleRange &ghost_particles) {
182  init_forces_icc(particles, ghost_particles);
183 
184  cell_structure.non_bonded_loop(
185  [](Particle &p1, Particle &p2, Distance const &d) {
186  /* calc non-bonded interactions */
187  add_non_bonded_pair_force_icc(p1, p2, d.vec21, sqrt(d.dist2), d.dist2);
188  });
189 
190  Coulomb::calc_long_range_force(particles);
191 }

220 void icc_set_params(int n_icc, double convergence, double relaxation,
221  Utils::Vector3d &ext_field, int max_iterations,
222  int first_id, double eps_out, std::vector<double> &areas,
223  std::vector<double> &e_in, std::vector<double> &sigma,
224  std::vector<Utils::Vector3d> &normals) {
225  if (n_icc < 0)
226  throw std::runtime_error("ICC: invalid number of particles. " +
227  std::to_string(n_icc));
228  if (convergence <= 0)
229  throw std::runtime_error("ICC: invalid convergence value. " +
230  std::to_string(convergence));
231  if (relaxation < 0 or relaxation > 2)
232  throw std::runtime_error("ICC: invalid relaxation value. " +
233  std::to_string(relaxation));
234  if (max_iterations <= 0)
235  throw std::runtime_error("ICC: invalid max_iterations. " +
236  std::to_string(max_iterations));
237  if (first_id < 0)
238  throw std::runtime_error("ICC: invalid first_id. " +
239  std::to_string(first_id));
240  if (eps_out <= 0)
241  throw std::runtime_error("ICC: invalid eps_out. " +
242  std::to_string(eps_out));
243  if (areas.size() != n_icc)
244  throw std::runtime_error("ICC: invalid areas vector.");
245  if (e_in.size() != n_icc)
246  throw std::runtime_error("ICC: invalid e_in vector.");
247  if (sigma.size() != n_icc)
248  throw std::runtime_error("ICC: invalid sigma vector.");
249  if (normals.size() != n_icc)
250  throw std::runtime_error("ICC: invalid normals vector.");
251 
252  icc_cfg.n_icc = n_icc;
253  icc_cfg.convergence = convergence;
254  icc_cfg.relax = relaxation;
255  icc_cfg.ext_field = ext_field;
256  icc_cfg.num_iteration = max_iterations;
257  icc_cfg.first_id = first_id;
258  icc_cfg.eout = eps_out;
259 
260  icc_cfg.areas = std::move(areas);
261  icc_cfg.ein = std::move(e_in);
262  icc_cfg.sigma = std::move(sigma);
263  icc_cfg.normals = std::move(normals);
264 
265  mpi_icc_init();
266 }
