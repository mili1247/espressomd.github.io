126 /**
127  * @brief Product quaternion and arithmetic type.
128  * @tparam T Data type of quaternion @p a.
129  * @tparam U Type of multiplier @p b.
130  * @param b Quaternion.
131  * @param a Multiplier.

135  std::enable_if_t<std::is_arithmetic<U>::value, bool> = true>

137  return boost::qvm::operator*(a, b);

140 /**
141  * @brief Convert quaternion to rotation matrix.
142  * @tparam T Data type of quaternion.
143  * @param q Quaternion.
144  * @return Rotation matrix.
145  */
146 template <typename T> Matrix<T, 3, 3> rotation_matrix(Quaternion<T> const &q) {
147  auto const normed_q = q.normalized();
148  auto const id_mat = Utils::identity_mat<double, 3, 3>();
149  auto const v1 = normed_q * id_mat.col<0>();
150  auto const v2 = normed_q * id_mat.col<1>();
151  auto const v3 = normed_q * id_mat.col<2>();
152  return {{v1[0], v2[0], v3[0]}, {v1[1], v2[1], v3[1]}, {v1[2], v2[2], v3[2]}};
153 }
