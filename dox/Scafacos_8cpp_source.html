119 void Scafacos::run(std::vector<double> &charges, std::vector<double> &positions,
120  std::vector<double> &fields,
121  std::vector<double> &potentials) {
122  const int local_n_part = charges.size();
123 
124  fields.resize(3 * local_n_part);
125  potentials.resize(local_n_part);
126 
127  handle_error(
128  fcs_tune(handle, local_n_part, positions.data(), charges.data()));
129 
130  handle_error(fcs_run(handle, local_n_part, positions.data(), charges.data(),
131  fields.data(), potentials.data()));
132 }

158 void Scafacos::set_common_parameters(const double *box_l,
159  const int *periodicity,
160  int total_particles) {
161  double boxa[3] = {0., 0., 0.}, boxb[3] = {0., 0., 0.}, boxc[3] = {0., 0., 0.},
162  off[3] = {0., 0., 0.};
163  boxa[0] = box_l[0];
164  boxb[1] = box_l[1];
165  boxc[2] = box_l[2];
166  // Does scafacos calculate the near field part?
167  // For charges, ESPResSo calculates near field if the method supports it.
168  // For dipoles, ScaFaCoS calculates near field.
169  int const sr = dipolar() || !has_near;
170  handle_error(fcs_set_common(handle, sr, boxa, boxb, boxc, off, periodicity,
171  total_particles));
172 #ifdef FCS_ENABLE_DIPOLES
173  if (m_dipolar)
174  handle_error(fcs_set_total_dipole_particles(handle, total_particles));
175 #endif
176 }
