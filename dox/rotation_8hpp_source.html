71 /**
72  * @brief Transform matrix from body- to space-fixed frame.
73  *
74  * Given a linear map represented by \f$ A \in \mathbb{R}^{3 \times 3}\f$
75  * in the body-fixed frame, this returns the matrix \f$ A \in \mathbb{R}^{3
76  * \times 3}\f$ representing the map in the space-fixed frame. They are related
77  * by the map between the space-fixed and body-fixed frame \f$O\f$ like
78  *
79  * \f[
80  * A' = O^T A O.
81  * \f]
82  *
83  * @tparam T Scalar type
84  * @param p Particle transforming from.
85  * @param A Matrix to transform
86  * @return Matrix representation in space-fixed coordinates.
87  */
88 template <class T>
89 auto convert_body_to_space(const Particle &p, const Utils::Matrix<T, 3, 3> &A) {
90  auto const O = rotation_matrix(p.r.quat);
91  return O.transposed() * A * O;
92 }

105 /** Rotate the particle p around the body-frame defined NORMALIZED axis
106  * @p aBodyFrame by amount @p phi.
107  */
108 inline Utils::Quaternion<double>
109 local_rotate_particle_body(Particle const &p,
110  const Utils::Vector3d &axis_body_frame,
111  const double phi) {
112  // Rotation turned off entirely?
113  if (!p.p.rotation)
114  return p.r.quat;
115  if (std::abs(phi) > std::numeric_limits<double>::epsilon())
116  return p.r.quat *
117  boost::qvm::rot_quat(mask(p.p.rotation, axis_body_frame), phi);
118  return p.r.quat;
119 }

121 /** Rotate the particle p around the NORMALIZED axis aSpaceFrame by amount phi
122  */
123 inline void local_rotate_particle(Particle &p,
124  const Utils::Vector3d &axis_space_frame,
125  const double phi) {
126  if (std::abs(phi) > std::numeric_limits<double>::epsilon()) {
127  // Convert rotation axis to body-fixed frame
128  Utils::Vector3d axis = convert_vector_space_to_body(p, axis_space_frame);
129  p.r.quat = local_rotate_particle_body(p, axis, phi);
130  }
131 }
