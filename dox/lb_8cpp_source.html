235 void lb_init(const LB_Parameters &lb_parameters) {
236  if (lb_parameters.agrid <= 0.0) {
237  runtimeErrorMsg()
238  << "Lattice Boltzmann agrid not set when initializing fluid";
239  }
240  if (check_runtime_errors(comm_cart))
241  return;
242 
243  /* initialize the local lattice domain */
244  try {
245  lblattice = Lattice(lb_parameters.agrid, 0.5 /*offset*/, 1 /*halo size*/,
246  local_geo.length(), local_geo.my_right(),
247  box_geo.length(), calc_node_pos(comm_cart), node_grid);
248  } catch (const std::runtime_error &e) {
249  runtimeErrorMsg() << e.what();
250  return;
251  }
252 
253  /* allocate memory for data structures */
254  lb_realloc_fluid(lbfluid_a, lbfluid_b, lblattice.halo_grid_volume, lbfluid,
255  lbfluid_post);
256 
257  lb_initialize_fields(lbfields, lbpar, lblattice);
258 
259  /* prepare the halo communication */
260  lb_prepare_communication(update_halo_comm, lblattice);
261 
262  /* initialize derived parameters */
263  lb_reinit_parameters(lbpar);
264 
265  lb_set_equilibrium_populations(lblattice, lbpar);
266 
267 #ifdef LB_BOUNDARIES
268  LBBoundaries::lb_init_boundaries();
269 #endif
270 }

279 void lb_reinit_parameters(LB_Parameters &lb_parameters) {
280  if (lb_parameters.viscosity > 0.0) {
281  /* Eq. (80) @cite dunweg07a. */
282  lb_parameters.gamma_shear = 1. - 2. / (6. * lb_parameters.viscosity + 1.);
283  }
284 
285  if (lb_parameters.bulk_viscosity > 0.0) {
286  /* Eq. (81) @cite dunweg07a. */
287  lb_parameters.gamma_bulk =
288  1. - 2. / (9. * lb_parameters.bulk_viscosity + 1.);
289  }
290 
291  if (lb_parameters.is_TRT) {
292  lb_parameters.gamma_bulk = lb_parameters.gamma_shear;
293  lb_parameters.gamma_even = lb_parameters.gamma_shear;
294  lb_parameters.gamma_odd = -(7.0 * lb_parameters.gamma_even + 1.0) /
295  (lb_parameters.gamma_even + 7.0);
296  // gamma_odd = lb_parameters.gamma_shear; //uncomment for BGK
297  }
298 
299  // lb_parameters.gamma_shear = 0.0; //uncomment for special case of BGK
300  // lb_parameters.gamma_bulk = 0.0;
301  // gamma_odd = 0.0;
302  // gamma_even = 0.0;
303 
304  if (lb_parameters.kT > 0.0) {
305  /* Eq. (51) @cite dunweg07a.
306  * Note that the modes are not normalized as in the paper here! */
307  double mu = lb_parameters.kT / D3Q19::c_sound_sq<double> *
308  lb_parameters.tau * lb_parameters.tau /
309  (lb_parameters.agrid * lb_parameters.agrid);
310 
311  for (int i = 0; i < 4; i++)
312  lb_parameters.phi[i] = 0.0;
313  lb_parameters.phi[4] =
314  sqrt(mu * D3Q19::w_k[4] * (1. - Utils::sqr(lb_parameters.gamma_bulk)));
315  for (int i = 5; i < 10; i++)
316  lb_parameters.phi[i] = sqrt(mu * D3Q19::w_k[i] *
317  (1. - Utils::sqr(lb_parameters.gamma_shear)));
318  for (int i = 10; i < 16; i++)
319  lb_parameters.phi[i] =
320  sqrt(mu * D3Q19::w_k[i] * (1 - Utils::sqr(lb_parameters.gamma_odd)));
321  for (int i = 16; i < 19; i++)
322  lb_parameters.phi[i] =
323  sqrt(mu * D3Q19::w_k[i] * (1 - Utils::sqr(lb_parameters.gamma_even)));
324  } else {
325  for (int i = 0; i < D3Q19::n_vel; i++)
326  lb_parameters.phi[i] = 0.0;
327  }
328 }

747 template <typename T>
748 std::array<T, 19> lb_relax_modes(const std::array<T, 19> &modes,
749  const Utils::Vector<T, 3> &force_density,
750  const LB_Parameters &parameters) {
751  using Utils::sqr;
752  using Utils::Vector;
753 
754  /* re-construct the real density
755  * remember that the populations are stored as differences to their
756  * equilibrium value */
757  auto const density = modes[0] + parameters.density;
758  auto const momentum_density =
759  Vector<T, 3>{modes[1], modes[2], modes[3]} + T{0.5} * force_density;
760  auto const momentum_density2 = momentum_density.norm2();
761 
762  /* equilibrium part of the stress modes */
763  auto const stress_eq =
764  Vector<T, 6>{momentum_density2,
765  (sqr(momentum_density[0]) - sqr(momentum_density[1])),
766  (momentum_density2 - 3.0 * sqr(momentum_density[2])),
767  momentum_density[0] * momentum_density[1],
768  momentum_density[0] * momentum_density[2],
769  momentum_density[1] * momentum_density[2]} /
770  density;
771 
772  return {{modes[0], modes[1], modes[2], modes[3],
773  /* relax the stress modes */
774  stress_eq[0] + parameters.gamma_bulk * (modes[4] - stress_eq[0]),
775  stress_eq[1] + parameters.gamma_shear * (modes[5] - stress_eq[1]),
776  stress_eq[2] + parameters.gamma_shear * (modes[6] - stress_eq[2]),
777  stress_eq[3] + parameters.gamma_shear * (modes[7] - stress_eq[3]),
778  stress_eq[4] + parameters.gamma_shear * (modes[8] - stress_eq[4]),
779  stress_eq[5] + parameters.gamma_shear * (modes[9] - stress_eq[5]),
780  /* relax the ghost modes (project them out) */
781  /* ghost modes have no equilibrium part due to orthogonality */
782  parameters.gamma_odd * modes[10], parameters.gamma_odd * modes[11],
783  parameters.gamma_odd * modes[12], parameters.gamma_odd * modes[13],
784  parameters.gamma_odd * modes[14], parameters.gamma_odd * modes[15],
785  parameters.gamma_even * modes[16], parameters.gamma_even * modes[17],
786  parameters.gamma_even * modes[18]}};
787 }

789 template <typename T>
790 std::array<T, 19> lb_thermalize_modes(
791  Lattice::index_t index, const std::array<T, 19> &modes,
792  const LB_Parameters &lb_parameters,
793  boost::optional<Utils::Counter<uint64_t>> const &rng_counter) {
794  if (lb_parameters.kT > 0.0) {
795  using Utils::uniform;
796  using rng_type = r123::Philox4x64;
797  using ctr_type = rng_type::ctr_type;
798 
799  const ctr_type c{
800  {rng_counter->value(), static_cast<uint64_t>(RNGSalt::FLUID)}};
801  const T rootdensity =
802  std::sqrt(std::fabs(modes[0] + lb_parameters.density));
803  auto const pref = std::sqrt(12.) * rootdensity;
804 
805  const ctr_type noise[4] = {
806  rng_type{}(c, {{static_cast<uint64_t>(index), 0ul}}),
807  rng_type{}(c, {{static_cast<uint64_t>(index), 1ul}}),
808  rng_type{}(c, {{static_cast<uint64_t>(index), 2ul}}),
809  rng_type{}(c, {{static_cast<uint64_t>(index), 3ul}})};
810 
811  auto rng = [&](int i) { return uniform(noise[i / 4][i % 4]) - 0.5; };
812 
813  return {/* conserved modes */
814  {modes[0], modes[1], modes[2], modes[3],
815  /* stress modes */
816  modes[4] + pref * lb_parameters.phi[4] * rng(0),
817  modes[5] + pref * lb_parameters.phi[5] * rng(1),
818  modes[6] + pref * lb_parameters.phi[6] * rng(2),
819  modes[7] + pref * lb_parameters.phi[7] * rng(3),
820  modes[8] + pref * lb_parameters.phi[8] * rng(4),
821  modes[9] + pref * lb_parameters.phi[9] * rng(5),
822 
823  /* ghost modes */
824  modes[10] + pref * lb_parameters.phi[10] * rng(6),
825  modes[11] + pref * lb_parameters.phi[11] * rng(7),
826  modes[12] + pref * lb_parameters.phi[12] * rng(8),
827  modes[13] + pref * lb_parameters.phi[13] * rng(9),
828  modes[14] + pref * lb_parameters.phi[14] * rng(10),
829  modes[15] + pref * lb_parameters.phi[15] * rng(11),
830  modes[16] + pref * lb_parameters.phi[16] * rng(12),
831  modes[17] + pref * lb_parameters.phi[17] * rng(13),
832  modes[18] + pref * lb_parameters.phi[18] * rng(14)}};
833  }
834  return modes;
835 }

837 template <typename T>
838 std::array<T, 19> lb_apply_forces(const std::array<T, 19> &modes,
839  const LB_Parameters &lb_parameters,
840  Utils::Vector<T, 3> const &f) {
841  auto const density = modes[0] + lb_parameters.density;
842 
843  /* hydrodynamic momentum density is redefined when external forces present */
844  auto const u =
845  Utils::Vector3d{modes[1], modes[2], modes[3]} + T{0.5} * f / density;
846 
847  auto const C = std::array<T, 6>{
848  (1. + lb_parameters.gamma_shear) * u[0] * f[0] +
849  1. / 3. * (lb_parameters.gamma_bulk - lb_parameters.gamma_shear) *
850  (u * f),
851  1. / 2. * (1. + lb_parameters.gamma_shear) * (u[0] * f[1] + u[1] * f[0]),
852  (1. + lb_parameters.gamma_shear) * u[1] * f[1] +
853  1. / 3. * (lb_parameters.gamma_bulk - lb_parameters.gamma_shear) *
854  (u * f),
855  1. / 2. * (1. + lb_parameters.gamma_shear) * (u[0] * f[2] + u[2] * f[0]),
856  1. / 2. * (1. + lb_parameters.gamma_shear) * (u[1] * f[2] + u[2] * f[1]),
857  (1. + lb_parameters.gamma_shear) * u[2] * f[2] +
858  1. / 3. * (lb_parameters.gamma_bulk - lb_parameters.gamma_shear) *
859  (u * f)};
860 
861  return {{modes[0],
862  /* update momentum modes */
863  modes[1] + f[0], modes[2] + f[1], modes[3] + f[2],
864  /* update stress modes */
865  modes[4] + C[0] + C[2] + C[5], modes[5] + C[0] - C[2],
866  modes[6] + C[0] + C[2] - 2. * C[5], modes[7] + C[1], modes[8] + C[3],
867  modes[9] + C[4], modes[10], modes[11], modes[12], modes[13],
868  modes[14], modes[15], modes[16], modes[17], modes[18]}};
869 }

900 /* Collisions and streaming (push scheme) */
901 void lb_integrate() {
902  ESPRESSO_PROFILER_CXX_MARK_FUNCTION;
903  /* loop over all lattice cells (halo excluded) */
904 #ifdef LB_BOUNDARIES
905  for (auto &lbboundary : LBBoundaries::lbboundaries) {
906  (*lbboundary).reset_force();
907  }
908 #endif // LB_BOUNDARIES
909 
910  auto const next_offsets = lb_next_offsets(lblattice, D3Q19::c);
911 
912  Lattice::index_t index = lblattice.halo_offset;
913  for (int z = 1; z <= lblattice.grid[2]; z++) {
914  for (int y = 1; y <= lblattice.grid[1]; y++) {
915  for (int x = 1; x <= lblattice.grid[0]; x++) {
916  // as we only want to apply this to non-boundary nodes we can throw out
917  // the if-clause if we have a non-bounded domain
918 #ifdef LB_BOUNDARIES
919  if (!lbfields[index].boundary)
920 #endif // LB_BOUNDARIES
921  {
922  /* calculate modes locally */
923  auto const modes = lb_calc_modes(index, lbfluid);
924 
925  /* deterministic collisions */
926  auto const relaxed_modes =
927  lb_relax_modes(modes, lbfields[index].force_density, lbpar);
928 
929  /* fluctuating hydrodynamics */
930  auto const thermalized_modes = lb_thermalize_modes(
931  index, relaxed_modes, lbpar, rng_counter_fluid);
932 
933  /* apply forces */
934  auto const modes_with_forces = lb_apply_forces(
935  thermalized_modes, lbpar, lbfields[index].force_density);
936 
937 #ifdef VIRTUAL_SITES_INERTIALESS_TRACERS
938  // Safeguard the node forces so that we can later use them for the IBM
939  // particle update
940  lbfields[index].force_density_buf = lbfields[index].force_density;
941 #endif
942 
943  /* reset the force density */
944  lbfields[index].force_density = lbpar.ext_force_density;
945 
946  /* transform back to populations and streaming */
947  auto const populations = lb_calc_n_from_m(modes_with_forces);
948  lb_stream(lbfluid_post, populations, index, next_offsets);
949  }
950 
951  ++index; /* next node */
952  }
953  index += 2; /* skip halo region */
954  }
955  index += 2 * lblattice.halo_grid[0]; /* skip halo region */
956  }
957 
958  /* exchange halo regions */
959  halo_push_communication(lbfluid_post, lblattice);
960 
961 #ifdef LB_BOUNDARIES
962  /* boundary conditions for links */
963  lb_bounce_back(lbfluid_post, lbpar, lbfields);
964 #endif // LB_BOUNDARIES
965 
966  /* swap the pointers for old and new population fields */
967  std::swap(lbfluid, lbfluid_post);
968 
969  halo_communication(update_halo_comm,
970  reinterpret_cast<char *>(lbfluid[0].data()));
971 
972 #ifdef ADDITIONAL_CHECKS
973  lb_check_halo_regions(lbfluid, lblattice);
974 #endif
975 }

1187 Utils::Vector6d lb_calc_pressure_tensor(std::array<double, 19> const &modes,
1188  Utils::Vector3d const &force_density,
1189  const LB_Parameters &lb_parameters) {
1190  auto const momentum_density = lb_calc_momentum_density(modes, force_density);
1191  auto const density = lb_calc_density(modes, lb_parameters);
1192  using Utils::sqr;
1193  auto const momentum_density2 = sqr(momentum_density[0]) +
1194  sqr(momentum_density[1]) +
1195  sqr(momentum_density[2]);
1196  /* equilibrium part of the stress modes */
1197  Utils::Vector6d modes_from_stress_eq{};
1198  modes_from_stress_eq[0] = momentum_density2 / density;
1199  modes_from_stress_eq[1] =
1200  (sqr(momentum_density[0]) - sqr(momentum_density[1])) / density;
1201  modes_from_stress_eq[2] =
1202  (momentum_density2 - 3.0 * sqr(momentum_density[2])) / density;
1203  modes_from_stress_eq[3] = momentum_density[0] * momentum_density[1] / density;
1204  modes_from_stress_eq[4] = momentum_density[0] * momentum_density[2] / density;
1205  modes_from_stress_eq[5] = momentum_density[1] * momentum_density[2] / density;
1206 
1207  /* Now we must predict the outcome of the next collision */
1208  /* We immediately average pre- and post-collision. */
1209 
1210  Utils::Vector6d avg_modes;
1211  avg_modes[0] =
1212  modes_from_stress_eq[0] + (0.5 + 0.5 * lb_parameters.gamma_bulk) *
1213  (modes[4] - modes_from_stress_eq[0]);
1214  avg_modes[1] =
1215  modes_from_stress_eq[1] + (0.5 + 0.5 * lb_parameters.gamma_shear) *
1216  (modes[5] - modes_from_stress_eq[1]);
1217  avg_modes[2] =
1218  modes_from_stress_eq[2] + (0.5 + 0.5 * lb_parameters.gamma_shear) *
1219  (modes[6] - modes_from_stress_eq[2]);
1220  avg_modes[3] =
1221  modes_from_stress_eq[3] + (0.5 + 0.5 * lb_parameters.gamma_shear) *
1222  (modes[7] - modes_from_stress_eq[3]);
1223  avg_modes[4] =
1224  modes_from_stress_eq[4] + (0.5 + 0.5 * lb_parameters.gamma_shear) *
1225  (modes[8] - modes_from_stress_eq[4]);
1226  avg_modes[5] =
1227  modes_from_stress_eq[5] + (0.5 + 0.5 * lb_parameters.gamma_shear) *
1228  (modes[9] - modes_from_stress_eq[5]);
1229 
1230  // Transform the stress tensor components according to the modes that
1231  // correspond to those used by U. Schiller. In terms of populations this
1232  // expression then corresponds exactly to those in eq. (116)-(121) in
1233  // @cite dunweg07a, when these are written out in populations.
1234  // But to ensure this, the expression in Schiller's modes has to be different!
1235 
1236  Utils::Vector6d stress;
1237  stress[0] =
1238  (2.0 * (modes[0] + avg_modes[0]) + avg_modes[2] + 3.0 * avg_modes[1]) /
1239  6.0; // xx
1240  stress[1] = avg_modes[3]; // xy
1241  stress[2] =
1242  (2.0 * (modes[0] + avg_modes[0]) + avg_modes[2] - 3.0 * avg_modes[1]) /
1243  6.0; // yy
1244  stress[3] = avg_modes[4]; // xz
1245  stress[4] = avg_modes[5]; // yz
1246  stress[5] = (modes[0] + avg_modes[0] - avg_modes[2]) / 3.0; // zz
1247  return stress;
1248 }
