42 double time_force_calc(int int_steps) {
43  Utils::Statistics::RunningAverage<double> running_average;
44 
45  if (mpi_integrate(0, 0))
46  return -1;
47 
48  /* perform force calculation test */
49  for (int i = 0; i < int_steps; i++) {
50  const double tick = MPI_Wtime();
51 
52  if (mpi_integrate(0, -1))
53  return -1;
54 
55  const double tock = MPI_Wtime();
56  running_average.add_sample((tock - tick));
57  }
58 
59  if (running_average.avg() <= 5 * MPI_Wtick()) {
60  runtimeWarningMsg()
61  << "Clock resolution is too low to reliably time integration.";
62  }
63 
64  if (running_average.sig() >= 0.1 * running_average.avg()) {
65  runtimeWarningMsg() << "Statistics of tuning samples is very bad.";
66  }
67 
68  /* MPI returns s, return value should be in ms. */
69  return 1000. * running_average.avg();
70 }

72 /**

77  * @param int_steps Number of steps to integrate.

80 static double time_calc(int int_steps) {

84  /* perform force calculation test */

86  if (mpi_integrate(int_steps, -1))
87  return -1;
88  const double tock = MPI_Wtime();
89 
90  /* MPI returns s, return value should be in ms. */

94 void tune_skin(double min_skin, double max_skin, double tol, int int_steps,
95  bool adjust_max_skin) {
96 
97  double a = min_skin;
98  double b = max_skin;
99  double time_a, time_b;
100 
101  /* The maximal skin is the remainder from the required cutoff to
102  * the maximal range that can be supported by the cell system, but
103  * never larger than half the box size. */
104  double const max_permissible_skin = std::min(
105  *boost::min_element(cell_structure.max_cutoff()) - maximal_cutoff(),
106  0.5 * *boost::max_element(box_geo.length()));
107 
108  if (adjust_max_skin and max_skin > max_permissible_skin)
109  b = max_permissible_skin;
110 
111  while (fabs(a - b) > tol) {
112  mpi_set_skin(a);
113  time_a = time_calc(int_steps);
114 
115  mpi_set_skin(b);
116  time_b = time_calc(int_steps);
117 
118  if (time_a > time_b) {
119  a = 0.5 * (a + b);
120  } else {
121  b = 0.5 * (a + b);
122  }
123  }
124  auto const new_skin = 0.5 * (a + b);
125  mpi_set_skin(new_skin);
126 }
