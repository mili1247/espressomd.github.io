95 int cuda_test_device_access() {
96  int *d = nullptr;
97  int h = 42;
98  cudaError_t err;
99 
100  err = cudaMalloc((void **)&d, sizeof(int));
101  if (err != cudaSuccess) {
102  throw cuda_runtime_error_cuda(err);
103  }
104  err = cudaMemcpy(d, &h, sizeof(int), cudaMemcpyHostToDevice);
105  if (err != cudaSuccess) {
106  cudaFree(d);
107  throw cuda_runtime_error_cuda(err);
108  }
109  h = 0;
110  err = cudaMemcpy(&h, d, sizeof(int), cudaMemcpyDeviceToHost);
111  cudaFree(d);
112  if (err != cudaSuccess) {
113  throw cuda_runtime_error_cuda(err);
114  }
115  if (h != 42) {
116  return ES_ERROR;
117  }
118  return ES_OK;
119 }
