33 /**
34  * @brief Bag of elements.
35  *
36  * A bag is a container in which the elements do not have
37  * a fixed order. It can be considered an unordered variant
38  * of vector, and implements the Container named requirement
39  * specified in C++11.
40  *
41  * Elements in the container do not have a stable position and
42  * removing elements can change the order of the other elements.
43  * The loser contract (compared to a vector) allows removing any
44  * element in the container in constant time.
45  *
46  * @tparam T Element Type, needs to be Swappable.
47  */
48 template <class T> class Bag {
49  /** Storage backend */
50  using storage_type = std::vector<T>;
51 
52 public:
53  using value_type = T;
54  using iterator = T *;
55  using const_iterator = const T *;
56  using pointer = T *;
57  using reference = T &;
58 
59  /**
60  * @brief Construct an empty container.
61  */
62  Bag() = default;
63 
64 private:
65  /** Underlying storage of the container */
66  std::vector<T> m_storage;
67 
68  friend boost::serialization::access;
69  /**
70  * @brief Serialize the container.
71  *
72  * Serialization requires T to be serializable.
73  */
74  template <class Archive> void serialize(Archive &ar, long int /* version */) {
75  ar &m_storage;
76  }

133  /**
134  * @brief Insert an element into the container.
135  *
136  * If before the call size() >= capacity(),
137  * this may reallocate, in which case all
138  * iterators into the container are invalidated.
139  * Otherwise only the end iterator is invalidated.
140  *
141  * @param v Element to add.
142  * @return Reference to the added element.
143  */
144  T &insert(T const &v) {
145  m_storage.push_back(v);
146 
147  return m_storage.back();
148  }

156  /**

160  * @return An iterator past the element that was removed.
161  */
162  iterator erase(iterator it) {
163  *it = std::move(m_storage.back());
164 
165  m_storage.pop_back();
166 
167  return it;
168  }

175  */
176  friend void swap(Bag &lhs, Bag &rhs) {

178
