53 template <typename Op>
54 void lattice_interpolation(Lattice const &lattice, Utils::Vector3d const &pos,
55  Op &&op) {
56  Utils::Vector<std::size_t, 8> node_index{};
57  Utils::Vector6d delta{};
58 
59  /* determine elementary lattice cell surrounding the particle
60  and the relative position of the particle in this cell */
61  lattice.map_position_to_lattice(pos, node_index, delta);
62  for (int z = 0; z < 2; z++) {
63  for (int y = 0; y < 2; y++) {
64  for (int x = 0; x < 2; x++) {
65  auto &index = node_index[(z * 2 + y) * 2 + x];
66  auto const w = delta[3 * x + 0] * delta[3 * y + 1] * delta[3 * z + 2];
67 
68  op(index, w);
69  }
70  }
71  }
72 }

74 Utils::Vector3d node_u(Lattice::index_t index) {
75 #ifdef LB_BOUNDARIES
76  if (lbfields[index].boundary) {
77  return lbfields[index].slip_velocity;
78  }
79 #endif // LB_BOUNDARIES
80  auto const modes = lb_calc_modes(index, lbfluid);
81  auto const local_density = lbpar.density + modes[0];
82  return Utils::Vector3d{modes[1], modes[2], modes[3]} / local_density;
83 }

85 double node_dens(Lattice::index_t index) {
86 #ifdef LB_BOUNDARIES
87  if (lbfields[index].boundary) {
88  return lbpar.density;
89  }
90 #endif // LB_BOUNDARIES
91  auto const modes = lb_calc_modes(index, lbfluid);
92  return lbpar.density + modes[0];
93 }

97 const Utils::Vector3d
98 lb_lbinterpolation_get_interpolated_velocity(const Utils::Vector3d &pos) {
99  Utils::Vector3d interpolated_u{};
100 
101  /* Calculate fluid velocity at particle's position.
102  This is done by linear interpolation (eq. (11) @cite ahlrichs99a) */
103  lattice_interpolation(lblattice, pos,
104  [&interpolated_u](Lattice::index_t index, double w) {
105  interpolated_u += w * node_u(index);
106  });
107 
108  return interpolated_u;
109 }

124 void lb_lbinterpolation_add_force_density(
125  const Utils::Vector3d &pos, const Utils::Vector3d &force_density) {
126  switch (interpolation_order) {
127  case (InterpolationOrder::quadratic):
128  throw std::runtime_error("The non-linear interpolation scheme is not "
129  "implemented for the CPU LB.");
130  case (InterpolationOrder::linear):
131  lattice_interpolation(lblattice, pos,
132  [&force_density](Lattice::index_t index, double w) {
133  auto &field = lbfields[index];
134  field.force_density += w * force_density;
135  });
136  break;
137  }
