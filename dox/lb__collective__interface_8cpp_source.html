45 template <typename Kernel>
46 auto lb_calc(Utils::Vector3i const &index, Kernel kernel) {
47  using R = decltype(kernel(index));
48  if (lblattice.is_local(index)) {
49  return boost::optional<R>(kernel(index));

51  return boost::optional<R>();

63 template <class Kernel>
64 auto lb_calc_fluid_kernel(Utils::Vector3i const &index, Kernel kernel) {
65  return lb_calc(index, [&](auto index) {
66  auto const linear_index =
67  get_linear_index(lblattice.local_index(index), lblattice.halo_grid);
68  auto const force_density = lbfields[linear_index].force_density;
69  auto const modes = lb_calc_modes(linear_index, lbfluid);
70  return kernel(modes, force_density);
71  });
72 }
