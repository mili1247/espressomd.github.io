80 static double far_error(int P, double minrad) {
81  auto const wavenumber = 2 * Utils::pi() * box_geo.length_inv()[2];
82  // this uses an upper bound to all force components and the potential
83  auto const rhores = wavenumber * minrad;
84  auto const pref = 4 * box_geo.length_inv()[2] * std::max(1.0, wavenumber);
85 
86  return pref * K1(rhores * P) * exp(rhores) / rhores * (P - 1 + 1 / rhores);
87 }

89 static double determine_minrad(double maxPWerror, int P) {
90  // bisection to search for where the error is maxPWerror
91  constexpr auto min_rad = 0.01;
92  double const rgranularity = min_rad * box_geo.length()[2];
93  double rmin = rgranularity;
94  double rmax = std::min(box_geo.length()[0], box_geo.length()[1]);
95  double const errmin = far_error(P, rmin);
96  double const errmax = far_error(P, rmax);
97  if (errmin < maxPWerror) {
98  // we can do almost all radii with this P
99  return rmin;
100  }
101  if (errmax > maxPWerror) {
102  // make sure that this switching radius cannot be reached
103  return 2 * std::max(box_geo.length()[0], box_geo.length()[1]);
104  }
105 
106  while (rmax - rmin > rgranularity) {
107  auto const c = 0.5 * (rmin + rmax);
108  auto const errc = far_error(P, c);
109  if (errc > maxPWerror) {
110  rmin = c;
111  } else {
112  rmax = c;
113  }
114  }
115  return 0.5 * (rmin + rmax);
116 }

118 static void determine_bessel_radii(double maxPWerror) {
119  for (int i = 0; i < MAXIMAL_B_CUT; ++i) {
120  bessel_radii[i] = determine_minrad(maxPWerror, i + 1);

124 static void prepare_polygamma_series(double maxPWerror, double maxrad2) {
125  /* polygamma, determine order */
126  double err;
127  auto const rhomax2 = uz2 * maxrad2;
128  /* rhomax2 < 1, so rhomax2m2 falls monotonously */
129  int n = 1;
130  auto rhomax2nm2 = 1.0;
131  do {
132  create_mod_psi_up_to(n + 1);
133 
134  /* |uz*z| <= 0.5 */
135  err = 2 * n * fabs(mod_psi_even(n, 0.5)) * rhomax2nm2;
136  rhomax2nm2 *= rhomax2;
137  n++;
138  } while (err > 0.1 * maxPWerror);
139 }

151  if (box_geo.periodic(0) || box_geo.periodic(1) || !box_geo.periodic(2)) {
152  runtimeErrorMsg() << "MMM1D requires periodicity (0, 0, 1)";

163  if (MMM1D_sanity_checks())
164  return ES_ERROR;

166  if (mmm1d_params.far_switch_radius_2 >= Utils::sqr(box_geo.length()[2]))
167  mmm1d_params.far_switch_radius_2 = 0.8 * Utils::sqr(box_geo.length()[2]);
168 
169  uz2 = Utils::sqr(box_geo.length_inv()[2]);
170  prefuz2 = coulomb.prefactor * uz2;
171  prefL3_i = prefuz2 * box_geo.length_inv()[2];
172 
173  determine_bessel_radii(mmm1d_params.maxPWerror);
174  prepare_polygamma_series(mmm1d_params.maxPWerror,
175  mmm1d_params.far_switch_radius_2);
176  return ES_OK;

179 void add_mmm1d_coulomb_pair_force(double chpref, Utils::Vector3d const &d,
180  double r, Utils::Vector3d &force) {
181  constexpr auto c_2pi = 2 * Utils::pi();
182  auto const n_modPsi = static_cast<int>(modPsi.size() >> 1);
183  auto const rxy2 = d[0] * d[0] + d[1] * d[1];
184  auto const rxy2_d = rxy2 * uz2;
185  auto const z_d = d[2] * box_geo.length_inv()[2];
186  Utils::Vector3d F;
187 
188  if (rxy2 <= mmm1d_params.far_switch_radius_2) {
189  /* polygamma summation */
190  auto sr = 0.;
191  auto sz = mod_psi_odd(0, z_d);
192  auto r2nm1 = 1.0;
193  for (int n = 1; n < n_modPsi; n++) {
194  auto const deriv = static_cast<double>(2 * n);
195  auto const mpe = mod_psi_even(n, z_d);
196  auto const mpo = mod_psi_odd(n, z_d);
197  auto const r2n = r2nm1 * rxy2_d;
198 
199  sz += r2n * mpo;
200  sr += deriv * r2nm1 * mpe;
201 
202  if (fabs(deriv * r2nm1 * mpe) < mmm1d_params.maxPWerror)
203  break;
204 
205  r2nm1 = r2n;
206  }
207 
208  double Fx = prefL3_i * sr * d[0];
209  double Fy = prefL3_i * sr * d[1];
210  double Fz = prefuz2 * sz;
211 
212  /* real space parts */
213 
214  double pref, rt, rt2, shift_z;
215 
216  pref = 1. / (r * r * r);
217  Fx += pref * d[0];
218  Fy += pref * d[1];
219  Fz += pref * d[2];
220 
221  shift_z = d[2] + box_geo.length()[2];
222  rt2 = rxy2 + shift_z * shift_z;
223  rt = sqrt(rt2);
224  pref = 1. / (rt2 * rt);
225  Fx += pref * d[0];
226  Fy += pref * d[1];
227  Fz += pref * shift_z;
228 
229  shift_z = d[2] - box_geo.length()[2];
230  rt2 = rxy2 + shift_z * shift_z;
231  rt = sqrt(rt2);
232  pref = 1. / (rt2 * rt);
233  Fx += pref * d[0];
234  Fy += pref * d[1];
235  Fz += pref * shift_z;
236 
237  F = {Fx, Fy, Fz};
238  } else {
239  /* far range formula */
240  auto const rxy = sqrt(rxy2);
241  auto const rxy_d = rxy * box_geo.length_inv()[2];
242  auto sr = 0., sz = 0.;
243 
244  for (int bp = 1; bp < MAXIMAL_B_CUT; bp++) {
245  if (bessel_radii[bp - 1] < rxy)
246  break;
247 
248  auto const fq = c_2pi * bp;
249  double k0, k1;
250 #ifdef MMM1D_MACHINE_PREC
251  k0 = K0(fq * rxy_d);
252  k1 = K1(fq * rxy_d);
253 #else
254  std::tie(k0, k1) = LPK01(fq * rxy_d);
255 #endif
256  sr += bp * k1 * cos(fq * z_d);
257  sz += bp * k0 * sin(fq * z_d);
258  }
259  sr *= uz2 * 4 * c_2pi;
260  sz *= uz2 * 4 * c_2pi;
261 
262  auto const pref = sr / rxy + 2 * box_geo.length_inv()[2] / rxy2;
263 
264  F = {pref * d[0], pref * d[1], sz};
265  }
266 
267  force += chpref * F;
268 }

270 double mmm1d_coulomb_pair_energy(double const chpref, Utils::Vector3d const &d,
271  double r2, double r) {
272  if (chpref == 0)
273  return 0;
274 
275  constexpr auto c_2pi = 2 * Utils::pi();
276  auto const n_modPsi = static_cast<int>(modPsi.size() >> 1);
277  auto const rxy2 = d[0] * d[0] + d[1] * d[1];
278  auto const rxy2_d = rxy2 * uz2;
279  auto const z_d = d[2] * box_geo.length_inv()[2];
280  double E;
281 
282  if (rxy2 <= mmm1d_params.far_switch_radius_2) {
283  /* near range formula */
284  E = -2 * Utils::gamma();
285 
286  /* polygamma summation */
287  double r2n = 1.0;
288  for (int n = 0; n < n_modPsi; n++) {
289  auto const add = mod_psi_even(n, z_d) * r2n;
290  E -= add;
291 
292  if (fabs(add) < mmm1d_params.maxPWerror)
293  break;
294 
295  r2n *= rxy2_d;
296  }
297  E *= box_geo.length_inv()[2];
298 
299  /* real space parts */
300 
301  double rt, shift_z;
302 
303  E += 1. / r;
304 
305  shift_z = d[2] + box_geo.length()[2];
306  rt = sqrt(rxy2 + shift_z * shift_z);
307  E += 1. / rt;
308 
309  shift_z = d[2] - box_geo.length()[2];
310  rt = sqrt(rxy2 + shift_z * shift_z);
311  E += 1. / rt;
312  } else {
313  /* far range formula */
314  auto const rxy = sqrt(rxy2);
315  auto const rxy_d = rxy * box_geo.length_inv()[2];
316  /* The first Bessel term will compensate a little bit the
317  log term, so add them close together */
318  E = -0.25 * log(rxy2_d) + 0.5 * (Utils::ln_2() - Utils::gamma());
319  for (int bp = 1; bp < MAXIMAL_B_CUT; bp++) {
320  if (bessel_radii[bp - 1] < rxy)
321  break;
322 
323  auto const fq = c_2pi * bp;
324  E += K0(fq * rxy_d) * cos(fq * z_d);
325  }
326  E *= 4 * box_geo.length_inv()[2];
327  }
328 
329  return chpref * E;
330 }

332 int mmm1d_tune(int timings, bool verbose) {
333  if (MMM1D_sanity_checks())
334  return ES_ERROR;
335 
336  if (mmm1d_params.far_switch_radius_2 < 0) {
337  auto const maxrad = box_geo.length()[2];
338  auto min_time = std::numeric_limits<double>::infinity();
339  auto min_rad = -1.;
340  double switch_radius;
341  /* determine besselcutoff and optimal switching radius. Should be around
342  * 0.33 */
343  for (switch_radius = 0.2 * maxrad; switch_radius < 0.4 * maxrad;
344  switch_radius += 0.025 * maxrad) {
345  if (switch_radius <= bessel_radii.back()) {
346  // this switching radius is too small for our Bessel series
347  continue;
348  }
349 
350  mmm1d_params.far_switch_radius_2 = Utils::sqr(switch_radius);
351 
352  coulomb.method = COULOMB_MMM1D;
353 
354  /* initialize mmm1d temporary structures */
355  mpi_bcast_coulomb_params();
356 
357  /* perform force calculation test */
358  double int_time = time_force_calc(timings);
359 
360  /* exit on errors */
361  if (int_time < 0)
362  return ES_ERROR;
363 
364  if (verbose) {
365  std::printf("r= %f t= %f ms\n", switch_radius, int_time);
366  }
367 
368  if (int_time < min_time) {
369  min_time = int_time;
370  min_rad = switch_radius;
371  }
372  /* stop if all hope is vain... */
373  else if (int_time > 2 * min_time)
374  break;
375  }
376  switch_radius = min_rad;
377  mmm1d_params.far_switch_radius_2 = Utils::sqr(switch_radius);
378  } else if (mmm1d_params.far_switch_radius_2 <=
379  Utils::sqr(bessel_radii.back())) {
380  // this switching radius is too small for our Bessel series
381  runtimeErrorMsg() << "could not find reasonable bessel cutoff";
382  return ES_ERROR;
383  }
384 
385  coulomb.method = COULOMB_MMM1D;
386 
387  mpi_bcast_coulomb_params();
388 
389  return ES_OK;
390 }
