73 void ClusterStructure::add_pair(const Particle &p1, const Particle &p2) {
74  // * check, if there's a neighbor
75  // * No: Then go on to the next particle
76  // * Yes: Then if
77  // * One of them belongs to a cluster, give the other one the same cluster
78  // id.
79  // * None of them belongs to a cluster: Give them both a new cluster id
80  // * Both belong to different clusters: Mark the clusters as identical
81  // * so that they can be put together later
82  if (!m_pair_criterion) {
83  runtimeErrorMsg() << "No cluster criterion defined";
84  return;
85  }
86  // If the two particles are neighbors...
87  if (m_pair_criterion->decide(p1, p2)) {
88 
89  if // None belongs to a cluster
90  ((!part_of_cluster(p1)) && (!part_of_cluster(p2))) {
91  // Both particles belong to the same, new cluster
92  const int cid = get_next_free_cluster_id();
93 
94  // assign the cluster_ids
95  cluster_id[p1.p.identity] = cid;
96  cluster_id[p2.p.identity] = cid;
97  } else if // p2 belongs to a cluster but p1 doesn't
98  (part_of_cluster(p2) && !part_of_cluster(p1)) {
99  // Give p1 the same cluster id as p2
100  cluster_id[p1.p.identity] = find_id_for(cluster_id.at(p2.p.identity));
101  } else if // i belongs to a cluster but j doesn't
102  (part_of_cluster(p1) && !part_of_cluster(p2)) {
103  // give p2 the cluster id from p1
104  cluster_id[p2.p.identity] = find_id_for(cluster_id.at(p1.p.identity));
105  } else if // Both belong to different clusters
106  (part_of_cluster(p1) && part_of_cluster(p2) &&
107  cluster_id.at(p1.p.identity) != cluster_id.at(p2.p.identity)) {
108  // Clusters of p1 and p2 are one and the same. Add an identity to the list
109  // The higher number must be inserted as first value of the pair
110  // because the substitutions later have to be done in descending order
111  const int cid1 = find_id_for(cluster_id.at(p1.p.identity));
112  const int cid2 = find_id_for(cluster_id.at(p2.p.identity));
113  if (cid1 > cid2) {
114  m_cluster_identities[cid1] = cid2;
115  } else if (cid1 < cid2) {
116  m_cluster_identities[cid2] = cid1;
117  }
118  // else do nothing. The clusters are already noted for merging.
119  // Connected clusters will be merged later
120  }
121  // The case for both particles being in the same cluster does not need to be
122  // treated.
123  }
124 }

126 void ClusterStructure::merge_clusters() {
127  // Relabel particles according to the cluster identities map
128  // Also create empty cluster objects for the final cluster id
129 
130  // Collect needed changes in a separate map, as doing the changes on the fly
131  // would screw up the iterators
132  std::vector<std::pair<int, int>> to_be_changed;
133 
134  for (auto it : cluster_id) {
135  // particle id is in it.first and cluster id in it.second
136  // We change the cluster id according to the cluster identities
137  // map
138  const int cid = find_id_for(it.second);
139  // We note the list of changes here, so we don't modify the map
140  // while iterating
141  to_be_changed.emplace_back(it.first, cid);
142  // Empty cluster object
143  if (clusters.find(cid) == clusters.end()) {
144  clusters[cid] = std::make_shared<Cluster>();
145  }
146  }
147 
148  // Now act on the changes marke in above iteration
149  for (auto it : to_be_changed) {
150  cluster_id[it.first] = it.second;
151  }
152 
153  // Now fill the cluster objects with particle ids
154  // Iterate over particles, fill in the cluster map
155  // to each cluster particle the corresponding cluster id
156  for (auto it : cluster_id) {
157  // If this is the first particle in this cluster, instance a new cluster
158  // object
159  if (clusters.find(it.second) == clusters.end()) {
160  clusters[it.second] = std::make_shared<Cluster>();
161  }
162  clusters[it.second]->particles.push_back(it.first);
163  }
164 
165  // Sort particles ids in the clusters
166  for (const auto &c : clusters) {
167  std::sort(c.second->particles.begin(), c.second->particles.end());
168  }
169 }

171 int ClusterStructure::find_id_for(int x) {
172  int tmp = x;
173  while (m_cluster_identities.find(tmp) != m_cluster_identities.end()) {
174  tmp = m_cluster_identities[tmp];
175  }
176  return tmp;
177 }

179 int ClusterStructure::get_next_free_cluster_id() {
180  // iterate over cluster_id'
181  int max_seen_cluster = 0;
182  for (auto it : cluster_id) {
183  int cid = it.second;
184  if (max_seen_cluster < cid) {
185  max_seen_cluster = cid;
186  }
187  }
188  return max_seen_cluster + 1;
