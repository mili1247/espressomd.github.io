51 /** Calculate Gay-Berne force and torques */
52 inline ParticleForce gb_pair_force(Utils::Vector3d const &ui,
53  Utils::Vector3d const &uj,
54  IA_parameters const &ia_params,
55  Utils::Vector3d const &d, double dist) {
56  using Utils::int_pow;
57  using Utils::sqr;
58 
59  if (dist >= ia_params.gay_berne.cut) {
60  return {};
61  }
62 
63  auto const e0 = ia_params.gay_berne.eps;
64  auto const s0 = ia_params.gay_berne.sig;
65  auto const chi1 = ia_params.gay_berne.chi1;
66  auto const chi2 = ia_params.gay_berne.chi2;
67  auto const mu = ia_params.gay_berne.mu;
68  auto const nu = ia_params.gay_berne.nu;
69  auto const r = Utils::Vector3d(d).normalize();
70  auto const dui = d * ui;
71  auto const duj = d * uj;
72  auto const rui = r * ui;
73  auto const ruj = r * uj;
74  auto const uij = ui * uj;
75  auto const oo1 = (dui + duj) / (1 + chi1 * uij);
76  auto const oo2 = (dui - duj) / (1 - chi1 * uij);
77  auto const tt1 = (dui + duj) / (1 + chi2 * uij);
78  auto const tt2 = (dui - duj) / (1 - chi2 * uij);
79  auto const o1 = sqr(rui + ruj) / (1 + chi1 * uij);
80  auto const o2 = sqr(rui - ruj) / (1 - chi1 * uij);
81  auto const t1 = sqr(rui + ruj) / (1 + chi2 * uij);
82  auto const t2 = sqr(rui - ruj) / (1 - chi2 * uij);
83  auto const Brhi1 = chi1 * (o1 + o2);
84  auto const Brhi2 = chi2 * (t1 + t2);
85 
86  auto const e1 = 1. / (1. - sqr(chi1 * uij));
87  auto const e2 = 1. - 0.5 * Brhi2;
88  auto const e = 4 * e0 * pow(e1, 0.5 * nu) * pow(e2, mu);
89 
90  auto const s1 = 1. / sqrt(1. - 0.5 * Brhi1);
91  auto const s = s0 * s1;
92  auto Koef1 = mu / e2;
93  auto Koef2 = int_pow<3>(s1) * 0.5;
94 
95  auto const X = s0 / (dist - s + s0);
96  auto const Xcut = s0 / (ia_params.gay_berne.cut - s + s0);
97 
98  auto const X6 = int_pow<6>(X);
99  auto const Xcut6 = int_pow<6>(Xcut);
100 
101  auto const Bra12 = 6 * X6 * X * (2 * X6 - 1);
102  auto const Bra12Cut = 6 * Xcut6 * Xcut * (2 * Xcut6 - 1);
103  auto const Brack = X6 * (X6 - 1);
104  auto const BrackCut = Xcut6 * (Xcut6 - 1);
105 
106  /*-------- Here we calculate derivatives -----------------------------*/
107 
108  auto const dU_dr = e *
109  (Koef1 * Brhi2 * (Brack - BrackCut) -
110  Koef2 * Brhi1 * (Bra12 - Bra12Cut) - Bra12 * dist / s0) /
111  sqr(dist);
112 
113  Koef1 *= chi2 / sqr(dist);
114  Koef2 *= chi1 / sqr(dist);
115 
116  auto const dU_da = e * (Koef1 * (tt1 + tt2) * (BrackCut - Brack) +
117  Koef2 * (oo1 + oo2) * (Bra12 - Bra12Cut));
118  auto const dU_db = e * (Koef1 * (tt2 - tt1) * (Brack - BrackCut) +
119  Koef2 * (oo1 - oo2) * (Bra12 - Bra12Cut));
120  auto const dU_dc =
121  e * ((Brack - BrackCut) * (nu * e1 * sqr(chi1) * uij +
122  0.5 * Koef1 * chi2 * (sqr(tt1) - sqr(tt2))) -
123  (Bra12 - Bra12Cut) * 0.5 * Koef2 * chi1 * (sqr(oo1) - sqr(oo2)));
124 
125  // torque = u_i x G
126  auto const G2 = -dU_da * d - dU_dc * uj;
127 
128  ParticleForce pf{-dU_dr * d - dU_da * ui - dU_db * uj,
129  vector_product(ui, G2)};
130 
131  return pf;
132 }

134 /** Calculate Gay-Berne energy */
135 inline double gb_pair_energy(Utils::Vector3d const &ui,
136  Utils::Vector3d const &uj,
137  IA_parameters const &ia_params,
138  Utils::Vector3d const &d, double dist) {
139  using Utils::int_pow;
140  using Utils::sqr;
141 
142  if (dist >= ia_params.gay_berne.cut) {
143  return 0.0;
144  }
145 
146  auto const e0 = ia_params.gay_berne.eps;
147  auto const s0 = ia_params.gay_berne.sig;
148  auto const chi1 = ia_params.gay_berne.chi1;
149  auto const chi2 = ia_params.gay_berne.chi2;
150  auto const mu = ia_params.gay_berne.mu;
151  auto const nu = ia_params.gay_berne.nu;
152  auto const r = Utils::Vector3d(d).normalize();
153 
154  auto const uij = ui * uj;
155  auto const rui = r * ui;
156  auto const ruj = r * uj;
157 
158  auto const o1 = sqr(rui + ruj) / (1 + chi1 * uij);
159  auto const o2 = sqr(rui - ruj) / (1 - chi1 * uij);
160  auto const t1 = sqr(rui + ruj) / (1 + chi2 * uij);
161  auto const t2 = sqr(rui - ruj) / (1 - chi2 * uij);
162 
163  auto const e1 = std::pow(1. - sqr(chi1 * uij), -0.5 * nu);
164  auto const e2 = std::pow(1. - 0.5 * chi2 * (t1 + t2), mu);
165  auto const e = e0 * e1 * e2;
166 
167  auto const s1 = 1. / std::sqrt(1. - 0.5 * chi1 * (o1 + o2));
168  auto const s = s0 * s1;
169 
170  auto r_eff = [=](double r) { return (r - s + s0) / s0; };
171  auto E = [=](double r) {
172  return 4. * e * (int_pow<12>(1. / r) - int_pow<6>(1. / r));
173  };
174 
175  return E(r_eff(dist)) - E(r_eff(ia_params.gay_berne.cut));
176 }
