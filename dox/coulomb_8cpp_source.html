113 double cutoff(const Utils::Vector3d &box_l) {
114  switch (coulomb.method) {
115  case COULOMB_MMM1D:
116  return std::numeric_limits<double>::infinity();
117 #ifdef P3M
118  case COULOMB_ELC_P3M:
119  return std::max(elc_params.space_layer, p3m.params.r_cut_iL * box_l[0]);
120  case COULOMB_P3M_GPU:
121  case COULOMB_P3M:
122  /* do not use precalculated r_cut here, might not be set yet */
123  return p3m.params.r_cut_iL * box_l[0];
124 #endif
125  case COULOMB_DH:
126  return dh_params.r_cut;
127  case COULOMB_RF:
128  return rf_params.r_cut;
129 #ifdef SCAFACOS
130  case COULOMB_SCAFACOS:
131  return Scafacos::fcs_coulomb()->get_r_cut();
132 #endif
133  default:
134  return -1.0;
135  }

256 void calc_long_range_force(const ParticleRange &particles) {
257  switch (coulomb.method) {
258 #ifdef P3M
259  case COULOMB_ELC_P3M:
260  if (elc_params.dielectric_contrast_on) {
261  ELC_P3M_modify_p3m_sums_both(particles);
262  ELC_p3m_charge_assign_both(particles);
263  ELC_P3M_self_forces(particles);
264  } else
265  p3m_charge_assign(particles);
266 
267  p3m_calc_kspace_forces(true, false, particles);
268 
269  if (elc_params.dielectric_contrast_on)
270  ELC_P3M_restore_p3m_sums(particles);
271 
272  ELC_add_force(particles);
273 
274  break;
275 #endif
276 #ifdef CUDA
277  case COULOMB_P3M_GPU:
278  if (this_node == 0) {
279  p3m_gpu_add_farfield_force();
280  }
281  /* there is no NPT handling here as long as we cannot compute energies.
282  This is checked in integrator_npt_sanity_checks() when integration
283  starts. */
284  break;
285 #endif
286 #ifdef P3M
287  case COULOMB_P3M:
288  p3m_charge_assign(particles);
289 #ifdef NPT
290  if (integ_switch == INTEG_METHOD_NPT_ISO) {
291  auto const energy = p3m_calc_kspace_forces(true, true, particles);
292  npt_add_virial_contribution(energy);
293  } else
294 #endif
295  p3m_calc_kspace_forces(true, false, particles);
296  break;
297 #endif
298 #ifdef SCAFACOS
299  case COULOMB_SCAFACOS:
300  Scafacos::fcs_coulomb()->add_long_range_force();
301  break;
302 #endif
303  default:
304  break;
305  }
306 
307  /* Add fields from EK if enabled */
308 #ifdef ELECTROKINETICS
309  if (this_node == 0) {
310  ek_calculate_electrostatic_coupling();
311  }
312 #endif
313 }

315 double calc_energy_long_range(const ParticleRange &particles) {
316  double energy = 0.0;
317  switch (coulomb.method) {
318 #ifdef P3M
319  case COULOMB_P3M_GPU:
320  runtimeWarningMsg()
321  << "long range energy calculation not implemented for GPU P3M";
322  break;
323  case COULOMB_P3M:
324  p3m_charge_assign(particles);
325  energy = p3m_calc_kspace_forces(false, true, particles);
326  break;
327  case COULOMB_ELC_P3M:
328  // assign the original charges first
329  // they may not have been assigned yet
330  p3m_charge_assign(particles);
331  if (!elc_params.dielectric_contrast_on)
332  energy = p3m_calc_kspace_forces(false, true, particles);
333  else {
334  energy = 0.5 * p3m_calc_kspace_forces(false, true, particles);
335  energy += 0.5 * coulomb.prefactor *
336  ELC_P3M_dielectric_layers_energy_self(particles);
337 
338  // assign both original and image charges now
339  ELC_p3m_charge_assign_both(particles);
340  ELC_P3M_modify_p3m_sums_both(particles);
341 
342  energy += 0.5 * p3m_calc_kspace_forces(false, true, particles);
343 
344  // assign only the image charges now
345  ELC_p3m_charge_assign_image(particles);
346  ELC_P3M_modify_p3m_sums_image(particles);
347 
348  energy -= 0.5 * p3m_calc_kspace_forces(false, true, particles);
349 
350  // restore modified sums
351  ELC_P3M_restore_p3m_sums(particles);
352  }
353  energy += ELC_energy(particles);
354  break;
355 #endif
356 #ifdef SCAFACOS
357  case COULOMB_SCAFACOS:
358  energy += Scafacos::fcs_coulomb()->long_range_energy();
359  break;
360 #endif
361  default:
362  break;
363  }
364  return energy;
365 }

367 int icc_sanity_check() {
368  switch (coulomb.method) {
369 #ifdef P3M
370  case COULOMB_ELC_P3M: {
371  if (elc_params.dielectric_contrast_on) {
372  runtimeErrorMsg() << "ICC conflicts with ELC dielectric contrast";
373  return 1;
374  }
375  break;
376  }
377 #endif
378  case COULOMB_DH: {
379  runtimeErrorMsg() << "ICC does not work with Debye-Hueckel.";
380  return 1;
381  }
382  case COULOMB_RF: {
383  runtimeErrorMsg() << "ICC does not work with COULOMB_RF.";
384  return 1;
385  }
386  default:
387  break;
388  }
389 
390 #ifdef NPT
391  if (integ_switch == INTEG_METHOD_NPT_ISO) {
392  runtimeErrorMsg() << "ICC does not work in the NPT ensemble";
393  return 1;
394  }
395 #endif
396 
397  return 0;
398 }
