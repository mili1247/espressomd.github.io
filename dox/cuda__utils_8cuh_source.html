74 /** Calculate @c dim_grid for CUDA kernel calls. */
75 inline dim3 calculate_dim_grid(unsigned const threads_x,
76  unsigned const blocks_per_grid_y,
77  unsigned const threads_per_block) {
78  assert(threads_x >= 1);
79  assert(blocks_per_grid_y >= 1);
80  assert(threads_per_block >= 1);
81  auto const threads_y = threads_per_block * blocks_per_grid_y;
82  auto const blocks_per_grid_x = (threads_x + threads_y - 1) / threads_y;
83  return make_uint3(blocks_per_grid_x, blocks_per_grid_y, 1);
84 }
