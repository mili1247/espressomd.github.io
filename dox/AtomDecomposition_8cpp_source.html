51 GhostCommunicator AtomDecomposition::prepare_comm() {
52  /* no need for comm for only 1 node */
53  if (comm.size() == 1) {
54  return GhostCommunicator{comm, 0};
55  }
56 
57  auto ghost_comm =
58  GhostCommunicator{comm, static_cast<std::size_t>(comm.size())};
59  /* every node has its dedicated comm step */
60  for (int n = 0; n < comm.size(); n++) {
61  ghost_comm.communications[n].part_lists.resize(1);
62  ghost_comm.communications[n].part_lists[0] = &(cells.at(n).particles());

68 void AtomDecomposition::configure_comms() {
69  m_exchange_ghosts_comm = prepare_comm();
70  m_collect_ghost_force_comm = prepare_comm();
71 
72  if (comm.size() > 1) {
73  for (int n = 0; n < comm.size(); n++) {
74  /* use the prefetched send buffers. Node 0 transmits first and never
75  * prefetches. */
76  if (comm.rank() == 0 || comm.rank() != n) {
77  m_exchange_ghosts_comm.communications[n].type = GHOST_BCST;
78  } else {
79  m_exchange_ghosts_comm.communications[n].type =
80  GHOST_BCST | GHOST_PREFETCH;
81  }
82  m_collect_ghost_force_comm.communications[n].type = GHOST_RDCE;
83  }
84  /* first round: all nodes except the first one prefetch their send data */
85  if (comm.rank() != 0) {
86  m_exchange_ghosts_comm.communications[0].type |= GHOST_PREFETCH;
87  }
88  }

99 void AtomDecomposition::resort(bool global_flag,
100  std::vector<ParticleChange> &diff) {
101  for (auto &p : local().particles()) {
102  fold_position(p.r.p, p.l.i, m_box);
103 
104  p.l.p_old = p.r.p;
105  }
106 
107  /* Local updates are a NoOp for this decomposition. */
108  if (not global_flag) {
109  return;
110  }
111 
112  /* Sort displaced particles by the node they belong to. */
113  std::vector<std::vector<Particle>> send_buf(comm.size());
114  for (auto it = local().particles().begin();
115  it != local().particles().end();) {
116  auto const target_node = id_to_rank(it->identity());
117  if (target_node != comm.rank()) {
118  diff.emplace_back(RemovedParticle{it->identity()});
119  send_buf.at(target_node).emplace_back(std::move(*it));
120  it = local().particles().erase(it);
121  } else {
122  ++it;
123  }
124  }
125 
126  /* Exchange particles */
127  std::vector<std::vector<Particle>> recv_buf(comm.size());
128  boost::mpi::all_to_all(comm, send_buf, recv_buf);
129 
130  diff.emplace_back(ModifiedList{local().particles()});
131 
132  /* Add new particles belonging to this node */
133  for (auto &parts : recv_buf) {
134  for (auto &p : parts) {
135  local().particles().insert(std::move(p));
136  }
137  }
138 }
