195 inline __device__ float4 random_wrapper_philox(unsigned int index,
196  unsigned int mode,
197  uint64_t philox_counter) {
198  // Split the 64 bit counter into two 32 bit ints.
199  auto const philox_counter_hi = static_cast<uint32_t>(philox_counter >> 32);
200  auto const philox_counter_low = static_cast<uint32_t>(philox_counter);
201  uint4 rnd_ints =
202  curand_Philox4x32_10(make_uint4(index, philox_counter_hi, 0, mode),
203  make_uint2(philox_counter_low, 0));
204  float4 rnd_floats;
205  rnd_floats.w = static_cast<float>(rnd_ints.w) * CURAND_2POW32_INV +
206  (CURAND_2POW32_INV / 2.0f);
207  rnd_floats.x = static_cast<float>(rnd_ints.x) * CURAND_2POW32_INV +
208  (CURAND_2POW32_INV / 2.0f);
209  rnd_floats.y = static_cast<float>(rnd_ints.y) * CURAND_2POW32_INV +
210  (CURAND_2POW32_INV / 2.0f);
211  rnd_floats.z = static_cast<float>(rnd_ints.z) * CURAND_2POW32_INV +
212  (CURAND_2POW32_INV / 2.0f);
213  return rnd_floats;
214 }
