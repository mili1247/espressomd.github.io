34 template <class CellRef> class Neighbors {
35  using storage_type = std::vector<CellRef>;
36 
37 public:
38  using value_type = typename storage_type::value_type;

43 private:
44  void copy(const Neighbors &rhs) {
45  m_neighbors = rhs.m_neighbors;
46  m_red_black_divider =
47  m_neighbors.begin() +
48  std::distance(rhs.m_neighbors.begin(),
49  const_iterator(rhs.m_red_black_divider));
50  }

60  Neighbors(Utils::Span<const CellRef> red_neighbors,
61  Utils::Span<const CellRef> black_neighbors) {
62  m_neighbors.resize(red_neighbors.size() + black_neighbors.size());
63  m_red_black_divider = std::copy(red_neighbors.begin(), red_neighbors.end(),
64  m_neighbors.begin());
65  std::copy(black_neighbors.begin(), black_neighbors.end(),
66  m_red_black_divider);
67  }
68 
69  /**
70  * @brief All neighbors.
71  */
72  cell_range all() { return {m_neighbors.begin(), m_neighbors.end()}; }
73  /**
74  * @brief Red partition of neighbors.
75  *
76  * An partition of the neighbors so that iterating over all
77  * neighbors_red of all cells visits every pair exactly once.
78  * Complement of neighbors_black.
79  */
80  cell_range red() { return {m_neighbors.begin(), m_red_black_divider}; }
81  /**
82  * @brief Black partition of neighbors.
83  *
84  * An partition of the neighbors so that iterating over all
85  * neighbors_black of all cells visits every pair exactly once.
86  * Complement of neighbors_red.
87  */
88  cell_range black() { return {m_red_black_divider, m_neighbors.end()}; }
89 
90 private:
91  /** Container with all the neighbors.
92  Red neighbors are first, black second. */
93  storage_type m_neighbors;
94  /** Iterator pointing to the first black neighbor

96  iterator m_red_black_divider;
