57 /** Layout of the lattice data.
58  * The description is similar to MPI datatypes but a bit more compact.
59  */
60 struct FieldType {
61  FieldType(int new_extent)
62  : count(0), disps({}), lengths({}), extent(new_extent), vblocks(0),
63  vstride(0), vskip(0), vflag(false), subtype(nullptr) {}
64  FieldType(int new_vblocks, int new_vstride, int new_vskip, bool new_vflag,
65  std::shared_ptr<FieldType> oldtype)
66  : count(oldtype->count), disps(oldtype->disps), lengths(oldtype->lengths),
67  extent(0), vblocks(new_vblocks), vstride(new_vstride), vskip(new_vskip),
68  vflag(new_vflag), subtype(oldtype) {
69  if (vflag) {
70  extent = oldtype->extent * ((vblocks - 1) * vskip + vstride);
71  } else {
72  extent = oldtype->extent * vstride + (vblocks - 1) * vskip;
73  }
74  }
75  int count; /**< number of subtypes in fieldtype */
76  std::vector<int> disps; /**< displacements of the subtypes */
77  std::vector<int> lengths; /**< lengths of the subtypes */
78  int extent; /**< extent of the complete fieldtype including gaps */
79  int vblocks; /**< number of blocks in field vectors */
80  int vstride; /**< size of strides in field vectors */
81  int vskip; /**< displacement between strides in field vectors */
82  bool vflag;
83  std::shared_ptr<FieldType> subtype;
84 };
