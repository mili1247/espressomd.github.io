59 /** Compute the Reaction Field pair force.
60  * @param[in] q1q2 Product of the charges on p1 and p2.
61  * @param[in] d Vector pointing from p1 to p2.
62  * @param[in] dist Distance between p1 and p2.
63  * @param[out] force Calculated force on p1.
64  */
65 inline void add_rf_coulomb_pair_force(double const q1q2,
66  Utils::Vector3d const &d,
67  double const dist,
68  Utils::Vector3d &force) {
69  if (dist < rf_params.r_cut) {
70  auto fac = 1.0 / Utils::int_pow<3>(dist) +
71  rf_params.B / Utils::int_pow<3>(rf_params.r_cut);
72  fac *= q1q2;
73  force += fac * d;
74  }
75 }
76 
77 /** Compute the Reaction Field pair energy.
78  * @param q1q2 Product of the charges on p1 and p2.
79  * @param dist Distance between p1 and p2.
80  */
81 inline double rf_coulomb_pair_energy(double const q1q2, double const dist) {
82  if (dist < rf_params.r_cut) {
83  auto fac = 1.0 / dist - (rf_params.B * dist * dist) /
84  (2 * Utils::int_pow<3>(rf_params.r_cut));
85  // remove discontinuity at dist = r_cut
86  fac -= (1 - rf_params.B / 2) / rf_params.r_cut;
87  fac *= q1q2;
88  return fac;
89  }
90  return 0.0;
91 }
