35 __device__ inline void get_mi_vector_dds(float res[3], float const a[3],
36  float const b[3], float const box_l[3],
37  int const periodic[3]) {
38  for (int i = 0; i < 3; i++) {
39  res[i] = a[i] - b[i];
40  if (periodic[i])
41  res[i] -= floor(res[i] / box_l[i] + 0.5f) * box_l[i];
42  }
43 }

45 #define scalar(a, b) ((a)[0] * (b)[0] + (a)[1] * (b)[1] + (a)[2] * (b)[2])

47 __device__ void dipole_ia_force(int id, float pf, float const *r1,
48  float const *r2, float const *dip1,
49  float const *dip2, float *f1, float *torque1,
50  float *torque2, float box_l[3],
51  int periodic[3]) {
52  float r_inv, pe1, pe2, pe3, pe4, r_sq, r3_inv, r5_inv, r_sq_inv, r7_inv, a, b,
53  cc, d, ab;
54 #ifdef ROTATION
55  float bx, by, bz, ax, ay, az;
56 #endif
57 
58  float dr[3];
59  float _r1[3], _r2[3], _dip1[3], _dip2[3];
60 
61  for (int i = 0; i < 3; i++) {
62  _r1[i] = r1[i];
63  _r2[i] = r2[i];
64  _dip1[i] = dip1[i];
65  _dip2[i] = dip2[i];
66  }
67 
68  // Distance between particles
69  get_mi_vector_dds(dr, _r1, _r2, box_l, periodic);
70 
71  // Powers of distance
72  r_sq = scalar(dr, dr);
73  r_sq_inv = 1 / r_sq;
74  r_inv = rsqrtf(r_sq);
75  r3_inv = 1 / r_sq * r_inv;
76  r5_inv = r3_inv * r_sq_inv;
77  r7_inv = r5_inv * r_sq_inv;
78 
79  // Dot products
80  pe1 = scalar(_dip1, _dip2);
81  pe2 = scalar(_dip1, dr);
82  pe3 = scalar(_dip2, dr);
83  pe4 = 3.0f * r5_inv;
84 
85  // Force
86  a = pe4 * pe1;
87  b = -15.0f * pe2 * pe3 * r7_inv;
88  ab = a + b;
89  cc = pe4 * pe3;
90  d = pe4 * pe2;
91 
92  f1[0] = (pf * (ab * dr[0] + cc * _dip1[0] + d * _dip2[0]));
93  f1[1] = (pf * (ab * dr[1] + cc * _dip1[1] + d * _dip2[1]));
94  f1[2] = (pf * (ab * dr[2] + cc * _dip1[2] + d * _dip2[2]));
95 
96 #ifdef ROTATION
97  // Torques
98  ax = _dip1[1] * _dip2[2] - _dip2[1] * _dip1[2];
99  ay = _dip2[0] * _dip1[2] - _dip1[0] * _dip2[2];
100  az = _dip1[0] * _dip2[1] - _dip2[0] * _dip1[1];
101 
102  bx = _dip1[1] * dr[2] - dr[1] * _dip1[2];
103  by = dr[0] * _dip1[2] - _dip1[0] * dr[2];
104  bz = _dip1[0] * dr[1] - dr[0] * _dip1[1];
105 
106  torque1[0] = (pf * (-ax * r3_inv + bx * cc));
107  torque1[1] = (pf * (-ay * r3_inv + by * cc));
108  torque1[2] = (pf * (-az * r3_inv + bz * cc));
109 
110  bx = _dip2[1] * dr[2] - dr[1] * _dip2[2];
111  by = dr[0] * _dip2[2] - _dip2[0] * dr[2];
112  bz = _dip2[0] * dr[1] - dr[0] * _dip2[1];
113 
114  torque2[0] = pf * (ax * r3_inv + bx * d);
115  torque2[1] = pf * (ay * r3_inv + by * d);
116  torque2[2] = pf * (az * r3_inv + bz * d);
117 #endif
118 }

120 __device__ float dipole_ia_energy(int id, float pf, float const *r1,
121  float const *r2, float const *dip1,
122  float const *dip2, float box_l[3],
123  int periodic[3]) {
124  float r_inv, pe1, pe2, pe3, pe4, r_sq, r3_inv, r5_inv, r_sq_inv;
125  float dr[3];
126  float _r1[3], _r2[3], _dip1[3], _dip2[3];
127 
128  for (int i = 0; i < 3; i++) {
129  _r1[i] = r1[i];
130  _r2[i] = r2[i];
131  _dip1[i] = dip1[i];
132  _dip2[i] = dip2[i];
133  }
134 
135  // Distance between particles
136  get_mi_vector_dds(dr, _r1, _r2, box_l, periodic);
137 
138  // Powers of distance
139  r_sq = scalar(dr, dr);
140  r_sq_inv = 1 / r_sq;
141  r_inv = rsqrtf(r_sq);
142  r3_inv = 1 / r_sq * r_inv;
143  r5_inv = r3_inv * r_sq_inv;
144 
145  // Dot products
146  pe1 = scalar(_dip1, _dip2);
147  pe2 = scalar(_dip1, dr);
148  pe3 = scalar(_dip2, dr);
149  pe4 = 3.0f * r5_inv;
150 
151  // Energy
152  return pf * (pe1 * r3_inv - pe4 * pe2 * pe3);
153 }

155 __global__ void DipolarDirectSum_kernel_force(float pf, int n, float *pos,
156  float *dip, float *f,
157  float *torque, float box_l[3],
158  int periodic[3]) {
159 
160  auto i = static_cast<int>(blockIdx.x * blockDim.x + threadIdx.x);
161 
162  if (i >= n)
163  return;
164 
165  // Kahan summation based on the wikipedia article
166  // Force
167  float fi[3], fsum[3], tj[3];
168 
169  // Torque
170  float ti[3], tsum[3];
171 
172  // There is one thread per particle. Each thread computes interactions
173  // with particles whose id is smaller than the thread id.
174  // The force and torque of all the interaction partners of the current thread
175  // is atomically added to global results ad once.
176  // The result for the particle id equal to the thread id is atomically added
177  // to global memory at the end.
178 
179  // Clear summation vars
180  for (int j = 0; j < 3; j++) {
181  // Force
182  fsum[j] = 0;
183  // Torque
184  tsum[j] = 0;
185  }
186 
187  for (int j = i + 1; j < n; j++) {
188  dipole_ia_force(i, pf, pos + 3 * i, pos + 3 * j, dip + 3 * i, dip + 3 * j,
189  fi, ti, tj, box_l, periodic);
190  for (int k = 0; k < 3; k++) {
191  // Add rhs to global memory
192  atomicAdd(f + 3 * j + k, -fi[k]);
193  atomicAdd((torque + 3 * j + k), tj[k]);
194  tsum[k] += ti[k];
195  fsum[k] += fi[k];
196  }
197  }
198 
199  // Add the left hand side result to global memory
200  for (int j = 0; j < 3; j++) {
201  atomicAdd(f + 3 * i + j, fsum[j]);
202  atomicAdd(torque + 3 * i + j, tsum[j]);
203  }
204 }

206 __device__ void dds_sumReduction(float *input, float *sum) {
207  auto tid = static_cast<int>(threadIdx.x);
208  for (auto i = static_cast<int>(blockDim.x); i > 1; i /= 2) {
209  __syncthreads();
210  if (tid < i / 2)
211  input[tid] += input[i / 2 + tid];
212  if ((i % 2 == 1) && (tid == 0))
213  input[tid] += input[i - 1];
214  }
215  __syncthreads();
216  if (threadIdx.x == 0) {
217  sum[0] = input[0];
218  }
219 }

221 __global__ void DipolarDirectSum_kernel_energy(float pf, int n, float *pos,
222  float *dip, float box_l[3],
223  int periodic[3],
224  float *energySum) {
225 
226  auto i = static_cast<int>(blockIdx.x * blockDim.x + threadIdx.x);
227  float sum = 0.0;
228  extern __shared__ float res[];
229 
230  // There is one thread per particle. Each thread computes interactions
231  // with particles whose id is larger than the thread id.
232  // The result for the particle id equal to the thread id is added
233  // to global memory at the end.
234 
235  if (i < n) {
236  // Summation for particle i
237  for (int j = i + 1; j < n; j++) {
238  sum += dipole_ia_energy(i, pf, pos + 3 * i, pos + 3 * j, dip + 3 * i,
239  dip + 3 * j, box_l, periodic);
240  }
241 
242  // Save per thread result into block shared mem
243  res[threadIdx.x] = sum;
244  } else
245  res[threadIdx.x] = 0;
246 
247  // Sum results within a block
248  __syncthreads(); // Wait till all threads in block are done
249  dds_sumReduction(res, &(energySum[blockIdx.x]));
250 }
