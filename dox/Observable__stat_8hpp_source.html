47  auto get_chunk_size() const { return m_chunk_size; }

49  /** Accumulate values.
50  * @param acc Initial value for the accumulator.
51  * @param column Which column to sum up (only relevant for multi-dimensional
52  * observables).
53  */
54  double accumulate(double acc = 0.0, std::size_t column = 0) const {
55  assert(column < m_chunk_size);
56  if (m_chunk_size == 1)
57  return boost::accumulate(m_data, acc);
58 
59  for (auto it = m_data.begin() + column; it < m_data.end();
60  it += m_chunk_size)
61  acc += *it;
62  return acc;
63  }
64 
65  /** Rescale values */
66  void rescale(double volume) {
67  auto const factor = 1. / volume;
68  for (auto &e : m_data) {
69  e *= factor;
70  }
71  }

96  void add_non_bonded_contribution(int type1, int type2,
97  Utils::Span<const double> data) {
98  assert(data.size() == m_chunk_size);
99  auto const source = (type1 == type2) ? non_bonded_intra : non_bonded_inter;
100  auto const dest = non_bonded_contribution(source, type1, type2);
101 
102  boost::transform(dest, data, dest.begin(), std::plus<>{});
103  }
104 
105  void add_non_bonded_contribution(int type1, int type2, double data) {
106  add_non_bonded_contribution(type1, type2, {&data, 1});
107  }
108
