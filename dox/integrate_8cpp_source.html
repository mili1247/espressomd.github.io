99 void integrator_sanity_checks() {
100  if (time_step < 0.0) {
101  runtimeErrorMsg() << "time_step not set";
102  }
103  switch (integ_switch) {
104  case INTEG_METHOD_STEEPEST_DESCENT:
105  if (thermo_switch != THERMO_OFF)
106  runtimeErrorMsg()
107  << "The steepest descent integrator is incompatible with thermostats";
108  break;
109  case INTEG_METHOD_NVT:
110  if (thermo_switch & (THERMO_NPT_ISO | THERMO_BROWNIAN | THERMO_SD))
111  runtimeErrorMsg() << "The VV integrator is incompatible with the "
112  "currently active combination of thermostats";
113  break;
114 #ifdef NPT
115  case INTEG_METHOD_NPT_ISO:
116  if (thermo_switch != THERMO_OFF and thermo_switch != THERMO_NPT_ISO)
117  runtimeErrorMsg() << "The NpT integrator requires the NpT thermostat";
118  break;
119 #endif
120  case INTEG_METHOD_BD:
121  if (thermo_switch != THERMO_BROWNIAN)
122  runtimeErrorMsg() << "The BD integrator requires the BD thermostat";
123  break;
124 #ifdef STOKESIAN_DYNAMICS
125  case INTEG_METHOD_SD:
126  if (thermo_switch != THERMO_OFF and thermo_switch != THERMO_SD)
127  runtimeErrorMsg() << "The SD integrator requires the SD thermostat";
128  break;
129 #endif
130  default:
131  runtimeErrorMsg() << "Unknown value for integ_switch";
132  }
133 }

135 /** @brief Calls the hook for propagation kernels before the force calculation
136  * @return whether or not to stop the integration loop early.
137  */
138 bool integrator_step_1(ParticleRange &particles) {
139  switch (integ_switch) {
140  case INTEG_METHOD_STEEPEST_DESCENT:
141  if (steepest_descent_step(particles))
142  return true; // early exit
143  break;
144  case INTEG_METHOD_NVT:
145  velocity_verlet_step_1(particles, time_step);
146  break;
147 #ifdef NPT
148  case INTEG_METHOD_NPT_ISO:
149  velocity_verlet_npt_step_1(particles, time_step);
150  break;
151 #endif
152  case INTEG_METHOD_BD:
153  // the Ermak-McCammon's Brownian Dynamics requires a single step
154  // so, just skip here
155  break;
156 #ifdef STOKESIAN_DYNAMICS
157  case INTEG_METHOD_SD:
158  stokesian_dynamics_step_1(particles, time_step);
159  break;
160 #endif // STOKESIAN_DYNAMICS
161  default:
162  throw std::runtime_error("Unknown value for integ_switch");
163  }
164  return false;
165 }

167 /** Calls the hook of the propagation kernels after force calculation */
168 void integrator_step_2(ParticleRange &particles, double kT) {
169  switch (integ_switch) {
170  case INTEG_METHOD_STEEPEST_DESCENT:
171  // Nothing
172  break;
173  case INTEG_METHOD_NVT:
174  velocity_verlet_step_2(particles, time_step);
175  break;
176 #ifdef NPT
177  case INTEG_METHOD_NPT_ISO:
178  velocity_verlet_npt_step_2(particles, time_step);
179  break;
180 #endif
181  case INTEG_METHOD_BD:
182  // the Ermak-McCammon's Brownian Dynamics requires a single step
183  brownian_dynamics_propagator(brownian, particles, time_step, kT);
184  break;
185 #ifdef STOKESIAN_DYNAMICS
186  case INTEG_METHOD_SD:
187  stokesian_dynamics_step_2(particles);
188  break;
189 #endif // STOKESIAN_DYNAMICS
190  default:
191  throw std::runtime_error("Unknown value for INTEG_SWITCH");
192  }

195 int integrate(int n_steps, int reuse_forces) {
196  ESPRESSO_PROFILER_CXX_MARK_FUNCTION;
197 
198  /* Prepare the integrator */
199  on_integration_start(time_step);
200 
201  /* if any method vetoes (e.g. P3M not initialized), immediately bail out */
202  if (check_runtime_errors(comm_cart))
203  return 0;
204 
205  /* Verlet list criterion */
206 
207  /* Integration Step: Preparation for first integration step:
208  * Calculate forces F(t) as function of positions x(t) (and velocities v(t))
209  */
210  if (reuse_forces == -1 || (recalc_forces && reuse_forces != 1)) {
211  ESPRESSO_PROFILER_MARK_BEGIN("Initial Force Calculation");
212  lb_lbcoupling_deactivate();
213 
214 #ifdef VIRTUAL_SITES
215  virtual_sites()->update();
216 #endif
217 
218  // Communication step: distribute ghost positions
219  cells_update_ghosts(global_ghost_flags());
220 
221  force_calc(cell_structure, time_step, temperature);
222 
223  if (integ_switch != INTEG_METHOD_STEEPEST_DESCENT) {
224 #ifdef ROTATION
225  convert_initial_torques(cell_structure.local_particles());
226 #endif
227  }
228 
229  ESPRESSO_PROFILER_MARK_END("Initial Force Calculation");
230  }
231 
232  lb_lbcoupling_activate();
233 
234  if (check_runtime_errors(comm_cart))
235  return 0;
236 
237  /* incremented if a Verlet update is done, aka particle resorting. */
238  int n_verlet_updates = 0;
239 
240 #ifdef VALGRIND_INSTRUMENTATION
241  CALLGRIND_START_INSTRUMENTATION;
242 #endif
243  /* Integration loop */
244  ESPRESSO_PROFILER_CXX_MARK_LOOP_BEGIN(integration_loop, "Integration loop");
245  int integrated_steps = 0;
246  for (int step = 0; step < n_steps; step++) {
247  ESPRESSO_PROFILER_CXX_MARK_LOOP_ITERATION(integration_loop, step);
248 
249  auto particles = cell_structure.local_particles();
250 
251 #ifdef BOND_CONSTRAINT
252  if (n_rigidbonds)
253  save_old_position(particles, cell_structure.ghost_particles());
254 #endif
255 
256  bool early_exit = integrator_step_1(particles);
257  if (early_exit)
258  break;
259 
260  /* Propagate philox rng counters */
261  philox_counter_increment();
262 
263 #ifdef BOND_CONSTRAINT
264  /* Correct those particle positions that participate in a rigid/constrained
265  * bond */
266  if (n_rigidbonds) {
267  correct_position_shake(cell_structure);
268  }
269 #endif
270 
271 #ifdef VIRTUAL_SITES
272  virtual_sites()->update();
273 #endif
274 
275  if (cell_structure.get_resort_particles() >= Cells::RESORT_LOCAL)
276  n_verlet_updates++;
277 
278  // Communication step: distribute ghost positions
279  cells_update_ghosts(global_ghost_flags());
280 
281  particles = cell_structure.local_particles();
282 
283  force_calc(cell_structure, time_step, temperature);
284 
285 #ifdef VIRTUAL_SITES
286  virtual_sites()->after_force_calc();
287 #endif
288  integrator_step_2(particles, temperature);
289 #ifdef BOND_CONSTRAINT
290  // SHAKE velocity updates
291  if (n_rigidbonds) {
292  correct_velocity_shake(cell_structure);
293  }
294 #endif
295 
296  // propagate one-step functionalities
297  if (integ_switch != INTEG_METHOD_STEEPEST_DESCENT) {
298  if (lb_lbfluid_get_lattice_switch() != ActiveLB::NONE) {
299  auto const tau = lb_lbfluid_get_tau();
300  auto const lb_steps_per_md_step =
301  static_cast<int>(std::round(tau / time_step));
302  fluid_step += 1;
303  if (fluid_step >= lb_steps_per_md_step) {
304  fluid_step = 0;
305  lb_lbfluid_propagate();
306  }
307  lb_lbcoupling_propagate();
308  }
309 
310 #ifdef VIRTUAL_SITES
311  virtual_sites()->after_lb_propagation(time_step);
312 #endif
313 
314 #ifdef COLLISION_DETECTION
315  handle_collisions();
316 #endif
317  }
318 
319  integrated_steps++;
320 
321  if (check_runtime_errors(comm_cart))
322  break;
323 
324  // Check if SIGINT has been caught.
325  if (ctrl_C == 1) {
326  notify_sig_int();
327  break;
328  }
329 
330  } // for-loop over integration steps
331  ESPRESSO_PROFILER_CXX_MARK_LOOP_END(integration_loop);
332 
333 #ifdef VALGRIND_INSTRUMENTATION
334  CALLGRIND_STOP_INSTRUMENTATION;
335 #endif
336 
337 #ifdef VIRTUAL_SITES
338  virtual_sites()->update();
339 #endif
340 
341  /* verlet list statistics */
342  if (n_verlet_updates > 0)
343  verlet_reuse = n_steps / (double)n_verlet_updates;
344  else
345  verlet_reuse = 0;
346 
347 #ifdef NPT
348  if (integ_switch == INTEG_METHOD_NPT_ISO) {
349  synchronize_npt_state();
350  }
351 #endif
352  return integrated_steps;
353 }

355 int python_integrate(int n_steps, bool recalc_forces_par,
356  bool reuse_forces_par) {
357  // Override the signal handler so that the integrator obeys Ctrl+C
358  SignalHandler sa(SIGINT, [](int) { ctrl_C = 1; });
359 
360  int reuse_forces = reuse_forces_par;
361 
362  if (recalc_forces_par) {
363  if (reuse_forces) {
364  runtimeErrorMsg() << "cannot reuse old forces and recalculate forces";
365  }
366  reuse_forces = -1;
367  }
368 
369  /* go on with integrate <n_steps> */
370  if (n_steps < 0) {
371  runtimeErrorMsg() << "illegal number of steps (must be >0)";
372  return ES_ERROR;
373  }
374 
375  /* if skin wasn't set, do an educated guess now */
376  if (!skin_set) {
377  auto const max_cut = maximal_cutoff();
378  if (max_cut <= 0.0) {
379  runtimeErrorMsg()
380  << "cannot automatically determine skin, please set it manually";
381  return ES_ERROR;
382  }
383  /* maximal skin that can be used without resorting is the maximal
384  * range of the cell system minus what is needed for interactions. */
385  auto const new_skin =
386  std::min(0.4 * max_cut,
387  *boost::min_element(cell_structure.max_cutoff()) - max_cut);
388  mpi_set_skin(new_skin);
389  }
390 
391  using Accumulators::auto_update;
392  using Accumulators::auto_update_next_update;
393 
394  for (int i = 0; i < n_steps;) {
395  /* Integrate to either the next accumulator update, or the
396  * end, depending on what comes first. */
397  auto const steps = std::min((n_steps - i), auto_update_next_update());
398  if (mpi_integrate(steps, reuse_forces))
399  return ES_ERROR;
400 
401  reuse_forces = 1;
402 
403  auto_update(steps);
404 
405  i += steps;
406  }
407 
408  if (n_steps == 0) {
409  if (mpi_integrate(0, reuse_forces))
410  return ES_ERROR;
411  }
412 
413  return ES_OK;
414 }
