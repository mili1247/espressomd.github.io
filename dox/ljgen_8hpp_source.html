59 /** Calculate Lennard-Jones force factor */
60 inline double ljgen_pair_force_factor(IA_parameters const &ia_params,
61  double dist) {
62  if (dist < (ia_params.ljgen.cut + ia_params.ljgen.offset)) {
63  auto r_off = dist - ia_params.ljgen.offset;
64 
65 #ifdef LJGEN_SOFTCORE
66  r_off *= r_off;
67  r_off += Utils::sqr(ia_params.ljgen.sig) * (1.0 - ia_params.ljgen.lambda1) *
68  ia_params.ljgen.softrad;
69  r_off = sqrt(r_off);
70 #else
71  r_off = std::abs(r_off);
72 #endif
73  auto const frac = ia_params.ljgen.sig / r_off;
74  auto const fac = ia_params.ljgen.eps
75 #ifdef LJGEN_SOFTCORE
76  * ia_params.ljgen.lambda1 *
77  (dist - ia_params.ljgen.offset) / r_off
78 #endif
79  * (ia_params.ljgen.b1 * ia_params.ljgen.a1 *
80  pow(frac, ia_params.ljgen.a1) -
81  ia_params.ljgen.b2 * ia_params.ljgen.a2 *
82  pow(frac, ia_params.ljgen.a2)) /
83  (r_off * dist);
84  return fac;
85  }
86  return 0.0;
87 }

89 /** Calculate Lennard-Jones energy */
90 inline double ljgen_pair_energy(IA_parameters const &ia_params, double dist) {
91  if (dist < (ia_params.ljgen.cut + ia_params.ljgen.offset)) {
92  auto r_off = dist - ia_params.ljgen.offset;
93 #ifdef LJGEN_SOFTCORE
94  r_off *= r_off;
95  r_off += pow(ia_params.ljgen.sig, 2) * (1.0 - ia_params.ljgen.lambda1) *
96  ia_params.ljgen.softrad;
97  r_off = sqrt(r_off);
98 #else
99  r_off = std::abs(r_off);
100 #endif
101  auto const frac = ia_params.ljgen.sig / r_off;
102  auto const fac = ia_params.ljgen.eps
103 #ifdef LJGEN_SOFTCORE
104  * ia_params.ljgen.lambda1
105 #endif
106  * (ia_params.ljgen.b1 * pow(frac, ia_params.ljgen.a1) -
107  ia_params.ljgen.b2 * pow(frac, ia_params.ljgen.a2) +
108  ia_params.ljgen.shift);
109  return fac;
110  }
111  return 0.0;
112 }
