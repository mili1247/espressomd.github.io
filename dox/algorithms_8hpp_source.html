38 template <class ValueOp, class WeightOp> struct WeightedSum {
39  template <class ParticleRange>
40  auto operator()(ParticleRange const &particles) const {
41  using particle_type = typename ParticleRange::value_type;
42  using value_op_type = decltype(ValueOp{}(std::declval<particle_type>()));
43  using weight_op_type = decltype(WeightOp{}(std::declval<particle_type>()));
44  auto func = [](auto sum, auto const &p) {
45  auto const w = WeightOp{}(p);
46  return std::make_pair(sum.first + ValueOp{}(p)*w, sum.second + w);
47  };
48 
49  return std::accumulate(std::begin(particles), std::end(particles),
50  std::pair<value_op_type, weight_op_type>(), func);
51  }
52 };
53 } // namespace detail
