88 /**
89  * @brief Check the boundary velocities.
90  * Sanity check if the velocity defined at LB boundaries is within the Mach
91  * number limits of the scheme i.e. u < 0.3.
92  */
93 void lb_boundary_mach_check() {
94  // Boundary velocities are stored in MD units, therefore we need to scale them
95  // in order to get lattice units.
96  auto const conv_fac = lb_lbfluid_get_tau() / lb_lbfluid_get_agrid();
97  double constexpr mach_limit = 0.3;
98  using LBBoundaries::lbboundaries;
99  if (std::any_of(lbboundaries.begin(), lbboundaries.end(),
100  [conv_fac, mach_limit](auto const &b) {
101  return (b->velocity() * conv_fac).norm() >= mach_limit;
102  })) {
103  runtimeErrorMsg() << "Lattice velocity exceeds the Mach number limit";
104  }
105 }

419 void check_tau_time_step_consistency(double tau, double time_step) {
420  auto const eps = std::numeric_limits<float>::epsilon();
421  if ((tau - time_step) / (tau + time_step) < -eps)
422  throw std::invalid_argument("LB tau (" + std::to_string(tau) +
423  ") must be >= MD time_step (" +
424  std::to_string(time_step) + ")");
425  auto const factor = tau / time_step;
426  if (fabs(round(factor) - factor) / factor > eps)
427  throw std::invalid_argument("LB tau (" + std::to_string(tau) +
428  ") must be integer multiple of "
429  "MD time_step (" +
430  std::to_string(time_step) + "). Factor is " +
431  std::to_string(factor));
432 }

540 void lb_lbfluid_print_vtk_velocity(const std::string &filename,
541  std::vector<int> bb1, std::vector<int> bb2) {
542  FILE *fp = fopen(filename.c_str(), "w");
543 
544  if (fp == nullptr) {
545  throw std::runtime_error("Could not open file for writing.");
546  }
547 
548  auto bb_low = Utils::Vector3i{};
549  auto bb_high = lb_lbfluid_get_shape();
550 
551  int it = 0;
552  for (auto val1 = bb1.begin(), val2 = bb2.begin();
553  val1 != bb1.end() && val2 != bb2.end(); ++val1, ++val2) {
554  if (*val1 == -1 || *val2 == -1) {
555  break;
556  }
557  auto const lower = std::min(*val1, *val2);
558  auto const upper = std::max(*val1, *val2);
559  if (lower < 0 or upper >= bb_high[it]) {
560  throw std::runtime_error(
561  "Tried to access index " + std::to_string(lower) + " and index " +
562  std::to_string(upper) + " on dimension " + std::to_string(it) +
563  " that has size " + std::to_string(bb_high[it]));
564  }
565  bb_low[it] = lower;
566  bb_high[it] = upper;
567  it++;
568  }
569 
570  Utils::Vector3i pos;
571  if (lattice_switch == ActiveLB::GPU) {
572 #ifdef CUDA
573  host_values.resize(lbpar_gpu.number_of_nodes);
574  lb_get_values_GPU(host_values.data());
575  auto const lattice_speed = lb_lbfluid_get_agrid() / lb_lbfluid_get_tau();

578  "ASCII\nDATASET STRUCTURED_POINTS\nDIMENSIONS %d %d %d\n"
579  "ORIGIN %f %f %f\nSPACING %f %f %f\nPOINT_DATA %d\n"

584  (bb_low[2] + 0.5) * lbpar_gpu.agrid, lbpar_gpu.agrid,
585  lbpar_gpu.agrid, lbpar_gpu.agrid,
586  (bb_high[0] - bb_low[0]) * (bb_high[1] - bb_low[1]) *
587  (bb_high[2] - bb_low[2]));
588  for (pos[2] = bb_low[2]; pos[2] < bb_high[2]; pos[2]++)
589  for (pos[1] = bb_low[1]; pos[1] < bb_high[1]; pos[1]++)
590  for (pos[0] = bb_low[0]; pos[0] < bb_high[0]; pos[0]++) {
591  auto const j =
592  static_cast<int>(lbpar_gpu.dim[0] * lbpar_gpu.dim[0] * pos[2] +
593  lbpar_gpu.dim[0] * pos[1] + pos[0]);

595  host_values[j].v[1] * lattice_speed,
596  host_values[j].v[2] * lattice_speed);

601  "# vtk DataFile Version 2.0\nlbfluid_cpu\n"
602  "ASCII\nDATASET STRUCTURED_POINTS\nDIMENSIONS %d %d %d\n"
603  "ORIGIN %f %f %f\nSPACING %f %f %f\nPOINT_DATA %d\n"
604  "SCALARS velocity float 3\nLOOKUP_TABLE default\n",
