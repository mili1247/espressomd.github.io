51 /** Calculate non-bonded energies between a pair of particles.
52  * @param p1 pointer to particle 1.
53  * @param p2 pointer to particle 2.
54  * @param d vector between p1 and p2.
55  * @param dist distance between p1 and p2.
56  * @param[in,out] obs_pressure pressure observable.
57  */
58 inline void add_non_bonded_pair_virials(Particle const &p1, Particle const &p2,
59  Utils::Vector3d const &d, double dist,
60  Observable_stat &obs_pressure) {
61 #ifdef EXCLUSIONS
62  if (do_nonbonded(p1, p2))
63 #endif
64  {
65  IA_parameters const &ia_params = *get_ia_param(p1.p.type, p2.p.type);
66  auto const force = calc_non_bonded_pair_force(p1, p2, ia_params, d, dist).f;
67  auto const stress = tensor_product(d, force);
68 
69  auto const type1 = p1.p.mol_id;
70  auto const type2 = p2.p.mol_id;
71  obs_pressure.add_non_bonded_contribution(type1, type2, flatten(stress));
72  }
73 
74 #ifdef ELECTROSTATICS
75  if (!obs_pressure.coulomb.empty()) {
76  /* real space Coulomb */
77  auto const p_coulomb = Coulomb::pair_pressure(p1, p2, d, dist);
78 
79  for (int i = 0; i < 3; i++) {
80  for (int j = 0; j < 3; j++) {
81  obs_pressure.coulomb[i * 3 + j] += p_coulomb(i, j);
82  }
83  }
84  }
85 #endif /*ifdef ELECTROSTATICS */
86 
87 #ifdef DIPOLES
88  /* real space magnetic dipole-dipole */
89  if (dipole.method != DIPOLAR_NONE) {
90  fprintf(stderr, "calculating pressure for magnetostatics which doesn't "
91  "have it implemented\n");
92  }
93 #endif /*ifdef DIPOLES */
94 }

96 boost::optional<Utils::Matrix<double, 3, 3>>
97 calc_bonded_virial_pressure_tensor(Bonded_IA_Parameters const &iaparams,
98  Particle const &p1, Particle const &p2) {
99  auto const dx = box_geo.get_mi_vector(p1.r.p, p2.r.p);
100  auto const result = calc_bond_pair_force(p1, p2, iaparams, dx);
101  if (result) {
102  auto const &force = result.get();
103 
104  return tensor_product(force, dx);
105  }
106 
107  return {};
108 }

110 boost::optional<Utils::Matrix<double, 3, 3>>
111 calc_bonded_three_body_pressure_tensor(Bonded_IA_Parameters const &iaparams,
112  Particle const &p1, Particle const &p2,
113  Particle const &p3) {
114  if ((boost::get<AngleHarmonicBond>(&iaparams) != nullptr) ||
115  (boost::get<AngleCosineBond>(&iaparams) != nullptr) ||
116 #ifdef TABULATED
117  (boost::get<TabulatedAngleBond>(&iaparams) != nullptr) ||
118 #endif
119  (boost::get<AngleCossquareBond>(&iaparams) != nullptr)) {
120  auto const dx21 = -box_geo.get_mi_vector(p1.r.p, p2.r.p);
121  auto const dx31 = box_geo.get_mi_vector(p3.r.p, p1.r.p);
122 
123  auto const result = calc_bonded_three_body_force(iaparams, p1, p2, p3);
124  if (result) {
125  Utils::Vector3d force2, force3;
126  std::tie(std::ignore, force2, force3) = result.get();
127 
128  return tensor_product(force2, dx21) + tensor_product(force3, dx31);
129  }
130  } else {
131  runtimeWarningMsg() << "Unsupported bond type " +
132  std::to_string(iaparams.which()) +
133  " in pressure calculation.";
134  return Utils::Matrix<double, 3, 3>{};
135  }
136 
137  return {};
138 }

140 inline boost::optional<Utils::Matrix<double, 3, 3>>
141 calc_bonded_pressure_tensor(Bonded_IA_Parameters const &iaparams,
142  Particle const &p1,
143  Utils::Span<Particle *> partners) {
144  switch (number_of_partners(iaparams)) {
145  case 1:
146  return calc_bonded_virial_pressure_tensor(iaparams, p1, *partners[0]);
147  case 2:
148  return calc_bonded_three_body_pressure_tensor(iaparams, p1, *partners[0],
149  *partners[1]);
150  default:
151  runtimeWarningMsg() << "Unsupported bond type " +
152  std::to_string(iaparams.which()) +
153  " in pressure calculation.";
154  return Utils::Matrix<double, 3, 3>{};
155  }

158 /** Calculate kinetic pressure (aka energy) for one particle.
159  * @param[in] p1 particle for which to calculate pressure
160  * @param[out] obs_pressure pressure observable
161  */
162 inline void add_kinetic_virials(Particle const &p1,
163  Observable_stat &obs_pressure) {
164  if (p1.p.is_virtual)
165  return;
166 
167  /* kinetic pressure */
168  for (int k = 0; k < 3; k++)
169  for (int l = 0; l < 3; l++)
170  obs_pressure.kinetic[k * 3 + l] += p1.m.v[k] * p1.m.v[l] * p1.p.mass;
171 }
