28 namespace Utils {
29 template <typename Key, typename Value> class Cache {
30  using map_type =
31  std::unordered_map<Key, typename std::add_const<Value>::type>;
32 
33 public:
34  using key_type = Key;
35  using value_type = const Value *;
36  using size_type = typename map_type::size_type;
37 
38  Cache() {
39  /* Default to maximal size of our map. */
40  m_max_size = m_cache.max_size();
41  }
42  explicit Cache(size_type max_size) : m_max_size(max_size), m_rand(max_size) {}
43 
44 private:
45  /** @brief The actual cache, maps Key -> Value.
46  */
47  map_type m_cache;
48  size_type m_max_size;
49 
50  /** Cheap linear-congruential PRNG. */
51  std::minstd_rand m_rand;
52 
53  /** @brief Drop an element.
54  *
55  * This function drops an element from the cache,
56  * it is used when a new item is to be cached, but then
57  * maximal size is reached. We use the bucket interface
58  * here to be able to pick a random element in constant
59  * time.
60  */
61  void drop_random_element() {
62  if (m_cache.empty())
63  return;
64 
65  auto const bucket_count = m_cache.bucket_count();
66 
67  /* Pick a random bucket, this has to terminate because
68  * the map is not empty. So there has to be a bucket with
69  * at least one element. */
70  auto bucket =
71  std::uniform_int_distribution<size_type>{0, bucket_count - 1}(m_rand);
72 
73  while (0 == m_cache.bucket_size(bucket)) {
74  /* Wrap around the end */
75  bucket = ((bucket + 1) % bucket_count);
76  }
77 
78  /* Pick a random element form that bucket. */
79  auto const elem_in_bucket = std::uniform_int_distribution<size_type>{
80  0, m_cache.bucket_size(bucket) - 1}(m_rand);
81 
82  /* Get the element in the bucket */
83  auto const drop_key =
84  std::next(m_cache.cbegin(bucket), elem_in_bucket)->first;
85 
86  /* And drop it. */
87  m_cache.erase(drop_key);
88  }
89 
90 public:
91  /** @brief Clear the cache.
92  *
93  * This invalidates all pointers into the cache.
94  */
95  void invalidate() { m_cache.clear(); }
96 
97  /** @brief Query if k is contained in the cache. */
98  bool has(Key const &k) const { return m_cache.find(k) != m_cache.end(); }
99 
100  /** @brief Number of elements currently cached. */
101  size_type size() const { return m_cache.size(); }
102 
103  /** @brief Maximal size of the cache. */
104  size_type max_size() const { return m_max_size; }
105 
106  /** @brief Put a value into the cache.
107  *
108  * If the value already exists, it is overwritten.
109  * When the size of the cache would grow below the
110  * maximal size, a random element is removed before
111  * putting the new one. */
112  template <typename ValueRef> Value const *put(Key const &k, ValueRef &&v) {
113  /* If there already is a value for k, overwriting it
114  * will not increase the size, so we don't have to
115  * make room. */
116  if ((m_cache.size() >= m_max_size) && !has(k))
117  drop_random_element();
118 
119  typename map_type::const_iterator it;
120  std::tie(it, std::ignore) = m_cache.emplace(k, std::forward<ValueRef>(v));
121 
122  return &(it->second);
123  }
124 
125  /** @brief Put a range of values into the cache.
126  *
127  * If the values already exists, it is overwritten.
128  * When the size of the cache would grow below the
129  * maximal size, a random elements are removed until
130  * all of the new values fit. If the given range is
131  * larger than max_size(), only the first max_size()
132  * elements are put into the cache.
133  *
134  * @tparam KeyInputIterator iterator of keys, at least InputIterator.
135  * @tparam ValueInputIterator iterator of value, at least InputIterator.
136  *
137  * @returns KeyInputIterator one past the last element that was put
138  * into the cache.
139  */
140  template <typename KeyInputIterator, typename ValueInputIterator>
141  KeyInputIterator put(KeyInputIterator kbegin, KeyInputIterator kend,
142  ValueInputIterator vbegin) {
143  auto const range_len = std::distance(kbegin, kend);
144  auto const len = (range_len > max_size()) ? max_size() : range_len;
145  kend = std::next(kbegin, len);
146 
147  /* Make some space. */
148  while ((max_size() - size()) < len) {
149  drop_random_element();
150  }
151 
152  while (kbegin != kend) {
153  put(*kbegin++, *vbegin++);
154  }
155 
156  return kend;
157  }
158 
159  /** @brief Get a value.
160  *
161  * The value is owned by the Cache and can not be modified.
162  * Pointers into the cache can be invalidated at any point
163  * and should not be stored beyond the calling function.
164  */
165  Value const *get(Key const &k) const {
166  auto const needle = m_cache.find(k);
167 
168  if (m_cache.end() != needle) {
169  return &(needle->second);
170  }
171  return nullptr;
172  }
173 };
174 } // namespace Utils
