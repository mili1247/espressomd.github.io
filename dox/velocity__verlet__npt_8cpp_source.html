46 void velocity_verlet_npt_propagate_vel_final(const ParticleRange &particles,
47  double time_step) {
48  extern IsotropicNptThermostat npt_iso;
49  nptiso.p_vel = {};
50 
51  for (auto &p : particles) {
52  // Virtual sites are not propagated during integration
53  if (p.p.is_virtual)
54  continue;
55  auto const noise = friction_therm0_nptiso<2>(npt_iso, p.m.v, p.p.identity);
56  for (int j = 0; j < 3; j++) {
57  if (!(p.p.ext_flag & COORD_FIXED(j))) {
58  if (nptiso.geometry & nptiso.nptgeom_dir[j]) {
59  nptiso.p_vel[j] += Utils::sqr(p.m.v[j] * time_step) * p.p.mass;
60  p.m.v[j] += (p.f.f[j] * time_step / 2.0 + noise[j]) / p.p.mass;
61  } else
62  // Propagate velocity: v(t+dt) = v(t+0.5*dt) + 0.5*dt * a(t+dt)
63  p.m.v[j] += p.f.f[j] * time_step / 2.0 / p.p.mass;
64  }
65  }
66  }
67 }

69 /** Scale and communicate instantaneous NpT pressure */
70 void velocity_verlet_npt_finalize_p_inst(double time_step) {
71  extern IsotropicNptThermostat npt_iso;
72  /* finalize derivation of p_inst */
73  nptiso.p_inst = 0.0;
74  for (int i = 0; i < 3; i++) {
75  if (nptiso.geometry & nptiso.nptgeom_dir[i]) {
76  nptiso.p_vel[i] /= Utils::sqr(time_step);
77  nptiso.p_inst += nptiso.p_vir[i] + nptiso.p_vel[i];
78  }
79  }
80 
81  double p_sum = 0.0;
82  boost::mpi::reduce(comm_cart, nptiso.p_inst, p_sum, std::plus<double>(), 0);
83  if (this_node == 0) {
84  nptiso.p_inst = p_sum / (nptiso.dimension * nptiso.volume);
85  nptiso.p_diff = nptiso.p_diff +
86  (nptiso.p_inst - nptiso.p_ext) * 0.5 * time_step +
87  friction_thermV_nptiso(npt_iso, nptiso.p_diff);
88  }
89 }

91 void velocity_verlet_npt_propagate_pos(const ParticleRange &particles,
92  double time_step) {
93  Utils::Vector3d scal{};
94  double L_new = 0.0;
95 
96  /* finalize derivation of p_inst */
97  velocity_verlet_npt_finalize_p_inst(time_step);
98 
99  /* adjust \ref NptIsoParameters::nptiso.volume; prepare pos- and
100  * vel-rescaling
101  */
102  if (this_node == 0) {
103  nptiso.volume += nptiso.inv_piston * nptiso.p_diff * 0.5 * time_step;
104  scal[2] = Utils::sqr(box_geo.length()[nptiso.non_const_dim]) /
105  pow(nptiso.volume, 2.0 / nptiso.dimension);
106  nptiso.volume += nptiso.inv_piston * nptiso.p_diff * 0.5 * time_step;
107  if (nptiso.volume < 0.0) {
108 
109  runtimeErrorMsg()
110  << "your choice of piston= " << nptiso.piston << ", dt= " << time_step
111  << ", p_diff= " << nptiso.p_diff
112  << " just caused the volume to become negative, decrease dt";
113  nptiso.volume = box_geo.volume();
114  scal[2] = 1;
115  }
116 
117  L_new = pow(nptiso.volume, 1.0 / nptiso.dimension);
118 
119  scal[1] = L_new * box_geo.length_inv()[nptiso.non_const_dim];
120  scal[0] = 1. / scal[1];
121  }
122  boost::mpi::broadcast(comm_cart, scal, 0);
123 
124  /* propagate positions while rescaling positions and velocities */
125  for (auto &p : particles) {
126  if (p.p.is_virtual)
127  continue;
128  for (int j = 0; j < 3; j++) {
129  if (!(p.p.ext_flag & COORD_FIXED(j))) {
130  if (nptiso.geometry & nptiso.nptgeom_dir[j]) {
131  {
132  p.r.p[j] = scal[1] * (p.r.p[j] + scal[2] * p.m.v[j] * time_step);
133  p.l.p_old[j] *= scal[1];
134  p.m.v[j] *= scal[0];
135  }
136  } else {
137  p.r.p[j] += p.m.v[j] * time_step;
138  }
139  }
140  }
141  }
142 
143  cell_structure.set_resort_particles(Cells::RESORT_LOCAL);
144 
145  /* Apply new volume to the box-length, communicate it, and account for
146  * necessary adjustments to the cell geometry */
147  Utils::Vector3d new_box;
148 
149  if (this_node == 0) {
150  new_box = box_geo.length();
151 
152  for (int i = 0; i < 3; i++) {
153  if (nptiso.geometry & nptiso.nptgeom_dir[i] || nptiso.cubic_box) {
154  new_box[i] = L_new;
155  }
156  }
157  }
158 
159  boost::mpi::broadcast(comm_cart, new_box, 0);
160 
161  box_geo.set_length(new_box);
162  // fast box length update
163  on_boxl_change(true);
164 }

166 void velocity_verlet_npt_propagate_vel(const ParticleRange &particles,
167  double time_step) {
168  extern IsotropicNptThermostat npt_iso;
169  nptiso.p_vel = {};
170 
171  for (auto &p : particles) {
172 #ifdef ROTATION
173  propagate_omega_quat_particle(p, time_step);
174 #endif
175 
176  // Don't propagate translational degrees of freedom of vs
177  if (p.p.is_virtual)
178  continue;
179  for (int j = 0; j < 3; j++) {
180  if (!(p.p.ext_flag & COORD_FIXED(j))) {
181  auto const noise =
182  friction_therm0_nptiso<1>(npt_iso, p.m.v, p.p.identity);
183  if (integ_switch == INTEG_METHOD_NPT_ISO &&
184  (nptiso.geometry & nptiso.nptgeom_dir[j])) {
185  p.m.v[j] += (p.f.f[j] * time_step / 2.0 + noise[j]) / p.p.mass;
186  nptiso.p_vel[j] += Utils::sqr(p.m.v[j] * time_step) * p.p.mass;
187  } else
188  // Propagate velocities: v(t+0.5*dt) = v(t) + 0.5*dt * a(t)
189  p.m.v[j] += p.f.f[j] * time_step / 2.0 / p.p.mass;
190  }
191  }
192  }
193 }
