79 void ek_init_boundaries() {
80 #if defined(CUDA) && defined(EK_BOUNDARIES)
81  int number_of_boundnodes = 0;
82 
83  std::vector<float> host_wallcharge_species_density;
84  float node_wallcharge = 0.0f;
85  int wallcharge_species = -1, charged_boundaries = 0;
86  bool node_charged = false;
87 
88  for (auto &lbboundary : lbboundaries) {
89  lbboundary->set_net_charge(0.0);
90  }
91 
92  if (ek_initialized) {
93  host_wallcharge_species_density.resize(ek_parameters.number_of_nodes);
94  for (auto &lbboundary : lbboundaries) {
95  if (lbboundary->charge_density() != 0.0) {
96  charged_boundaries = 1;
97  break;
98  }
99  }
100 
101  for (int n = 0; n < int(ek_parameters.number_of_species); n++)
102  if (ek_parameters.valency[n] != 0.0) {
103  wallcharge_species = n;
104  break;
105  }
106 
107  ek_gather_wallcharge_species_density(host_wallcharge_species_density.data(),
108  wallcharge_species);
109 
110  if (wallcharge_species == -1 && charged_boundaries) {
111  runtimeErrorMsg()
112  << "no charged species available to create wall charge\n";
113  }
114 
115  for (int z = 0; z < int(lbpar_gpu.dim[2]); z++) {
116  for (int y = 0; y < int(lbpar_gpu.dim[1]); y++) {
117  for (int x = 0; x < int(lbpar_gpu.dim[0]); x++) {
118  auto const pos = static_cast<double>(lbpar_gpu.agrid) *
119  (Utils::Vector3d{1. * x, 1. * y, 1. * z} +
120  Utils::Vector3d::broadcast(0.5));
121  node_charged = false;
122  node_wallcharge = 0.0f;
123 
124  std::vector<std::shared_ptr<LBBoundary>> boundaries;
125  std::copy_if(lbboundaries.begin(), lbboundaries.end(),
126  std::back_inserter(boundaries), [&pos](auto const lbb) {
127  return not lbb->shape().is_inside(pos);
128  });
129  for (auto lbb : boundaries) {
130  if ((*lbb).charge_density() != 0.0f) {
131  node_charged = true;
132  node_wallcharge += (*lbb).charge_density() * ek_parameters.agrid *
133  ek_parameters.agrid * ek_parameters.agrid;
134  (*lbb).set_net_charge(
135  (*lbb).net_charge() +
136  (*lbb).charge_density() * ek_parameters.agrid *
137  ek_parameters.agrid * ek_parameters.agrid);
138  }
139  }
140  if (not boundaries.empty()) {
141  number_of_boundnodes++;
142  }
143  ek_parameters.number_of_boundary_nodes = number_of_boundnodes;
144 
145  if (wallcharge_species != -1) {
146  if (node_charged)
147  host_wallcharge_species_density[ek_parameters.dim_y *
148  ek_parameters.dim_x * z +
149  ek_parameters.dim_x * y + x] =
150  node_wallcharge / ek_parameters.valency[wallcharge_species];
151  }
152  }
153  }
154  }
155  ek_init_species_density_wallcharge(host_wallcharge_species_density.data(),
156  wallcharge_species);
157  }
158 #endif
159 }

162 void lb_init_boundaries() {
163  if (lattice_switch == ActiveLB::GPU) {
164 #if defined(CUDA) && defined(LB_BOUNDARIES_GPU)
165  if (this_node != 0) {
166  return;
167  }
168  ek_init_boundaries();
169  unsigned number_of_boundnodes = 0;
170  std::vector<int> host_boundary_node_list;
171  std::vector<int> host_boundary_index_list;
172  std::size_t size_of_index;
173 
174  for (int z = 0; z < int(lbpar_gpu.dim[2]); z++) {
175  for (int y = 0; y < int(lbpar_gpu.dim[1]); y++) {
176  for (int x = 0; x < int(lbpar_gpu.dim[0]); x++) {
177  auto const pos = static_cast<double>(lbpar_gpu.agrid) *
178  (Utils::Vector3d{1. * x, 1. * y, 1. * z} +
179  Utils::Vector3d::broadcast(0.5));
180 
181  // take last boundary containing the node
182  auto const boundary = boost::find_if(
183  lbboundaries | boost::adaptors::reversed, [&pos](auto const lbb) {
184  return not lbb->shape().is_inside(pos);
185  });
186 
187  if (boundary != boost::rend(lbboundaries)) {
188  size_of_index = (number_of_boundnodes + 1) * sizeof(int);
189  host_boundary_node_list.resize(size_of_index);
190  host_boundary_index_list.resize(size_of_index);
191  host_boundary_node_list[number_of_boundnodes] =
192  x + lbpar_gpu.dim[0] * y +
193  lbpar_gpu.dim[0] * lbpar_gpu.dim[1] * z;
194  auto const boundary_number =
195  std::distance(lbboundaries.begin(), boundary.base()) - 1;
196  host_boundary_index_list[number_of_boundnodes] =
197  boundary_number + 1;
198  number_of_boundnodes++;
199  }
200  }
201  }
202  }
203  lbpar_gpu.number_of_boundnodes = number_of_boundnodes;
204  /* call of cuda fkt */
205  std::vector<float> boundary_velocity(3 * (lbboundaries.size() + 1));
206  int n = 0;
207  for (auto lbb = lbboundaries.begin(); lbb != lbboundaries.end();
208  ++lbb, n++) {
209  boundary_velocity[3 * n + 0] = static_cast<float>((**lbb).velocity()[0]);
210  boundary_velocity[3 * n + 1] = static_cast<float>((**lbb).velocity()[1]);
211  boundary_velocity[3 * n + 2] = static_cast<float>((**lbb).velocity()[2]);
212  }
213 
214  boundary_velocity[3 * lbboundaries.size() + 0] = 0.0f;
215  boundary_velocity[3 * lbboundaries.size() + 1] = 0.0f;
216  boundary_velocity[3 * lbboundaries.size() + 2] = 0.0f;
217 
218  lb_init_boundaries_GPU(lbboundaries.size(), number_of_boundnodes,
219  host_boundary_node_list.data(),
220  host_boundary_index_list.data(),
221  boundary_velocity.data());
222 
223 #endif /* defined (CUDA) && defined (LB_BOUNDARIES_GPU) */
224  } else if (lattice_switch == ActiveLB::CPU) {
225 #if defined(LB_BOUNDARIES)
226  auto const lblattice = lb_lbfluid_get_lattice();
227 
228  boost::for_each(lbfields, [](auto &f) { f.boundary = 0; });
229 
230  auto const node_pos = calc_node_pos(comm_cart);
231  auto const offset = Utils::hadamard_product(node_pos, lblattice.grid);
232 
233  for (int z = 0; z < lblattice.grid[2] + 2; z++) {
234  for (int y = 0; y < lblattice.grid[1] + 2; y++) {
235  for (int x = 0; x < lblattice.grid[0] + 2; x++) {
236  auto const pos =
237  (offset + Utils::Vector3d{x - 0.5, y - 0.5, z - 0.5}) *
238  lblattice.agrid;
239 
240  auto const boundary = boost::find_if(
241  lbboundaries | boost::adaptors::reversed, [&pos](auto const lbb) {
242  return not lbb->shape().is_inside(pos);
243  });
244  if (boundary != boost::rend(lbboundaries)) {
245  auto const index = get_linear_index(x, y, z, lblattice.halo_grid);
246  auto const boundary_number =
247  std::distance(lbboundaries.begin(), boundary.base()) - 1;
248  auto &node = lbfields[index];
249  node.boundary = static_cast<int>(boundary_number) + 1;
250  node.slip_velocity =
251  (*boundary)->velocity() *
252  (lb_lbfluid_get_tau() / lb_lbfluid_get_agrid());
253  } else {
254  lbfields[get_linear_index(x, y, z, lblattice.halo_grid)].boundary =
255  0;
256  }
257  }
258  }
259  }
260 #endif
261  }

272 Utils::Vector3d lbboundary_get_force(LBBoundary const *lbb) {
273  Utils::Vector3d force{};
274 #if defined(LB_BOUNDARIES) || defined(LB_BOUNDARIES_GPU)
275  auto const it =
276  boost::find_if(lbboundaries, [lbb](std::shared_ptr<LBBoundary> const &i) {
277  return i.get() == lbb;
278  });
279  if (it == lbboundaries.end())
280  throw std::runtime_error("You probably tried to get the force of an "
281  "lbboundary that was not added to "
282  "system.lbboundaries.");
283  std::vector<double> forces(3 * lbboundaries.size());
284  if (lattice_switch == ActiveLB::GPU) {
285 #if defined(LB_BOUNDARIES_GPU) && defined(CUDA)
286  lb_gpu_get_boundary_forces(forces);
287 #endif
288  } else if (lattice_switch == ActiveLB::CPU) {
289 #if defined(LB_BOUNDARIES)
290  mpi_call(lb_collect_boundary_forces_local, -1, -1);
291  lb_collect_boundary_forces(forces.data());
292 #endif
293  }
294  auto const container_index = std::distance(lbboundaries.begin(), it);
295  force[0] = forces[3 * container_index + 0];
296  force[1] = forces[3 * container_index + 1];
297  force[2] = forces[3 * container_index + 2];
298 #endif
299  return force;
300 }
