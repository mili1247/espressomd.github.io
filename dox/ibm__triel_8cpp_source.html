37 /** Rotate calculated trielastic forces in the 2d plane back to the 3d plane.
38  * Use knowledge that the x-axis in rotated system is parallel to r(p1->p2) in
39  * original system; To find the corresponding unit vector to y in the rotated
40  * system, construct vector perpendicular to r(p1->p2); note that f3 is not
41  * calculated here but is implicitly calculated by f3 = -(f1+f2) which is
42  * consistent with the literature
43  * @return forces on particles 1, 2, 3
44  */
45 std::tuple<Utils::Vector3d, Utils::Vector3d, Utils::Vector3d>
46 RotateForces(Utils::Vector2d const &f1_rot, Utils::Vector2d const &f2_rot,
47  Utils::Vector3d const &v12, Utils::Vector3d const &v13) {
48  // fRot is in the rotated system, i.e. in a system where the side lPrime of
49  // the triangle (i.e. v12) is parallel to the x-axis, and the y-axis is
50  // perpendicular to the x-axis (cf. Krueger, Fig. 7.1c).
51  // I.e. fRot[XX] is the component parallel to the x-axis, fRot[YY]] the
52  // component parallel to the y-axis. Now, the idea is to get these unit
53  // vectors for the x- and y-axis in the real coordinate system. They are named
54  // xu and yu below. The x-component of the force in the real coordinate system
55  // is therefore: fRot[XX]*xu
56 
57  // xu is simple: The x-axis in the rotated system is parallel to v12 --> xu =
58  // v12 (+ normalization)
59  auto const xu = Utils::Vector3d(v12).normalize();
60 
61  // yu needs to be orthogonal to xu, and point in the direction of node 3 in
62  // Krueger, Fig. 7.1b. Therefore: First get the projection of v13 onto v12:
63  // The direction is defined by xu, the length by the scalar product (scalar
64  // product can be interpreted as a projection, after all). --> sca * xu Then:
65  // v13 - sca * xu gives the component of v13 orthogonal to v12, i.e.
66  // perpendicular to the x-axis --> yu Last: Normalize yu.
67  auto const yu = (v13 - (v13 * xu) * xu).normalize();
68 
69  // Calculate forces in 3D
70  auto const force1 = f1_rot[0] * xu + f1_rot[1] * yu;
71  auto const force2 = f2_rot[0] * xu + f2_rot[1] * yu;
72  auto const force3 = -(force1 + force2);
73  return std::make_tuple(force1, force2, force3);
74 }

77 boost::optional<std::tuple<Utils::Vector3d, Utils::Vector3d, Utils::Vector3d>>
78 IBMTriel::calc_forces(Particle const &p1, Particle const &p2,
79  Particle const &p3) const {
80 
81  // Calculate the current shape of the triangle (l,lp,cos(phi),sin(phi));
82  // l = length between 1 and 3
83  // get_mi_vector is an ESPResSo function which considers PBC
84  auto const vec2 = box_geo.get_mi_vector(p3.r.p, p1.r.p);
85  auto const l = vec2.norm();
86 
87  // lp = length between 1 and 2
88  auto const vec1 = box_geo.get_mi_vector(p2.r.p, p1.r.p);
89  auto const lp = vec1.norm();
90 
91  // Check for sanity
92  if ((lp - lp0 > maxDist) || (l - l0 > maxDist)) {
93  return {};
94  }
95 
96  // angles between these vectors; calculated directly via the products
97  auto const cosPhi = (vec1 * vec2) / (lp * l);
98  auto const vecpro = vector_product(vec1, vec2);
99  auto const sinPhi = vecpro.norm() / (l * lp);
100 
101  // Displacement gradient tensor D: eq. (C.9) in @cite kruger12a
102  const double Dxx = lp / lp0;
103  const double Dxy = ((l / l0 * cosPhi) - (lp / lp0 * cosPhi0)) / sinPhi0;
104  const double Dyx = 0.0;
105  const double Dyy = l / l0 * sinPhi / sinPhi0;
106 
107  // Tensor G: eq. (C.12)
108  const double Gxx = Utils::sqr(Dxx) + Utils::sqr(Dyx);
109  const double Gxy = Dxx * Dxy + Dyx * Dyy;
110  const double Gyx = Dxx * Dxy + Dyy * Dyx; // = Gxy because of symmetry
111  const double Gyy = Utils::sqr(Dxy) + Utils::sqr(Dyy);
112 
113  // Strain invariants: eq. (C.11) and (C.12)
114  const double i1 = (Gxx + Gyy) - 2;
115  const double i2 = ((Gxx * Gyy) - (Gxy * Gyx)) - 1;
116 
117  // Derivatives of energy density E used in chain rule below: eq. (C.14)
118  double dEdI1;
119  double dEdI2;
120  if (elasticLaw == tElasticLaw::NeoHookean) {
121  // Neo-Hookean
122  dEdI1 = k1 / 6.0;
123  dEdI2 = -k1 / (6.0 * (i2 + 1.0) * (i2 + 1.0));
124  } else {
125  // Skalak
126  dEdI1 = k1 * (i1 + 1) / 6.0;
127  dEdI2 = -k1 / 6.0 + k2 * i2 / 6.0;
128  }
129 
130  // Derivatives of Is: eq. (C.15)
131  const double dI1dGxx = 1;
132  const double dI1dGxy = 0;
133  const double dI1dGyx = 0;
134  const double dI1dGyy = 1;
135 
136  const double dI2dGxx = Gyy;
137  const double dI2dGxy = -Gyx; // Note: Krueger has a factor 2 here, because he
138  // uses the symmetry of the G-matrix.
139  const double dI2dGyx = -Gxy; // But we don't use it. So, Krueger is missing
140  // the yx term, whereas we have it.
141  const double dI2dGyy = Gxx;
142 
143  // Derivatives of G: eq. (C.16)
144  const double dGxxdV1x = 2 * a1 * Dxx;
145  const double dGxxdV1y = 0;
146  const double dGxxdV2x = 2 * a2 * Dxx;
147  const double dGxxdV2y = 0;
148 
149  const double dGxydV1x = a1 * Dxy + b1 * Dxx;
150  const double dGxydV1y = a1 * Dyy;
151  const double dGxydV2x = a2 * Dxy + b2 * Dxx;
152  const double dGxydV2y = a2 * Dyy;
153 
154  const double dGyxdV1x = a1 * Dxy + b1 * Dxx;
155  const double dGyxdV1y = a1 * Dyy;
156  const double dGyxdV2x = a2 * Dxy + b2 * Dxx;
157  const double dGyxdV2y = a2 * Dyy;
158 
159  const double dGyydV1x = 2 * b1 * Dxy;
160  const double dGyydV1y = 2 * b1 * Dyy;
161  const double dGyydV2x = 2 * b2 * Dxy;
162  const double dGyydV2y = 2 * b2 * Dyy;
163 
164  // Calculate forces per area in rotated system: chain rule as in appendix C of
165  // Krüger (chain rule applied in eq. (C.13), but for the energy density). Only
166  // two nodes are needed, third one is calculated from momentum conservation
167  // Note: If you calculate the derivatives in a straightforward manner, you get
168  // 8 terms (done here). Krueger exploits the symmetry of the G-matrix, which
169  // results in 6 elements, but with an additional factor 2 for the xy-elements
170  // (see also above at the definition of dI2dGxy).
171  Utils::Vector2d f1_rot{};
172  Utils::Vector2d f2_rot{};
173  f1_rot[0] = -(dEdI1 * dI1dGxx * dGxxdV1x) - (dEdI1 * dI1dGxy * dGxydV1x) -
174  (dEdI1 * dI1dGyx * dGyxdV1x) - (dEdI1 * dI1dGyy * dGyydV1x) -
175  (dEdI2 * dI2dGxx * dGxxdV1x) - (dEdI2 * dI2dGxy * dGxydV1x) -
176  (dEdI2 * dI2dGyx * dGyxdV1x) - (dEdI2 * dI2dGyy * dGyydV1x);
177  f1_rot[1] = -(dEdI1 * dI1dGxx * dGxxdV1y) - (dEdI1 * dI1dGxy * dGxydV1y) -
178  (dEdI1 * dI1dGyx * dGyxdV1y) - (dEdI1 * dI1dGyy * dGyydV1y) -
179  (dEdI2 * dI2dGxx * dGxxdV1y) - (dEdI2 * dI2dGxy * dGxydV1y) -
180  (dEdI2 * dI2dGyx * dGyxdV1y) - (dEdI2 * dI2dGyy * dGyydV1y);
181  f2_rot[0] = -(dEdI1 * dI1dGxx * dGxxdV2x) - (dEdI1 * dI1dGxy * dGxydV2x) -
182  (dEdI1 * dI1dGyx * dGyxdV2x) - (dEdI1 * dI1dGyy * dGyydV2x) -
183  (dEdI2 * dI2dGxx * dGxxdV2x) - (dEdI2 * dI2dGxy * dGxydV2x) -
184  (dEdI2 * dI2dGyx * dGyxdV2x) - (dEdI2 * dI2dGyy * dGyydV2x);
185  f2_rot[1] = -(dEdI1 * dI1dGxx * dGxxdV2y) - (dEdI1 * dI1dGxy * dGxydV2y) -
186  (dEdI1 * dI1dGyx * dGyxdV2y) - (dEdI1 * dI1dGyy * dGyydV2y) -
187  (dEdI2 * dI2dGxx * dGxxdV2y) - (dEdI2 * dI2dGxy * dGxydV2y) -
188  (dEdI2 * dI2dGyx * dGyxdV2y) - (dEdI2 * dI2dGyy * dGyydV2y);
189 
190  // Multiply by undeformed area
191  f1_rot *= area0;
192  f2_rot *= area0;
193 
194  // Rotate forces back into original position of triangle
195  auto forces = RotateForces(f1_rot, f2_rot, vec1, vec2);
196 
197  return forces;
198 }

200 IBMTriel::IBMTriel(const int ind1, const int ind2, const int ind3,
201  const double maxDist, const tElasticLaw elasticLaw,
202  const double k1, const double k2) {
203 
204  // Get data (especially location) of three particles
205  auto part1 = get_particle_data(ind1);
206  auto part2 = get_particle_data(ind2);
207  auto part3 = get_particle_data(ind3);
208 
209  // Calculate equilibrium lengths and angle; Note the sequence of the points!
210  // l0 = length between 1 and 3
211  auto const temp_l0 = box_geo.get_mi_vector(part3.r.p, part1.r.p);
212  l0 = temp_l0.norm();
213  // lp0 = length between 1 and 2
214  auto const temp_lp0 = box_geo.get_mi_vector(part2.r.p, part1.r.p);
215  lp0 = temp_lp0.norm();
216 
217  // cospo / sinpo angle functions between these vectors; calculated directly
218  // via the products
219  cosPhi0 = (temp_l0 * temp_lp0) / (l0 * lp0);
220  auto const vecpro = vector_product(temp_l0, temp_lp0);
221  sinPhi0 = vecpro.norm() / (l0 * lp0);
222 
223  // Use the values determined above for further constants of the stretch-force
224  // calculation
225  const double area2 = l0 * lp0 * sinPhi0;
226  a1 = -(l0 * sinPhi0) / area2;
227  a2 = -a1;
228  b1 = (l0 * cosPhi0 - lp0) / area2;
229  b2 = -(l0 * cosPhi0) / area2;
230  area0 = 0.5 * area2;
231  this->maxDist = maxDist;
232  this->elasticLaw = elasticLaw;
233  // Always store two constants, for NeoHookean only k1 is used
234  this->k1 = k1;
235  this->k2 = k2;
236 }
