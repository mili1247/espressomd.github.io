31 /** Evaluate forces and energies using a custom potential profile.
32  *
33  * Forces and energies are evaluated by linear interpolation. The curves
34  * @ref force_tab and @ref energy_tab must be sampled uniformly between
35  * @ref minval and @ref maxval.
36  */
37 struct TabulatedPotential {
38  /** Position on the x-axis of the first tabulated value. */
39  double minval = -1.0;
40  /** Position on the x-axis of the last tabulated value. */
41  double maxval = -1.0;
42  /** %Distance on the x-axis between tabulated values. */
43  double invstepsize = 0.0;
44  /** Tabulated forces. */
45  std::vector<double> force_tab;
46  /** Tabulated energies. */
47  std::vector<double> energy_tab;
48 
49  /** Evaluate the force at position @p x.
50  * @param x Bond length/angle
51  * @return Interpolated force.
52  */
53  double force(double x) const {
54  using boost::algorithm::clamp;
55  return Utils::linear_interpolation(force_tab, invstepsize, minval,
56  clamp(x, minval, maxval));
57  }
58 
59  /** Evaluate the energy at position @p x.
60  * @param x Bond length/angle
61  * @return Interpolated energy.
62  */
63  double energy(double x) const {
64  using boost::algorithm::clamp;
65  return Utils::linear_interpolation(energy_tab, invstepsize, minval,
66  clamp(x, minval, maxval));
67  }
68 
69  double cutoff() const { return maxval; }
70 
71 private:
72  friend boost::serialization::access;
73  template <typename Archive>
74  void serialize(Archive &ar, long int /* version */) {
75  ar &minval;
76  ar &maxval;
77  ar &invstepsize;
78  ar &force_tab;
79  ar &energy_tab;
80  }
81 };
