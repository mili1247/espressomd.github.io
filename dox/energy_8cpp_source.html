46 static std::shared_ptr<Observable_stat> calculate_energy_local() {
47 
48  auto obs_energy_ptr = std::make_shared<Observable_stat>(1);
49 
50  if (!interactions_sanity_checks())
51  return obs_energy_ptr;
52 
53  auto &obs_energy = *obs_energy_ptr;
54 
55 #ifdef CUDA
56  clear_energy_on_GPU();
57 #endif
58 
59  EspressoSystemInterface::Instance().update();
60 
61  // Compute the energies from the energyActors
62  for (auto &energyActor : energyActors)
63  energyActor->computeEnergy(espressoSystemInterface);
64 
65  on_observable_calc();
66 
67  auto const local_parts = cell_structure.local_particles();
68 
69  for (auto const &p : local_parts) {
70  obs_energy.kinetic[0] += calc_kinetic_energy(p);
71  }
72 
73  short_range_loop(
74  [&obs_energy](Particle const &p1, int bond_id,
75  Utils::Span<Particle *> partners) {
76  auto const &iaparams = bonded_ia_params[bond_id];
77  auto const result = calc_bonded_energy(iaparams, p1, partners);
78  if (result) {
79  obs_energy.bonded_contribution(bond_id)[0] += result.get();
80  return false;
81  }
82  return true;
83  },
84  [&obs_energy](Particle const &p1, Particle const &p2, Distance const &d) {
85  add_non_bonded_pair_energy(p1, p2, d.vec21, sqrt(d.dist2), d.dist2,
86  obs_energy);
87  },
88  maximal_cutoff(), maximal_cutoff_bonded());
89 
90 #ifdef ELECTROSTATICS
91  /* calculate k-space part of electrostatic interaction. */
92  obs_energy.coulomb[1] = Coulomb::calc_energy_long_range(local_parts);
93 #endif
94 
95 #ifdef DIPOLES
96  /* calculate k-space part of magnetostatic interaction. */
97  obs_energy.dipolar[1] = Dipole::calc_energy_long_range(local_parts);
98 #endif
99 
100  Constraints::constraints.add_energy(local_parts, get_sim_time(), obs_energy);
101 
102 #ifdef CUDA
103  auto const energy_host = copy_energy_from_GPU();
104  if (!obs_energy.coulomb.empty())
105  obs_energy.coulomb[1] += energy_host.coulomb;
106  if (!obs_energy.dipolar.empty())
107  obs_energy.dipolar[1] += energy_host.dipolar;
108 #endif
109 
110  obs_energy.mpi_reduce();
111  return obs_energy_ptr;
112 }
