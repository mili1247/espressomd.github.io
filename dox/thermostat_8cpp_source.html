99 void thermo_init(double time_step) {
100  // initialize thermalized bond regardless of the current thermostat
101  if (n_thermalized_bonds) {
102  thermalized_bond_init(time_step);
103  }
104  if (thermo_switch == THERMO_OFF) {
105  return;
106  }
107  if (thermo_switch & THERMO_LANGEVIN)
108  langevin.recalc_prefactors(temperature, time_step);
109 #ifdef DPD
110  if (thermo_switch & THERMO_DPD)
111  dpd_init(temperature, time_step);
112 #endif
113 #ifdef NPT
114  if (thermo_switch & THERMO_NPT_ISO) {
115  npt_iso.recalc_prefactors(temperature, nptiso.piston, time_step);
116  }
117 #endif
118  if (thermo_switch & THERMO_BROWNIAN)
119  brownian.recalc_prefactors(temperature);
120 }
