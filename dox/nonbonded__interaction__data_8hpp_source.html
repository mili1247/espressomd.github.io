300 /**
301  * @brief Get interaction parameters between particle types i and j
302  *
303  * This is symmetric, e.g. it holds that get_ia_param(i, j) and
304  * get_ia_param(j, i) point to the same data.
305  *
306  * @param i First type, has to be smaller than @ref max_seen_particle_type.
307  * @param j Second type, has to be smaller than @ref max_seen_particle_type.
308  *
309  * @return Pointer to interaction parameters for the type pair.
310  */
311 inline IA_parameters *get_ia_param(int i, int j) {
312  assert(i >= 0 && i < max_seen_particle_type);
313  assert(j >= 0 && j < max_seen_particle_type);
314 
315  return &ia_params[Utils::upper_triangular(std::min(i, j), std::max(i, j),
316  max_seen_particle_type)];
317 }
