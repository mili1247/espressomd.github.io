37 void CellStructure::check_particle_index() {
38  auto const max_id = get_max_local_particle_id();
39 
40  for (auto const &p : local_particles()) {
41  auto const id = p.identity();
42 
43  if (id < 0 || id > max_id) {
44  throw std::runtime_error("Particle id out of bounds.");
45  }
46 
47  if (get_local_particle(id) != &p) {
48  throw std::runtime_error("Invalid local particle index entry.");
49  }
50  }
51 
52  /* checks: local particle id */
53  int local_part_cnt = 0;
54  for (int n = 0; n < get_max_local_particle_id() + 1; n++) {
55  if (get_local_particle(n) != nullptr) {
56  local_part_cnt++;
57  if (get_local_particle(n)->p.identity != n) {
58  throw std::runtime_error("local_particles part has corrupted id.");
59  }
60  }
61  }
62 
63  if (local_part_cnt != local_particles().size()) {
64  throw std::runtime_error(
65  std::to_string(local_particles().size()) + " parts in cells but " +
66  std::to_string(local_part_cnt) + " parts in local_particles");
67  }
68 }

85 void CellStructure::remove_particle(int id) {
86  auto remove_all_bonds_to = [id](BondList &bl) {
87  for (auto it = bl.begin(); it != bl.end();) {
88  if (Utils::contains(it->partner_ids(), id)) {
89  it = bl.erase(it);
90  } else {
91  std::advance(it, 1);
92  }
93  }
94  };
95 
96  for (auto c : decomposition().local_cells()) {
97  auto &parts = c->particles();
98 
99  for (auto it = parts.begin(); it != parts.end();) {
100  if (it->identity() == id) {
101  it = parts.erase(it);
102  update_particle_index(id, nullptr);
103  update_particle_index(parts);
104  } else {
105  remove_all_bonds_to(it->bonds());
106  it++;
107  }
108  }
109  }
110 }

123 Particle *CellStructure::add_particle(Particle &&p) {
124  auto const sort_cell = particle_to_cell(p);
125  /* There is always at least one cell, so if the particle
126  * does not belong to a cell on this node we can put it there. */
127  auto cell = sort_cell ? sort_cell : local_cells()[0];
128 
129  /* If the particle isn't local a global resort may be
130  * needed, otherwise a local resort if sufficient. */
131  set_resort_particles(sort_cell ? Cells::RESORT_LOCAL : Cells::RESORT_GLOBAL);
132 
133  return std::addressof(
134  append_indexed_particle(cell->particles(), std::move(p)));
135 }
