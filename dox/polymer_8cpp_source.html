52 template <class RNG> static Utils::Vector3d random_position(RNG &rng) {
53  Utils::Vector3d v;
54  for (int i = 0; i < 3; ++i)
55  v[i] = box_geo.length()[i] * rng();
56  return v;

59 template <class RNG> static Utils::Vector3d random_unit_vector(RNG &rng) {
60  Utils::Vector3d v;
61  double const phi = acos(1. - 2. * rng());
62  double const theta = 2. * Utils::pi() * rng();
63  v[0] = sin(phi) * cos(theta);
64  v[1] = sin(phi) * sin(theta);
65  v[2] = cos(phi);
66  v /= v.norm();
67  return v;
68 }

70 /** Determines whether a given position @p pos is valid, i.e., it doesn't
71  * collide with existing or buffered particles, nor with existing constraints
72  * (if @c respect_constraints).
73  * @param pos the trial position in question
74  * @param positions buffered positions to respect
75  * @param partCfg existing particles to respect
76  * @param min_distance threshold for the minimum distance between
77  * trial position and buffered/existing particles
78  * @param respect_constraints whether to respect constraints
79  * @return true if valid position, false if not.
80  */
81 static bool
82 is_valid_position(Utils::Vector3d const &pos,
83  std::vector<std::vector<Utils::Vector3d>> const &positions,
84  PartCfg &partCfg, double const min_distance,
85  int const respect_constraints) {
86  // check if constraint is violated
87  if (respect_constraints) {
88  Utils::Vector3d const folded_pos = folded_position(pos, box_geo);
89 
90  for (auto &c : Constraints::constraints) {
91  auto cs =
92  std::dynamic_pointer_cast<const Constraints::ShapeBasedConstraint>(c);
93  if (cs) {
94  double d;
95  Utils::Vector3d v;
96 
97  cs->calc_dist(folded_pos, d, v);
98 
99  if (d <= 0) {
100  return false;
101  }
102  }
103  }
104  }
105 
106  if (min_distance > 0) {
107  // check for collision with existing particles
108  if (distto(partCfg, pos, -1) < min_distance) {
109  return false;
110  }
111 
112  for (auto const &p : positions) {
113  for (auto const &m : p) {
114  if (box_geo.get_mi_vector(pos, m).norm() < min_distance) {
115  return false;
116  }
117  }
118  }
119  }
120  return true;
121 }

123 std::vector<std::vector<Utils::Vector3d>>
124 draw_polymer_positions(PartCfg &partCfg, int const n_polymers,
125  int const beads_per_chain, double const bond_length,
126  std::vector<Utils::Vector3d> const &start_positions,
127  double const min_distance, int const max_tries,
128  int const use_bond_angle, double const bond_angle,
129  int const respect_constraints, int const seed) {
130  auto rng = [mt = Random::mt19937(static_cast<unsigned>(seed)),
131  dist = std::uniform_real_distribution<double>(
132  0.0, 1.0)]() mutable { return dist(mt); };
133 
134  std::vector<std::vector<Utils::Vector3d>> positions(n_polymers);
135  for (auto &p : positions) {
136  p.reserve(beads_per_chain);
137  }
138 
139  auto is_valid_pos = [&positions, &partCfg, min_distance,
140  respect_constraints](Utils::Vector3d const &v) {
141  return is_valid_position(v, positions, partCfg, min_distance,
142  respect_constraints);
143  };
144 
145  for (std::size_t p = 0; p < start_positions.size(); p++) {
146  if (is_valid_pos(start_positions[p])) {
147  positions[p].push_back(start_positions[p]);
148  } else {
149  throw std::runtime_error("Invalid start positions.");
150  }
151  }
152 
153  /* Draw a monomer position, obeying angle and starting position
154  * constraints where appropriate. */
155  auto draw_monomer_position = [&](int p, int m) {
156  if (m == 0) {
157  return (p < start_positions.size()) ? start_positions[p]
158  : random_position(rng);
159  }
160 
161  if (not use_bond_angle or m < 2) {
162  return positions[p][m - 1] + bond_length * random_unit_vector(rng);
163  }
164 
165  auto const last_vec = positions[p][m - 1] - positions[p][m - 2];
166  return positions[p][m - 1] +
167  Utils::vec_rotate(vector_product(last_vec, random_unit_vector(rng)),
168  bond_angle, -last_vec);
169  };
170 
171  /* Try up to max_tries times to draw a valid position */
172  auto draw_valid_monomer_position =
173  [&](int p, int m) -> boost::optional<Utils::Vector3d> {
174  for (unsigned _ = 0; _ < max_tries; _++) {
175  auto const trial_pos = draw_monomer_position(p, m);
176 
177  if (is_valid_pos(trial_pos)) {
178  return trial_pos;
179  }
180  }
181 
182  return {};
183  };
184 
185  // create remaining monomers' positions by backtracking.
186  for (int p = 0; p < n_polymers; ++p) {
187  for (int attempts_poly = 0; attempts_poly < max_tries; attempts_poly++) {
188  int rejections = 0;
189  while (positions[p].size() < beads_per_chain) {
190  auto pos = draw_valid_monomer_position(p, positions[p].size());
191 
192  if (pos) {
193  /* Move on one position */
194  positions[p].push_back(*pos);
195  } else if (not positions[p].empty()) {
196  /* Go back one position and try again */
197  positions[p].pop_back();
198  rejections++;
199  if (rejections > max_tries) {
200  /* Give up for this try. */
201  break;
202  }
203  } else {
204  /* Give up for this try. */
205  break;
206  }
207  }
208 
209  /* If the polymer has not full length, we try again. Otherwise we can
210  * move on to the next polymer. */
211  if (positions[p].size() == beads_per_chain) {
212  break;
213  }
214  }
215 
216  /* We did not get a complete polymer, but have exceeded the maximal
217  * number of tries, which means failure. */
218  if (positions[p].size() < beads_per_chain)
219  throw std::runtime_error("Failed to create polymer positions.");
220  }
221  return positions;
222 }
