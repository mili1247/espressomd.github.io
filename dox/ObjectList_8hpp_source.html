42 template <

44  class = std::enable_if_t<std::is_base_of<ObjectHandle, ManagedType>::value>>
45 class ObjectList : public BaseType {
46 private:

48  virtual void remove_in_core(const std::shared_ptr<ManagedType> &obj_ptr) = 0;

50 public:
51  /**
52  * @brief Add an element to the list.
53  *
54  * @param element The element to add.
55  */
56  void add(std::shared_ptr<ManagedType> const &element) {
57  add_in_core(element);
58  m_elements.push_back(element);
59  }

61  /**

66  void remove(std::shared_ptr<ManagedType> const &element) {
67  remove_in_core(element);
68  m_elements.erase(std::remove(m_elements.begin(), m_elements.end(), element),
69  m_elements.end());
70  }

77  /**

79  */
80  void clear() {

83  }
84 
85  m_elements.clear();
86  }

153  }
154  }
