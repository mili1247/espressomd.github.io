23 /** permute an integer array field of size size about permute positions. */
24 inline void permute_ifield(int *field, int size, int permute) {
25  if (permute == 0)
26  return;
27  if (permute < 0)
28  permute = (size + permute);
29  while (permute > 0) {
30  int tmp = field[0];
31  for (int i = 1; i < size; i++)
32  field[i - 1] = field[i];
33  field[size - 1] = tmp;
34  permute--;
35  }
36 }
