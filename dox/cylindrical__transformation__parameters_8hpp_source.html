30 /**
31  * @brief A class to hold and validate parameters for a cylindrical coordinate
32  * transformations.
33  *
34  * @param center The origin of the cylindrical coordinates.
35  * @param axis The "z"-axis. Must be normalized.
36  * @param orientation The axis along which phi = 0. Must be normalized and
37  * orthogonal to axis.
38  */
39 class CylindricalTransformationParameters {
40 public:
41  CylindricalTransformationParameters() = default;
42  CylindricalTransformationParameters(Utils::Vector3d const &center,
43  Utils::Vector3d const &axis,
44  Utils::Vector3d const &orientation)
45  : m_center(center), m_axis(axis), m_orientation(orientation) {
46  validate();
47  }
48  /**
49  * @brief if you only provide center and axis, an orientation will be
50  * generated automatically such that it is orthogonal to axis
51  */
52  CylindricalTransformationParameters(Utils::Vector3d const &center,
53  Utils::Vector3d const &axis)
54  : m_center(center), m_axis(axis),
55  m_orientation(Utils::calc_orthonormal_vector(axis)) {}
56 
57  Utils::Vector3d center() const { return m_center; }
58  Utils::Vector3d axis() const { return m_axis; }
59  Utils::Vector3d orientation() const { return m_orientation; }
60 
61 private:
62  void validate() const {
63  auto constexpr eps = 10 * std::numeric_limits<double>::epsilon();
64  if (Utils::abs(m_orientation * m_axis) > eps) {
65  throw std::runtime_error(
66  "CylindricalTransformationParameters: Axis and orientation must be "
67  "orthogonal. Scalar product is " +
68  std::to_string(m_orientation * m_axis));
69  }
70  if (Utils::abs(m_axis.norm() - 1) > eps) {
71  throw std::runtime_error("CylindricalTransformationParameters: Axis must "
72  "be normalized. Norm is " +
73  std::to_string(m_axis.norm()));
74  }
75  if (Utils::abs(m_orientation.norm() - 1) > eps) {
76  throw std::runtime_error("CylindricalTransformationParameters: "
77  "orientation must be normalized. Norm is " +
78  std::to_string(m_orientation.norm()));
79  }
80  }
81 
82  const Utils::Vector3d m_center{};
83  const Utils::Vector3d m_axis{0, 0, 1};
84  const Utils::Vector3d m_orientation{1, 0, 0};
85 };
