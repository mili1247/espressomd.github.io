27 double factorial_Ni0_divided_by_factorial_Ni0_plus_nu_i(int Ni0, int nu_i) {
28  double value = 1.0;
29  if (nu_i) {
30  if (nu_i > 0) {
31  for (int i = 1; i <= nu_i; i++) {
32  value /= Ni0 + i;
33  }
34  } else {
35  auto const abs_nu_i = std::abs(nu_i);
36  for (int i = 0; i < abs_nu_i; i++) {
37  value *= Ni0 - i;
38  }
39  }
40  }
41  return value;
42 }

45 calculate_factorial_expression(SingleReaction const &current_reaction,
46  std::map<int, int> const &old_particle_numbers) {
47  double factorial_expr = 1.0;
48  // factorial contribution of reactants
49  for (int i = 0; i < current_reaction.reactant_types.size(); i++) {
50  int nu_i = -1 * current_reaction.reactant_coefficients[i];
51  int N_i0 = old_particle_numbers.at(current_reaction.reactant_types[i]);
52  factorial_expr *= factorial_Ni0_divided_by_factorial_Ni0_plus_nu_i(
53  N_i0, nu_i); // zeta = 1 (see @cite smith94c)
54  // since we only perform one reaction
55  // at one call of the function
56  }
57  // factorial contribution of products
58  for (int i = 0; i < current_reaction.product_types.size(); i++) {
59  int nu_i = current_reaction.product_coefficients[i];
60  int N_i0 = old_particle_numbers.at(current_reaction.product_types[i]);
61  factorial_expr *= factorial_Ni0_divided_by_factorial_Ni0_plus_nu_i(
62  N_i0, nu_i); // zeta = 1 (see @cite smith94c)
63  // since we only perform one reaction
64  // at one call of the function
65  }
66  return factorial_expr;
67 }
