296 /**
297  * @brief Visitor for message evaluation.
298  *
299  * This visitor either recurses into the active type
300  * if it is a variant type, or otherwise calls
301  * operator()(Particle&) on the active type with
302  * the particle that ought to be updated. This construction
303  * allows to nest message variants to allow for sub-
304  * categories. Those are mostly used here to differentiate
305  * the updates for the substructs of Particle.
306  */
307 struct UpdateVisitor : public boost::static_visitor<void> {
308  explicit UpdateVisitor(int id) : id(id) {}
309 
310  const int id;
311 
312  /* Recurse into sub-variants */
313  template <class... Message>
314  void operator()(const boost::variant<Message...> &msg) const {
315  boost::apply_visitor(*this, msg);
316  }
317  /* Plain messages are just called. */
318  template <typename Message> void operator()(const Message &msg) const {
319  assert(cell_structure.get_local_particle(id));
320  msg(*cell_structure.get_local_particle(id));
321  }
322 };
323 } // namespace

337 /**
338  * @brief Send a particle update message.
339  *
340  * This sends the message to the node that is responsible for the particle,
341  * where
342  * @p msg is called with the particle as argument. The message then performs the
343  * change to the particle that is encoded in it. The mechanism to call a functor
344  * based on the active type of a variant is called visitation. Here we can use
345  * @c UpdateVisitor with a single templated call operator on the visitor because
346  * all message (eventually) provide the same interface. Overall this is
347  * logically equivalent to nested switch statements over the message types,
348  * where the case statements play the role of the call of the messages in our
349  * case. A general introduction can be found in the documentation of
350  * boost::variant.
351  *
352  * @param id Id of the particle to update
353  * @param msg The message
354  */
355 void mpi_send_update_message(int id, const UpdateMessage &msg) {
356  auto const pnode = get_particle_node(id);
357 
358  mpi_call(mpi_send_update_message_local, pnode, id);
359 
360  /* If the particle is remote, send the
361  * message to the target, otherwise we
362  * can just apply the update directly. */
363  if (pnode != comm_cart.rank()) {
364  comm_cart.send(pnode, SOME_TAG, msg);
365  } else {
366  boost::apply_visitor(UpdateVisitor{id}, msg);
367  }
368 
369  on_particle_change();
370 }

1087 void auto_exclusions(int distance) {
1088  /* partners is a list containing the currently found excluded particles for
1089  each particle, and their distance, as an interleaved list */
1090  std::unordered_map<int, std::vector<int>> partners;
1091 
1092  /* determine initial connectivity */
1093  for (auto const &part1 : partCfg()) {
1094  auto const p1 = part1.p.identity;
1095  for (auto const bond : part1.bonds()) {
1096  if ((bond.partner_ids().size() == 1) and (bond.partner_ids()[0] != p1)) {
1097  auto const p2 = bond.partner_ids()[0];
1098  add_partner(partners[p1], p1, p2, 1);
1099  add_partner(partners[p2], p2, p1, 1);
1100  }
1101  }
1102  }
1103 
1104  /* calculate transient connectivity. For each of the current neighbors,
1105  also exclude their close enough neighbors.
1106  */
1107  for (int count = 1; count < distance; count++) {
1108  for (auto const &p : partCfg()) {
1109  auto const p1 = p.identity();
1110  for (int i = 0; i < partners[p1].size(); i += 2) {
1111  auto const p2 = partners[p1][i];
1112  auto const dist1 = partners[p1][i + 1];
1113  if (dist1 > distance)
1114  continue;
1115  /* loop over all partners of the partner */
1116  for (int j = 0; j < partners[p2].size(); j += 2) {
1117  auto const p3 = partners[p2][j];
1118  auto const dist2 = dist1 + partners[p2][j + 1];
1119  if (dist2 > distance)
1120  continue;
1121  add_partner(partners[p1], p1, p3, dist2);
1122  add_partner(partners[p3], p3, p1, dist2);
1123  }
1124  }
1125  }
1126  }
1127 
1128  /* setup the exclusions and clear the arrays. We do not setup the exclusions
1129  up there, since on_part_change clears the partCfg, so that we would have to
1130  restore it continuously. Of course this could be optimized by bundling the
1131  exclusions, but this is only done once and the overhead is as much as for
1132  setting the bonds, which the user apparently accepted.
1133  */
1134  for (auto &kv : partners) {
1135  auto const id = kv.first;
1136  auto const partner_list = kv.second;
1137  for (int j : partner_list)
1138  if (id < j)
1139  change_exclusion(id, j, 0);
1140  }
1141 }
