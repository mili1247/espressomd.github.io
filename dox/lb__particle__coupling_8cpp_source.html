122  * @brief Add a force to the lattice force density.

127 void add_md_force(Utils::Vector3d const &pos, Utils::Vector3d const &force,
128  double time_step) {
129  /* transform momentum transfer to lattice units
130  (eq. (12) @cite ahlrichs99a) */
131  auto const delta_j = -(time_step / lb_lbfluid_get_lattice_speed()) * force;
132  lb_lbinterpolation_add_force_density(pos, delta_j);

136 /** Coupling of a single particle to viscous fluid with Stokesian friction.
137  *
138  * Section II.C. @cite ahlrichs99a
139  *
140  * @param[in] p The coupled particle.
141  * @param[in] f_random Additional force to be included.
142  * @param[in] time_step MD time step.
143  *
144  * @return The viscous coupling force plus @p f_random.
145  */
146 Utils::Vector3d lb_viscous_coupling(Particle const &p,
147  Utils::Vector3d const &f_random,
148  double time_step) {
149  /* calculate fluid velocity at particle's position
150  this is done by linear interpolation (eq. (11) @cite ahlrichs99a) */
151  auto const interpolated_u =
152  lb_lbinterpolation_get_interpolated_velocity(p.r.p) *
153  lb_lbfluid_get_lattice_speed();
154 
155  Utils::Vector3d v_drift = interpolated_u;
156 #ifdef ENGINE
157  if (p.p.swim.swimming) {
158  v_drift += p.p.swim.v_swim * p.r.calc_director();
159  }
160 #endif
161 
162 #ifdef LB_ELECTROHYDRODYNAMICS
163  v_drift += p.p.mu_E;
164 #endif
165 
166  /* calculate viscous force (eq. (9) @cite ahlrichs99a) */
167  auto const force = -lb_lbcoupling_get_gamma() * (p.m.v - v_drift) + f_random;
168 
169  add_md_force(p.r.p, force, time_step);
170 
171  return force;
172 }

181 /**
182  * @brief Check if a position is in a box.
183  *
184  * The left boundary belong to the box, the
185  * right one does not. Periodic boundaries are
186  * not considered.
187  *
188  * @param pos Position to check
189  * @param box Box to check
190  *
191  * @return True iff the point is inside of the box.
192  */
193 template <class T, std::size_t N>
194 bool in_box(Vector<T, N> const &pos, Box<T, N> const &box) {
195  return (pos >= box.first) and (pos < box.second);
196 }

198 /**
199  * @brief Check if a position is within the local box + halo.
200  *
201  * @param pos Position to check
202  * @param local_box Geometry to check
203  * @param halo Halo
204  *
205  * @return True iff the point is inside of the box up to halo.
206  */
207 template <class T>
208 bool in_local_domain(Vector<T, 3> const &pos, LocalBox<T> const &local_box,
209  T const &halo = {}) {
210  auto const halo_vec = Vector<T, 3>::broadcast(halo);
211 
212  return in_box(
213  pos, {local_geo.my_left() - halo_vec, local_geo.my_right() + halo_vec});
214 }

231 void add_swimmer_force(Particle const &p, double time_step) {
232  if (p.p.swim.swimming) {
233  // calculate source position
234  const double direction =
235  double(p.p.swim.push_pull) * p.p.swim.dipole_length;
236  auto const director = p.r.calc_director();
237  auto const source_position = p.r.p + direction * director;
238 
239  if (not in_local_halo(source_position)) {
240  return;
241  }
242 
243  add_md_force(source_position, p.p.swim.f_swim * director, time_step);
244  }

250 void lb_lbcoupling_calc_particle_lattice_ia(bool couple_virtual,
251  const ParticleRange &particles,
252  const ParticleRange &more_particles,
253  double time_step) {
254  ESPRESSO_PROFILER_CXX_MARK_FUNCTION;
255  if (lattice_switch == ActiveLB::GPU) {
256 #ifdef CUDA
257  if (lb_particle_coupling.couple_to_md && this_node == 0) {
258  switch (lb_lbinterpolation_get_interpolation_order()) {
259  case (InterpolationOrder::linear):
260  lb_calc_particle_lattice_ia_gpu<8>(
261  couple_virtual, lb_lbcoupling_get_gamma(), time_step);
262  break;
263  case (InterpolationOrder::quadratic):
264  lb_calc_particle_lattice_ia_gpu<27>(
265  couple_virtual, lb_lbcoupling_get_gamma(), time_step);
266  break;
267  }
268  }
269 #endif
270  } else if (lattice_switch == ActiveLB::CPU) {
271  if (lb_particle_coupling.couple_to_md) {
272  switch (lb_lbinterpolation_get_interpolation_order()) {
273  case (InterpolationOrder::quadratic):
274  throw std::runtime_error("The non-linear interpolation scheme is not "
275  "implemented for the CPU LB.");
276  case (InterpolationOrder::linear): {
277  auto const kT = lb_lbfluid_get_kT();
278  /* Eq. (16) @cite ahlrichs99a.
279  * The factor 12 comes from the fact that we use random numbers
280  * from -0.5 to 0.5 (equally distributed) which have variance 1/12.
281  * time_step comes from the discretization.
282  */
283  auto const noise_amplitude =
284  (kT > 0.) ? std::sqrt(12. * 2. * lb_lbcoupling_get_gamma() * kT /
285  time_step)
286  : 0.0;
287 
288  auto f_random = [noise_amplitude](int id) -> Utils::Vector3d {
289  if (noise_amplitude > 0.0) {
290  return Random::noise_uniform<RNGSalt::PARTICLES>(
291  lb_particle_coupling.rng_counter_coupling->value(), 0, id);
292  }
293  return {};
294  };
295 
296  auto couple_particle = [&](Particle &p) -> void {
297  if (p.p.is_virtual and !couple_virtual)
298  return;
299 
300  /* Particle is in our LB volume, so this node
301  * is responsible to adding its force */
302  if (in_local_domain(p.r.p, local_geo)) {
303  auto const force = lb_viscous_coupling(
304  p, noise_amplitude * f_random(p.identity()), time_step);
305  /* add force to the particle */
306  p.f.f += force;
307  /* Particle is not in our domain, but adds to the force
308  * density in our domain, only calculate contribution to
309  * the LB force density. */
310  } else if (in_local_halo(p.r.p)) {
311  lb_viscous_coupling(p, noise_amplitude * f_random(p.identity()),
312  time_step);
313  }
314 
315 #ifdef ENGINE
316  add_swimmer_force(p, time_step);
317 #endif
318  };
319 
320  /* Couple particles ranges */
321  for (auto &p : particles) {
322  couple_particle(p);
323  }
324 
325  for (auto &p : more_particles) {
326  couple_particle(p);
327  }
328 
329  break;
330  }
331  }
332  }
333  }
334 }
