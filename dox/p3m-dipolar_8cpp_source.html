146 /** Correction of the dipolar p3m-energy. */
147 double dp3m_average_dipolar_self_energy() {
148  auto const start = Utils::Vector3i{dp3m.fft.plan[3].start};
149  auto const size = Utils::Vector3i{dp3m.fft.plan[3].new_mesh};
150 
151  auto const node_phi = grid_influence_function_self_energy(
152  dp3m.params, start, start + size, dp3m.g_energy);
153 
154  double phi = 0.0;
155  boost::mpi::reduce(comm_cart, node_phi, phi, std::plus<>(), 0);
156  phi /= 3. * box_geo.length()[0] * Utils::int_pow<3>(dp3m.params.mesh[0]);
157  return phi * Utils::pi();
158 }

268  Dipole::set_method_local(DIPOLAR_P3M);

270  dp3m.params.r_cut = r_cut;

272  dp3m.params.mesh[2] = dp3m.params.mesh[1] = dp3m.params.mesh[0] = mesh;
273  dp3m.params.cao = cao;
274  dp3m.params.alpha = alpha;
275  dp3m.params.alpha_L = alpha * box_geo.length()[0];
276  dp3m.params.accuracy = accuracy;

313  dp3m.rs_mesh_dip[1][ind] += w * dip[1];
314  dp3m.rs_mesh_dip[2][ind] += w * dip[2];
315  });
316 
317  dp3m.inter_weights.store<cao>(weights);
318  }

339 template <std::size_t cao> struct AssignTorques {
340  void operator()(double prefac, int d_rs,
341  const ParticleRange &particles) const {
342  /* particle counter */
343  int cp_cnt = 0;
344  for (auto &p : particles) {
345  auto const w = dp3m.inter_weights.load<cao>(cp_cnt++);
346 
347  Utils::Vector3d E{};
348  p3m_interpolate(dp3m.local_mesh, w, [&E, d_rs](int ind, double w) {
349  E[d_rs] += w * dp3m.rs_mesh[ind];
350  });
351 
352  p.f.torque -= vector_product(p.calc_dip(), prefac * E);
353  }
354  }
355 };

372 
373  p.f.f[d_rs] += p.calc_dip() * prefac * E;
374  }

381 double dp3m_calc_kspace_forces(bool force_flag, bool energy_flag,
382  const ParticleRange &particles) {
383  int i, d, d_rs, ind, j[3];
384  /* k-space energy */
385  double k_space_energy_dip = 0.0;
386  double tmp0, tmp1;
387 
388  auto const dipole_prefac =
389  dipole.prefactor / Utils::int_pow<3>(dp3m.params.mesh[0]);
390 
391  if (dp3m.sum_mu2 > 0) {
392  /* Gather information for FFT grid inside the nodes domain (inner local
393  * mesh) and perform forward 3D FFT (Charge Assignment Mesh). */
394  std::array<double *, 3> meshes = {dp3m.rs_mesh_dip[0].data(),
395  dp3m.rs_mesh_dip[1].data(),
396  dp3m.rs_mesh_dip[2].data()};
397 
398  dp3m.sm.gather_grid(Utils::make_span(meshes), comm_cart,
399  dp3m.local_mesh.dim);
400 
401  fft_perform_forw(dp3m.rs_mesh_dip[0].data(), dp3m.fft, comm_cart);
402  fft_perform_forw(dp3m.rs_mesh_dip[1].data(), dp3m.fft, comm_cart);
403  fft_perform_forw(dp3m.rs_mesh_dip[2].data(), dp3m.fft, comm_cart);
404  // Note: after these calls, the grids are in the order yzx and not xyz
405  // anymore!!!
406  }
407 
408  /* === k-space calculations === */
409 
410  /* === k-space energy calculation === */
411  if (energy_flag) {
412  /*********************
413  Dipolar energy
414  **********************/
415  if (dp3m.sum_mu2 > 0) {
416  /* i*k differentiation for dipolar gradients:
417  * |(\Fourier{\vect{mu}}(k)\cdot \vect{k})|^2 */
418  ind = 0;
419  i = 0;
420  double node_k_space_energy_dip = 0.0;
421  for (j[0] = 0; j[0] < dp3m.fft.plan[3].new_mesh[0]; j[0]++) {
422  for (j[1] = 0; j[1] < dp3m.fft.plan[3].new_mesh[1]; j[1]++) {
423  for (j[2] = 0; j[2] < dp3m.fft.plan[3].new_mesh[2]; j[2]++) {
424  node_k_space_energy_dip +=
425  dp3m.g_energy[i] *
426  (Utils::sqr(
427  dp3m.rs_mesh_dip[0][ind] *
428  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
429  dp3m.rs_mesh_dip[1][ind] *
430  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
431  dp3m.rs_mesh_dip[2][ind] *
432  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]]) +
433  Utils::sqr(
434  dp3m.rs_mesh_dip[0][ind + 1] *
435  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
436  dp3m.rs_mesh_dip[1][ind + 1] *
437  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
438  dp3m.rs_mesh_dip[2][ind + 1] *
439  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]]));
440  ind += 2;
441  i++;
442  }
443  }
444  }
445  node_k_space_energy_dip *=
446  dipole_prefac * Utils::pi() * box_geo.length_inv()[0];
447  boost::mpi::reduce(comm_cart, node_k_space_energy_dip, k_space_energy_dip,
448  std::plus<>(), 0);
449 
450  dp3m_compute_constants_energy_dipolar();
451 
452  if (this_node == 0) {
453  /* self energy correction */
454  k_space_energy_dip -=
455  dipole.prefactor *
456  (dp3m.sum_mu2 * 2 *
457  pow(dp3m.params.alpha_L * box_geo.length_inv()[0], 3) *
458  Utils::sqrt_pi_i() / 3.0);
459 
460  auto const volume = box_geo.volume();
461  k_space_energy_dip += dipole.prefactor * dp3m.energy_correction /
462  volume; /* add the dipolar energy correction due
463  to systematic Madelung-Self effects */
464  }
465  }
466  } // if (energy_flag)
467 
468  /* === k-space force calculation === */
469  if (force_flag) {
470  /****************************
471  * DIPOLAR TORQUES (k-space)
472  ****************************/
473  if (dp3m.sum_mu2 > 0) {
474  /* fill in ks_mesh array for torque calculation */
475  ind = 0;
476  i = 0;
477 
478  for (j[0] = 0; j[0] < dp3m.fft.plan[3].new_mesh[0]; j[0]++) { // j[0]=n_y
479  for (j[1] = 0; j[1] < dp3m.fft.plan[3].new_mesh[1];
480  j[1]++) { // j[1]=n_z
481  for (j[2] = 0; j[2] < dp3m.fft.plan[3].new_mesh[2];
482  j[2]++) { // j[2]=n_x
483  // tmp0 = Re(mu)*k, tmp1 = Im(mu)*k
484 
485  tmp0 = dp3m.rs_mesh_dip[0][ind] *
486  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
487  dp3m.rs_mesh_dip[1][ind] *
488  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
489  dp3m.rs_mesh_dip[2][ind] *
490  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]];
491 
492  tmp1 = dp3m.rs_mesh_dip[0][ind + 1] *
493  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
494  dp3m.rs_mesh_dip[1][ind + 1] *
495  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
496  dp3m.rs_mesh_dip[2][ind + 1] *
497  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]];
498 
499  /* the optimal influence function is the same for torques
500  and energy */
501 
502  dp3m.ks_mesh[ind] = tmp0 * dp3m.g_energy[i];
503  dp3m.ks_mesh[ind + 1] = tmp1 * dp3m.g_energy[i];
504  ind += 2;
505  i++;
506  }
507  }
508  }
509 
510  /* Force component loop */
511  for (d = 0; d < 3; d++) {
512  d_rs = (d + dp3m.ks_pnum) % 3;
513  ind = 0;
514  for (j[0] = 0; j[0] < dp3m.fft.plan[3].new_mesh[0]; j[0]++) {
515  for (j[1] = 0; j[1] < dp3m.fft.plan[3].new_mesh[1]; j[1]++) {
516  for (j[2] = 0; j[2] < dp3m.fft.plan[3].new_mesh[2]; j[2]++) {
517  dp3m.rs_mesh[ind] =
518  dp3m.d_op[0][j[d] + dp3m.fft.plan[3].start[d]] *
519  dp3m.ks_mesh[ind];
520  ind++;
521  dp3m.rs_mesh[ind] =
522  dp3m.d_op[0][j[d] + dp3m.fft.plan[3].start[d]] *
523  dp3m.ks_mesh[ind];
524  ind++;
525  }
526  }
527  }
528 
529  /* Back FFT force component mesh */
530  fft_perform_back(dp3m.rs_mesh.data(), false, dp3m.fft, comm_cart);
531  /* redistribute force component mesh */
532  dp3m.sm.spread_grid(dp3m.rs_mesh.data(), comm_cart,
533  dp3m.local_mesh.dim);
534  /* Assign force component from mesh to particle */
535  Utils::integral_parameter<AssignTorques, 1, 7>(
536  dp3m.params.cao,
537  dipole_prefac * (2 * Utils::pi() * box_geo.length_inv()[0]), d_rs,
538  particles);
539  }
540 
541  /***************************
542  DIPOLAR FORCES (k-space)
543  ****************************/
544 
545  // Compute forces after torques because the algorithm below overwrites the
546  // grids dp3m.rs_mesh_dip !
547  // Note: I'll do here 9 inverse FFTs. By symmetry, we can reduce this
548  // number to 6 !
549  /* fill in ks_mesh array for force calculation */
550  ind = 0;
551  i = 0;
552  for (j[0] = 0; j[0] < dp3m.fft.plan[3].new_mesh[0]; j[0]++) { // j[0]=n_y
553  for (j[1] = 0; j[1] < dp3m.fft.plan[3].new_mesh[1];
554  j[1]++) { // j[1]=n_z
555  for (j[2] = 0; j[2] < dp3m.fft.plan[3].new_mesh[2];
556  j[2]++) { // j[2]=n_x
557  // tmp0 = Im(mu)*k, tmp1 = -Re(mu)*k
558  tmp0 = dp3m.rs_mesh_dip[0][ind + 1] *
559  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
560  dp3m.rs_mesh_dip[1][ind + 1] *
561  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
562  dp3m.rs_mesh_dip[2][ind + 1] *
563  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]];
564  tmp1 = dp3m.rs_mesh_dip[0][ind] *
565  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] +
566  dp3m.rs_mesh_dip[1][ind] *
567  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] +
568  dp3m.rs_mesh_dip[2][ind] *
569  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]];
570  dp3m.ks_mesh[ind] = tmp0 * dp3m.g_force[i];
571  dp3m.ks_mesh[ind + 1] = -tmp1 * dp3m.g_force[i];
572  ind += 2;
573  i++;
574  }
575  }
576  }
577 
578  /* Force component loop */
579  for (d = 0; d < 3; d++) { /* direction in k-space: */
580  d_rs = (d + dp3m.ks_pnum) % 3;
581  ind = 0;
582  for (j[0] = 0; j[0] < dp3m.fft.plan[3].new_mesh[0];
583  j[0]++) { // j[0]=n_y
584  for (j[1] = 0; j[1] < dp3m.fft.plan[3].new_mesh[1];
585  j[1]++) { // j[1]=n_z
586  for (j[2] = 0; j[2] < dp3m.fft.plan[3].new_mesh[2];
587  j[2]++) { // j[2]=n_x
588  tmp0 = dp3m.d_op[0][j[d] + dp3m.fft.plan[3].start[d]] *
589  dp3m.ks_mesh[ind];
590  dp3m.rs_mesh_dip[0][ind] =
591  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] * tmp0;
592  dp3m.rs_mesh_dip[1][ind] =
593  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] * tmp0;
594  dp3m.rs_mesh_dip[2][ind] =
595  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]] * tmp0;
596  ind++;
597  tmp0 = dp3m.d_op[0][j[d] + dp3m.fft.plan[3].start[d]] *
598  dp3m.ks_mesh[ind];
599  dp3m.rs_mesh_dip[0][ind] =
600  dp3m.d_op[0][j[2] + dp3m.fft.plan[3].start[2]] * tmp0;
601  dp3m.rs_mesh_dip[1][ind] =
602  dp3m.d_op[0][j[0] + dp3m.fft.plan[3].start[0]] * tmp0;
603  dp3m.rs_mesh_dip[2][ind] =
604  dp3m.d_op[0][j[1] + dp3m.fft.plan[3].start[1]] * tmp0;
605  ind++;
606  }
607  }
608  }
609  /* Back FFT force component mesh */
610  fft_perform_back(dp3m.rs_mesh_dip[0].data(), false, dp3m.fft,
611  comm_cart);
612  fft_perform_back(dp3m.rs_mesh_dip[1].data(), false, dp3m.fft,
613  comm_cart);
614  fft_perform_back(dp3m.rs_mesh_dip[2].data(), false, dp3m.fft,
615  comm_cart);
616  /* redistribute force component mesh */
617  std::array<double *, 3> meshes = {dp3m.rs_mesh_dip[0].data(),
618  dp3m.rs_mesh_dip[1].data(),
619  dp3m.rs_mesh_dip[2].data()};
620 
621  dp3m.sm.spread_grid(Utils::make_span(meshes), comm_cart,
622  dp3m.local_mesh.dim);
623  /* Assign force component from mesh to particle */
624  Utils::integral_parameter<AssignForces, 1, 7>(
625  dp3m.params.cao,
626  dipole_prefac * pow(2 * Utils::pi() * box_geo.length_inv()[0], 2),
627  d_rs, particles);
628  }
629  } /* if (dp3m.sum_mu2 > 0) */
630  } /* if (force_flag) */
631 
632  if (dp3m.params.epsilon != P3M_EPSILON_METALLIC) {
633  auto const surface_term =
634  calc_surface_term(force_flag, energy_flag, particles);
635  if (this_node == 0)
636  k_space_energy_dip += surface_term;
637  }
638 
639  return k_space_energy_dip;
640 }

643 
644 double calc_surface_term(bool force_flag, bool energy_flag,
645  const ParticleRange &particles) {
646  auto const pref = dipole.prefactor * 4 * Utils::pi() / box_geo.volume() /
647  (2 * dp3m.params.epsilon + 1);
648  auto const n_local_part = particles.size();
649 
650  // We put all the dipolar momenta in a the arrays mx,my,mz according to the
651  // id-number of the particles
652  std::vector<double> mx(n_local_part);
653  std::vector<double> my(n_local_part);
654  std::vector<double> mz(n_local_part);
655 
656  int ip = 0;
657  for (auto const &p : particles) {
658  auto const dip = p.calc_dip();
659  mx[ip] = dip[0];
660  my[ip] = dip[1];
661  mz[ip] = dip[2];
662  ip++;
663  }
664 
665  // we will need the sum of all dipolar momenta vectors
666  auto local_dip = Utils::Vector3d{};
667  for (int i = 0; i < n_local_part; i++) {
668  local_dip[0] += mx[i];
669  local_dip[1] += my[i];
670  local_dip[2] += mz[i];
671  }
672  auto const box_dip =
673  boost::mpi::all_reduce(comm_cart, local_dip, std::plus<>());
674 
675  double energy = 0.0;
676  if (energy_flag) {
677  double sum_e = 0.0;
678  for (int i = 0; i < n_local_part; i++) {
679  sum_e += mx[i] * box_dip[0] + my[i] * box_dip[1] + mz[i] * box_dip[2];
680  }
681  energy =
682  0.5 * pref * boost::mpi::all_reduce(comm_cart, sum_e, std::plus<>());
683  }
684 
685  if (force_flag) {
686 
687  std::vector<double> sumix(n_local_part);
688  std::vector<double> sumiy(n_local_part);
689  std::vector<double> sumiz(n_local_part);
690 
691  for (int i = 0; i < n_local_part; i++) {
692  sumix[i] = my[i] * box_dip[2] - mz[i] * box_dip[1];
693  sumiy[i] = mz[i] * box_dip[0] - mx[i] * box_dip[2];
694  sumiz[i] = mx[i] * box_dip[1] - my[i] * box_dip[0];
695  }
696 
697  ip = 0;
698  for (auto &p : particles) {
699  p.f.torque[0] -= pref * sumix[ip];
700  p.f.torque[1] -= pref * sumiy[ip];
701  p.f.torque[2] -= pref * sumiz[ip];
702  ip++;
703  }
704  }
705 
706  return energy;
707 }

729 /** @copybrief p3m_get_accuracy
730  *
731  * The real space error is tuned such that it contributes half of the
732  * total error, and then the Fourier space error is calculated.
733  * If an optimal alpha is not found, the value 0.1 is used as fallback.
734  * @param[in] mesh @copybrief P3MParameters::mesh
735  * @param[in] cao @copybrief P3MParameters::cao
736  * @param[in] r_cut_iL @copybrief P3MParameters::r_cut_iL
737  * @param[out] _alpha_L @copybrief P3MParameters::alpha_L
738  * @param[out] _rs_err real space error
739  * @param[out] _ks_err Fourier space error
740  * @returns Error magnitude
741  */
742 double dp3m_get_accuracy(int mesh, int cao, double r_cut_iL, double *_alpha_L,
743  double *_rs_err, double *_ks_err) {
744  double rs_err, ks_err;
745  double alpha_L;
746 
747  /* calc maximal real space error for setting */
748 
749  // Alpha cannot be zero in the dipolar case because real_space formula breaks
750  // down
751  rs_err =
752  dp3m_real_space_error(box_geo.length()[0], dipole.prefactor, r_cut_iL,
753  dp3m.sum_dip_part, dp3m.sum_mu2, 0.001);
754 
755  if (Utils::sqrt_2() * rs_err > dp3m.params.accuracy) {
756  /* assume rs_err = ks_err -> rs_err = accuracy/sqrt(2.0) -> alpha_L */
757  alpha_L = dp3m_rtbisection(
758  box_geo.length()[0], dipole.prefactor, r_cut_iL, dp3m.sum_dip_part,
759  dp3m.sum_mu2, 0.0001 * box_geo.length()[0], 5.0 * box_geo.length()[0],
760  0.0001, dp3m.params.accuracy);
761  if (alpha_L == -DP3M_RTBISECTION_ERROR) {
762  *_rs_err = -1;
763  *_ks_err = -1;
764  return -DP3M_RTBISECTION_ERROR;
765  }
766  }
767 
768  else
769  /* even alpha=0 is ok, however, we cannot choose it since it kills the
770  k-space error formula.
771  Anyways, this very likely NOT the optimal solution */
772  alpha_L = 0.1;
773 
774  *_alpha_L = alpha_L;
775  /* calculate real space and k-space error for this alpha_L */
776 
777  rs_err =
778  dp3m_real_space_error(box_geo.length()[0], dipole.prefactor, r_cut_iL,
779  dp3m.sum_dip_part, dp3m.sum_mu2, alpha_L);
780  ks_err = dp3m_k_space_error(box_geo.length()[0], dipole.prefactor, mesh, cao,
781  dp3m.sum_dip_part, dp3m.sum_mu2, alpha_L);
782 
783  *_rs_err = rs_err;
784  *_ks_err = ks_err;
785  return sqrt(Utils::sqr(rs_err) + Utils::sqr(ks_err));
786 }

788 /** @copybrief p3m_mcr_time
789  *
790  * @param[in] mesh @copybrief P3MParameters::mesh
791  * @param[in] cao @copybrief P3MParameters::cao
792  * @param[in] r_cut_iL @copybrief P3MParameters::r_cut_iL
793  * @param[in] alpha_L @copybrief P3MParameters::alpha_L
794  * @param[in] timings Number of test force calculations
795  *
796  * @returns The integration time in case of success, otherwise
797  * -@ref P3M_TUNE_FAIL
798  */
799 static double dp3m_mcr_time(int mesh, int cao, double r_cut_iL, double alpha_L,
800  int timings) {
801 
802  /* broadcast p3m parameters for test run */
803  if (dipole.method != DIPOLAR_P3M && dipole.method != DIPOLAR_MDLC_P3M)
804  Dipole::set_method_local(DIPOLAR_P3M);
805  dp3m.params.r_cut_iL = r_cut_iL;
806  dp3m.params.mesh[0] = dp3m.params.mesh[1] = dp3m.params.mesh[2] = mesh;
807  dp3m.params.cao = cao;
808  dp3m.params.alpha_L = alpha_L;
809  /* initialize p3m structures */
810  mpi_bcast_coulomb_params();
811  /* perform force calculation test */
812  double int_time = time_force_calc(timings);
813  if (int_time == -1) {
814  return -P3M_TUNE_FAIL;
815  }
816  return int_time;
817 }
818 
819 /** @copybrief p3m_mc_time
820  *
821  * The @p _r_cut_iL is determined via a simple bisection.
822  *
823  * @param[in] mesh @copybrief P3MParameters::mesh
824  * @param[in] cao @copybrief P3MParameters::cao
825  * @param[in] r_cut_iL_min lower bound for @p _r_cut_iL
826  * @param[in] r_cut_iL_max upper bound for @p _r_cut_iL
827  * @param[out] _r_cut_iL @copybrief P3MParameters::r_cut_iL
828  * @param[out] _alpha_L @copybrief P3MParameters::alpha_L
829  * @param[out] _accuracy @copybrief P3MParameters::accuracy
830  * @param[in] timings Number of test force calculations
831  * @param[in] verbose printf output
832  *
833  * @returns The integration time in case of success, otherwise
834  * -@ref P3M_TUNE_FAIL, -@ref P3M_TUNE_ACCURACY_TOO_LARGE,
835  * -@ref P3M_TUNE_CAO_TOO_LARGE, -@ref P3M_TUNE_ELCTEST, or
836  * -@ref P3M_TUNE_CUTOFF_TOO_LARGE
837  */
838 static double dp3m_mc_time(int mesh, int cao, double r_cut_iL_min,
839  double r_cut_iL_max, double *_r_cut_iL,
840  double *_alpha_L, double *_accuracy, int timings,
841  bool verbose) {
842  double r_cut_iL;
843  double rs_err, ks_err;
844 
845  /* initial checks. */
846  auto const mesh_size = box_geo.length()[0] / static_cast<double>(mesh);
847  auto const k_cut = mesh_size * cao / 2.0;
848 
849  auto const min_box_l = *boost::min_element(box_geo.length());
850  auto const min_local_box_l = *boost::min_element(local_geo.length());
851 
852  if (cao >= mesh || k_cut >= std::min(min_box_l, min_local_box_l) - skin) {
853  /* print result */
854  if (verbose) {
855  std::printf("%-4d %-3d cao too large for this mesh\n", mesh, cao);
856  }
857  return -P3M_TUNE_CAO_TOO_LARGE;
858  }
859 
860  /* Either low and high boundary are equal (for fixed cut), or the low border
861  is initially 0 and therefore
862  has infinite error estimate, as required. Therefore if the high boundary
863  fails, there is no possible r_cut */
864  *_accuracy =
865  dp3m_get_accuracy(mesh, cao, r_cut_iL_max, _alpha_L, &rs_err, &ks_err);
866  if (*_accuracy == -DP3M_RTBISECTION_ERROR)
867  return *_accuracy;
868  if (*_accuracy > dp3m.params.accuracy) {
869  /* print result */
870  if (verbose) {
871  std::printf("%-4d %-3d %.5e %.5e %.5e %.3e %.3e accuracy not achieved\n",
872  mesh, cao, r_cut_iL_max, *_alpha_L, *_accuracy, rs_err,
873  ks_err);
874  }
875  return -P3M_TUNE_ACCURACY_TOO_LARGE;
876  }
877 
878  for (;;) {
879  r_cut_iL = 0.5 * (r_cut_iL_min + r_cut_iL_max);
880 
881  if (r_cut_iL_max - r_cut_iL_min < P3M_RCUT_PREC)
882  break;
883 
884  /* bisection */
885  auto const tmp_accuracy =
886  dp3m_get_accuracy(mesh, cao, r_cut_iL, _alpha_L, &rs_err, &ks_err);
887  if (tmp_accuracy == -DP3M_RTBISECTION_ERROR)
888  return tmp_accuracy;
889  if (tmp_accuracy > dp3m.params.accuracy)
890  r_cut_iL_min = r_cut_iL;
891  else
892  r_cut_iL_max = r_cut_iL;
893  }
894  /* final result is always the upper interval boundary, since only there
895  we know that the desired minimal accuracy is obtained */
896  *_r_cut_iL = r_cut_iL = r_cut_iL_max;
897 
898  /* check whether we are running P3M+DLC, and whether we leave a reasonable gap
899  * space */
900  if (dipole.method == DIPOLAR_MDLC_P3M) {
901  runtimeErrorMsg() << "dipolar P3M: tuning when dlc needs to be fixed";
902  }
903 
904  double const int_time =
905  dp3m_mcr_time(mesh, cao, r_cut_iL, *_alpha_L, timings);
906  if (int_time == -P3M_TUNE_FAIL) {
907  if (verbose) {
908  std::printf("tuning failed, test integration not possible\n");
909  }
910  return int_time;
911  }
912 
913  *_accuracy =
914  dp3m_get_accuracy(mesh, cao, r_cut_iL, _alpha_L, &rs_err, &ks_err);
915  if (*_accuracy == -DP3M_RTBISECTION_ERROR) {
916  return *_accuracy;
917  }
918 
919  /* print result */
920  if (verbose) {
921  std::printf("%-4d %-3d %.5e %.5e %.5e %.3e %.3e %-8.0f\n", mesh, cao,
922  r_cut_iL, *_alpha_L, *_accuracy, rs_err, ks_err, int_time);
923  }
924  return int_time;
925 }
926 
927 /** @copybrief p3m_m_time
928  *
929  * @p _cao should contain an initial guess, which is then adapted by stepping
930  * up and down.
931  *
932  * @param[in] mesh @copybrief P3MParameters::mesh
933  * @param[in] cao_min lower bound for @p _cao
934  * @param[in] cao_max upper bound for @p _cao
935  * @param[in,out] _cao initial guess for the
936  * @copybrief P3MParameters::cao
937  * @param[in] r_cut_iL_min lower bound for @p _r_cut_iL
938  * @param[in] r_cut_iL_max upper bound for @p _r_cut_iL
939  * @param[out] _r_cut_iL @copybrief P3MParameters::r_cut_iL
940  * @param[out] _alpha_L @copybrief P3MParameters::alpha_L
941  * @param[out] _accuracy @copybrief P3MParameters::accuracy
942  * @param[in] timings Number of test force calculations
943  * @param[in] verbose printf output
944  *
945  * @returns The integration time in case of success, otherwise
946  * -@ref P3M_TUNE_FAIL or -@ref P3M_TUNE_CAO_TOO_LARGE */
947 static double dp3m_m_time(int mesh, int cao_min, int cao_max, int *_cao,
948  double r_cut_iL_min, double r_cut_iL_max,
949  double *_r_cut_iL, double *_alpha_L,
950  double *_accuracy, int timings, bool verbose) {
951  double best_time = -1, tmp_r_cut_iL = -1., tmp_alpha_L = 0.0,
952  tmp_accuracy = 0.0;
953  /* in which direction improvement is possible. Initially, we don't know it
954  * yet.
955  */
956  int final_dir = 0;
957  int cao = *_cao;
958 
959  /* the initial step sets a timing mark. If there is no valid r_cut, we can
960  only try
961  to increase cao to increase the obtainable precision of the far formula. */
962  double tmp_time;
963  do {
964  tmp_time =
965  dp3m_mc_time(mesh, cao, r_cut_iL_min, r_cut_iL_max, &tmp_r_cut_iL,
966  &tmp_alpha_L, &tmp_accuracy, timings, verbose);
967  /* bail out if the force evaluation is not working */
968  if (tmp_time == -P3M_TUNE_FAIL || tmp_time == -DP3M_RTBISECTION_ERROR)
969  return tmp_time;
970  /* cao is too large for this grid, but still the accuracy cannot be
971  * achieved, give up */
972  if (tmp_time == -P3M_TUNE_CAO_TOO_LARGE) {
973  return tmp_time;
974  }
975  /* we have a valid time, start optimising from there */
976  if (tmp_time >= 0) {
977  best_time = tmp_time;
978  *_r_cut_iL = tmp_r_cut_iL;
979  *_alpha_L = tmp_alpha_L;
980  *_accuracy = tmp_accuracy;
981  *_cao = cao;
982  break;
983  }
984  /* the required accuracy could not be obtained, try higher caos */
985  cao++;
986  final_dir = 1;
987  } while (cao <= cao_max);
988  /* with this mesh, the required accuracy cannot be obtained. */
989  if (cao > cao_max)
990  return -P3M_TUNE_CAO_TOO_LARGE;
991 
992  /* at the boundaries, only the opposite direction can be used for optimisation
993  */
994  if (cao == cao_min)
995  final_dir = 1;
996  else if (cao == cao_max)
997  final_dir = -1;
998 
999  if (final_dir == 0) {
1000  /* check in which direction we can optimise. Both directions are possible */
1001  double dir_times[3];
1002  for (final_dir = -1; final_dir <= 1; final_dir += 2) {
1003  dir_times[final_dir + 1] = tmp_time = dp3m_mc_time(
1004  mesh, cao + final_dir, r_cut_iL_min, r_cut_iL_max, &tmp_r_cut_iL,
1005  &tmp_alpha_L, &tmp_accuracy, timings, verbose);
1006  /* bail out on errors, as usual */
1007  if (tmp_time == -P3M_TUNE_FAIL || tmp_time == -DP3M_RTBISECTION_ERROR)
1008  return tmp_time;
1009  /* in this direction, we cannot optimise, since we get into precision
1010  * trouble */
1011  if (tmp_time < 0)
1012  continue;
1013 
1014  if (tmp_time < best_time) {
1015  best_time = tmp_time;
1016  *_r_cut_iL = tmp_r_cut_iL;
1017  *_alpha_L = tmp_alpha_L;
1018  *_accuracy = tmp_accuracy;
1019  *_cao = cao + final_dir;
1020  }
1021  }
1022  /* choose the direction which was optimal, if any of the two */
1023  if (dir_times[0] == best_time) {
1024  final_dir = -1;
1025  } else if (dir_times[2] == best_time) {
1026  final_dir = 1;
1027  } else {
1028  /* no improvement in either direction, however if one is only marginally
1029  * worse, we can still try*/
1030  /* down is possible and not much worse, while up is either illegal or even
1031  * worse */
1032  if ((dir_times[0] >= 0 && dir_times[0] < best_time + P3M_TIME_GRAN) &&
1033  (dir_times[2] < 0 || dir_times[2] > dir_times[0]))
1034  final_dir = -1;
1035  /* same for up */
1036  else if ((dir_times[2] >= 0 &&
1037  dir_times[2] < best_time + P3M_TIME_GRAN) &&
1038  (dir_times[0] < 0 || dir_times[0] > dir_times[2]))
1039  final_dir = 1;
1040  else {
1041  /* really no chance for optimisation */
1042  return best_time;
1043  }
1044  }
1045  /* we already checked the initial cao and its neighbor */
1046  cao += 2 * final_dir;
1047  } else {
1048  /* here some constraint is active, and we only checked the initial cao
1049  * itself */
1050  cao += final_dir;
1051  }
1052

1248 /** Calculate the k-space error of dipolar-P3M */
1249 static double dp3m_k_space_error(double box_size, double prefac, int mesh,
1250  int cao, int n_c_part, double sum_q2,
1251  double alpha_L) {
1252  double he_q = 0.0;
1253  auto const mesh_i = 1. / mesh;
1254  auto const alpha_L_i = 1. / alpha_L;
1255 
1256  for (int nx = -mesh / 2; nx < mesh / 2; nx++)
1257  for (int ny = -mesh / 2; ny < mesh / 2; ny++)
1258  for (int nz = -mesh / 2; nz < mesh / 2; nz++)
1259  if ((nx != 0) || (ny != 0) || (nz != 0)) {
1260  auto const n2 = Utils::sqr(nx) + Utils::sqr(ny) + Utils::sqr(nz);
1261  auto const cs = p3m_analytic_cotangent_sum(nx, mesh_i, cao) *
1262  p3m_analytic_cotangent_sum(ny, mesh_i, cao) *
1263  p3m_analytic_cotangent_sum(nz, mesh_i, cao);
1264  double alias1, alias2;
1265  dp3m_tune_aliasing_sums(nx, ny, nz, mesh, mesh_i, cao, alpha_L_i,
1266  &alias1, &alias2);
1267  double d = alias1 - Utils::sqr(alias2 / cs) /
1268  Utils::int_pow<3>(static_cast<double>(n2));
1269  /* at high precision, d can become negative due to extinction;
1270  also, don't take values that have no significant digits left*/
1271  if (d > 0 && (fabs(d / alias1) > ROUND_ERROR_PREC))
1272  he_q += d;
1273  }
1274 
1275  return 8. * Utils::sqr(Utils::pi()) / 3. * sum_q2 * sqrt(he_q / n_c_part) /
1276  Utils::int_pow<4>(box_size);
1277 }

1310 /** Calculate the value of the errors for the REAL part of the force in terms
1311  * of the splitting parameter alpha of Ewald. Based on eq. (33) @cite wang01a.
1312  *
1313  * Please note that in this more refined approach we don't use
1314  * eq. (37), but eq. (33) which maintains all the powers in alpha.
1315  */
1316 double dp3m_real_space_error(double box_size, double prefac, double r_cut_iL,
1317  int n_c_part, double sum_q2, double alpha_L) {
1318  double d_error_f, d_cc, d_dc, d_rcut2, d_con;
1319  double d_a2, d_c, d_RCUT;
1320 
1321  d_RCUT = r_cut_iL * box_size;
1322  d_rcut2 = d_RCUT * d_RCUT;
1323 
1324  d_a2 = Utils::sqr(alpha_L) / Utils::sqr(box_size);
1325 
1326  d_c = sum_q2 * exp(-d_a2 * Utils::sqr(d_RCUT));
1327 
1328  d_cc =
1329  4.0 * Utils::sqr(d_a2) * Utils::sqr(d_rcut2) + 6.0 * d_a2 * d_rcut2 + 3.0;
1330 
1331  d_dc = 8.0 * Utils::int_pow<3>(d_a2) * Utils::int_pow<3>(d_rcut2) +
1332  20.0 * Utils::sqr(d_a2) * Utils::sqr(d_rcut2) + 30.0 * d_a2 * d_rcut2 +
1333  15.0;
1334 
1335  d_con = 1.0 / sqrt(Utils::int_pow<3>(box_size) * Utils::sqr(d_a2) *
1336  Utils::int_pow<4>(d_rcut2) * d_RCUT * (double)n_c_part);
1337 
1338  d_error_f = d_c * d_con *
1339  sqrt((13. / 6.) * Utils::sqr(d_cc) +
1340  (2. / 15.) * Utils::sqr(d_dc) - (13. / 15.) * d_cc * d_dc);
1341 
1342  return d_error_f;
1343 }

1345 /** Using bisection, find the root of a function "func-tuned_accuracy/sqrt(2.)"
1346  * known to lie between x1 and x2. The root, returned as rtbis, will be
1347  * refined until its accuracy is \f$\pm\f$ @p xacc.
1348  */
1349 double dp3m_rtbisection(double box_size, double prefac, double r_cut_iL,
1350  int n_c_part, double sum_q2, double x1, double x2,
1351  double xacc, double tuned_accuracy) {
1352  constexpr int JJ_RTBIS_MAX = 40;
1353 
1354  auto const constant = tuned_accuracy / Utils::sqrt_2();
1355 
1356  auto const f1 =
1357  dp3m_real_space_error(box_size, prefac, r_cut_iL, n_c_part, sum_q2, x1) -
1358  constant;
1359  auto const f2 =
1360  dp3m_real_space_error(box_size, prefac, r_cut_iL, n_c_part, sum_q2, x2) -
1361  constant;
1362  if (f1 * f2 >= 0.0) {
1363  runtimeErrorMsg()
1364  << "Root must be bracketed for bisection in dp3m_rtbisection";
1365  return -DP3M_RTBISECTION_ERROR;
1366  }
1367  // Orient the search dx, and set rtb to x1 or x2 ...
1368  double dx;
1369  double rtb = f1 < 0.0 ? (dx = x2 - x1, x1) : (dx = x1 - x2, x2);
1370  for (int j = 1; j <= JJ_RTBIS_MAX; j++) {
1371  auto const xmid = rtb + (dx *= 0.5);
1372  auto const fmid = dp3m_real_space_error(box_size, prefac, r_cut_iL,
1373  n_c_part, sum_q2, xmid) -
1374  constant;
1375  if (fmid <= 0.0)
1376  rtb = xmid;
1377  if (fabs(dx) < xacc || fmid == 0.0)
1378  return rtb;
1379  }
1380  runtimeErrorMsg() << "Too many bisections in dp3m_rtbisection";
1381  return -DP3M_RTBISECTION_ERROR;
1382 }
