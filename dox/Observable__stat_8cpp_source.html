45 Observable_stat::Observable_stat(std::size_t chunk_size)
46  : m_chunk_size(chunk_size) {
47  // number of chunks for different interaction types
48  constexpr std::size_t n_coulomb = 2;
49  constexpr std::size_t n_dipolar = 2;
50 #ifdef VIRTUAL_SITES
51  constexpr std::size_t n_vs = 1;
52 #else
53  constexpr std::size_t n_vs = 0;
54 #endif
55  auto const n_bonded = bonded_ia_params.size();
56  auto const n_non_bonded = max_non_bonded_pairs();
57  constexpr std::size_t n_ext_fields = 1; // reduction over all fields
58  constexpr std::size_t n_kinetic = 1; // linear+angular kinetic contributions
59 
60  auto const n_elements = n_kinetic + n_bonded + 2 * n_non_bonded + n_coulomb +
61  n_dipolar + n_vs + n_ext_fields;
62  m_data = std::vector<double>(m_chunk_size * n_elements);
63 
64  // spans for the different contributions
65  kinetic = Utils::Span<double>(m_data.data(), m_chunk_size);
66  bonded = Utils::Span<double>(kinetic.end(), n_bonded * m_chunk_size);
67  coulomb = Utils::Span<double>(bonded.end(), n_coulomb * m_chunk_size);
68  dipolar = Utils::Span<double>(coulomb.end(), n_dipolar * m_chunk_size);
69  virtual_sites = Utils::Span<double>(dipolar.end(), n_vs * m_chunk_size);
70  external_fields =
71  Utils::Span<double>(virtual_sites.end(), n_ext_fields * m_chunk_size);
72  non_bonded_intra =
73  Utils::Span<double>(external_fields.end(), n_non_bonded * m_chunk_size);
74  non_bonded_inter =
75  Utils::Span<double>(non_bonded_intra.end(), n_non_bonded * m_chunk_size);
76  assert(non_bonded_inter.end() == (m_data.data() + m_data.size()));
77 }
