33 IBMTribend::calc_forces(Particle const &p1, Particle const &p2,
34  Particle const &p3, Particle const &p4) const {
35 
36  // Get vectors making up the two triangles
37  auto const dx1 = box_geo.get_mi_vector(p1.r.p, p3.r.p);
38  auto const dx2 = box_geo.get_mi_vector(p2.r.p, p3.r.p);
39  auto const dx3 = box_geo.get_mi_vector(p4.r.p, p3.r.p);
40 
41  // Get normals on triangle; pointing outwards by definition of indices
42  // sequence
43  auto n1 = vector_product(dx1, dx2);
44  auto n2 = vector_product(dx3, dx1);
45 
46  // Get 2*area of triangles out of the magnitude of the resulting normals and
47  // make the latter unity
48  auto const Ai = n1.norm();
49  n1 /= Ai;
50 
51  auto const Aj = n2.norm();
52  n2 /= Aj;
53 
54  // Get the prefactor for the force term
55  auto const sc = std::min(1.0, n1 * n2);
56 
57  // Get theta as angle between normals
58  auto theta = acos(sc);
59 
60  auto const direc = vector_product(n1, n2);
61  auto const desc = (dx1 * direc);
62 
63  if (desc < 0)
64  theta *= -1;
65 
66  auto const DTh = theta - theta0;
67 
68  auto Pre = kb * DTh;
69  // Correct version with linearized sin
70  if (theta < 0)
71  Pre *= -1;
72 
73  auto const v1 = (n2 - sc * n1).normalize();
74  auto const v2 = (n1 - sc * n2).normalize();
75 
76  // Force on particles: eq. (C.28-C.31)
77  auto const force1 =
78  Pre * (vector_product(box_geo.get_mi_vector(p2.r.p, p3.r.p), v1) / Ai +
79  vector_product(box_geo.get_mi_vector(p3.r.p, p4.r.p), v2) / Aj);
80  auto const force2 =
81  Pre * (vector_product(box_geo.get_mi_vector(p3.r.p, p1.r.p), v1) / Ai);
82  auto const force3 =
83  Pre * (vector_product(box_geo.get_mi_vector(p1.r.p, p2.r.p), v1) / Ai +
84  vector_product(box_geo.get_mi_vector(p4.r.p, p1.r.p), v2) / Aj);
85  auto const force4 =
86  Pre * (vector_product(box_geo.get_mi_vector(p1.r.p, p3.r.p), v2) / Aj);
87  return std::make_tuple(force1, force2, force3, force4);
88 }

105  auto const dx2 = box_geo.get_mi_vector(p2.r.p, p3.r.p);
106  auto const dx3 = box_geo.get_mi_vector(p4.r.p, p3.r.p);
107 
108  // Get normals on triangle; pointing outwards by definition of indices
109  // sequence
110  auto const n1l = vector_product(dx1, dx2);
111  auto const n2l = -vector_product(dx1, dx3);
112 
113  auto const n1 = n1l / n1l.norm();
114  auto const n2 = n2l / n2l.norm();
115 
116  // calculate theta0 by taking the acos of the scalar n1*n2
117  auto const sc = std::min(1.0, n1 * n2);
118 
119  theta0 = acos(sc);
120 
121  auto const desc = dx1 * vector_product(n1, n2);
122  if (desc < 0)
123  theta0 = 2.0 * Utils::pi() - theta0;
124  }
125

127  // If triangle pairs appear only once, the total bending force should get a
128  // factor 2. For the numerical model, a factor sqrt(3) should be added, see
129  // @cite gompper96a and @cite kruger12a. This is an approximation,
130  // it holds strictly only for a sphere
131  this->kb = kb;
