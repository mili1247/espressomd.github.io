63 /**
64  * @brief Calculate the positional correction for the particles.
65  *
66  * @param ia_params Parameters
67  * @param p1 First particle.
68  * @param p2 Second particle.
69  * @return True if there was a correction.
70  */
71 static bool calculate_positional_correction(RigidBond const &ia_params,
72  Particle &p1, Particle &p2) {
73  auto const r_ij = box_geo.get_mi_vector(p1.r.p, p2.r.p);
74  auto const r_ij2 = r_ij.norm2();
75 
76  if (std::abs(1.0 - r_ij2 / ia_params.d2) > ia_params.p_tol) {
77  auto const r_ij_t =
78  box_geo.get_mi_vector(p1.r.p_last_timestep, p2.r.p_last_timestep);
79  auto const r_ij_dot = r_ij_t * r_ij;
80  auto const G =
81  0.50 * (ia_params.d2 - r_ij2) / r_ij_dot / (p1.p.mass + p2.p.mass);
82 
83  auto const pos_corr = G * r_ij_t;
84  p1.rattle.correction += pos_corr * p2.p.mass;
85  p2.rattle.correction -= pos_corr * p1.p.mass;
86 
87  return true;
88  }
89 
90  return false;
91 }

93 /**
94  * @brief Compute the correction vectors using given kernel.
95  *
96  * @param cs cell structure
97  * @param kernel kernel function
98  * @return True if correction is necessary
99  */
100 template <typename Kernel>
101 static bool compute_correction_vector(CellStructure &cs, Kernel kernel) {
102  bool correction = false;
103  cs.bond_loop([&correction, &kernel](Particle &p1, int bond_id,
104  Utils::Span<Particle *> partners) {
105  auto const &iaparams = bonded_ia_params[bond_id];
106 
107  if (auto const *bond = boost::get<RigidBond>(&iaparams)) {
108  auto const corrected = kernel(*bond, p1, *partners[0]);
109  if (corrected)
110  correction = true;
111  }
112 
113  /* Rigid bonds cannot break */
114  return false;
115  });
116 
117  return correction;
118 }

132 void correct_position_shake(CellStructure &cs) {
133  cells_update_ghosts(Cells::DATA_PART_POSITION | Cells::DATA_PART_PROPERTIES);
134 
135  auto particles = cs.local_particles();
136  auto ghost_particles = cs.ghost_particles();
137 
138  int cnt;
139  for (cnt = 0; cnt < SHAKE_MAX_ITERATIONS; ++cnt) {
140  init_correction_vector(particles, ghost_particles);
141  bool const repeat_ =
142  compute_correction_vector(cs, calculate_positional_correction);
143  bool const repeat =
144  boost::mpi::all_reduce(comm_cart, repeat_, std::logical_or<bool>());
145 
146  // no correction is necessary, skip communication and bail out
147  if (!repeat)
148  break;
149 
150  cell_structure.ghosts_reduce_rattle_correction();
151 
152  apply_positional_correction(particles);
153  cs.ghosts_update(Cells::DATA_PART_POSITION | Cells::DATA_PART_MOMENTUM);
154  }
155  if (cnt >= SHAKE_MAX_ITERATIONS) {
156  runtimeErrorMsg() << "RATTLE failed to converge after " << cnt
157  << " iterations";
158  }
159 
160  check_resort_particles();
161 }

163 /**
164  * @brief Calculate the velocity correction for the particles.
165  *
166  * The position correction is accumulated in the forces
167  * of the particles so that it can be reduced over the ghosts.
168  *
169  * @param ia_params Parameters
170  * @param p1 First particle.
171  * @param p2 Second particle.
172  * @return True if there was a correction.
173  */
174 static bool calculate_velocity_correction(RigidBond const &ia_params,
175  Particle &p1, Particle &p2) {
176  auto const v_ij = p1.m.v - p2.m.v;
177  auto const r_ij = box_geo.get_mi_vector(p1.r.p, p2.r.p);
178 
179  auto const v_proj = v_ij * r_ij;
180  if (std::abs(v_proj) > ia_params.v_tol) {
181  auto const K = v_proj / ia_params.d2 / (p1.p.mass + p2.p.mass);
182 
183  auto const vel_corr = K * r_ij;
184 
185  p1.rattle.correction -= vel_corr * p2.p.mass;
186  p2.rattle.correction += vel_corr * p1.p.mass;
187 
188  return true;
189  }
190 
191  return false;
192 }
