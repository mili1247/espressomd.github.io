65 /** Compute the harmonic bond force.
66  * @param[in] dx %Distance between the particles.
67  */
68 inline boost::optional<Utils::Vector3d>
69 HarmonicBond::force(Utils::Vector3d const &dx) const {
70  auto const dist = dx.norm();
71 
72  if ((r_cut > 0.0) && (dist > r_cut)) {
73  return {};
74  }
75 
76  auto const dr = dist - r;
77  auto fac = -k * dr;
78  if (dist > ROUND_ERROR_PREC) { /* Regular case */
79  fac /= dist;
80  } else {
81  fac = 0;
82  }
83  return fac * dx;
84 }

86 /** Compute the harmonic bond energy.
87  * @param[in] dx %Distance between the particles.
88  */
89 inline boost::optional<double>
90 HarmonicBond::energy(Utils::Vector3d const &dx) const {
91  auto const dist = dx.norm();
92 
93  if ((r_cut > 0.0) && (dist > r_cut)) {
94  return {};
95  }
96 
97  return 0.5 * k * Utils::sqr(dist - r);
98 }
