33 std::pair<double, double> Cylinder::dist_half_pore(double r, double z) const {
34  assert(z >= 0.0);
35  assert(r >= 0.0);
36 
37  if (z >= m_half_length || r >= m_rad) {
38  /* Outside */
39  if (!m_open && z >= m_half_length && r < m_rad) {
40  /* Closest feature: cap */
41  return {0, -(z - m_half_length)};
42  }
43  if (z >= m_half_length && (m_open || r >= m_rad)) {
44  /* Closest feature: ring */
45  return {-(r - m_rad), -(z - m_half_length)};
46  }
47  /* Closest feature: cylinder */
48  return {-(r - m_rad), 0};
49  }
50  /* Inside */
51  if (!m_open && z >= m_half_length - m_rad &&
52  r < (z - (m_half_length - m_rad))) {
53  /* Closest feature: cap */
54  return {0, m_half_length - z};
55  }
56  /* Closest feature: cylinder */
57  return {m_rad - r, 0};
58 }

60 void Cylinder::calculate_dist(const Utils::Vector3d &pos, double &dist,
61  Utils::Vector3d &vec) const {
62  /* Coordinate transform to cylinder coords
63  with origin at m_center. */
64  Utils::Vector3d const c_dist = pos - m_center;
65  auto const z = e_z * c_dist;
66  auto const r_vec = c_dist - z * e_z;
67  auto const r = r_vec.norm();
68 
69  /* If exactly on the axis, chose e_r orthogonal
70  to e_z. */
71  auto const e_r = (r == 0) ? e_r_axis : r_vec / r;
72 
73  /* The pore has mirror symmetry in z with regard to
74  the center in the {r,z} system. We calculate always
75  for the z > 0 case, and flip the result if appropriate. */
76  double dr, dz;
77  std::tie(dr, dz) = dist_half_pore(r, std::abs(z));
78 
79  double side = -1;
80  if (std::abs(z) >= m_half_length || r >= m_rad) /* outside */
81  side = 1;
82 
83  if (z <= 0.0) {
84  dz *= -1;
85  }
86 
87  dist = std::sqrt(dr * dr + dz * dz) * m_direction * side;
88  vec = -dr * e_r - dz * e_z;
89 }
