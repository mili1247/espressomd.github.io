45 /**
46  * Calculate dipolar energy and optionally force between two particles.
47  * @param[in,out] p1 First particle
48  * @param[in] dip1 Cached dipole moment of the first particle
49  * @param[in,out] p2 Second particle
50  * @param[in] force_flag If true, update the particle forces and torques
51  */
52 static double calc_dipole_dipole_ia(Particle &p1, Utils::Vector3d const &dip1,
53  Particle &p2, bool force_flag) {
54 
55  // Cache dipole moment
56  auto const dip2 = p2.calc_dip();
57 
58  // Distance between particles
59  auto const dr = box_geo.get_mi_vector(p1.r.p, p2.r.p);
60 
61  // Powers of distance
62  auto const r2 = dr.norm2();
63  auto const r = sqrt(r2);
64  auto const r3 = r2 * r;
65  auto const r5 = r3 * r2;
66  auto const r7 = r5 * r2;
67 
68  // Dot products
69  auto const pe1 = dip1 * dip2;
70  auto const pe2 = dip1 * dr;
71  auto const pe3 = dip2 * dr;
72  auto const pe4 = 3.0 / r5;
73 
74  // Energy
75  auto const energy = dipole.prefactor * (pe1 / r3 - pe4 * pe2 * pe3);
76 
77  // Forces, if requested
78  if (force_flag) {
79  auto const a = pe4 * pe1;
80  auto const b = -15.0 * pe2 * pe3 / r7;
81  auto const ab = a + b;
82  auto const cc = pe4 * pe3;
83  auto const dd = pe4 * pe2;
84 
85  // Forces
86  auto const ff = ab * dr + cc * dip1 + dd * dip2;
87  p1.f.f += dipole.prefactor * ff;
88  p2.f.f -= dipole.prefactor * ff;
89 
90  // Torques
91  auto const aa = vector_product(dip1, dip2);
92  auto const b1 = vector_product(dip1, dr);
93  auto const b2 = vector_product(dip2, dr);
94  p1.f.torque += dipole.prefactor * (-aa / r3 + b1 * cc);
95  p2.f.torque += dipole.prefactor * (aa / r3 + b2 * dd);
96  }
97 
98  return energy;
99 }

106 double dawaanr_calculations(bool force_flag, bool energy_flag,
107  const ParticleRange &particles) {
108 
109  assert(n_nodes == 1);
110  assert(force_flag || energy_flag);
111 
112  double energy = 0.0;
113  // Iterate over all particles
114  for (auto it = particles.begin(), end = particles.end(); it != end; ++it) {
115  // If the particle has no dipole moment, ignore it
116  if (it->p.dipm == 0.0)
117  continue;
118 
119  const Utils::Vector3d dip1 = it->calc_dip();
120  auto jt = it;
121  /* Skip diagonal */
122  ++jt;
123  for (; jt != end; ++jt) {
124  // If the particle has no dipole moment, ignore it
125  if (jt->p.dipm == 0.0)
126  continue;
127  // Calculate energy and/or force between the particles
128  energy += calc_dipole_dipole_ia(*it, dip1, *jt, force_flag);
129  }
130  }
131 
132  return energy;
133 }

150 double
151 magnetic_dipolar_direct_sum_calculations(bool force_flag, bool energy_flag,
152  ParticleRange const &particles) {
153 
154  assert(n_nodes == 1);
155  assert(force_flag || energy_flag);
156 
157  std::vector<double> x, y, z;
158  std::vector<double> mx, my, mz;
159  std::vector<double> fx, fy, fz;
160  std::vector<double> tx, ty, tz;
161 
162  auto const n_part = particles.size();
163 
164  x.resize(n_part);
165  y.resize(n_part);
166  z.resize(n_part);
167 
168  mx.resize(n_part);
169  my.resize(n_part);
170  mz.resize(n_part);
171 
172  if (force_flag) {
173  fx.resize(n_part);
174  fy.resize(n_part);
175  fz.resize(n_part);
176  tx.resize(n_part);
177  ty.resize(n_part);
178  tz.resize(n_part);
179  }
180 
181  int dip_particles = 0;
182  for (auto const &p : particles) {
183  if (p.p.dipm != 0.0) {
184  const Utils::Vector3d dip = p.calc_dip();
185 
186  mx[dip_particles] = dip[0];
187  my[dip_particles] = dip[1];
188  mz[dip_particles] = dip[2];
189 
190  /* here we wish the coordinates to be folded into the primary box */
191  auto const ppos = folded_position(p.r.p, box_geo);
192  x[dip_particles] = ppos[0];
193  y[dip_particles] = ppos[1];
194  z[dip_particles] = ppos[2];
195 
196  if (force_flag) {
197  fx[dip_particles] = 0;
198  fy[dip_particles] = 0;
199  fz[dip_particles] = 0;
200  tx[dip_particles] = 0;
201  ty[dip_particles] = 0;
202  tz[dip_particles] = 0;
203  }
204 
205  dip_particles++;
206  }
207  }
208 
209  /* energy calculation */
210  double energy = 0.;
211 
212  int NCUT[3];
213  for (int i = 0; i < 3; i++) {
214  NCUT[i] = box_geo.periodic(i) ? mdds_n_replica : 0;
215  }
216  auto const NCUT2 = Utils::sqr(mdds_n_replica);
217 
218  for (int i = 0; i < dip_particles; i++) {
219  for (int j = 0; j < dip_particles; j++) {
220  auto const pe1 = mx[i] * mx[j] + my[i] * my[j] + mz[i] * mz[j];
221  auto const rx = x[i] - x[j];
222  auto const ry = y[i] - y[j];
223  auto const rz = z[i] - z[j];
224 
225  for (int nx = -NCUT[0]; nx <= NCUT[0]; nx++) {
226  auto const rnx = rx + nx * box_geo.length()[0];
227  auto const rnx2 = rnx * rnx;
228  for (int ny = -NCUT[1]; ny <= NCUT[1]; ny++) {
229  auto const rny = ry + ny * box_geo.length()[1];
230  auto const rny2 = rny * rny;
231  for (int nz = -NCUT[2]; nz <= NCUT[2]; nz++) {
232  if (!(i == j && nx == 0 && ny == 0 && nz == 0) and
233  (nx * nx + ny * ny + nz * nz <= NCUT2)) {
234  auto const rnz = rz + nz * box_geo.length()[2];
235  auto const r2 = rnx2 + rny2 + rnz * rnz;
236  auto const r = sqrt(r2);
237  auto const r3 = r2 * r;
238  auto const r5 = r3 * r2;
239  auto const r7 = r5 * r2;
240 
241  auto const pe2 = mx[i] * rnx + my[i] * rny + mz[i] * rnz;
242  auto const pe3 = mx[j] * rnx + my[j] * rny + mz[j] * rnz;
243  auto const pe4 = 3.0 / r5;
244 
245  // Energy
246  energy += pe1 / r3 - pe4 * pe2 * pe3;
247 
248  if (force_flag) {
249  // Forces
250  auto const a = pe4 * pe1;
251  auto const b = -15.0 * pe2 * pe3 / r7;
252  auto const c = pe4 * pe3;
253  auto const d = pe4 * pe2;
254 
255  fx[i] += (a + b) * rnx + c * mx[i] + d * mx[j];
256  fy[i] += (a + b) * rny + c * my[i] + d * my[j];
257  fz[i] += (a + b) * rnz + c * mz[i] + d * mz[j];
258 
259  // Torques
260  auto const ax = my[i] * mz[j] - my[j] * mz[i];
261  auto const ay = mx[j] * mz[i] - mx[i] * mz[j];
262  auto const az = mx[i] * my[j] - mx[j] * my[i];
263 
264  auto const bx = my[i] * rnz - rny * mz[i];
265  auto const by = rnx * mz[i] - mx[i] * rnz;
266  auto const bz = mx[i] * rny - rnx * my[i];
267 
268  tx[i] += -ax / r3 + bx * c;
269  ty[i] += -ay / r3 + by * c;
270  tz[i] += -az / r3 + bz * c;
271  } /* if force_flag */
272  } /* if distance criterion */
273  } /* for nz */
274  } /* for ny */
275  } /* for nx */
276  } /* for j */
277  } /* for i */
278 
279  /* update particle forces and torques */
280  if (force_flag) {
281 
282  dip_particles = 0;
283 
284  for (auto &p : particles) {
285  if (p.p.dipm != 0.0) {
286 
287  p.f.f[0] += dipole.prefactor * fx[dip_particles];
288  p.f.f[1] += dipole.prefactor * fy[dip_particles];
289  p.f.f[2] += dipole.prefactor * fz[dip_particles];
290 
291  p.f.torque[0] += dipole.prefactor * tx[dip_particles];
292  p.f.torque[1] += dipole.prefactor * ty[dip_particles];
293  p.f.torque[2] += dipole.prefactor * tz[dip_particles];
294 
295  dip_particles++;
296  }
297  }
298  } /* if force_flag */
299 
300  return 0.5 * dipole.prefactor * energy;
301 }
