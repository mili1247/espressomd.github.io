36 /** @brief Type trait to indicate that a type is
37  * serializable with a static size, e.g. is
38  * suitable for memcpy serialization. Only
39  * specialize this to std::true_type if it is
40  * guaranteed that serializing this type always
41  * returns the same number of bytes, independent
42  * of object state.
43  *
44  * @tparam T type under consideration.
45  */
46 template <class T>
47 struct is_statically_serializable
48  : std::integral_constant<
49  bool, std::is_trivially_copyable<T>::value or
50  boost::serialization::is_bitwise_serializable<T>::value> {};

52 namespace detail {
53 /* Use memcpy for packing */
54 template <class T>
55 using use_memcpy = std::integral_constant<
56  bool, std::is_trivially_copyable<T>::value or
57  boost::serialization::is_bitwise_serializable<T>::value>;
58 /* Use serialize function only if the type is opt-in but not
59  * trivially copyable, in which case memcpy is more efficient. */
60 template <class T>
61 using use_serialize =
62  std::integral_constant<bool, not use_memcpy<T>::value and
63  is_statically_serializable<T>::value>;

65 template <class Derived> class BasicMemcpyArchive {
66  /** Buffer to write to */
67  Utils::Span<char> buf;
68  /** Current position in the buffer */
69  char *insert;
70 
71 public:
72  explicit BasicMemcpyArchive(Utils::Span<char> buf)
73  : buf(buf), insert(buf.data()) {}
74 
75  std::size_t get_library_version() const { return 4; }
76 
77  std::size_t bytes_processed() const { return insert - buf.data(); }
78  void skip(std::size_t bytes) {
79  assert((insert + bytes) <= buf.end());
80  insert += bytes;
81  }
82 
83 private:
84  void read(void *data, std::size_t bytes) {
85  /* check that there is enough space left in the buffer */
86  assert((insert + bytes) <= buf.end());
87  std::memcpy(data, insert, bytes);
88  insert += bytes;
89  }
90 
91  void write(const void *data, std::size_t bytes) {
92  /* check that there is enough space left in the buffer */
93  assert((insert + bytes) <= buf.end());
94  std::memcpy(insert, data, bytes);
95  insert += bytes;
96  }
97 
98 public:
99  template <typename T>
100  auto operator>>(T &value) -> std::enable_if_t<use_memcpy<T>::value> {
101  read(&value, sizeof(T));
102  }
103 
104  template <typename T>
105  auto operator<<(T const &value) -> std::enable_if_t<use_memcpy<T>::value> {
106  write(&value, sizeof(T));
107  }
108 
109 private:
110  template <class T> void process(T &value) {
111  auto const old_pos = insert;
112  boost::serialization::serialize_adl(*static_cast<Derived *>(this), value,
113  4);
114  auto const new_pos = insert;
115  assert((new_pos - old_pos) <= sizeof(T));
116 
117  auto const padding_size = sizeof(T) - (new_pos - old_pos);
118  skip(padding_size);
119  }
120 
121 public:
122  template <class T>
123  auto operator>>(T &value)
124  -> std::enable_if_t<detail::use_serialize<T>::value> {
125  process(value);
126  }
127 
128  template <class T>
129  auto operator<<(T &value)
130  -> std::enable_if_t<detail::use_serialize<T>::value> {
131  process(value);
132  }
133 
134  template <class T> void operator<<(const boost::serialization::nvp<T> &nvp) {
135  operator<<(nvp.const_value());
136  }
137 
138  template <class T> void operator>>(const boost::serialization::nvp<T> &nvp) {
139  operator>>(nvp.value());
140  }
