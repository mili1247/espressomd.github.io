85 template <std::size_t I, std::size_t N> struct find_if_impl {
86  template <class Pred, class Tuple, class F>
87  static constexpr bool eval(Pred &&pred, Tuple const &t, F &&f) {
88  using Utils::get;
89  auto const &e = get<I>(t);
90 
91  if (pred(e)) {
92  (void)f(e);
93  return true;
94  }
95 
96  return find_if_impl<I + 1, N>::eval(std::forward<Pred>(pred), t,
97  std::forward<F>(f));
98  }
99 };

109 /**
110  * @brief Find first element in tuple for which a predicate holds,
111  * and call a Callable with the element.
112  *
113  * Any return value of the Callable is ignored.
114  *
115  * @tparam Pred Must be callable with all types occurring as elements
116  * in Tuple, returning a value convertible to bool.
117  * @tparam Tuple a tuple-like type supporting Utils::get
118  * @tparam F Must be callable with all types occurring as elements
119  * in Tuple.
120  * @param pred The predicate.
121  * @param t Tuple for search in.
122  * @param f Is called for first found element.
123  * @return true iff the predicate was true for at least one element.
124  */
125 template <class Pred, class Tuple, class F>
126 constexpr auto find_if(Pred &&pred, Tuple const &t, F &&f) {
127  return detail::find_if_impl<0, Utils::tuple_size<Tuple>::value>::eval(
128  std::forward<Pred>(pred), t, std::forward<F>(f));
129 }

147  using Utils::tuple_element_t;

149  constexpr bool pred = Predicate<element_type>::value;
150 
151  return std::tuple_cat(

153  filter_impl<Predicate, I + 1, N>::eval(t));

166 /**
167  * @brief Filter a tuple by a static predicate.
168  *
169  * E.g. filter(std::is_integral, std::make_tuple(1, 1.5, 2u))
170  * -> std::tuple<int, unsigned>(1, 2u).
171  *
172  * @tparam Predicate Metafunction that returns a bool for each type.
173  * @tparam Tuple tuple-like type that implements the tuple interface.
174  * @param in The input tuple-like
175  *
176  * @return std::tuple with the elements selected by the predicate.
177  */
178 template <template <class> class Predicate, class Tuple>
179 constexpr auto filter(Tuple const &in) {
180  using Utils::tuple_size;
181 
182  return detail::filter_impl<Predicate, 0, tuple_size<Tuple>::value>::eval(in);
183 }
