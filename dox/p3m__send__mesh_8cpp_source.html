36 void p3m_send_mesh::resize(const boost::mpi::communicator &comm,
37  const P3MLocalMesh &local_mesh) {
38  int done[3] = {0, 0, 0};
39  /* send grids */
40  for (int i = 0; i < 3; i++) {
41  for (int j = 0; j < 3; j++) {
42  /* left */
43  s_ld[i * 2][j] = 0 + done[j] * local_mesh.margin[j * 2];
44  if (j == i)
45  s_ur[i * 2][j] = local_mesh.margin[j * 2];
46  else
47  s_ur[i * 2][j] =
48  local_mesh.dim[j] - done[j] * local_mesh.margin[(j * 2) + 1];
49  /* right */
50  if (j == i)
51  s_ld[(i * 2) + 1][j] = local_mesh.in_ur[j];
52  else
53  s_ld[(i * 2) + 1][j] = 0 + done[j] * local_mesh.margin[j * 2];
54  s_ur[(i * 2) + 1][j] =
55  local_mesh.dim[j] - done[j] * local_mesh.margin[(j * 2) + 1];
56  }
57  done[i] = 1;
58  }
59  max = 0;
60  for (int i = 0; i < 6; i++) {
61  s_size[i] = 1;
62  for (int j = 0; j < 3; j++) {
63  s_dim[i][j] = s_ur[i][j] - s_ld[i][j];
64  s_size[i] *= s_dim[i][j];
65  }
66  if (s_size[i] > max)
67  max = s_size[i];
68  }
69  /* communication */
70  auto const node_neighbors = Utils::Mpi::cart_neighbors<3>(comm);
71 
72  int r_margin[6];
73  for (int i = 0; i < 6; i++) {
74  auto const j = (i % 2 == 0) ? i + 1 : i - 1;
75 
76  if (node_neighbors[i] != comm.rank()) {
77  MPI_Sendrecv(&(local_mesh.margin[i]), 1, MPI_INT, node_neighbors[i],
78  REQ_P3M_INIT, &(r_margin[j]), 1, MPI_INT, node_neighbors[j],
79  REQ_P3M_INIT, comm, MPI_STATUS_IGNORE);
80  } else {
81  r_margin[j] = local_mesh.margin[i];
82  }
83  }
84  /* recv grids */
85  for (int i = 0; i < 3; i++)
86  for (int j = 0; j < 3; j++) {
87  if (j == i) {
88  r_ld[i * 2][j] = s_ld[i * 2][j] + local_mesh.margin[2 * j];
89  r_ur[i * 2][j] = s_ur[i * 2][j] + r_margin[2 * j];
90  r_ld[(i * 2) + 1][j] = s_ld[(i * 2) + 1][j] - r_margin[(2 * j) + 1];
91  r_ur[(i * 2) + 1][j] =
92  s_ur[(i * 2) + 1][j] - local_mesh.margin[(2 * j) + 1];
93  } else {
94  r_ld[i * 2][j] = s_ld[i * 2][j];
95  r_ur[i * 2][j] = s_ur[i * 2][j];
96  r_ld[(i * 2) + 1][j] = s_ld[(i * 2) + 1][j];
97  r_ur[(i * 2) + 1][j] = s_ur[(i * 2) + 1][j];
98  }
99  }
100  for (int i = 0; i < 6; i++) {
101  r_size[i] = 1;
102  for (int j = 0; j < 3; j++) {
103  r_dim[i][j] = r_ur[i][j] - r_ld[i][j];
104  r_size[i] *= r_dim[i][j];
105  }
106  if (r_size[i] > max)
107  max = r_size[i];
108  }
109 }

111 void p3m_send_mesh::gather_grid(Utils::Span<double *> meshes,
112  const boost::mpi::communicator &comm,
113  const Utils::Vector3i &dim) {
114  auto const node_neighbors = Utils::Mpi::cart_neighbors<3>(comm);
115  send_grid.resize(max * meshes.size());
116  recv_grid.resize(max * meshes.size());
117 
118  /* direction loop */
119  for (int s_dir = 0; s_dir < 6; s_dir++) {
120  auto const r_dir = (s_dir % 2 == 0) ? s_dir + 1 : s_dir - 1;
121 
122  /* pack send block */
123  if (s_size[s_dir] > 0)
124  for (std::size_t i = 0; i < meshes.size(); i++) {
125  fft_pack_block(meshes[i], send_grid.data() + i * s_size[s_dir],
126  s_ld[s_dir], s_dim[s_dir], dim.data(), 1);
127  }
128 
129  /* communication */
130  if (node_neighbors[s_dir] != comm.rank()) {
131  MPI_Sendrecv(
132  send_grid.data(), static_cast<int>(meshes.size()) * s_size[s_dir],
133  MPI_DOUBLE, node_neighbors[s_dir], REQ_P3M_GATHER, recv_grid.data(),
134  static_cast<int>(meshes.size()) * r_size[r_dir], MPI_DOUBLE,
135  node_neighbors[r_dir], REQ_P3M_GATHER, comm, MPI_STATUS_IGNORE);
136  } else {
137  std::swap(send_grid, recv_grid);
138  }
139  /* add recv block */
140  if (r_size[r_dir] > 0) {
141  for (std::size_t i = 0; i < meshes.size(); i++) {
142  p3m_add_block(recv_grid.data() + i * r_size[r_dir], meshes[i],
143  r_ld[r_dir], r_dim[r_dir], dim.data());
144  }
145  }
146  }
147 }

149 void p3m_send_mesh::spread_grid(Utils::Span<double *> meshes,
150  const boost::mpi::communicator &comm,
151  const Utils::Vector3i &dim) {
152  auto const node_neighbors = Utils::Mpi::cart_neighbors<3>(comm);
153  send_grid.resize(max * meshes.size());
154  recv_grid.resize(max * meshes.size());
155 
156  /* direction loop */
157  for (int s_dir = 5; s_dir >= 0; s_dir--) {
158  auto const r_dir = (s_dir % 2 == 0) ? s_dir + 1 : s_dir - 1;
159 
160  /* pack send block */
161  if (r_size[r_dir] > 0)
162  for (std::size_t i = 0; i < meshes.size(); i++) {
163  fft_pack_block(meshes[i], send_grid.data() + i * r_size[r_dir],
164  r_ld[r_dir], r_dim[r_dir], dim.data(), 1);
165  }
166  /* communication */
167  if (node_neighbors[r_dir] != comm.rank()) {
168  MPI_Sendrecv(
169  send_grid.data(), r_size[r_dir] * static_cast<int>(meshes.size()),
170  MPI_DOUBLE, node_neighbors[r_dir], REQ_P3M_SPREAD, recv_grid.data(),
171  s_size[s_dir] * static_cast<int>(meshes.size()), MPI_DOUBLE,
172  node_neighbors[s_dir], REQ_P3M_SPREAD, comm, MPI_STATUS_IGNORE);
173  } else {
174  std::swap(send_grid, recv_grid);
175  }
176  /* un pack recv block */
177  if (s_size[s_dir] > 0) {
178  for (std::size_t i = 0; i < meshes.size(); i++) {
179  fft_unpack_block(recv_grid.data() + i * s_size[s_dir], meshes[i],
180  s_ld[s_dir], s_dim[s_dir], dim.data(), 1);
181  }
182  }
183  }
