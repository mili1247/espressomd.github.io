31 /** Returns true if the particles are to be considered for short range
32  * interactions.
33  */
34 class VerletCriterion {
35  const double m_skin;
36  const double m_eff_max_cut2;
37  const double m_eff_coulomb_cut2 = 0.;
38  const double m_eff_dipolar_cut2 = 0.;
39  const double m_collision_cut2 = 0.;
40  double eff_cutoff_sqr(double x) const {
41  if (x == INACTIVE_CUTOFF)
42  return INACTIVE_CUTOFF;
43  return Utils::sqr(x + m_skin);
44  }
45 
46 public:
47  VerletCriterion(double skin, double max_cut, double coulomb_cut = 0.,
48  double dipolar_cut = 0.,
49  double collision_detection_cutoff = 0.)
50  : m_skin(skin), m_eff_max_cut2(eff_cutoff_sqr(max_cut)),
51  m_eff_coulomb_cut2(eff_cutoff_sqr(coulomb_cut)),
52  m_eff_dipolar_cut2(eff_cutoff_sqr(dipolar_cut)),
53  m_collision_cut2(eff_cutoff_sqr(collision_detection_cutoff)) {}
54 
55  template <typename Distance>
56  bool operator()(const Particle &p1, const Particle &p2,
57  Distance const &dist) const {
58  auto const &dist2 = dist.dist2;
59  if (dist2 > m_eff_max_cut2)
60  return false;
61 
62 #ifdef ELECTROSTATICS
63  // Within real space cutoff of electrostatics and both charged
64  if ((dist2 <= m_eff_coulomb_cut2) && (p1.p.q != 0) && (p2.p.q != 0))
65  return true;
66 #endif
67 
68 #ifdef DIPOLES
69  // Within dipolar cutoff and both carry magnetic moments
70  if ((dist2 <= m_eff_dipolar_cut2) && (p1.p.dipm != 0) && (p2.p.dipm != 0))
71  return true;
72 #endif
73 
74 #ifdef COLLISION_DETECTION
75  // Collision detection
76  if (dist2 <= m_collision_cut2)
77  return true;
78 #endif
79 
80  // Within short-range distance (including dpd and the like)
81  auto const max_cut = get_ia_param(p1.p.type, p2.p.type)->max_cut;
82  return (max_cut != INACTIVE_CUTOFF) &&
83  (dist2 <= Utils::sqr(max_cut + m_skin));
84  }
85 };
