50 /** Initialize volume conservation */
51 void ImmersedBoundaries::init_volume_conservation(CellStructure &cs) {
52  // Check since this function is called at the start of every integrate loop
53  // Also check if volume has been set due to reading of a checkpoint
54  if (not BoundariesFound) {
55  BoundariesFound = boost::algorithm::any_of(
56  bonded_ia_params, [](auto const &bonded_ia_param) {
57  return (boost::get<IBMVolCons>(&bonded_ia_param) != nullptr);
58  });
59  }
60 
61  if (!VolumeInitDone && BoundariesFound) {
62  // Calculate volumes
63  calc_volumes(cs);
64 
65  // Loop through all bonded interactions and check if we need to set the
66  // reference volume
67  for (auto &bonded_ia_param : bonded_ia_params) {
68  if (auto *v = boost::get<IBMVolCons>(&bonded_ia_param)) {
69  // This check is important because InitVolumeConservation may be called
70  // accidentally during the integration. Then we must not reset the
71  // reference
72  BoundariesFound = true;
73  if (v->volRef == 0.) {
74  v->volRef = VolumesCurrent[v->softID];
75  }
76  }
77  }
78 
79  VolumeInitDone = true;
80  }
81 }

84  auto it = boost::find_if(p1.bonds(), [](auto const &bond) {
85  return boost::get<IBMVolCons>(&bonded_ia_params[bond.bond_id()]) != nullptr;

93 /** Calculate partial volumes on all compute nodes and call MPI to sum up.
94  * See @cite zhang01b, @cite dupin08a, @cite kruger12a.
95  */
96 void ImmersedBoundaries::calc_volumes(CellStructure &cs) {
97 
98  if (!BoundariesFound)
99  return;
100 
101  // Partial volumes for each soft particle, to be summed up
102  std::vector<double> tempVol(VolumesCurrent.size());
103 
104  // Loop over all particles on local node
105  cs.bond_loop(
106  [&tempVol](Particle &p1, int bond_id, Utils::Span<Particle *> partners) {
107  auto const &iaparams = bonded_ia_params[bond_id];
108  auto vol_cons_params = vol_cons_parameters(p1);
109 
110  if (vol_cons_params &&
111  (boost::get<IBMTriel>(&bonded_ia_params[bond_id]) != nullptr)) {
112  // Our particle is the leading particle of a triel
113  // Get second and third particle of the triangle
114  Particle &p2 = *partners[0];
115  Particle &p3 = *partners[1];
116 
117  // Unfold position of first node.
118  // This is to get a continuous trajectory with no jumps when box
119  // boundaries are crossed.
120  auto const x1 = unfolded_position(p1.r.p, p1.l.i, box_geo.length());
121  auto const x2 = x1 + box_geo.get_mi_vector(p2.r.p, x1);
122  auto const x3 = x1 + box_geo.get_mi_vector(p3.r.p, x1);
123 
124  // Volume of this tetrahedron
125  // See @cite zhang01b
126  // The volume can be negative, but it is not necessarily the
127  // "signed volume" in the above paper (the sign of the real
128  // "signed volume" must be calculated using the normal vector; the
129  // result of the calculation here is simply a term in the sum
130  // required to calculate the volume of a particle). Again, see the
131  // paper. This should be equivalent to the formulation using
132  // vector identities in @cite kruger12a
133 
134  const double v321 = x3[0] * x2[1] * x1[2];
135  const double v231 = x2[0] * x3[1] * x1[2];
136  const double v312 = x3[0] * x1[1] * x2[2];
137  const double v132 = x1[0] * x3[1] * x2[2];
138  const double v213 = x2[0] * x1[1] * x3[2];
139  const double v123 = x1[0] * x2[1] * x3[2];
140 
141  tempVol[vol_cons_params->softID] +=
142  1.0 / 6.0 * (-v321 + v231 + v312 - v132 - v213 + v123);
143  }
144  return false;
145  });
146 
147  // Sum up and communicate
148  boost::mpi::all_reduce(comm_cart, tempVol.data(),
149  static_cast<int>(tempVol.size()),
150  VolumesCurrent.data(), std::plus<double>());
151 }

153 /** Calculate and add the volume force to each node */
154 void ImmersedBoundaries::calc_volume_force(CellStructure &cs) {
155  if (!BoundariesFound)
156  return;
157 
158  cs.bond_loop(
159  [this](Particle &p1, int bond_id, Utils::Span<Particle *> partners) {
160  if (boost::get<IBMTriel>(&bonded_ia_params[bond_id]) != nullptr) {
161  // Check if particle has an IBM Triel bonded interaction and an
162  // IBM VolCons bonded interaction. Basically this loops over all
163  // triangles, not all particles. First round to check for volume
164  // conservation.
165  const IBMVolCons *ibmVolConsParameters = vol_cons_parameters(p1);
166  if (not ibmVolConsParameters)
167  return false;
168 
169  auto current_volume = VolumesCurrent[ibmVolConsParameters->softID];
170 
171  // Our particle is the leading particle of a triel
172  // Get second and third particle of the triangle
173  Particle &p2 = *partners[0];
174  Particle &p3 = *partners[1];
175 
176  // Unfold position of first node.
177  // This is to get a continuous trajectory with no jumps when box
178  // boundaries are crossed.
179  auto const x1 = unfolded_position(p1.r.p, p1.l.i, box_geo.length());
180 
181  // Unfolding seems to work only for the first particle of a triel
182  // so get the others from relative vectors considering PBC
183  auto const a12 = box_geo.get_mi_vector(p2.r.p, x1);
184  auto const a13 = box_geo.get_mi_vector(p3.r.p, x1);
185 
186  // Now we have the true and good coordinates
187  // This is eq. (9) in @cite dupin08a.
188  auto const n = vector_product(a12, a13);
189  const double ln = n.norm();
190  const double A = 0.5 * ln;
191  const double fact = ibmVolConsParameters->kappaV *
192  (current_volume - ibmVolConsParameters->volRef) /
193  current_volume;
194 
195  auto const nHat = n / ln;
196  auto const force = -fact * A * nHat;
197 
198  p1.f.f += force;
199  p2.f.f += force;
200  p3.f.f += force;
201  }
202  return false;
203  });
204 }
