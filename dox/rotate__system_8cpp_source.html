39 void local_rotate_system(double phi, double theta, double alpha,
40  const ParticleRange &particles) {
41  // Calculate center of mass
42  Utils::Vector3d local_com{};
43  double local_mass = 0.0;
44 
45  for (auto const &p : particles) {
46  if (not p.p.is_virtual) {
47  local_com += p.p.mass * p.r.p;
48  local_mass += p.p.mass;
49  }
50  }
51 
52  auto const total_mass = mpi::all_reduce(comm_cart, local_mass, std::plus<>());
53  auto const com =
54  mpi::all_reduce(comm_cart, local_com, std::plus<>()) / total_mass;
55 
56  // Rotation axis in Cartesian coordinates
57  Utils::Vector3d axis;
58  axis[0] = sin(theta) * cos(phi);
59  axis[1] = sin(theta) * sin(phi);
60  axis[2] = cos(theta);
61 
62  // Rotate particle coordinates
63  for (auto &p : particles) {
64  // Move the center of mass of the system to the origin
65  for (int j = 0; j < 3; j++) {
66  p.r.p[j] -= com[j];
67  }
68 
69  p.r.p = com + Utils::vec_rotate(axis, alpha, p.r.p);
70 #ifdef ROTATION
71  local_rotate_particle(p, axis, alpha);
72 #endif
73  }
74 
75  cell_structure.set_resort_particles(Cells::RESORT_GLOBAL);
76  on_particle_change();
77  update_dependent_particles();
78 }
