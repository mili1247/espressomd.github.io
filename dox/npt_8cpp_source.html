76 void integrator_npt_coulomb_dipole_sanity_checks(
77  NptIsoParameters const &params) {
78 #ifdef ELECTROSTATICS
79  if (params.dimension < 3 && !params.cubic_box && coulomb.prefactor > 0) {
80  throw std::runtime_error("If electrostatics is being used you must "
81  "use the cubic box NpT.");
82  }
83 #endif
84 
85 #ifdef DIPOLES
86  if (params.dimension < 3 && !params.cubic_box && dipole.prefactor > 0) {
87  throw std::runtime_error("If magnetostatics is being used you must "
88  "use the cubic box NpT.");
89  }
90 #endif
91 
92 #if defined(ELECTROSTATICS) && defined(CUDA)
93  if (coulomb.method == COULOMB_P3M_GPU) {
94  throw std::runtime_error("NpT virial cannot be calculated on P3M GPU");
95  }
96 #endif
97 }

99 void nptiso_init(double ext_pressure, double piston, bool xdir_rescale,
100  bool ydir_rescale, bool zdir_rescale, bool cubic_box) {
101 
102  if (ext_pressure < 0.0) {
103  throw std::runtime_error("The external pressure must be positive.");
104  }
105  if (piston <= 0.0) {
106  throw std::runtime_error("The piston mass must be positive.");
107  }
108 
109  NptIsoParameters new_nptiso = {piston,
110  nptiso.inv_piston,
111  nptiso.volume,
112  ext_pressure,
113  nptiso.p_inst,
114  nptiso.p_diff,
115  nptiso.p_vir,
116  nptiso.p_vel,
117  0,
118  nptiso.nptgeom_dir,
119  0,
120  cubic_box,
121  -1};
122 
123  /* set the NpT geometry */
124  if (xdir_rescale) {
125  new_nptiso.geometry |= NPTGEOM_XDIR;
126  new_nptiso.dimension += 1;
127  new_nptiso.non_const_dim = 0;
128  }
129  if (ydir_rescale) {
130  new_nptiso.geometry |= NPTGEOM_YDIR;
131  new_nptiso.dimension += 1;
132  new_nptiso.non_const_dim = 1;
133  }
134  if (zdir_rescale) {
135  new_nptiso.geometry |= NPTGEOM_ZDIR;
136  new_nptiso.dimension += 1;
137  new_nptiso.non_const_dim = 2;
138  }
139 
140  if (new_nptiso.dimension == 0 || new_nptiso.non_const_dim == -1) {
141  throw std::runtime_error(
142  "You must enable at least one of the x y z components "
143  "as fluctuating dimension(s) for box length motion!");
144  }
145 
146  integrator_npt_coulomb_dipole_sanity_checks(new_nptiso);
147 
148  nptiso = new_nptiso;
149 
150  mpi_bcast_nptiso_geom_barostat();
151 }

153 void npt_ensemble_init(const BoxGeometry &box) {
154  if (integ_switch == INTEG_METHOD_NPT_ISO) {
155  /* prepare NpT-integration */
156  nptiso.inv_piston = 1 / nptiso.piston;
157  nptiso.volume = pow(box.length()[nptiso.non_const_dim], nptiso.dimension);
158  if (recalc_forces) {
159  nptiso.p_inst = 0.0;
160  nptiso.p_vir = Utils::Vector3d{};
161  nptiso.p_vel = Utils::Vector3d{};
162  }
163  }
164 }
