41 Lattice::Lattice(double agrid, double offset, int halo_size,
42  Utils::Vector3d const &local_box,
43  Utils::Vector3d const &my_right,
44  Utils::Vector3d const &box_length,
45  Utils::Vector3i const &node_pos,
46  Utils::Vector3i const &node_grid)
47  : agrid(agrid), halo_size(halo_size), offset(offset), node_grid(node_grid),
48  local_box(local_box), my_right(my_right) {
49  /* determine the number of local lattice nodes */
50  auto const epsilon = std::numeric_limits<double>::epsilon();
51  for (int d = 0; d < 3; d++) {
52  grid[d] = static_cast<int>(round(local_box[d] / agrid));
53  global_grid[d] = node_grid[d] * grid[d];
54  local_index_offset[d] = node_pos[d] * grid[d];
55  }
56 
57  // sanity checks
58  for (int dir = 0; dir < 3; dir++) {
59  // check if local_box_l is compatible with lattice spacing
60  auto diff = fabs(local_box[dir] - grid[dir] * agrid);
61  if (diff > epsilon * box_length[dir]) {
62  throw std::runtime_error(
63  "Lattice spacing agrid[" + std::to_string(dir) +
64  "]=" + std::to_string(agrid) + " is incompatible with local_box_l[" +
65  std::to_string(dir) + "]=" + std::to_string(local_box[dir]) +
66  " ( box_l[" + std::to_string(dir) +
67  "]=" + std::to_string(box_length[dir]) +
68  " ). Mismatch: " + std::to_string(diff));
69  }
70  }
71 
72  /* determine the number of total nodes including halo */
73  halo_grid = grid + Utils::Vector3i::broadcast(2 * halo_size);
74  halo_grid_volume = boost::accumulate(halo_grid, 1, std::multiplies<int>());
75  halo_offset =
76  Utils::get_linear_index(halo_size, halo_size, halo_size, halo_grid);
77 }

84 void Lattice::map_position_to_lattice(const Utils::Vector3d &pos,
85  Utils::Vector<std::size_t, 8> &node_index,
86  Utils::Vector6d &delta) const {
87  Utils::Vector3i ind{};
88  auto const epsilon = std::numeric_limits<double>::epsilon();
89 
90  /* determine the elementary lattice cell containing the particle
91  and the relative position of the particle in this cell */
92  for (int dir = 0; dir < 3; dir++) {
93  auto const lpos = pos[dir] - (my_right[dir] - local_box[dir]);
94  auto const rel = lpos / agrid + offset;
95  ind[dir] = static_cast<int>(floor(rel));
96 
97  /* surrounding elementary cell is not completely inside this box,
98  adjust if this is due to round off errors */
99  if (ind[dir] < 0) {
100  if (fabs(rel) < epsilon) {
101  ind[dir] = 0; // TODO
102  } else {
103  throw std::runtime_error("position not inside a local plaquette");
104  }
105  } else if (ind[dir] > grid[dir]) {
106  if (lpos - local_box[dir] < epsilon * local_box[dir])
107  ind[dir] = grid[dir];
108  else
109  throw std::runtime_error("position not inside a local plaquette");
110  }
111 
112  delta[3 + dir] = rel - ind[dir]; // delta_x/a
113  delta[dir] = 1.0 - delta[3 + dir];
114  }
115  node_index[0] = Utils::get_linear_index(ind, halo_grid);
116  node_index[1] = node_index[0] + 1u;
117  node_index[2] = node_index[0] + static_cast<std::size_t>(halo_grid[0]);
118  node_index[3] = node_index[0] + static_cast<std::size_t>(halo_grid[0]) + 1u;
119  node_index[4] = node_index[0] + static_cast<std::size_t>(halo_grid[0]) *
120  static_cast<std::size_t>(halo_grid[1]);
121  node_index[5] = node_index[4] + 1u;
122  node_index[6] = node_index[4] + static_cast<std::size_t>(halo_grid[0]);
123  node_index[7] = node_index[4] + static_cast<std::size_t>(halo_grid[0]) + 1u;
124 }
