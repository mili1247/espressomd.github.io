51 static void backup_file(const std::string &from, const std::string &to) {
52  /*
53  * If the file itself *and* a backup file exists, something must
54  * have gone wrong.
55  */
56  boost::filesystem::path pfrom(from), pto(to);
57  try {
58  boost::filesystem::copy_file(
59  pfrom, pto, boost::filesystem::copy_option::fail_if_exists);
60  } catch (const boost::filesystem::filesystem_error &e) {
61  throw left_backupfile();
62  }
63 }

65 template <typename extent_type>
66 static void extend_dataset(h5xx::dataset &dataset,
67  extent_type const &change_extent) {
68  auto const rank = static_cast<h5xx::dataspace>(dataset).rank();
69  auto extents = static_cast<h5xx::dataspace>(dataset).extents();
70  /* Extend the dataset for another timestep */
71  for (int i = 0; i < rank; i++) {
72  extents[i] += change_extent[i];
73  }
74  H5Dset_extent(dataset.hid(), extents.data()); // extend all dims is collective
75 }

98 void File::init_file(std::string const &file_path) {
99  m_backup_filename = file_path + ".bak";
100  boost::filesystem::path script_path(m_script_path);
101  m_absolute_script_path = boost::filesystem::canonical(script_path);
102  bool file_exists = boost::filesystem::exists(file_path);
103  bool backup_file_exists = boost::filesystem::exists(m_backup_filename);
104  /* Perform a barrier synchronization. Otherwise one process might already
105  * create the file while another still checks for its existence. */
106  m_comm.barrier();
107  if (file_exists) {
108  if (H5MD_Specification::is_compliant(file_path)) {
109  /*
110  * If the file exists and has a valid H5MD structure, let's create a
111  * backup of it. This has the advantage, that the new file can
112  * just be deleted if the simulation crashes at some point and we
113  * still have a valid trajectory, we can start from.
114  */
115  if (m_comm.rank() == 0)
116  backup_file(file_path, m_backup_filename);
117  load_file(file_path);
118  } else {
119  throw incompatible_h5mdfile();
120  }
121  } else {
122  if (backup_file_exists)
123  throw left_backupfile();
124  create_file(file_path);
125  }
126 }

143 static std::vector<hsize_t> create_dims(hsize_t rank, hsize_t data_dim) {
144  switch (rank) {
145  case 3:
146  return std::vector<hsize_t>{0, 0, data_dim};
147  case 2:
148  return std::vector<hsize_t>{0, data_dim};

153  "H5MD Error: datasets with this dimension are not implemented\n");
154  }

157 static std::vector<hsize_t> create_chunk_dims(hsize_t rank, hsize_t data_dim) {
158  hsize_t chunk_size = (rank > 1) ? 1000 : 1;
159  switch (rank) {
160  case 3:
161  return {1, chunk_size, data_dim};

164  case 1:
165  return {chunk_size};
166  default:
167  throw std::runtime_error(
168  "H5MD Error: datasets with this dimension are not implemented\n");

172 void File::create_datasets() {
173  namespace hps = h5xx::policy::storage;
174  for (const auto &d : H5MD_Specification::DATASETS) {
175  if (d.is_link)
176  continue;
177  auto maxdims = std::vector<hsize_t>(d.rank, H5S_UNLIMITED);
178  auto dataspace = h5xx::dataspace(create_dims(d.rank, d.data_dim), maxdims);
179  auto storage = hps::chunked(create_chunk_dims(d.rank, d.data_dim))
180  .set(hps::fill_value(-10));
181  datasets[d.path()] = h5xx::dataset(m_h5md_file, d.path(), d.type, dataspace,
182  storage, H5P_DEFAULT, H5P_DEFAULT);
183  }
184 }

288 void write_td_particle_property(hsize_t prefix, hsize_t n_part_global,
289  ParticleRange const &particles,
290  h5xx::dataset &dataset, Op op) {
291  auto const old_extents = static_cast<h5xx::dataspace>(dataset).extents();
292  auto const extent_particle_number =
293  std::max(n_part_global, old_extents[1]) - old_extents[1];
294  extend_dataset(dataset,
295  detail::slice_info<dim>::extent(extent_particle_number));
296  auto const count = detail::slice_info<dim>::count();
297  auto offset = detail::slice_info<dim>::offset(old_extents[0], prefix);
298  for (auto const &p : particles) {
299  h5xx::write_dataset(dataset, op(p), h5xx::slice(offset, count));
300  // advance in the particle dimension
301  offset[1] += 1;
302  }
303 }

305 void File::write(const ParticleRange &particles, double time, int step,
306  BoxGeometry const &geometry) {
307  write_box(geometry, m_h5md_file, datasets["particles/atoms/box/edges/value"]);
308  write_connectivity(particles);
309 
310  int const n_part_local = particles.size();
311  // calculate count and offset
312  int prefix = 0;
313  // calculate prefix for write of the current process
314  BOOST_MPI_CHECK_RESULT(MPI_Exscan,
315  (&n_part_local, &prefix, 1, MPI_INT, MPI_SUM, m_comm));
316  auto const extents =
317  static_cast<h5xx::dataspace>(datasets["particles/atoms/id/value"])
318  .extents();
319 
320  auto const n_part_global =
321  boost::mpi::all_reduce(m_comm, n_part_local, std::plus<int>());
322 
323  write_td_particle_property<2>(
324  prefix, n_part_global, particles, datasets["particles/atoms/id/value"],
325  [](auto const &p) { return Utils::Vector<int, 1>{p.p.identity}; });
326  write_dataset(Utils::Vector<double, 1>{time},
327  datasets["particles/atoms/id/time"], Vector1hs{1},
328 
329  Vector1hs{extents[0]}, Vector1hs{1});
330  write_dataset(Utils::Vector<int, 1>{step},
331  datasets["particles/atoms/id/step"], Vector1hs{1},
332  Vector1hs{extents[0]}, Vector1hs{1});
333 
334  write_td_particle_property<2>(
335  prefix, n_part_global, particles,
336  datasets["particles/atoms/species/value"],
337  [](auto const &p) { return Utils::Vector<int, 1>{p.p.type}; });
338 
339  write_td_particle_property<2>(
340  prefix, n_part_global, particles, datasets["particles/atoms/mass/value"],
341  [](auto const &p) { return Utils::Vector<double, 1>{p.p.mass}; });
342 
343  write_td_particle_property<3>(
344  prefix, n_part_global, particles,
345  datasets["particles/atoms/position/value"],
346  [&](auto const &p) { return folded_position(p.r.p, geometry); });
347  write_td_particle_property<3>(prefix, n_part_global, particles,
348  datasets["particles/atoms/image/value"],
349  [](auto const &p) { return p.l.i; });
350 
351  write_td_particle_property<3>(prefix, n_part_global, particles,
352  datasets["particles/atoms/velocity/value"],
353  [](auto const &p) { return p.m.v; });
354 
355  write_td_particle_property<3>(prefix, n_part_global, particles,
356  datasets["particles/atoms/force/value"],
357  [](auto const &p) { return p.f.f; });
358  write_td_particle_property<2>(
359  prefix, n_part_global, particles,
360  datasets["particles/atoms/charge/value"],
361  [](auto const &p) { return Utils::Vector<double, 1>{p.p.q}; });
362 }
363 void File::write_connectivity(const ParticleRange &particles) {
364  MultiArray3i bond(boost::extents[0][0][0]);
365  int particle_index = 0;
366  for (auto const &p : particles) {
367  int nbonds_local = bond.shape()[1];
368  for (auto const b : p.bonds()) {
369  auto const partner_ids = b.partner_ids();
370  if (partner_ids.size() == 1) {
371  bond.resize(boost::extents[1][nbonds_local + 1][2]);
372  bond[0][nbonds_local][0] = p.p.identity;
373  bond[0][nbonds_local][1] = partner_ids[0];
374  nbonds_local++;
375  }
376  }
377  particle_index++;
378  }
379 
380  int n_bonds_local = bond.shape()[1];
381  int prefix_bonds = 0;
382  BOOST_MPI_CHECK_RESULT(
383  MPI_Exscan, (&n_bonds_local, &prefix_bonds, 1, MPI_INT, MPI_SUM, m_comm));
384  auto const n_bonds_total =
385  boost::mpi::all_reduce(m_comm, n_bonds_local, std::plus<int>());
386  auto const extents =
387  static_cast<h5xx::dataspace>(datasets["connectivity/atoms/value"])
388  .extents();
389  Vector3hs offset_bonds = {extents[0], static_cast<hsize_t>(prefix_bonds), 0};
390  Vector3hs count_bonds = {1, static_cast<hsize_t>(n_bonds_local), 2};
391  auto const n_bond_diff =
392  std::max(static_cast<hsize_t>(n_bonds_total), extents[1]) - extents[1];
393  Vector3hs change_extent_bonds = {1, static_cast<hsize_t>(n_bond_diff), 0};
394  write_dataset(bond, datasets["connectivity/atoms/value"], change_extent_bonds,
395  offset_bonds, count_bonds);
396 }
