68  void calculate_sampling_positions() {
69  auto const lim = limits();
70  sampling_positions.clear();
71  assert(sampling_delta[0] > 0. and sampling_delta[1] > 0. and
72  sampling_delta[2] > 0.);
73  assert(sampling_offset[0] >= 0. and sampling_offset[1] >= 0. and
74  sampling_offset[2] >= 0.);
75  const auto n_samples_x = static_cast<std::size_t>(
76  std::rint((lim[0].second - lim[0].first) / sampling_delta[0]));
77  const auto n_samples_y = static_cast<std::size_t>(
78  std::rint((lim[1].second - lim[1].first) / sampling_delta[1]));
79  const auto n_samples_z = static_cast<std::size_t>(
80  std::rint((lim[2].second - lim[2].first) / sampling_delta[2]));
81  for (std::size_t x = 0; x < n_samples_x; ++x) {
82  for (std::size_t y = 0; y < n_samples_y; ++y) {
83  for (std::size_t z = 0; z < n_samples_z; ++z) {
84  sampling_positions.push_back(Utils::Vector3d{
85  {lim[0].first + sampling_offset[0] +
86  static_cast<double>(x) * sampling_delta[0],
87  lim[1].first + sampling_offset[1] +
88  static_cast<double>(y) * sampling_delta[1],
89  lim[2].first + sampling_offset[2] +
90  static_cast<double>(z) * sampling_delta[2]}});
91  }
92  }
93  }
94  }
