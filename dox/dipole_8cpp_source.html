170 void calc_long_range_force(const ParticleRange &particles) {
171  switch (dipole.method) {
172 #ifdef DP3M
173  case DIPOLAR_MDLC_P3M:
174  add_mdlc_force_corrections(particles);
175  // fall through
176  case DIPOLAR_P3M:
177  dp3m_dipole_assign(particles);
178 #ifdef NPT
179  if (integ_switch == INTEG_METHOD_NPT_ISO) {
180  auto const energy = dp3m_calc_kspace_forces(true, true, particles);
181  npt_add_virial_contribution(energy);
182  fprintf(stderr, "dipolar_P3M at this moment is added to p_vir[0]\n");
183  } else
184 #endif
185  dp3m_calc_kspace_forces(true, false, particles);
186 
187  break;
188 #endif
189  case DIPOLAR_ALL_WITH_ALL_AND_NO_REPLICA:
190  dawaanr_calculations(true, false, particles);
191  break;
192 #ifdef DP3M
193  case DIPOLAR_MDLC_DS:
194  add_mdlc_force_corrections(particles);
195  // fall through
196 #endif
197  case DIPOLAR_DS:
198  magnetic_dipolar_direct_sum_calculations(true, false, particles);
199  break;
200  case DIPOLAR_DS_GPU: // NOLINT(bugprone-branch-clone)
201  // do nothing: it's an actor
202  break;
203 #ifdef DIPOLAR_BARNES_HUT
204  case DIPOLAR_BH_GPU:
205  // do nothing: it's an actor
206  break;
207 #endif
208 #ifdef SCAFACOS_DIPOLES
209  case DIPOLAR_SCAFACOS:
210  Scafacos::fcs_dipoles()->add_long_range_force();
211 #endif
212  case DIPOLAR_NONE:
213  break;
214  default:
215  runtimeErrorMsg() << "unknown dipolar method";
216  break;
217  }
218 }

220 double calc_energy_long_range(const ParticleRange &particles) {
221  double energy = 0.;
222  switch (dipole.method) {
223 #ifdef DP3M
224  case DIPOLAR_P3M:
225  dp3m_dipole_assign(particles);
226  energy = dp3m_calc_kspace_forces(false, true, particles);
227  break;
228  case DIPOLAR_MDLC_P3M:
229  dp3m_dipole_assign(particles);
230  energy = dp3m_calc_kspace_forces(false, true, particles);
231  energy += add_mdlc_energy_corrections(particles);
232  break;
233 #endif
234  case DIPOLAR_ALL_WITH_ALL_AND_NO_REPLICA:
235  energy = dawaanr_calculations(false, true, particles);
236  break;
237 #ifdef DP3M
238  case DIPOLAR_MDLC_DS:
239  energy = magnetic_dipolar_direct_sum_calculations(false, true, particles);
240  energy += add_mdlc_energy_corrections(particles);
241  break;
242 #endif
243  case DIPOLAR_DS:
244  energy = magnetic_dipolar_direct_sum_calculations(false, true, particles);
245  break;
246  case DIPOLAR_DS_GPU: // NOLINT(bugprone-branch-clone)
247  // do nothing: it's an actor
248  break;
249 #ifdef DIPOLAR_BARNES_HUT
250  case DIPOLAR_BH_GPU:
251  // do nothing: it's an actor.
252  break;
253 #endif
254 #ifdef SCAFACOS_DIPOLES
255  case DIPOLAR_SCAFACOS:
256  energy = Scafacos::fcs_dipoles()->long_range_energy();
257  break;
258 #endif
259  case DIPOLAR_NONE:
260  break;
261  default:
262  runtimeErrorMsg()
263  << "energy calculation not implemented for dipolar method.";
264  break;
265  }
266  return energy;
267 }
