56 std::vector<double>
57 RDF::evaluate(Utils::Span<const Particle *const> particles1,
58  Utils::Span<const Particle *const> particles2) const {
59  auto const bin_width = (max_r - min_r) / static_cast<double>(n_r_bins);
60  auto const inv_bin_width = 1.0 / bin_width;
61  std::vector<double> res(n_values(), 0.0);
62  long int cnt = 0;
63  auto op = [=, &cnt, &res](const Particle *const p1,
64  const Particle *const p2) {
65  auto const dist = box_geo.get_mi_vector(p1->r.p, p2->r.p).norm();
66  if (dist > min_r && dist < max_r) {
67  auto const ind =
68  static_cast<int>(std::floor((dist - min_r) * inv_bin_width));
69  res[ind]++;
70  }
71  cnt++;
72  };
73  if (particles2.empty()) {
74  Utils::for_each_pair(particles1, op);
75  } else {
76  auto cmp = std::not_equal_to<const Particle *const>();
77  Utils::for_each_cartesian_pair_if(particles1, particles2, op, cmp);
78  }
79  if (cnt == 0)
80  return res;
81  // normalization
82  auto const volume = box_geo.volume();
83  for (int i = 0; i < n_r_bins; ++i) {
84  auto const r_in = i * bin_width + min_r;
85  auto const r_out = r_in + bin_width;
86  auto const bin_volume =
87  (4.0 / 3.0) * Utils::pi() *
88  (Utils::int_pow<3>(r_out) - Utils::int_pow<3>(r_in));
89  res[i] *= volume / (bin_volume * static_cast<double>(cnt));
90  }
91 
92  return res;
93 }
