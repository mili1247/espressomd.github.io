40 template <typename T> T g_ewald(T alpha, T k2) {
41  auto constexpr limit = T{30};
42  auto const exponent = Utils::sqr(1. / (2. * alpha)) * k2;
43  return (exponent < limit) ? std::exp(-exponent) * (4. * Utils::pi<T>() / k2)
44  : 0.0;
45 }

47 template <std::size_t S, std::size_t m>
48 std::pair<double, double> aliasing_sums_ik(std::size_t cao, double alpha,
49  const Utils::Vector3d &k,
50  const Utils::Vector3d &h) {
51  using namespace detail::FFT_indexing;
52  using Utils::int_pow;
53  using Utils::sinc;
54  using Utils::Vector3d;
55 
56  constexpr double two_pi = 2 * Utils::pi();
57  constexpr double two_pi_i = 1 / two_pi;
58 
59  double numerator = 0.0;
60  double denominator = 0.0;
61 
62  for (int mx = -m; mx <= m; mx++) {
63  for (int my = -m; my <= m; my++) {
64  for (int mz = -m; mz <= m; mz++) {
65  auto const km =
66  k + two_pi * Vector3d{mx / h[RX], my / h[RY], mz / h[RZ]};
67  auto const U2 = std::pow(sinc(km[RX] * h[RX] * two_pi_i) *
68  sinc(km[RY] * h[RY] * two_pi_i) *
69  sinc(km[RZ] * h[RZ] * two_pi_i),
70  2 * cao);
71 
72  numerator += U2 * g_ewald(alpha, km.norm2()) * int_pow<S>(k * km);
73  denominator += U2;
74  }
75  }
76  }
77  return {numerator, denominator};
78 }
79 } // namespace detail

81 /**
82  * @brief Hockney/Eastwood/Ballenegger optimal influence function.
83  *
84  * This implements Eq. 30 of @cite cerda08d, which can be used
85  * for monopole and dipole P3M by choosing the appropriate S factor.
86  *
87  * @tparam S Order of the differential operator, e.g. 0 for potential,
88  * 1 for electric field, ...
89  * @tparam m Number of aliasing terms to take into account.
90  * @tparam T Floating-point type.
91  *
92  * @param cao Charge assignment order.
93  * @param alpha Ewald splitting parameter.
94  * @param k k Vector to evaluate the function for.
95  * @param h Grid spacing.
96  */
97 template <std::size_t S, std::size_t m, class T>
98 double G_opt(std::size_t cao, T alpha, const Utils::Vector3<T> &k,
99  const Utils::Vector3<T> &h) {
100  using Utils::int_pow;
101  using Utils::sqr;
102 
103  auto const k2 = k.norm2();
104  if (k2 == 0.0) {
105  return 0.0;
106  }
107 
108  const auto as = detail::aliasing_sums_ik<S, m>(cao, alpha, k, h);
109  return as.first / (int_pow<S>(k2) * sqr(as.second));
110 }

112 /**
113  * @brief Map influence function over a grid.
114  *
115  * This evaluates the optimal influence function @ref G_opt
116  * over a regular grid of k vectors, and returns the values as a vector.
117  *
118  * @tparam S Order of the differential operator, e.g. 0 for potential,
119  * 1 for electric field...
120  * @tparam m Number of aliasing terms to take into account.
121  *
122  * @param params P3M parameters
123  * @param n_start Lower left corner of the grid
124  * @param n_end Upper right corner of the grid.
125  * @param box_l Box size
126  * @return Values of G_opt at regular grid points.
127  */
128 template <std::size_t S, std::size_t m = 0>
129 std::vector<double> grid_influence_function(const P3MParameters &params,
130  const Utils::Vector3i &n_start,
131  const Utils::Vector3i &n_end,
132  const Utils::Vector3d &box_l) {
133  using namespace detail::FFT_indexing;
134 
135  auto const shifts =
136  detail::calc_meshift({params.mesh[0], params.mesh[1], params.mesh[2]});
137 
138  auto const size = n_end - n_start;
139 
140  /* The influence function grid */
141  auto g =
142  std::vector<double>(boost::accumulate(size, 1, std::multiplies<>()), 0.);
143 
144  /* Skip influence function calculation in tuning mode,
145  the results need not be correct for timing. */
146  if (params.tuning) {
147  return g;
148  }
149 
150  auto const h = Utils::Vector3d{params.a};
151 
152  Utils::Vector3i n{};
153  for (n[0] = n_start[0]; n[0] < n_end[0]; n[0]++) {
154  for (n[1] = n_start[1]; n[1] < n_end[1]; n[1]++) {
155  for (n[2] = n_start[2]; n[2] < n_end[2]; n[2]++) {
156  auto const ind = Utils::get_linear_index(n - n_start, size,
157  Utils::MemoryOrder::ROW_MAJOR);
158  if ((n[KX] % (params.mesh[RX] / 2) == 0) &&
159  (n[KY] % (params.mesh[RY] / 2) == 0) &&
160  (n[KZ] % (params.mesh[RZ] / 2) == 0)) {
161  g[ind] = 0.0;
162  } else {
163  auto const k = 2 * Utils::pi() *
164  Utils::Vector3d{shifts[RX][n[KX]] / box_l[RX],
165  shifts[RY][n[KY]] / box_l[RY],
166  shifts[RZ][n[KZ]] / box_l[RZ]};
167 
168  g[ind] = G_opt<S, m>(params.cao, params.alpha, k, h);
169  }
170  }
171  }
172  }
173 
174  return g;
175 }
