54 /** Return a random uniform 3D vector with the Philox thermostat.
55  * Random numbers depend on
56  * 1. dpd_rng_counter (initialized by seed) which is increased on integration
57  * 2. Salt (decorrelates different counters)
58  * 3. Two particle IDs (order-independent, decorrelates particles, gets rid of
59  * seed-per-node)
60  */
61 Vector3d dpd_noise(uint32_t pid1, uint32_t pid2) {
62  return Random::noise_uniform<RNGSalt::SALT_DPD>(
63  dpd.rng_counter(), dpd.rng_seed(), (pid1 < pid2) ? pid2 : pid1,
64  (pid1 < pid2) ? pid1 : pid2);
65 }

80 void dpd_init(double kT, double time_step) {
81  for (int type_a = 0; type_a < max_seen_particle_type; type_a++) {
82  for (int type_b = 0; type_b < max_seen_particle_type; type_b++) {
83  IA_parameters &ia_params = *get_ia_param(type_a, type_b);
84 
85  ia_params.dpd_radial.pref =
86  sqrt(24.0 * kT * ia_params.dpd_radial.gamma / time_step);
87  ia_params.dpd_trans.pref =
88  sqrt(24.0 * kT * ia_params.dpd_trans.gamma / time_step);
89  }
90  }
91 }

100 Vector3d dpd_pair_force(DPDParameters const &params, Vector3d const &v,
101  double dist, Vector3d const &noise) {
102  if (dist < params.cutoff) {
103  auto const omega = weight(params.wf, params.cutoff, params.k, dist);
104  auto const omega2 = Utils::sqr(omega);
105 
106  auto const f_d = params.gamma * omega2 * v;
107  auto const f_r = params.pref * omega * noise;
108 
109  return f_r - f_d;
110  }
111 
112  return {};
113 }

115 Utils::Vector3d dpd_pair_force(Particle const &p1, Particle const &p2,
116  IA_parameters const &ia_params,
117  Utils::Vector3d const &d, double dist,
118  double dist2) {
119  if (ia_params.dpd_radial.cutoff <= 0.0 && ia_params.dpd_trans.cutoff <= 0.0) {
120  return {};
121  }
122 
123  auto const v21 = p1.m.v - p2.m.v;
124  auto const noise_vec =
125  (ia_params.dpd_radial.pref > 0.0 || ia_params.dpd_trans.pref > 0.0)
126  ? dpd_noise(p1.p.identity, p2.p.identity)
127  : Vector3d{};
128 
129  auto const f_r = dpd_pair_force(ia_params.dpd_radial, v21, dist, noise_vec);
130  auto const f_t = dpd_pair_force(ia_params.dpd_trans, v21, dist, noise_vec);
131 
132  /* Projection operator to radial direction */
133  auto const P = tensor_product(d / dist2, d);
134  /* This is equivalent to P * f_r + (1 - P) * f_t, but with
135  * doing only one matrix-vector multiplication */
136  auto const force = P * (f_r - f_t) + f_t;
137  return force;
138 }

140 static auto dpd_viscous_stress_local() {
141  on_observable_calc();
142 
143  Utils::Matrix<double, 3, 3> stress{};
144  cell_structure.non_bonded_loop(
145  [&stress](const Particle &p1, const Particle &p2, Distance const &d) {
146  auto const v21 = p1.m.v - p2.m.v;
147 
148  IA_parameters const &ia_params = *get_ia_param(p1.p.type, p2.p.type);
149  auto const dist = std::sqrt(d.dist2);
150 
151  auto const f_r = dpd_pair_force(ia_params.dpd_radial, v21, dist, {});
152  auto const f_t = dpd_pair_force(ia_params.dpd_trans, v21, dist, {});
153 
154  /* Projection operator to radial direction */
155  auto const P = tensor_product(d.vec21 / d.dist2, d.vec21);
156  /* This is equivalent to P * f_r + (1 - P) * f_t, but with
157  * doing only one matrix-vector multiplication */
158  auto const f = P * (f_r - f_t) + f_t;
159 
160  stress += tensor_product(d.vec21, f);
161  });
162 
163  return stress;
164 }

167 /**
168  * @brief Viscous stress tensor of the DPD interaction.
169  *
170  * This calculates the total viscous stress contribution of the
171  * DPD interaction. It contains only the dissipative contributions
172  * of the interaction without noise. It's calculated as the
173  * sum over all pair virials as
174  * \f[
175  * \sigma^{\nu\mu} = V^{-1}\sum_i \sum_{j < i} r_{i,j}^{\nu} (- \gamma_{i,j}
176  * v_{i,j})^{\mu} \f] where \f$\gamma_{i,j}\f$ is the (in general tensor valued)
177  * DPD friction coefficient for particles i and j, \f$v_{i,j}\f$, \f$r_{i,j}\f$
178  * are their relative velocity and distance and \f$V\f$ is the box volume.
179  *
180  * @return Stress tensor contribution.
181  */
182 Utils::Vector9d dpd_stress() {
183  auto const stress = mpi_call(Communication::Result::reduction, std::plus<>(),
184  dpd_viscous_stress_local);
185  auto const volume = box_geo.volume();
186 
187  return Utils::flatten(stress) / volume;
188 }
