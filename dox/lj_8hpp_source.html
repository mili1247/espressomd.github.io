44 /** Calculate Lennard-Jones force factor */
45 inline double lj_pair_force_factor(IA_parameters const &ia_params,
46  double dist) {
47  if ((dist < ia_params.lj.cut + ia_params.lj.offset) &&
48  (dist > ia_params.lj.min + ia_params.lj.offset)) {
49  auto const r_off = dist - ia_params.lj.offset;
50  auto const frac6 = Utils::int_pow<6>(ia_params.lj.sig / r_off);
51  return 48.0 * ia_params.lj.eps * frac6 * (frac6 - 0.5) / (r_off * dist);
52  }
53  return 0.0;
54 }
55 
56 /** Calculate Lennard-Jones energy */
57 inline double lj_pair_energy(IA_parameters const &ia_params, double dist) {
58  if ((dist < ia_params.lj.cut + ia_params.lj.offset) &&
59  (dist > ia_params.lj.min + ia_params.lj.offset)) {
60  auto const r_off = dist - ia_params.lj.offset;
61  auto const frac6 = Utils::int_pow<6>(ia_params.lj.sig / r_off);
62  return 4.0 * ia_params.lj.eps *
63  (Utils::sqr(frac6) - frac6 + ia_params.lj.shift);
64  }
65  return 0.0;
66 }
