21 /** \file
22  *
23  * Concerning the file layouts.
24  * - Scalar arrays are written like this:
25  * rank0 --- rank1 --- rank2 ...
26  * where each rank dumps its scalars in the ordering of the particles.
27  * - Vector arrays are written in the rank ordering like scalar arrays.
28  * The ordering of the vector data is: v[0] v[1] v[2], so the data
29  * looks like this:
30  * v1[0] v1[1] v1[2] v2[0] v2[1] v2[2] v3[0] ...
31  *
32  * To be able to determine the rank boundaries (a multiple of
33  * nlocalparts), the file 1.pref is written, which dumps the Exscan
34  * results of nlocalparts, i.e. the prefixes in scalar arrays:
35  * - 1.prefs looks like this:
36  * 0 nlocalpats_rank0 nlocalparts_rank0+nlocalparts_rank1 ...
37  *
38  * Bonds are dumped as two arrays, namely 1.bond which stores the
39  * bonding partners of the particles and 1.boff which stores the
40  * iteration indices for each particle.
41  * - 1.boff is a scalar array of size (nlocalpart + 1) per rank.
42  * - The last element (at index nlocalpart) of 1.boff's subpart
43  * [rank * (nlocalpart + 1) : (rank + 1) * (nlocalpart + 1)]
44  * determines the number of bonds for processor "rank".
45  * - In this subarray one can find the bonding partners of particle
46  * id[i]. The iteration indices for local part of 1.bonds are:
47  * subarray[i] : subarray[i+1]
48  * - Take a look at the bond input code. It's easy to understand.
49  */

80 /** Dumps arr of size len starting from prefix pref of type T using
81  * MPI_T as MPI datatype. Beware, that T and MPI_T have to match!
82  *
83  * \param fn The file name to dump to. Must not exist already
84  * \param arr The array to dump
85  * \param len The number of elements to dump
86  * \param pref The prefix for this process
87  * \param MPI_T The MPI_Datatype corresponding to the template parameter T.
88  */
89 template <typename T>
90 static void mpiio_dump_array(const std::string &fn, T *arr, std::size_t len,
91  std::size_t pref, MPI_Datatype MPI_T) {
92  MPI_File f;
93  int ret;
94 
95  ret = MPI_File_open(MPI_COMM_WORLD, const_cast<char *>(fn.c_str()),
96  // MPI_MODE_EXCL: Prohibit overwriting
97  MPI_MODE_WRONLY | MPI_MODE_CREATE | MPI_MODE_EXCL,
98  MPI_INFO_NULL, &f);
99  if (ret) {
100  char buf[MPI_MAX_ERROR_STRING];
101  int buf_len;
102  MPI_Error_string(ret, buf, &buf_len);
103  buf[buf_len] = '\0';
104  fprintf(stderr, "MPI-IO Error: Could not open file \"%s\": %s\n",
105  fn.c_str(), buf);
106  errexit();
107  }
108  ret = MPI_File_set_view(f, pref * sizeof(T), MPI_T, MPI_T,
109  const_cast<char *>("native"), MPI_INFO_NULL);
110  ret |= MPI_File_write_all(f, arr, len, MPI_T, MPI_STATUS_IGNORE);
111  MPI_File_close(&f);
112  if (ret) {
113  fprintf(stderr, "MPI-IO Error: Could not write file \"%s\".\n", fn.c_str());
114  errexit();
115  }
116 }

118 /** Dumps some generic infos like the dumped fields and info to process
119  * the bond information offline (without ESPResSo). To be called by the
120  * master node only.
121  *
122  * \param fn The filename to write to
123  * \param fields The dumped fields
124  */
125 static void dump_info(const std::string &fn, unsigned fields) {
126  FILE *f = fopen(fn.c_str(), "wb");
127  if (!f) {
128  fprintf(stderr, "MPI-IO Error: Could not open %s for writing.\n",
129  fn.c_str());
130  errexit();
131  }
132  static std::vector<int> npartners;
133  int success = (fwrite(&fields, sizeof(fields), 1, f) == 1);
134  // Pack the necessary information of bonded_ia_params:
135  // The number of partners. This is needed to interpret the bond IntList.
136  if (bonded_ia_params.size() > npartners.size())
137  npartners.resize(bonded_ia_params.size());
138 
139  for (int i = 0; i < bonded_ia_params.size(); ++i) {
140  npartners[i] = number_of_partners(bonded_ia_params[i]);
141  }
142  auto ia_params_size = static_cast<std::size_t>(bonded_ia_params.size());
143  success =
144  success && (fwrite(&ia_params_size, sizeof(std::size_t), 1, f) == 1);
145  success =
146  success && (fwrite(npartners.data(), sizeof(int), bonded_ia_params.size(),
147  f) == bonded_ia_params.size());
148  fclose(f);
149  if (!success) {
150  fprintf(stderr, "MPI-IO Error: Failed to write %s.\n", fn.c_str());
151  errexit();
152  }
153 }

155 void mpi_mpiio_common_write(const char *filename, unsigned fields,
156  const ParticleRange &particles) {
157  std::string fnam(filename);
158  int const nlocalpart = static_cast<int>(particles.size());
159  // Keep static buffers in order not having to allocate them on every
160  // function call
161  static std::vector<double> pos, vel;
162  static std::vector<int> id, type;
163 
164  // Nlocalpart prefixes
165  // Prefixes based for arrays: 3 * pref for vel, pos.
166  int pref = 0, bpref = 0;
167  MPI_Exscan(&nlocalpart, &pref, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
168 
169  // Realloc static buffers if necessary
170  if (nlocalpart > id.size())
171  id.resize(nlocalpart);
172  if (fields & MPIIO_OUT_POS && 3 * nlocalpart > pos.size())
173  pos.resize(3 * nlocalpart);
174  if (fields & MPIIO_OUT_VEL && 3 * nlocalpart > vel.size())
175  vel.resize(3 * nlocalpart);
176  if (fields & MPIIO_OUT_TYP && nlocalpart > type.size())
177  type.resize(nlocalpart);
178 
179  // Pack the necessary information
180  // Esp. rescale the velocities.
181  int i1 = 0, i3 = 0;
182  for (auto const &p : particles) {
183  id[i1] = p.p.identity;
184  if (fields & MPIIO_OUT_POS) {
185  pos[i3] = p.r.p[0];
186  pos[i3 + 1] = p.r.p[1];
187  pos[i3 + 2] = p.r.p[2];
188  }
189  if (fields & MPIIO_OUT_VEL) {
190  vel[i3] = p.m.v[0];
191  vel[i3 + 1] = p.m.v[1];
192  vel[i3 + 2] = p.m.v[2];
193  }
194  if (fields & MPIIO_OUT_TYP) {
195  type[i1] = p.p.type;
196  }
197  i1++;
198  i3 += 3;
199  }
200 
201  int rank;
202  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
203  if (rank == 0)
204  dump_info(fnam + ".head", fields);
205  mpiio_dump_array<int>(fnam + ".pref", &pref, 1, rank, MPI_INT);
206  mpiio_dump_array<int>(fnam + ".id", id.data(), nlocalpart, pref, MPI_INT);
207  if (fields & MPIIO_OUT_POS)
208  mpiio_dump_array<double>(fnam + ".pos", pos.data(), 3 * nlocalpart,
209  3 * pref, MPI_DOUBLE);
210  if (fields & MPIIO_OUT_VEL)
211  mpiio_dump_array<double>(fnam + ".vel", vel.data(), 3 * nlocalpart,
212  3 * pref, MPI_DOUBLE);
213  if (fields & MPIIO_OUT_TYP)
214  mpiio_dump_array<int>(fnam + ".type", type.data(), nlocalpart, pref,
215  MPI_INT);
216 
217  if (fields & MPIIO_OUT_BND) {
218  std::vector<char> bonds;
219 
220  /* Construct archive that pushes back to the bond buffer */
221  {
222  namespace io = boost::iostreams;
223  io::stream_buffer<io::back_insert_device<std::vector<char>>> os{
224  io::back_inserter(bonds)};
225  boost::archive::binary_oarchive bond_archiver{os};
226 
227  for (auto const &p : particles) {
228  bond_archiver << p.bonds();
229  }
230  }
231 
232  // Determine the prefixes in the bond file
233  int bonds_size = static_cast<int>(bonds.size());
234  MPI_Exscan(&bonds_size, &bpref, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
235 
236  mpiio_dump_array<int>(fnam + ".boff", &bonds_size, 1, rank, MPI_INT);
237  mpiio_dump_array<char>(fnam + ".bond", bonds.data(), bonds.size(), bpref,
238  MPI_CHAR);
239  }
240 }

242 /** Get the number of elements in a file by its file size and
243  * elem_sz. I.e. query the file size using stat(2) and divide it by
244  * elem_sz.
245  *
246  * \param fn The filename
247  * \param elem_sz Sizeof a single element
248  * \return The number of elements stored binary in the file
249  */
250 static int get_num_elem(const std::string &fn, std::size_t elem_sz) {
251  // Could also be done via MPI_File_open, MPI_File_get_size,
252  // MPI_File_close.
253  struct stat st;
254  errno = 0;
255  if (stat(fn.c_str(), &st) != 0) {
256  fprintf(stderr, "MPI-IO Input Error: Could not get file size of %s: %s\n",
257  fn.c_str(), strerror(errno));
258  errexit();
259  }
260  return static_cast<int>(st.st_size / elem_sz);
261 }

269  std::size_t pref, MPI_Datatype MPI_T) {

274  MPI_MODE_RDONLY, MPI_INFO_NULL, &f);
275 
276  if (ret) {
277  char buf[MPI_MAX_ERROR_STRING];
278  int buf_len;
279  MPI_Error_string(ret, buf, &buf_len);
280  buf[buf_len] = '\0';
281  fprintf(stderr, "MPI-IO Error: Could not open file \"%s\": %s\n",

283  errexit();

285  ret = MPI_File_set_view(f, pref * sizeof(T), MPI_T, MPI_T,
286  const_cast<char *>("native"), MPI_INFO_NULL);

288  ret |= MPI_File_read_all(f, arr, len, MPI_T, MPI_STATUS_IGNORE);
289  MPI_File_close(&f);

296 /** Read the header file and store the information in the pointer
297  * "field". To be called by all processes.
298  *
299  * \param fn Filename of the head file
300  * \param rank The rank of the current process in MPI_COMM_WORLD
301  * \param fields Pointer to store the fields to
302  */
303 static void read_head(const std::string &fn, int rank, unsigned *fields) {
304  FILE *f = nullptr;
305  if (rank == 0) {
306  if (!(f = fopen(fn.c_str(), "rb"))) {
307  fprintf(stderr, "MPI-IO: Could not open %s.head.\n", fn.c_str());
308  errexit();
309  }
310  if (fread((void *)fields, sizeof(unsigned), 1, f) != 1) {
311  fprintf(stderr, "MPI-IO: Read on %s.head failed.\n", fn.c_str());
312  errexit();
313  }
314  MPI_Bcast(fields, 1, MPI_UNSIGNED, 0, MPI_COMM_WORLD);
315  fclose(f);
316  } else {
317  MPI_Bcast(fields, 1, MPI_UNSIGNED, 0, MPI_COMM_WORLD);
318  }
319 }

321 /** Reads the pref file and fills pref and nlocalpart with their
322  * corresponding values. Needs to be called by all processes.
323  *
324  * \param fn The file name of the prefs file
325  * \param rank The rank of the current process in MPI_COMM_WORLD
326  * \param size The size of MPI_COMM_WORLD
327  * \param nglobalpart The global amount of particles
328  * \param pref Pointer to store the prefix to
329  * \param nlocalpart Pointer to store the amount of local particles to
330  */
331 static void read_prefs(const std::string &fn, int rank, int size,
332  int nglobalpart, int *pref, int *nlocalpart) {
333  mpiio_read_array<int>(fn, pref, 1, rank, MPI_INT);
334  if (rank > 0)
335  MPI_Send(pref, 1, MPI_INT, rank - 1, 0, MPI_COMM_WORLD);
336  if (rank < size - 1)
337  MPI_Recv(nlocalpart, 1, MPI_INT, rank + 1, MPI_ANY_TAG, MPI_COMM_WORLD,
338  MPI_STATUS_IGNORE);
339  else
340  *nlocalpart = nglobalpart;
341  *nlocalpart -= *pref;
342 }

344 void mpi_mpiio_common_read(const char *filename, unsigned fields) {
345  std::string fnam(filename);
346 
347  cell_structure.remove_all_particles();
348 
349  int size, rank;
350  MPI_Comm_size(MPI_COMM_WORLD, &size);
351  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
352  auto const nproc = get_num_elem(fnam + ".pref", sizeof(int));
353  auto const nglobalpart = get_num_elem(fnam + ".id", sizeof(int));
354 
355  if (rank == 0 && nproc != size) {
356  fprintf(stderr, "MPI-IO Error: Trying to read a file with a different COMM "
357  "size than at point of writing.\n");
358  errexit();
359  }
360 
361  // 1.head on master node:
362  // Read head to determine fields at time of writing.
363  // Compare this var to the current fields.
364  unsigned avail_fields;
365  read_head(fnam + ".head", rank, &avail_fields);
366  if (rank == 0 && (fields & avail_fields) != fields) {
367  fprintf(stderr,
368  "MPI-IO Error: Requesting to read fields which were not dumped.\n");
369  errexit();
370  }
371 
372  // 1.pref on all nodes:
373  // Read own prefix (1 int at prefix rank).
374  // Communicate own prefix to rank-1
375  // Determine nlocalpart (prefix of rank+1 - own prefix) on every node.
376  int pref, nlocalpart;
377  read_prefs(fnam + ".pref", rank, size, nglobalpart, &pref, &nlocalpart);
378 
379  std::vector<Particle> particles(nlocalpart);
380 
381  {
382  // 1.id on all nodes:
383  // Read nlocalpart ints at defined prefix.
384  std::vector<int> id(nlocalpart);
385  mpiio_read_array<int>(fnam + ".id", id.data(), nlocalpart, pref, MPI_INT);
386 
387  for (int i = 0; i < nlocalpart; ++i) {
388  particles[i].p.identity = id[i];
389  }
390  }
391 
392  if (fields & MPIIO_OUT_POS) {
393  // 1.pos on all nodes:
394  // Read nlocalpart * 3 doubles at defined prefix * 3
395  std::vector<double> pos(3 * nlocalpart);
396  mpiio_read_array<double>(fnam + ".pos", pos.data(), 3 * nlocalpart,
397  3 * pref, MPI_DOUBLE);
398 
399  for (int i = 0; i < nlocalpart; ++i) {
400  particles[i].r.p =
401  Utils::Vector3d{pos[3 * i + 0], pos[3 * i + 1], pos[3 * i + 2]};
402  }
403  }
404 
405  if (fields & MPIIO_OUT_TYP) {
406  // 1.type on all nodes:
407  // Read nlocalpart ints at defined prefix.
408  std::vector<int> type(nlocalpart);
409  mpiio_read_array<int>(fnam + ".type", type.data(), nlocalpart, pref,
410  MPI_INT);
411 
412  for (int i = 0; i < nlocalpart; ++i)
413  particles[i].p.type = type[i];
414  }
415 
416  if (fields & MPIIO_OUT_VEL) {
417  // 1.vel on all nodes:
418  // Read nlocalpart * 3 doubles at defined prefix * 3
419  std::vector<double> vel(3 * nlocalpart);
420  mpiio_read_array<double>(fnam + ".vel", vel.data(), 3 * nlocalpart,
421  3 * pref, MPI_DOUBLE);
422 
423  for (int i = 0; i < nlocalpart; ++i)
424  particles[i].m.v =
425  Utils::Vector3d{vel[3 * i + 0], vel[3 * i + 1], vel[3 * i + 2]};
426  }
427 
428  if (fields & MPIIO_OUT_BND) {
429  // 1.boff
430  // 1 int per process
431  int bonds_size = 0;
432  mpiio_read_array<int>(fnam + ".boff", &bonds_size, 1, rank, MPI_INT);
433  int bpref = 0;
434  MPI_Exscan(&bonds_size, &bpref, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
435 
436  // 1.bond
437  // nlocalbonds ints per process
438  std::vector<char> bond(bonds_size);
439  mpiio_read_array<char>(fnam + ".bond", bond.data(), bonds_size, bpref,
440  MPI_CHAR);
441 
442  namespace io = boost::iostreams;
443  io::array_source src(bond.data(), bond.size());
444  io::stream<io::array_source> ss(src);
445  boost::archive::binary_iarchive ia(ss);
446 
447  for (auto &p : particles) {
448  ia >> p.bonds();
449  }
450  }
451 
452  for (auto &p : particles) {
453  cell_structure.add_particle(std::move(p));
454  }
455 }
