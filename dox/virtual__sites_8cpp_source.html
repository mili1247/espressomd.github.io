56 /** Calculate the rotation quaternion and distance between two particles */
57 inline std::tuple<Utils::Quaternion<double>, double>
58 calculate_vs_relate_to_params(Particle const &p_current,
59  Particle const &p_relate_to) {
60  // get the distance between the particles
61  Utils::Vector3d d = box_geo.get_mi_vector(p_current.r.p, p_relate_to.r.p);
62 
63  // Check if the distance between virtual and non-virtual particles is larger
64  // than minimum global cutoff. If so, warn user.
65  auto const dist = d.norm();
66  if (dist > min_global_cut && n_nodes > 1) {
67  runtimeErrorMsg()
68  << "Warning: The distance between virtual and non-virtual particle ("
69  << dist << ") is\nlarger than the minimum global cutoff ("
70  << min_global_cut
71  << "). This may lead to incorrect simulations\nunder certain "
72  "conditions. Set the \"System()\" "
73  "class property \"min_global_cut\" to\nincrease the minimum "
74  "cutoff.\n";
75  }
76 
77  // Now, calculate the quaternions which specify the angle between
78  // the director of the particle we relate to and the vector
79  // (particle_we_relate_to - this_particle)
80  // The vs_relative implementation later obtains the director by multiplying
81  // the quaternions representing the orientation of the real particle
82  // with those in the virtual particle. The resulting quaternion is then
83  // converted to a director.
84  // We have quat_(real particle) * quat_(virtual particle)
85  // = quat_(obtained from desired director)
86  // Resolving this for the quat_(virtual particle)
87 
88  Utils::Quaternion<double> quat;
89  // If the distance between real & virtual particle is 0 we just set the
90  // relative orientation to {1 0 0 0}, as it is irrelevant but needs to be
91  // a valid quaternion
92  if (dist == 0) {
93  quat = Utils::Quaternion<double>::identity();
94  } else {
95  d.normalize();
96 
97  // Obtain quaternion from desired director
98  Utils::Quaternion<double> quat_director =
99  Utils::convert_director_to_quaternion(d);
100 
101  // Define quaternion as described above
102  quat =
103  Utils::Quaternion<double>{Utils::dot(p_relate_to.r.quat, quat_director),
104  -quat_director[0] * p_relate_to.r.quat[1] +
105  quat_director[1] * p_relate_to.r.quat[0] +
106  quat_director[2] * p_relate_to.r.quat[3] -
107  quat_director[3] * p_relate_to.r.quat[2],
108  p_relate_to.r.quat[1] * quat_director[3] +
109  p_relate_to.r.quat[0] * quat_director[2] -
110  p_relate_to.r.quat[3] * quat_director[1] -
111  p_relate_to.r.quat[2] * quat_director[0],
112  quat_director[3] * p_relate_to.r.quat[0] -
113  p_relate_to.r.quat[3] * quat_director[0] +
114  p_relate_to.r.quat[2] * quat_director[1] -
115  p_relate_to.r.quat[1] * quat_director[2]};
116  quat /= p_relate_to.r.quat.norm2();
117 
118  // Verify result
119  Utils::Quaternion<double> qtemp = p_relate_to.r.quat * quat;
120  for (int i = 0; i < 4; i++)
121  if (fabs(qtemp[i] - quat_director[i]) > 1E-9)
122  fprintf(stderr, "vs_relate_to: component %d: %f instead of %f\n", i,
123  qtemp[i], quat_director[i]);
124  }
125  return std::make_tuple(quat, dist);
126 }

128 void local_vs_relate_to(Particle &p_current, Particle const &p_relate_to) {
129  // Set the particle id of the particle we want to relate to, the distance
130  // and the relative orientation
131  p_current.p.vs_relative.to_particle_id = p_relate_to.identity();
132  std::tie(p_current.p.vs_relative.rel_orientation,
133  p_current.p.vs_relative.distance) =
134  calculate_vs_relate_to_params(p_current, p_relate_to);

142

146 
147  // Set the particle id of the particle we want to relate to, the distance
148  // and the relative orientation
149  set_particle_vs_relative(part_num, relate_to, dist, quat);
150  set_particle_virtual(part_num, true);
