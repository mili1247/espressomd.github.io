64 /** This ugly function does the bookkeeping: which nodes have to
65  * communicate to each other, when you change the node grid.
66  * Changing the domain decomposition requires communication. This
67  * function finds (hopefully) the best way to do this. As input it
68  * needs the two grids (@p grid1, @p grid2) and a linear list (@p node_list1)
69  * with the node identities for @p grid1. The linear list (@p node_list2)
70  * for the second grid is calculated. For the communication group of
71  * the calling node it calculates a list (@c group) with the node
72  * identities and the positions (@p my_pos, @p pos) of that nodes in @p grid1
73  * and @p grid2. The return value is the size of the communication
74  * group. It gives -1 if the two grids do not fit to each other
75  * (@p grid1 and @p grid2 have to be component-wise multiples of each
76  * other, see e.g. \ref calc_2d_grid for how to do this).
77  *
78  * \param[in] grid1 The node grid you start with.
79  * \param[in] grid2 The node grid you want to have.
80  * \param[in] node_list1 Linear node index list for grid1.
81  * \param[out] node_list2 Linear node index list for grid2.
82  * \param[out] pos Positions of the nodes in grid2
83  * \param[out] my_pos Position of comm.rank() in grid2.
84  * \param[in] comm MPI communicator.
85  * \return Size of the communication group.
86  */
87 boost::optional<std::vector<int>>
88 find_comm_groups(Utils::Vector3i const &grid1, Utils::Vector3i const &grid2,
89  Utils::Span<const int> node_list1, Utils::Span<int> node_list2,
90  Utils::Span<int> pos, Utils::Span<int> my_pos,
91  boost::mpi::communicator const &comm) {
92  int i;
93  /* communication group cell size on grid1 and grid2 */
94  int s1[3], s2[3];
95  /* The communication group cells build the same super grid on grid1 and grid2
96  */
97  int ds[3];
98  /* communication group size */
99  int g_size = 1;
100  /* comm. group cell index */
101  int gi[3];
102  /* position of a node in a grid */
103  int p1[3], p2[3];
104  /* node identity */
105  int n;
106  /* comm.rank() position in the communication group. */
107  int c_pos = -1;
108  /* flag for group identification */
109  int my_group = 0;
110 
111  /* calculate dimension of comm. group cells for both grids */
112  if ((grid1[0] * grid1[1] * grid1[2]) != (grid2[0] * grid2[1] * grid2[2]))
113  return boost::none; /* unlike number of nodes */
114  for (i = 0; i < 3; i++) {
115  s1[i] = grid1[i] / grid2[i];
116  if (s1[i] == 0)
117  s1[i] = 1;
118  else if (grid1[i] != grid2[i] * s1[i])
119  return boost::none; /* grids do not match!!! */
120 
121  s2[i] = grid2[i] / grid1[i];
122  if (s2[i] == 0)
123  s2[i] = 1;
124  else if (grid2[i] != grid1[i] * s2[i])
125  return boost::none; /* grids do not match!!! */
126 
127  ds[i] = grid2[i] / s2[i];
128  g_size *= s2[i];
129  }
130 
131  std::vector<int> group(g_size);
132 
133  /* calc node_list2 */
134  /* loop through all comm. group cells */
135  for (gi[2] = 0; gi[2] < ds[2]; gi[2]++)
136  for (gi[1] = 0; gi[1] < ds[1]; gi[1]++)
137  for (gi[0] = 0; gi[0] < ds[0]; gi[0]++) {
138  /* loop through all nodes in that comm. group cell */
139  for (i = 0; i < g_size; i++) {
140  p1[0] = (gi[0] * s1[0]) + (i % s1[0]);
141  p1[1] = (gi[1] * s1[1]) + ((i / s1[0]) % s1[1]);
142  p1[2] = (gi[2] * s1[2]) + (i / (s1[0] * s1[1]));
143 
144  p2[0] = (gi[0] * s2[0]) + (i % s2[0]);
145  p2[1] = (gi[1] * s2[1]) + ((i / s2[0]) % s2[1]);
146  p2[2] = (gi[2] * s2[2]) + (i / (s2[0] * s2[1]));
147 
148  n = node_list1[get_linear_index(p1[0], p1[1], p1[2], grid1)];
149  node_list2[get_linear_index(p2[0], p2[1], p2[2], grid2)] = n;
150 
151  pos[3 * n + 0] = p2[0];
152  pos[3 * n + 1] = p2[1];
153  pos[3 * n + 2] = p2[2];
154  if (my_group == 1)
155  group[i] = n;
156  if (n == comm.rank() && my_group == 0) {
157  my_group = 1;
158  c_pos = i;
159  my_pos[0] = p2[0];
160  my_pos[1] = p2[1];
161  my_pos[2] = p2[2];
162  i = -1; /* restart the loop */
163  }
164  }
165  my_group = 0;
166  }
167 
168  /* permute comm. group according to the nodes position in the group */
169  /* This is necessary to have matching node pairs during communication! */
170  while (c_pos > 0) {
171  n = group[g_size - 1];
172  for (i = g_size - 1; i > 0; i--)
173  group[i] = group[i - 1];
174  group[0] = n;
175  c_pos--;
176  }
177  return group;
178 }

180 /** Calculate the local fft mesh. Calculate the local mesh (@p loc_mesh)
181  * of a node at position (@p n_pos) in a node grid (@p n_grid) for a global
182  * mesh of size (@p mesh) and a mesh offset (@p mesh_off (in mesh units))
183  * and store also the first point (@p start) of the local mesh.
184  *
185  * \param[in] n_pos Position of the node in @p n_grid.
186  * \param[in] n_grid node grid.
187  * \param[in] mesh global mesh dimensions.
188  * \param[in] mesh_off global mesh offset (see \ref p3m_data_struct).
189  * \param[out] loc_mesh local mesh dimension.
190  * \param[out] start first point of local mesh in global mesh.
191  * \return Number of mesh points in local mesh.
192  */
193 int calc_local_mesh(const int *n_pos, const int *n_grid, const int *mesh,
194  const double *mesh_off, int *loc_mesh, int *start) {
195  int last[3], size = 1;
196 
197  for (int i = 0; i < 3; i++) {
198  auto const ratio = mesh[i] / static_cast<double>(n_grid[i]);
199  start[i] = static_cast<int>(ceil(ratio * n_pos[i] - mesh_off[i]));
200  last[i] = static_cast<int>(floor(ratio * (n_pos[i] + 1) - mesh_off[i]));
201  /* correct round off errors */
202  if (ratio * (n_pos[i] + 1) - mesh_off[i] - last[i] < 1.0e-15)
203  last[i]--;
204  if (1.0 + ratio * n_pos[i] - mesh_off[i] - start[i] < 1.0e-15)
205  start[i]--;
206  loc_mesh[i] = last[i] - start[i] + 1;
207  size *= loc_mesh[i];
208  }
209  return size;
210 }

212 /** Calculate a send (or recv.) block for grid communication during a
213  * decomposition change. Calculate the send block specification
214  * (block = lower left corner and upper right corner) which a node at
215  * position (@p pos1) in the actual node grid (@p grid1) has to send to
216  * another node at position (@p pos2) in the desired node grid (@p grid2).
217  * The global mesh, subject to communication, is specified via its size
218  * (@p mesh) and its mesh offset (@p mesh_off (in mesh units)).
219  *
220  * For the calculation of a receive block you have to change the arguments in
221  * the following way:
222  * - @p pos1: position of receiving node in the desired node grid.
223  * - @p grid1: desired node grid.
224  * - @p pos2: position of the node you intend to receive the data from in the
225  * actual node grid.
226  * - @p grid2: actual node grid.
227  *
228  * \param[in] pos1 Position of send node in @p grid1.
229  * \param[in] grid1 node grid 1.
230  * \param[in] pos2 Position of recv node in @p grid2.
231  * \param[in] grid2 node grid 2.
232  * \param[in] mesh global mesh dimensions.
233  * \param[in] mesh_off global mesh offset (see \ref p3m_data_struct).
234  * \param[out] block send block specification.
235  * \return Size of the send block.
236  */
237 int calc_send_block(const int *pos1, const int *grid1, const int *pos2,
238  const int *grid2, const int *mesh, const double *mesh_off,
239  int *block) {
240  int size = 1;
241  int mesh1[3], first1[3], last1[3];
242  int mesh2[3], first2[3], last2[3];
243 
244  calc_local_mesh(pos1, grid1, mesh, mesh_off, mesh1, first1);
245  calc_local_mesh(pos2, grid2, mesh, mesh_off, mesh2, first2);
246 
247  for (int i = 0; i < 3; i++) {
248  last1[i] = first1[i] + mesh1[i] - 1;
249  last2[i] = first2[i] + mesh2[i] - 1;
250  block[i] = std::max(first1[i], first2[i]) - first1[i];
251  block[i + 3] = (std::min(last1[i], last2[i]) - first1[i]) - block[i] + 1;
252  size *= block[i + 3];
253  }
254  return size;
255 }

257 /** Pack a block with dimensions <tt>size[0] * size[1] * size[2]</tt> starting
258  * at @p start of an input 3D-grid with dimension @p dim into an output
259  * 3D-grid with dimensions <tt>size[2] * size[0] * size[1]</tt> with
260  * a simultaneous one-fold permutation of the indices. The permutation is
261  * defined as: slow_in -> fast_out, mid_in ->slow_out, fast_in -> mid_out.
262  *
263  * An element <tt>(i0_in, i1_in, i2_in)</tt> is then
264  * <tt>(i0_out = i1_in-start[1], i1_out = i2_in-start[2],
265  * i2_out = i0_in-start[0])</tt> and for the linear indices we have:
266  * - <tt>li_in = i2_in + size[2] * (i1_in + (size[1]*i0_in))</tt>
267  * - <tt>li_out = i2_out + size[0] * (i1_out + (size[2]*i0_out))</tt>
268  *
269  * For index definition see \ref fft_pack_block.
270  *
271  * \param[in] in input 3D-grid.
272  * \param[out] out output 3D-grid (block).
273  * \param[in] start start index of the block in the in-grid.
274  * \param[in] size size of the block (=dimension of the out-grid).
275  * \param[in] dim size of the in-grid.
276  * \param[in] element size of a grid element (e.g. 1 for Real, 2 for Complex).
277  */
278 void pack_block_permute1(double const *const in, double *const out,
279  const int *start, const int *size, const int *dim,
280  int element) {
281 
282  /* offsets for indices in input grid */
283  auto const m_in_offset = element * (dim[2] - size[2]);
284  auto const s_in_offset = element * (dim[2] * (dim[1] - size[1]));
285  /* offset for mid changing indices of output grid */
286  auto const m_out_offset = (element * size[0]) - element;
287  /* linear index of in grid */
288  int li_in = element * (start[2] + dim[2] * (start[1] + dim[1] * start[0]));
289 
290  for (int s = 0; s < size[0]; s++) { /* fast changing out */
291  /* linear index of out grid */
292  int li_out = element * s;
293  for (int m = 0; m < size[1]; m++) { /* slow changing out */
294  for (int f = 0; f < size[2]; f++) { /* mid changing out */
295  for (int e = 0; e < element; e++)
296  out[li_out++] = in[li_in++];
297  li_out += m_out_offset;
298  }
299  li_in += m_in_offset;
300  }
301  li_in += s_in_offset;
302  }
303 }

305 /** Pack a block with dimensions <tt>size[0] * size[1] * size[2]</tt> starting
306  * at @p start of an input 3D-grid with dimension @p dim into an output
307  * 3D-grid with dimensions <tt>size[2] * size[0] * size[1]</tt> with
308  * a simultaneous two-fold permutation of the indices. The permutation is
309  * defined as: slow_in -> mid_out, mid_in ->fast_out, fast_in -> slow_out.
310  *
311  * An element <tt>(i0_in, i1_in, i2_in)</tt> is then
312  * <tt>(i0_out = i2_in-start[2], i1_out = i0_in-start[0],
313  * i2_out = i1_in-start[1])</tt> and for the linear indices we have:
314  * - <tt>li_in = i2_in + size[2] * (i1_in + (size[1]*i0_in))</tt>
315  * - <tt>li_out = i2_out + size[0] * (i1_out + (size[2]*i0_out))</tt>
316  *
317  * For index definition see \ref fft_pack_block.
318  *
319  * \param[in] in input 3D-grid.
320  * \param[out] out output 3D-grid (block).
321  * \param[in] start start index of the block in the in-grid.
322  * \param[in] size size of the block (=dimension of the out-grid).
323  * \param[in] dim size of the in-grid.
324  * \param[in] element size of a grid element (e.g. 1 for Real, 2 for Complex).
325  */
326 void pack_block_permute2(double const *const in, double *const out,
327  const int *start, const int *size, const int *dim,
328  int element) {
329 
330  /* offsets for indices in input grid */
331  auto const m_in_offset = element * (dim[2] - size[2]);
332  auto const s_in_offset = element * (dim[2] * (dim[1] - size[1]));
333  /* offset for slow changing index of output grid */
334  auto const s_out_offset = (element * size[0] * size[1]) - element;
335  /* linear index of in grid */
336  int li_in = element * (start[2] + dim[2] * (start[1] + dim[1] * start[0]));
337 
338  for (int s = 0; s < size[0]; s++) { /* mid changing out */
339  auto const m_out_start = element * (s * size[1]);
340  for (int m = 0; m < size[1]; m++) { /* fast changing out */
341  /* linear index of out grid */
342  int li_out = m_out_start + element * m;
343  for (int f = 0; f < size[2]; f++) { /* slow changing out */
344  for (int e = 0; e < element; e++)
345  out[li_out++] = in[li_in++];
346  li_out += s_out_offset;
347  }
348  li_in += m_in_offset;
349  }
350  li_in += s_in_offset;
351  }
352 }

354 /** Communicate the grid data according to the given forward FFT plan.
355  * \param plan FFT communication plan.
356  * \param in input mesh.
357  * \param out output mesh.
358  * \param fft FFT communication plan.
359  * \param comm MPI communicator.
360  */
361 void forw_grid_comm(fft_forw_plan plan, const double *in, double *out,
362  fft_data_struct &fft,
363  const boost::mpi::communicator &comm) {
364  for (int i = 0; i < plan.group.size(); i++) {
365  plan.pack_function(in, fft.send_buf.data(), &(plan.send_block[6 * i]),
366  &(plan.send_block[6 * i + 3]), plan.old_mesh,
367  plan.element);
368 
369  if (plan.group[i] != comm.rank()) {
370  MPI_Sendrecv(fft.send_buf.data(), plan.send_size[i], MPI_DOUBLE,
371  plan.group[i], REQ_FFT_FORW, fft.recv_buf.data(),
372  plan.recv_size[i], MPI_DOUBLE, plan.group[i], REQ_FFT_FORW,
373  comm, MPI_STATUS_IGNORE);
374  } else { /* Self communication... */
375  std::swap(fft.send_buf, fft.recv_buf);
376  }
377  fft_unpack_block(fft.recv_buf.data(), out, &(plan.recv_block[6 * i]),
378  &(plan.recv_block[6 * i + 3]), plan.new_mesh,
379  plan.element);
380  }
381 }

383 /** Communicate the grid data according to the given backward FFT plan.
384  * \param plan_f Forward FFT plan.
385  * \param plan_b Backward FFT plan.
386  * \param in input mesh.
387  * \param out output mesh.
388  * \param fft FFT communication plan.
389  * \param comm MPI communicator.
390  */
391 void back_grid_comm(fft_forw_plan plan_f, fft_back_plan plan_b,
392  const double *in, double *out, fft_data_struct &fft,
393  const boost::mpi::communicator &comm) {
394  /* Back means: Use the send/receive stuff from the forward plan but
395  replace the receive blocks by the send blocks and vice
396  versa. Attention then also new_mesh and old_mesh are exchanged */
397 
398  for (int i = 0; i < plan_f.group.size(); i++) {
399  plan_b.pack_function(in, fft.send_buf.data(), &(plan_f.recv_block[6 * i]),
400  &(plan_f.recv_block[6 * i + 3]), plan_f.new_mesh,
401  plan_f.element);
402 
403  if (plan_f.group[i] != comm.rank()) { /* send first, receive second */
404  MPI_Sendrecv(fft.send_buf.data(), plan_f.recv_size[i], MPI_DOUBLE,
405  plan_f.group[i], REQ_FFT_BACK, fft.recv_buf.data(),
406  plan_f.send_size[i], MPI_DOUBLE, plan_f.group[i],
407  REQ_FFT_BACK, comm, MPI_STATUS_IGNORE);
408  } else { /* Self communication... */
409  std::swap(fft.send_buf, fft.recv_buf);
410  }
411  fft_unpack_block(fft.recv_buf.data(), out, &(plan_f.send_block[6 * i]),
412  &(plan_f.send_block[6 * i + 3]), plan_f.old_mesh,
413  plan_f.element);
414  }
415 }

417 /** Calculate 'best' mapping between a 2D and 3D grid.
418  * Required for the communication from 3D domain decomposition
419  * to 2D row decomposition.
420  * The dimensions of the 2D grid are resorted, if necessary, in a way
421  * that they are multiples of the 3D grid dimensions.
422  * \param g3d 3D grid.
423  * \param g2d 2D grid.
424  * \param mult factors between 3D and 2D grid dimensions
425  * \return index of the row direction [0,1,2].
426  */
427 int map_3don2d_grid(int const g3d[3], int g2d[3], int mult[3]) {
428  int row_dir = -1;
429  /* trivial case */
430  if (g3d[2] == 1) {
431  for (int i = 0; i < 3; i++)
432  mult[i] = 1;
433  return 2;
434  }
435  if (g2d[0] % g3d[0] == 0) {
436  if (g2d[1] % g3d[1] == 0) {
437  row_dir = 2;
438  } else if (g2d[1] % g3d[2] == 0) {
439  row_dir = 1;
440  g2d[2] = g2d[1];
441  g2d[1] = 1;
442  }
443  } else if (g2d[0] % g3d[1] == 0) {
444  if (g2d[1] % g3d[0] == 0) {
445  row_dir = 2;
446  int const tmp = g2d[0];
447  g2d[0] = g2d[1];
448  g2d[1] = tmp;
449  } else if (g2d[1] % g3d[2] == 0) {
450  row_dir = 0;
451  g2d[2] = g2d[1];
452  g2d[1] = g2d[0];
453  g2d[0] = 1;
454  }
455  } else if (g2d[0] % g3d[2] == 0) {
456  if (g2d[1] % g3d[0] == 0) {
457  row_dir = 1;
458  g2d[2] = g2d[0];
459  g2d[0] = g2d[1];
460  g2d[1] = 1;
461  } else if (g2d[1] % g3d[1] == 0) {
462  row_dir = 0;
463  g2d[2] = g2d[0];
464  g2d[0] = 1;
465  }
466  }
467  for (int i = 0; i < 3; i++)
468  mult[i] = g2d[i] / g3d[i];
469  return row_dir;
470 }

472 /** Calculate most square 2D grid. */
473 void calc_2d_grid(int n, int grid[3]) {
474  for (auto i = static_cast<int>(std::sqrt(n)); i >= 1; i--) {
475  if (n % i == 0) {
476  grid[0] = n / i;
477  grid[1] = i;
478  grid[2] = 1;
479  return;
480  }
481  }

485 int fft_init(const Utils::Vector3i &ca_mesh_dim, int const *ca_mesh_margin,
486  int const *global_mesh_dim, double const *global_mesh_off,
487  int &ks_pnum, fft_data_struct &fft, const Utils::Vector3i &grid,
488  const boost::mpi::communicator &comm) {
489  int i, j;
490  /* helpers */
491  int mult[3];
492 
493  int n_grid[4][3]; /* The four node grids. */
494  int my_pos[4][3]; /* The position of comm.rank() in the node grids. */
495  std::vector<int> n_id[4]; /* linear node identity lists for the node grids. */
496  std::vector<int> n_pos[4]; /* positions of nodes in the node grids. */
497 
498  int node_pos[3];
499  MPI_Cart_coords(comm, comm.rank(), 3, node_pos);
500 
501  fft.max_comm_size = 0;
502  fft.max_mesh_size = 0;
503  for (i = 0; i < 4; i++) {
504  n_id[i].resize(1 * comm.size());
505  n_pos[i].resize(3 * comm.size());
506  }
507 
508  /* === node grids === */
509  /* real space node grid (n_grid[0]) */
510  for (i = 0; i < 3; i++) {
511  n_grid[0][i] = grid[i];

684 void fft_perform_forw(double *data, fft_data_struct &fft,
685  const boost::mpi::communicator &comm) {

687

690 
691  /* communication to current dir row format (in is data) */

757 void fft_pack_block(double const *const in, double *const out,
758  int const start[3], int const size[3], int const dim[3],
759  int element) {
760 
761  auto const copy_size = element * size[2] * static_cast<int>(sizeof(double));
762  /* offsets for indices in input grid */
763  auto const m_in_offset = element * dim[2];
764  auto const s_in_offset = element * (dim[2] * (dim[1] - size[1]));
765  /* offsets for indices in output grid */
766  auto const m_out_offset = element * size[2];
767  /* linear index of in grid, linear index of out grid */
768  int li_in = element * (start[2] + dim[2] * (start[1] + dim[1] * start[0]));
769  int li_out = 0;
770 
771  for (int s = 0; s < size[0]; s++) {
772  for (int m = 0; m < size[1]; m++) {
773  memmove(&(out[li_out]), &(in[li_in]), copy_size);
774  li_in += m_in_offset;
775  li_out += m_out_offset;
776  }
777  li_in += s_in_offset;
778  }
779 }
