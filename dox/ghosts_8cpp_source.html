87 static std::size_t calc_transmit_size(unsigned data_parts) {
88  std::size_t size = {};
89  if (data_parts & GHOSTTRANS_PROPRTS)
90  size += Utils::MemcpyOArchive::packing_size<ParticleProperties>();
91  if (data_parts & GHOSTTRANS_POSITION)
92  size += Utils::MemcpyOArchive::packing_size<ParticlePosition>();
93  if (data_parts & GHOSTTRANS_MOMENTUM)
94  size += Utils::MemcpyOArchive::packing_size<ParticleMomentum>();
95  if (data_parts & GHOSTTRANS_FORCE)
96  size += Utils::MemcpyOArchive::packing_size<ParticleForce>();
97 #ifdef BOND_CONSTRAINT
98  if (data_parts & GHOSTTRANS_RATTLE)
99  size += Utils::MemcpyOArchive::packing_size<ParticleRattle>();
100 #endif
101 
102  return size;
103 }

105 static std::size_t calc_transmit_size(const GhostCommunication &ghost_comm,
106  unsigned int data_parts) {
107  if (data_parts & GHOSTTRANS_PARTNUM)
108  return sizeof(int) * ghost_comm.part_lists.size();
109 
110  auto const n_part = boost::accumulate(
111  ghost_comm.part_lists, 0ul,
112  [](std::size_t sum, auto part_list) { return sum + part_list->size(); });
113 
114  return n_part * calc_transmit_size(data_parts);
115 }

117 static void prepare_send_buffer(CommBuf &send_buffer,
118  const GhostCommunication &ghost_comm,
119  unsigned int data_parts) {
120  /* reallocate send buffer */
121  send_buffer.resize(calc_transmit_size(ghost_comm, data_parts));
122  send_buffer.bonds().clear();
123 
124  auto archiver = Utils::MemcpyOArchive{Utils::make_span(send_buffer)};
125 
126  /* Construct archive that pushes back to the bond buffer */
127  namespace io = boost::iostreams;
128  io::stream<io::back_insert_device<std::vector<char>>> os{
129  io::back_inserter(send_buffer.bonds())};
130  boost::archive::binary_oarchive bond_archiver{os};
131 
132  /* put in data */
133  for (auto part_list : ghost_comm.part_lists) {
134  if (data_parts & GHOSTTRANS_PARTNUM) {
135  int np = part_list->size();
136  archiver << np;
137  } else {
138  for (Particle &part : *part_list) {
139  if (data_parts & GHOSTTRANS_PROPRTS) {
140  archiver << part.p;
141  }
142  if (data_parts & GHOSTTRANS_POSITION) {
143  /* ok, this is not nice, but perhaps fast */
144  auto pp = part.r;
145  pp.p += ghost_comm.shift;
146  archiver << pp;
147  }
148  if (data_parts & GHOSTTRANS_MOMENTUM) {
149  archiver << part.m;
150  }
151  if (data_parts & GHOSTTRANS_FORCE) {
152  archiver << part.f;
153  }
154 #ifdef BOND_CONSTRAINT
155  if (data_parts & GHOSTTRANS_RATTLE) {
156  archiver << part.rattle;
157  }
158 #endif
159  if (data_parts & GHOSTTRANS_BONDS) {
160  bond_archiver << part.bonds();
161  }
162  }
163  }
164  }
165 
166  assert(archiver.bytes_written() == send_buffer.size());
167 }

188 static void put_recv_buffer(CommBuf &recv_buffer,
189  const GhostCommunication &ghost_comm,
190  unsigned int data_parts) {
191  /* put back data */
192  auto archiver = Utils::MemcpyIArchive{Utils::make_span(recv_buffer)};
193 
194  if (data_parts & GHOSTTRANS_PARTNUM) {
195  for (auto part_list : ghost_comm.part_lists) {
196  int np;
197  archiver >> np;
198  prepare_ghost_cell(part_list, np);
199  }
200  } else {
201  for (auto part_list : ghost_comm.part_lists) {
202  for (Particle &part : *part_list) {
203  if (data_parts & GHOSTTRANS_PROPRTS) {
204  archiver >> part.p;
205  }
206  if (data_parts & GHOSTTRANS_POSITION) {
207  archiver >> part.r;
208  }
209  if (data_parts & GHOSTTRANS_MOMENTUM) {
210  archiver >> part.m;
211  }
212  if (data_parts & GHOSTTRANS_FORCE) {
213  archiver >> part.f;
214  }
215 #ifdef BOND_CONSTRAINT
216  if (data_parts & GHOSTTRANS_RATTLE) {
217  archiver >> part.rattle;
218  }
219 #endif
220  }
221  }
222  if (data_parts & GHOSTTRANS_BONDS) {
223  namespace io = boost::iostreams;
224  io::stream<io::array_source> bond_stream(io::array_source{
225  recv_buffer.bonds().data(), recv_buffer.bonds().size()});
226  boost::archive::binary_iarchive bond_archive(bond_stream);
227 
228  for (auto part_list : ghost_comm.part_lists) {
229  for (Particle &part : *part_list) {
230  bond_archive >> part.bonds();
231  }
232  }
233  }
234  }
235 
236  assert(archiver.bytes_read() == recv_buffer.size());
237 
238  recv_buffer.bonds().clear();
239 }

270 static void cell_cell_transfer(const GhostCommunication &ghost_comm,
271  unsigned int data_parts) {
272  /* transfer data */
273  auto const offset = ghost_comm.part_lists.size() / 2;
274  for (std::size_t pl = 0; pl < offset; pl++) {
275  const auto *src_list = ghost_comm.part_lists[pl];
276  auto *dst_list = ghost_comm.part_lists[pl + offset];
277 
278  if (data_parts & GHOSTTRANS_PARTNUM) {
279  prepare_ghost_cell(dst_list, src_list->size());
280  } else {
281  auto const &src_part = *src_list;
282  auto &dst_part = *dst_list;
283  assert(src_part.size() == dst_part.size());
284 
285  for (std::size_t i = 0; i < src_part.size(); i++) {
286  auto const &part1 = src_part.begin()[i];
287  auto &part2 = dst_part.begin()[i];
288 
289  if (data_parts & GHOSTTRANS_PROPRTS) {
290  part2.p = part1.p;
291  }
292  if (data_parts & GHOSTTRANS_BONDS) {
293  part2.bonds() = part1.bonds();
294  }
295  if (data_parts & GHOSTTRANS_POSITION) {
296  /* ok, this is not nice, but perhaps fast */
297  part2.r = part1.r;
298  part2.r.p += ghost_comm.shift;
299  }
300  if (data_parts & GHOSTTRANS_MOMENTUM) {
301  part2.m = part1.m;
302  }
303  if (data_parts & GHOSTTRANS_FORCE)
304  part2.f += part1.f;
305 #ifdef BOND_CONSTRAINT
306  if (data_parts & GHOSTTRANS_RATTLE)
307  part2.rattle += part1.rattle;
308 #endif
309  }
310  }
311  }

341 void ghost_communicator(const GhostCommunicator &gcr, unsigned int data_parts) {
342  if (GHOSTTRANS_NONE == data_parts)
343  return;
344 
345  static CommBuf send_buffer, recv_buffer;
346 
347  auto const &comm = gcr.mpi_comm;
348 
349  for (auto it = gcr.communications.begin(); it != gcr.communications.end();
350  ++it) {
351  const GhostCommunication &ghost_comm = *it;
352  int const comm_type = ghost_comm.type & GHOST_JOBMASK;
353 
354  if (comm_type == GHOST_LOCL) {
355  cell_cell_transfer(ghost_comm, data_parts);
356  continue;
357  }
358 
359  int const prefetch = ghost_comm.type & GHOST_PREFETCH;
360  int const poststore = ghost_comm.type & GHOST_PSTSTORE;
361  int const node = ghost_comm.node;
362 
363  /* prepare send buffer if necessary */
364  if (is_send_op(comm_type, node, comm.rank())) {
365  /* ok, we send this step, prepare send buffer if not yet done */
366  if (!prefetch) {
367  prepare_send_buffer(send_buffer, ghost_comm, data_parts);
368  }
369  // Check prefetched send buffers (must also hold for buffers allocated
370  // in the previous lines.)
371  assert(send_buffer.size() == calc_transmit_size(ghost_comm, data_parts));
372  } else if (prefetch) {
373  /* we do not send this time, let's look for a prefetch */
374  auto prefetch_ghost_comm = std::find_if(
375  std::next(it), gcr.communications.end(),
376  [this_node = comm.rank()](GhostCommunication const &ghost_comm) {
377  return is_prefetchable(ghost_comm, this_node);
378  });
379 
380  if (prefetch_ghost_comm != gcr.communications.end())
381  prepare_send_buffer(send_buffer, *prefetch_ghost_comm, data_parts);
382  }
383 
384  /* recv buffer for recv and multinode operations to this node */
385  if (is_recv_op(comm_type, node, comm.rank()))
386  prepare_recv_buffer(recv_buffer, ghost_comm, data_parts);
387 
388  /* transfer data */
389  // Use two send/recvs in order to avoid having to serialize CommBuf
390  // (which consists of already serialized data).
391  switch (comm_type) {
392  case GHOST_RECV:
393  comm.recv(node, REQ_GHOST_SEND, recv_buffer.data(), recv_buffer.size());
394  comm.recv(node, REQ_GHOST_SEND, recv_buffer.bonds());
395  break;
396  case GHOST_SEND:
397  comm.send(node, REQ_GHOST_SEND, send_buffer.data(), send_buffer.size());
398  comm.send(node, REQ_GHOST_SEND, send_buffer.bonds());
399  break;
400  case GHOST_BCST:
401  if (node == comm.rank()) {
402  boost::mpi::broadcast(comm, send_buffer.data(), send_buffer.size(),
403  node);
404  boost::mpi::broadcast(comm, send_buffer.bonds(), node);
405  } else {
406  boost::mpi::broadcast(comm, recv_buffer.data(), recv_buffer.size(),
407  node);
408  boost::mpi::broadcast(comm, recv_buffer.bonds(), node);
409  }
410  break;
411  case GHOST_RDCE:
412  if (node == comm.rank())
413  boost::mpi::reduce(
414  comm, reinterpret_cast<double *>(send_buffer.data()),
415  static_cast<int>(send_buffer.size() / sizeof(double)),
416  reinterpret_cast<double *>(recv_buffer.data()), std::plus<double>{},
417  node);
418  else
419  boost::mpi::reduce(
420  comm, reinterpret_cast<double *>(send_buffer.data()),
421  static_cast<int>(send_buffer.size() / sizeof(double)),
422  std::plus<double>{}, node);
423  break;
424  }
425 
426  // recv op; write back data directly, if no PSTSTORE delay is requested.
427  if (is_recv_op(comm_type, node, comm.rank())) {
428  if (!poststore) {
429  /* forces have to be added, the rest overwritten. Exception is RDCE,
430  * where the addition is integrated into the communication. */
431  if (data_parts == GHOSTTRANS_FORCE && comm_type != GHOST_RDCE)
432  add_forces_from_recv_buffer(recv_buffer, ghost_comm);
433 #ifdef BOND_CONSTRAINT
434  else if (data_parts == GHOSTTRANS_RATTLE && comm_type != GHOST_RDCE)
435  add_rattle_correction_from_recv_buffer(recv_buffer, ghost_comm);
436 #endif
437  else
438  put_recv_buffer(recv_buffer, ghost_comm, data_parts);
439  }
440  } else if (poststore) {
441  /* send op; write back delayed data from last recv, when this was a
442  * prefetch send. */
443  /* find previous action where we recv and which has PSTSTORE set */
444  auto poststore_ghost_comm = std::find_if(
445  std::make_reverse_iterator(it), gcr.communications.rend(),
446  [this_node = comm.rank()](GhostCommunication const &ghost_comm) {
447  return is_poststorable(ghost_comm, this_node);
448  });
449 
450  if (poststore_ghost_comm != gcr.communications.rend()) {
451  assert(recv_buffer.size() ==
452  calc_transmit_size(*poststore_ghost_comm, data_parts));
453  /* as above */
454  if (data_parts == GHOSTTRANS_FORCE && comm_type != GHOST_RDCE)
455  add_forces_from_recv_buffer(recv_buffer, *poststore_ghost_comm);
456 #ifdef BOND_CONSTRAINT
457  else if (data_parts == GHOSTTRANS_RATTLE && comm_type != GHOST_RDCE)
458  add_rattle_correction_from_recv_buffer(recv_buffer,
459  *poststore_ghost_comm);
460 #endif
461  else
462  put_recv_buffer(recv_buffer, *poststore_ghost_comm, data_parts);
463  }
464  }
465  }
