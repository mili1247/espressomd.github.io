68 /** Compute the three-body angle interaction force.
69  * @param[in] r_mid Position of second/middle particle.
70  * @param[in] r_left Position of first/left particle.
71  * @param[in] r_right Position of third/right particle.
72  * @return Forces on the second, first and third particles, in that order.
73  */
74 inline std::tuple<Utils::Vector3d, Utils::Vector3d, Utils::Vector3d>
75 AngleCossquareBond::forces(Utils::Vector3d const &r_mid,
76  Utils::Vector3d const &r_left,
77  Utils::Vector3d const &r_right) const {
78 
79  auto forceFactor = [this](double const cos_phi) {
80  return bend * (cos_phi - cos_phi0);
81  };
82 
83  return angle_generic_force(r_mid, r_left, r_right, forceFactor, false);
84 }

86 /** Computes the three-body angle interaction energy.
87  * @param[in] r_mid Position of second/middle particle.
88  * @param[in] r_left Position of first/left particle.
89  * @param[in] r_right Position of third/right particle.
90  */
91 inline double AngleCossquareBond::energy(Utils::Vector3d const &r_mid,
92  Utils::Vector3d const &r_left,
93  Utils::Vector3d const &r_right) const {
94  auto const vectors = calc_vectors_and_cosine(r_mid, r_left, r_right, true);
95  auto const cos_phi = std::get<4>(vectors);
96  return 0.5 * bend * Utils::sqr(cos_phi - cos_phi0);
97 }
