42  * explicitly. This saves a lot of multiplications with 1's and 0's
43  * thus making the code more efficient. */

46 /** Parameters for the lattice Boltzmann system for GPU. */
47 struct LB_parameters_gpu {
48  /** number density (LB units) */
49  float rho;
50  /** mu (LJ units) */
51  float mu;
52  /** viscosity (LJ) units */
53  float viscosity;
54  /** relaxation rate of shear modes */
55  float gamma_shear;
56  /** relaxation rate of bulk modes */
57  float gamma_bulk;
58  /** */
59  float gamma_odd;
60  float gamma_even;
61  /** flag determining whether gamma_shear, gamma_odd, and gamma_even are
62  * calculated from gamma_shear in such a way to yield a TRT LB with minimized
63  * slip at bounce-back boundaries
64  */
65  bool is_TRT;
66 
67  float bulk_viscosity;
68 
69  /** lattice spacing (LJ units) */
70  float agrid;
71 
72  /** time step for fluid propagation (LJ units)
73  * Note: Has to be larger than MD time step!
74  */
75  float tau;
76 
77  Utils::Array<unsigned int, 3> dim;
78 
79  unsigned int number_of_nodes;
80 #ifdef LB_BOUNDARIES_GPU
81  unsigned int number_of_boundnodes;
82 #endif
83  /** to calculate and print out physical values */
84  int calc_val;
85 
86  int external_force_density;
87 
88  Utils::Array<float, 3> ext_force_density;
89 
90  unsigned int reinit;
91  // Thermal energy
92  float kT;
93 };
