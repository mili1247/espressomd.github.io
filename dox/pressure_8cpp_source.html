55 static std::shared_ptr<Observable_stat> calculate_pressure_local() {
56 
57  auto obs_pressure_ptr = std::make_shared<Observable_stat>(9);
58 
59  if (!interactions_sanity_checks())
60  return obs_pressure_ptr;
61 
62  auto &obs_pressure = *obs_pressure_ptr;
63 
64  on_observable_calc();
65 
66  auto const volume = box_geo.volume();
67  auto const local_parts = cell_structure.local_particles();
68 
69  for (auto const &p : local_parts) {
70  add_kinetic_virials(p, obs_pressure);
71  }
72 
73  short_range_loop(
74  [&obs_pressure](Particle const &p1, int bond_id,
75  Utils::Span<Particle *> partners) {
76  auto const &iaparams = bonded_ia_params[bond_id];
77  auto const result = calc_bonded_pressure_tensor(iaparams, p1, partners);
78  if (result) {
79  auto const &tensor = result.get();
80  /* pressure tensor part */
81  for (int k = 0; k < 3; k++)
82  for (int l = 0; l < 3; l++)
83  obs_pressure.bonded_contribution(bond_id)[k * 3 + l] +=
84  tensor(k, l);
85 
86  return false;
87  }
88  return true;
89  },
90  [&obs_pressure](Particle const &p1, Particle const &p2,
91  Distance const &d) {
92  add_non_bonded_pair_virials(p1, p2, d.vec21, sqrt(d.dist2),
93  obs_pressure);
94  },
95  maximal_cutoff(), maximal_cutoff_bonded());
96 
97 #ifdef ELECTROSTATICS
98  /* calculate k-space part of electrostatic interaction. */
99  auto const coulomb_pressure = Coulomb::calc_pressure_long_range(local_parts);
100  boost::copy(coulomb_pressure, obs_pressure.coulomb.begin() + 9);
101 #endif
102 #ifdef DIPOLES
103  /* calculate k-space part of magnetostatic interaction. */
104  Dipole::calc_pressure_long_range();
105 #endif
106 
107 #ifdef VIRTUAL_SITES
108  if (!obs_pressure.virtual_sites.empty()) {
109  auto const vs_pressure = virtual_sites()->pressure_tensor();
110  boost::copy(flatten(vs_pressure), obs_pressure.virtual_sites.begin());
111  }
112 #endif
113 
114  obs_pressure.rescale(volume);
115 
116  obs_pressure.mpi_reduce();
117  return obs_pressure_ptr;
118 }
