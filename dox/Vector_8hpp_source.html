116  Vector<U, N> ret;

134  Vector &normalize() {
135  auto const l = norm();
136  if (l > T(0)) {
137  for (int i = 0; i < N; i++)

183 template <std::size_t N, typename T, typename Op>
184 constexpr bool all_of(Vector<T, N> const &a, Vector<T, N> const &b, Op op) {
185  for (int i = 0; i < a.size(); i++) {
186  /* Short circuit */
187  if (!static_cast<bool>(op(a[i], b[i]))) {
188  return false;
189  }
190  }
191 
192  return true;
