105 /**
106  * @brief Calculate cached sin/cos values for one direction.
107  *
108  * @tparam dir Index of the dimension to consider (e.g. 0 for x ...).
109  *
110  * @param particles Particle to calculate values for
111  * @param n_freq Number of frequencies to calculate per particle
112  * @param u Inverse box length
113  * @return Calculated values.
114  */
115 template <std::size_t dir>
116 static std::vector<SCCache> calc_sc_cache(const ParticleRange &particles,
117  std::size_t n_freq, double u) {
118  constexpr double c_2pi = 2 * Utils::pi();
119  auto const n_part = particles.size();
120  std::vector<SCCache> ret(n_freq * n_part);
121 
122  for (std::size_t freq = 1; freq <= n_freq; freq++) {
123  auto const pref = c_2pi * u * static_cast<double>(freq);
124 
125  std::size_t o = (freq - 1) * n_part;
126  for (auto const &part : particles) {
127  auto const arg = pref * part.r.p[dir];
128  ret[o++] = {sin(arg), cos(arg)};
129  }
130  }
131 
132  return ret;
133 }

203 /** Calculate the dipole force.
204  * See @cite yeh99a.
205  */
206 static void add_dipole_force(const ParticleRange &particles) {
207  double const pref = coulomb.prefactor * 4 * Utils::pi() *
208  box_geo.length_inv()[0] * box_geo.length_inv()[1] *
209  box_geo.length_inv()[2];
210  constexpr std::size_t size = 3;
211 
212  auto local_particles = particles;
213 
214  /* for nonneutral systems, this shift gives the background contribution
215  (rsp. for this shift, the DM of the background is zero) */
216  double const shift = box_geo.length_half()[2];
217 
218  // collect moments
219 
220  gblcblk[0] = 0; // sum q_i (z_i - L/2)
221  gblcblk[1] = 0; // sum q_i z_i
222  gblcblk[2] = 0; // sum q_i
223 
224  for (auto const &p : local_particles) {
225  check_gap_elc(p);
226 
227  gblcblk[0] += p.p.q * (p.r.p[2] - shift);
228  gblcblk[1] += p.p.q * p.r.p[2];
229  gblcblk[2] += p.p.q;
230 
231  if (elc_params.dielectric_contrast_on) {
232  if (p.r.p[2] < elc_params.space_layer) {
233  gblcblk[0] += elc_params.delta_mid_bot * p.p.q * (-p.r.p[2] - shift);
234  gblcblk[2] += elc_params.delta_mid_bot * p.p.q;
235  }
236  if (p.r.p[2] > (elc_params.h - elc_params.space_layer)) {
237  gblcblk[0] += elc_params.delta_mid_top * p.p.q *
238  (2 * elc_params.h - p.r.p[2] - shift);
239  gblcblk[2] += elc_params.delta_mid_top * p.p.q;
240  }
241  }
242  }
243 
244  gblcblk[0] *= pref;
245  gblcblk[1] *= pref / elc_params.h * box_geo.length()[2];
246  gblcblk[2] *= pref;
247 
248  distribute(size);
249 
250  // Yeh + Berkowitz dipole term @cite yeh99a
251  double field_tot = gblcblk[0];
252 
253  // Const. potential contribution
254  if (elc_params.const_pot) {
255  coulomb.field_induced = gblcblk[1];
256  coulomb.field_applied = elc_params.pot_diff / elc_params.h;
257  field_tot -= coulomb.field_applied + coulomb.field_induced;
258  }
259 
260  for (auto &p : local_particles) {
261  p.f.f[2] -= field_tot * p.p.q;
262 
263  if (!elc_params.neutralize) {
264  // SUBTRACT the forces of the P3M homogeneous neutralizing background
265  p.f.f[2] += gblcblk[2] * p.p.q * (p.r.p[2] - shift);
266  }
267  }
268 }

270 /** Calculate the dipole energy.
271  * See @cite yeh99a.
272  */
273 static double dipole_energy(const ParticleRange &particles) {
274  double const pref = coulomb.prefactor * 2 * Utils::pi() *
275  box_geo.length_inv()[0] * box_geo.length_inv()[1] *
276  box_geo.length_inv()[2];
277  constexpr std::size_t size = 7;
278  /* for nonneutral systems, this shift gives the background contribution
279  (rsp. for this shift, the DM of the background is zero) */
280  double const shift = box_geo.length_half()[2];
281 
282  // collect moments
283 
284  gblcblk[0] = 0; // sum q_i primary box
285  gblcblk[1] = 0; // sum q_i boundary layers
286  gblcblk[2] = 0; // sum q_i (z_i - L/2) primary box
287  gblcblk[3] = 0; // sum q_i (z_i - L/2) boundary layers
288  gblcblk[4] = 0; // sum q_i (z_i - L/2)^2 primary box
289  gblcblk[5] = 0; // sum q_i (z_i - L/2)^2 boundary layers
290  gblcblk[6] = 0; // sum q_i z_i primary box
291 
292  for (auto &p : particles) {
293  check_gap_elc(p);
294 
295  gblcblk[0] += p.p.q;
296  gblcblk[2] += p.p.q * (p.r.p[2] - shift);
297  gblcblk[4] += p.p.q * (Utils::sqr(p.r.p[2] - shift));
298  gblcblk[6] += p.p.q * p.r.p[2];
299 
300  if (elc_params.dielectric_contrast_on) {
301  if (p.r.p[2] < elc_params.space_layer) {
302  gblcblk[1] += elc_params.delta_mid_bot * p.p.q;
303  gblcblk[3] += elc_params.delta_mid_bot * p.p.q * (-p.r.p[2] - shift);
304  gblcblk[5] +=
305  elc_params.delta_mid_bot * p.p.q * (Utils::sqr(-p.r.p[2] - shift));
306  }
307  if (p.r.p[2] > (elc_params.h - elc_params.space_layer)) {
308  gblcblk[1] += elc_params.delta_mid_top * p.p.q;
309  gblcblk[3] += elc_params.delta_mid_top * p.p.q *
310  (2 * elc_params.h - p.r.p[2] - shift);
311  gblcblk[5] += elc_params.delta_mid_top * p.p.q *
312  (Utils::sqr(2 * elc_params.h - p.r.p[2] - shift));
313  }
314  }
315  }
316 
317  distribute(size);
318 
319  // Yeh + Berkowitz term @cite yeh99a
320  double energy = 2 * pref * (Utils::sqr(gblcblk[2]) + gblcblk[2] * gblcblk[3]);
321 
322  if (!elc_params.neutralize) {
323  // SUBTRACT the energy of the P3M homogeneous neutralizing background
324  energy += 2 * pref *
325  (-gblcblk[0] * gblcblk[4] -
326  (.25 - .5 / 3.) * Utils::sqr(gblcblk[0] * box_geo.length()[2]));
327  }
328 
329  if (elc_params.dielectric_contrast_on) {
330  if (elc_params.const_pot) {
331  // zero potential difference contribution
332  energy +=
333  pref / elc_params.h * box_geo.length()[2] * Utils::sqr(gblcblk[6]);
334  // external potential shift contribution
335  energy -= 2 * elc_params.pot_diff / elc_params.h * gblcblk[6];
336  }
337 
338  /* counter the P3M homogeneous background contribution to the
339  boundaries. We never need that, since a homogeneous background
340  spanning the artificial boundary layers is aphysical. */
341  energy += pref * (-(gblcblk[1] * gblcblk[4] + gblcblk[0] * gblcblk[5]) -
342  (1. - 2. / 3.) * gblcblk[0] * gblcblk[1] *
343  Utils::sqr(box_geo.length()[2]));
344  }
345 
346  return this_node == 0 ? energy : 0;
347 }

351 inline double image_sum_b(double q, double z) {
352  double const shift = box_geo.length_half()[2];
353  double const fac = elc_params.delta_mid_top * elc_params.delta_mid_bot;
354  double const image_sum =
355  (q / (1.0 - fac) * (z - 2.0 * fac * box_geo.length()[2] / (1.0 - fac))) -
356  q * shift / (1 - fac);
357  return image_sum;
358 }

360 inline double image_sum_t(double q, double z) {
361  double const shift = box_geo.length_half()[2];
362  double const fac = elc_params.delta_mid_top * elc_params.delta_mid_bot;
363  double const image_sum =
364  (q / (1.0 - fac) * (z + 2.0 * fac * box_geo.length()[2] / (1.0 - fac))) -
365  q * shift / (1 - fac);
366  return image_sum;
367 }

370 static double z_energy(const ParticleRange &particles) {
371  double const pref = coulomb.prefactor * 2 * Utils::pi() *
372  box_geo.length_inv()[0] * box_geo.length_inv()[1];
373  constexpr std::size_t size = 4;
374 
375  /* for nonneutral systems, this shift gives the background contribution
376  (rsp. for this shift, the DM of the background is zero) */
377  double const shift = box_geo.length_half()[2];
378 
379  if (elc_params.dielectric_contrast_on) {
380  if (elc_params.const_pot) {
381  clear_vec(gblcblk, size);
382  for (auto &p : particles) {
383  gblcblk[0] += p.p.q;
384  gblcblk[1] += p.p.q * (p.r.p[2] - shift);
385  if (p.r.p[2] < elc_params.space_layer) {
386  gblcblk[2] -= elc_params.delta_mid_bot * p.p.q;
387  gblcblk[3] -= elc_params.delta_mid_bot * p.p.q * (-p.r.p[2] - shift);
388  }
389  if (p.r.p[2] > (elc_params.h - elc_params.space_layer)) {
390  gblcblk[2] += elc_params.delta_mid_top * p.p.q;
391  gblcblk[3] += elc_params.delta_mid_top * p.p.q *
392  (2 * elc_params.h - p.r.p[2] - shift);
393  }
394  }
395  } else {
396  double const delta = elc_params.delta_mid_top * elc_params.delta_mid_bot;
397  double const fac_delta_mid_bot = elc_params.delta_mid_bot / (1 - delta);
398  double const fac_delta_mid_top = elc_params.delta_mid_top / (1 - delta);
399  double const fac_delta = delta / (1 - delta);
400 
401  clear_vec(gblcblk, size);
402  for (auto &p : particles) {
403  gblcblk[0] += p.p.q;
404  gblcblk[1] += p.p.q * (p.r.p[2] - shift);
405  if (elc_params.dielectric_contrast_on) {
406  if (p.r.p[2] < elc_params.space_layer) {
407  gblcblk[2] += fac_delta * (elc_params.delta_mid_bot + 1) * p.p.q;
408  gblcblk[3] +=
409  p.p.q * (image_sum_b(elc_params.delta_mid_bot * delta,
410  -(2 * elc_params.h + p.r.p[2])) +
411  image_sum_b(delta, -(2 * elc_params.h - p.r.p[2])));
412  } else {
413  gblcblk[2] +=
414  fac_delta_mid_bot * (1 + elc_params.delta_mid_top) * p.p.q;
415  gblcblk[3] +=
416  p.p.q * (image_sum_b(elc_params.delta_mid_bot, -p.r.p[2]) +
417  image_sum_b(delta, -(2 * elc_params.h - p.r.p[2])));
418  }
419  if (p.r.p[2] > (elc_params.h - elc_params.space_layer)) {
420  // note the minus sign here which is required due to |z_i-z_j|
421  gblcblk[2] -= fac_delta * (elc_params.delta_mid_top + 1) * p.p.q;
422  gblcblk[3] -=
423  p.p.q * (image_sum_t(elc_params.delta_mid_top * delta,
424  4 * elc_params.h - p.r.p[2]) +
425  image_sum_t(delta, 2 * elc_params.h + p.r.p[2]));
426  } else {
427  // note the minus sign here which is required due to |z_i-z_j|
428  gblcblk[2] -=
429  fac_delta_mid_top * (1 + elc_params.delta_mid_bot) * p.p.q;
430  gblcblk[3] -=
431  p.p.q * (image_sum_t(elc_params.delta_mid_top,
432  2 * elc_params.h - p.r.p[2]) +
433  image_sum_t(delta, 2 * elc_params.h + p.r.p[2]));
434  }
435  }
436  }
437  }
438  }
439  distribute(size);
440 
441  double energy = 0;
442  if (this_node == 0)
443  energy -= gblcblk[1] * gblcblk[2] - gblcblk[0] * gblcblk[3];
444 
445  return pref * energy;
446 }

448 /*****************************************************************/
449 static void add_z_force(const ParticleRange &particles) {
450  double const pref = coulomb.prefactor * 2 * Utils::pi() *
451  box_geo.length_inv()[0] * box_geo.length_inv()[1];
452  constexpr std::size_t size = 1;
453 
454  if (elc_params.dielectric_contrast_on) {
455  auto local_particles = particles;
456  if (elc_params.const_pot) {
457  clear_vec(gblcblk, size);
458  /* just counter the 2 pi |z| contribution stemming from P3M */
459  for (auto &p : local_particles) {
460  if (p.r.p[2] < elc_params.space_layer)
461  gblcblk[0] -= elc_params.delta_mid_bot * p.p.q;
462  if (p.r.p[2] > (elc_params.h - elc_params.space_layer))
463  gblcblk[0] += elc_params.delta_mid_top * p.p.q;
464  }
465  } else {
466  double const delta = elc_params.delta_mid_top * elc_params.delta_mid_bot;
467  double const fac_delta_mid_bot = elc_params.delta_mid_bot / (1 - delta);
468  double const fac_delta_mid_top = elc_params.delta_mid_top / (1 - delta);
469  double const fac_delta = delta / (1 - delta);
470 
471  clear_vec(gblcblk, size);
472  for (auto &p : local_particles) {
473  if (p.r.p[2] < elc_params.space_layer) {
474  gblcblk[0] += fac_delta * (elc_params.delta_mid_bot + 1) * p.p.q;
475  } else {
476  gblcblk[0] +=
477  fac_delta_mid_bot * (1 + elc_params.delta_mid_top) * p.p.q;
478  }
479 
480  if (p.r.p[2] > (elc_params.h - elc_params.space_layer)) {
481  // note the minus sign here which is required due to |z_i-z_j|
482  gblcblk[0] -= fac_delta * (elc_params.delta_mid_top + 1) * p.p.q;
483  } else {
484  // note the minus sign here which is required due to |z_i-z_j|
485  gblcblk[0] -=
486  fac_delta_mid_top * (1 + elc_params.delta_mid_bot) * p.p.q;
487  }
488  }
489  }
490 
491  gblcblk[0] *= pref;
492 
493  distribute(size);
494 
495  for (auto &p : local_particles) {
496  p.f.f[2] += gblcblk[0] * p.p.q;
497  }
498  }

505 /** \name q=0 or p=0 per frequency code */
506 /**@{*/
507 template <PoQ axis>
508 void setup_PoQ(std::size_t index, double omega,
509  const ParticleRange &particles) {
510  assert(index >= 1);
511  double const pref_di = coulomb.prefactor * 4 * Utils::pi() *
512  box_geo.length_inv()[0] * box_geo.length_inv()[1];
513  double const pref = -pref_di / expm1(omega * box_geo.length()[2]);
514  constexpr std::size_t size = 4;
515  double lclimgebot[4], lclimgetop[4], lclimge[4];
516  double fac_delta_mid_bot = 1, fac_delta_mid_top = 1, fac_delta = 1;
517 
518  if (elc_params.dielectric_contrast_on) {
519  double const fac_elc =
520  1.0 / (1 - elc_params.delta_mid_top * elc_params.delta_mid_bot *
521  exp(-omega * 2 * elc_params.h));
522  fac_delta_mid_bot = elc_params.delta_mid_bot * fac_elc;
523  fac_delta_mid_top = elc_params.delta_mid_top * fac_elc;
524  fac_delta = fac_delta_mid_bot * elc_params.delta_mid_top;
525  }
526 
527  clear_vec(lclimge, size);
528  clear_vec(gblcblk, size);
529  auto &sc_cache = (axis == PoQ::P) ? scxcache : scycache;
530 
531  std::size_t ic = 0;
532  auto const o = (index - 1) * particles.size();
533  for (auto &p : particles) {
534  double e = exp(omega * p.r.p[2]);
535 
536  partblk[size * ic + POQESM] = p.p.q * sc_cache[o + ic].s / e;

965 double ELC_tune_far_cut(ELCParameters const &params) {
966  // Largest reasonable cutoff for far formula
967  constexpr auto maximal_far_cut = 50.;
968  double const h = params.h;
969  double lz = box_geo.length()[2];
970  double const min_inv_boxl =
971  std::min(box_geo.length_inv()[0], box_geo.length_inv()[1]);
972 
973  if (params.dielectric_contrast_on) {
974  // adjust lz according to dielectric layer method
975  lz = params.h + params.space_layer;
976  }
977 
978  if (h < 0.) {
979  throw std::runtime_error("gap size too large");
980  }
981 
982  auto far_cut = min_inv_boxl;
983  double err;
984  do {
985  const auto prefactor = 2 * Utils::pi() * far_cut;
986 
987  const auto sum =
988  prefactor + 2 * (box_geo.length_inv()[0] + box_geo.length_inv()[1]);
989  const auto den = -expm1(-prefactor * lz);
990  const auto num1 = exp(prefactor * (h - lz));
991  const auto num2 = exp(-prefactor * (h + lz));
992 
993  err = 0.5 / den *
994  (num1 * (sum + 1 / (lz - h)) / (lz - h) +
995  num2 * (sum + 1 / (lz + h)) / (lz + h));
996
