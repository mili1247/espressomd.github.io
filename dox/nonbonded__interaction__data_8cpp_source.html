102 static double recalc_maximal_cutoff(const IA_parameters &data) {
103  auto max_cut_current = INACTIVE_CUTOFF;
104 
105 #ifdef LENNARD_JONES
106  max_cut_current = std::max(max_cut_current, (data.lj.cut + data.lj.offset));
107 #endif
108 
109 #ifdef WCA
110  max_cut_current = std::max(max_cut_current, data.wca.cut);
111 #endif
112 
113 #ifdef DPD
114  max_cut_current = std::max(
115  max_cut_current, std::max(data.dpd_radial.cutoff, data.dpd_trans.cutoff));
116 #endif
117 
118 #ifdef LENNARD_JONES_GENERIC
119  max_cut_current =
120  std::max(max_cut_current, (data.ljgen.cut + data.ljgen.offset));
121 #endif
122 
123 #ifdef SMOOTH_STEP
124  max_cut_current = std::max(max_cut_current, data.smooth_step.cut);
125 #endif
126 
127 #ifdef HERTZIAN
128  max_cut_current = std::max(max_cut_current, data.hertzian.sig);
129 #endif
130 
131 #ifdef GAUSSIAN
132  max_cut_current = std::max(max_cut_current, data.gaussian.cut);
133 #endif
134 
135 #ifdef BMHTF_NACL
136  max_cut_current = std::max(max_cut_current, data.bmhtf.cut);
137 #endif
138 
139 #ifdef MORSE
140  max_cut_current = std::max(max_cut_current, data.morse.cut);
141 #endif
142 
143 #ifdef BUCKINGHAM
144  max_cut_current = std::max(max_cut_current, data.buckingham.cut);
145 #endif
146 
147 #ifdef SOFT_SPHERE
148  max_cut_current = std::max(max_cut_current,
149  (data.soft_sphere.cut + data.soft_sphere.offset));
150 #endif
151 
152 #ifdef HAT
153  max_cut_current = std::max(max_cut_current, data.hat.r);
154 #endif
155 
156 #ifdef LJCOS
157  max_cut_current =
158  std::max(max_cut_current, (data.ljcos.cut + data.ljcos.offset));
159 #endif
160 
161 #ifdef LJCOS2
162  max_cut_current =
163  std::max(max_cut_current, (data.ljcos2.cut + data.ljcos2.offset));
164 #endif
165 
166 #ifdef GAY_BERNE
167  max_cut_current = std::max(max_cut_current, data.gay_berne.cut);
168 #endif
169 
170 #ifdef TABULATED
171  max_cut_current = std::max(max_cut_current, data.tab.cutoff());
172 #endif
173 
174 #ifdef THOLE
175  // If THOLE is active, use p3m cutoff
176  if (data.thole.scaling_coeff != 0)
177  max_cut_current =
178  std::max(max_cut_current, Coulomb::cutoff(box_geo.length()));
179 #endif
180 
181  return max_cut_current;
182 }

184 double maximal_cutoff_nonbonded() {
185  auto max_cut_nonbonded = INACTIVE_CUTOFF;
186 
187  for (auto &data : ia_params) {
188  data.max_cut = recalc_maximal_cutoff(data);
189  max_cut_nonbonded = std::max(max_cut_nonbonded, data.max_cut);
190  }
191 
192  return max_cut_nonbonded;
193 }

195 double maximal_cutoff() {
196  auto max_cut = min_global_cut;
197  auto const max_cut_long_range = recalc_long_range_cutoff();
198  auto const max_cut_bonded = maximal_cutoff_bonded();
199  auto const max_cut_nonbonded = maximal_cutoff_nonbonded();
200 
201  max_cut = std::max(max_cut, max_cut_long_range);
202  max_cut = std::max(max_cut, max_cut_bonded);
203  max_cut = std::max(max_cut, max_cut_nonbonded);
204  max_cut = std::max(max_cut, collision_detection_cutoff());
205 
206  return max_cut;
207 }

209 /** Increase the LOCAL @ref ia_params field for non-bonded interactions
210  * to the given size. This function is not exported since it does not
211  * do this on all nodes. Use @ref make_particle_type_exist for that.
212  */
213 void realloc_ia_params(int nsize) {
214  if (nsize <= max_seen_particle_type)
215  return;
216 
217  auto new_params = std::vector<IA_parameters>(nsize * (nsize + 1) / 2);
218 
219  /* if there is an old field, move entries */
220  for (int i = 0; i < max_seen_particle_type; i++)
221  for (int j = i; j < max_seen_particle_type; j++) {
222  new_params.at(Utils::upper_triangular(i, j, nsize)) =
223  std::move(*get_ia_param(i, j));
224  }
225 
226  max_seen_particle_type = nsize;
227  std::swap(ia_params, new_params);
228 }
