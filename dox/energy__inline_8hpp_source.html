71 /** Calculate non-bonded energies between a pair of particles.
72  * @param p1 particle 1.
73  * @param p2 particle 2.
74  * @param ia_params the interaction parameters between the two particles
75  * @param d vector between p1 and p2.
76  * @param dist distance between p1 and p2.
77  * @return the short-range interaction energy between the two particles
78  */
79 inline double calc_non_bonded_pair_energy(Particle const &p1,
80  Particle const &p2,
81  IA_parameters const &ia_params,
82  Utils::Vector3d const &d,
83  double const dist) {
84 
85  double ret = 0;
86 
87 #ifdef LENNARD_JONES
88  /* Lennard-Jones */
89  ret += lj_pair_energy(ia_params, dist);
90 #endif
91 #ifdef WCA
92  /* WCA */
93  ret += wca_pair_energy(ia_params, dist);
94 #endif
95 
96 #ifdef LENNARD_JONES_GENERIC
97  /* Generic Lennard-Jones */
98  ret += ljgen_pair_energy(ia_params, dist);
99 #endif
100 
101 #ifdef SMOOTH_STEP
102  /* smooth step */
103  ret += SmSt_pair_energy(ia_params, dist);
104 #endif
105 
106 #ifdef HERTZIAN
107  /* Hertzian potential */
108  ret += hertzian_pair_energy(ia_params, dist);
109 #endif
110 
111 #ifdef GAUSSIAN
112  /* Gaussian potential */
113  ret += gaussian_pair_energy(ia_params, dist);
114 #endif
115 
116 #ifdef BMHTF_NACL
117  /* BMHTF NaCl */
118  ret += BMHTF_pair_energy(ia_params, dist);
119 #endif
120 
121 #ifdef MORSE
122  /* Morse */
123  ret += morse_pair_energy(ia_params, dist);
124 #endif
125 
126 #ifdef BUCKINGHAM
127  /* Buckingham */
128  ret += buck_pair_energy(ia_params, dist);
129 #endif
130 
131 #ifdef SOFT_SPHERE
132  /* soft-sphere */
133  ret += soft_pair_energy(ia_params, dist);
134 #endif
135 
136 #ifdef HAT
137  /* hat */
138  ret += hat_pair_energy(ia_params, dist);
139 #endif
140 
141 #ifdef LJCOS2
142  /* Lennard-Jones */
143  ret += ljcos2_pair_energy(ia_params, dist);
144 #endif
145 
146 #ifdef THOLE
147  /* Thole damping */
148  ret += thole_pair_energy(p1, p2, ia_params, d, dist);
149 #endif
150 
151 #ifdef TABULATED
152  /* tabulated */
153  ret += tabulated_pair_energy(ia_params, dist);
154 #endif
155 
156 #ifdef LJCOS
157  /* Lennard-Jones cosine */
158  ret += ljcos_pair_energy(ia_params, dist);
159 #endif
160 
161 #ifdef GAY_BERNE
162  /* Gay-Berne */
163  ret += gb_pair_energy(p1.r.calc_director(), p2.r.calc_director(), ia_params,
164  d, dist);
165 #endif
166 
167  return ret;
168 }

170 /** Add non-bonded and short-range Coulomb energies between a pair of particles
171  * to the energy observable.
172  * @param p1 particle 1.
173  * @param p2 particle 2.
174  * @param d vector between p1 and p2.
175  * @param dist distance between p1 and p2.
176  * @param dist2 distance squared between p1 and p2.
177  * @param[in,out] obs_energy energy observable.
178  */
179 inline void add_non_bonded_pair_energy(Particle const &p1, Particle const &p2,
180  Utils::Vector3d const &d,
181  double const dist, double const dist2,
182  Observable_stat &obs_energy) {
183  IA_parameters const &ia_params = *get_ia_param(p1.p.type, p2.p.type);
184 
185 #ifdef EXCLUSIONS
186  if (do_nonbonded(p1, p2))
187 #endif
188  obs_energy.add_non_bonded_contribution(
189  p1.p.type, p2.p.type,
190  calc_non_bonded_pair_energy(p1, p2, ia_params, d, dist));
191 
192 #ifdef ELECTROSTATICS
193  if (!obs_energy.coulomb.empty())
194  obs_energy.coulomb[0] +=
195  Coulomb::pair_energy(p1, p2, p1.p.q * p2.p.q, d, dist, dist2);
196 #endif
197 
198 #ifdef DIPOLES
199  if (!obs_energy.dipolar.empty())
200  obs_energy.dipolar[0] += Dipole::pair_energy(p1, p2, d, dist, dist2);
201 #endif
202 }

204 inline boost::optional<double>
205 calc_bonded_energy(Bonded_IA_Parameters const &iaparams, Particle const &p1,
206  Utils::Span<Particle *> partners) {
207  auto const n_partners = partners.size();
208 
209  auto p2 = (n_partners > 0) ? partners[0] : nullptr;
210  auto p3 = (n_partners > 1) ? partners[1] : nullptr;
211  auto p4 = (n_partners > 2) ? partners[2] : nullptr;
212 
213  if (n_partners == 1) {
214  auto const dx = box_geo.get_mi_vector(p1.r.p, p2->r.p);
215  if (auto const *iap = boost::get<FeneBond>(&iaparams)) {
216  return iap->energy(dx);
217  }
218  if (auto const *iap = boost::get<HarmonicBond>(&iaparams)) {
219  return iap->energy(dx);
220  }
221  if (auto const *iap = boost::get<QuarticBond>(&iaparams)) {
222  return iap->energy(dx);
223  }
224 #ifdef ELECTROSTATICS
225  if (auto const *iap = boost::get<BondedCoulomb>(&iaparams)) {
226  return iap->energy(p1.p.q * p2->p.q, dx);
227  }
228  if (auto const *iap = boost::get<BondedCoulombSR>(&iaparams)) {
229  return iap->energy(p1, *p2, dx);
230  }
231 #endif
232 #ifdef BOND_CONSTRAINT
233  if (auto const *iap = boost::get<RigidBond>(&iaparams)) {
234  return boost::optional<double>(0);
235  }
236 #endif
237 #ifdef TABULATED
238  if (auto const *iap = boost::get<TabulatedDistanceBond>(&iaparams)) {
239  return iap->energy(dx);
240  }
241 #endif
242  if (auto const *iap = boost::get<VirtualBond>(&iaparams)) {
243  return boost::optional<double>(0);
244  }
245  throw BondUnknownTypeError();
246  } // 1 partner
247  if (n_partners == 2) {
248  if (auto const *iap = boost::get<AngleHarmonicBond>(&iaparams)) {
249  return iap->energy(p1.r.p, p2->r.p, p3->r.p);
250  }
251  if (auto const *iap = boost::get<AngleCosineBond>(&iaparams)) {
252  return iap->energy(p1.r.p, p2->r.p, p3->r.p);
253  }
254  if (auto const *iap = boost::get<AngleCossquareBond>(&iaparams)) {
255  return iap->energy(p1.r.p, p2->r.p, p3->r.p);
256  }
257  if (auto const *iap = boost::get<TabulatedAngleBond>(&iaparams)) {
258  return iap->energy(p1.r.p, p2->r.p, p3->r.p);
259  }
260  if (auto const *iap = boost::get<IBMTriel>(&iaparams)) {
261  runtimeWarningMsg() << "Unsupported bond type " +
262  std::to_string(iaparams.which()) +
263  " in energy calculation.";
264  return 0.;
265  }
266  throw BondUnknownTypeError();
267  } // 2 partners
268  if (n_partners == 3) {
269  if (auto const *iap = boost::get<DihedralBond>(&iaparams)) {
270  return iap->energy(p2->r.p, p1.r.p, p3->r.p, p4->r.p);
271  }
272  if (auto const *iap = boost::get<TabulatedDihedralBond>(&iaparams)) {
273  return iap->energy(p2->r.p, p1.r.p, p3->r.p, p4->r.p);
274  }
275  if (auto const *iap = boost::get<IBMTribend>(&iaparams)) {
276  runtimeWarningMsg() << "Unsupported bond type " +
277  std::to_string(iaparams.which()) +
278  " in energy calculation.";
279  return 0.;
280  }
281  throw BondUnknownTypeError();
282  } // 3 partners
283  if (n_partners == 0) {
284  return 0.;
285  }
286 
287  throw BondInvalidSizeError(n_partners);
288 }

290 /** Calculate kinetic energies from translation for one particle.
291  * @param p particle for which to calculate energies
292  */
293 inline double translational_kinetic_energy(Particle const &p) {
294  return p.p.is_virtual ? 0. : 0.5 * p.p.mass * p.m.v.norm2();
295 }
296 
297 /** Calculate kinetic energies from rotation for one particle.
298  * @param p particle for which to calculate energies
299  */
300 inline double rotational_kinetic_energy(Particle const &p) {
301 #ifdef ROTATION
302  return p.p.rotation
303  ? 0.5 * (hadamard_product(p.m.omega, p.m.omega) * p.p.rinertia)
304  : 0.0;
305 #else
306  return 0.0;
307 #endif
308 }
309 
310 /** Calculate kinetic energies for one particle.
311  * @param p particle for which to calculate energies
312  */
313 inline double calc_kinetic_energy(Particle const &p) {
314  return translational_kinetic_energy(p) + rotational_kinetic_energy(p);
315 }
