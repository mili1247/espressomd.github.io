114 /** %Thermostat for Langevin dynamics. */
115 struct LangevinThermostat : public BaseThermostat {
116 private:
117  using GammaType = Thermostat::GammaType;
118 
119 public:
120  /** Recalculate prefactors.
121  * Needs to be called every time the parameters are changed.
122  */
123  void recalc_prefactors(double kT, double time_step) {
124  pref_friction = -gamma;
125  pref_noise = sigma(kT, time_step, gamma);
126  // If gamma_rotation is not set explicitly, use the translational one.
127  if (gamma_rotation < GammaType{}) {
128  gamma_rotation = gamma;
129  }
130  pref_noise_rotation = sigma(kT, time_step, gamma_rotation);
131  }
132  /** Calculate the noise prefactor.
133  * Evaluates the quantity @f$ \sqrt{2 k_B T \gamma / dt} / \sigma_\eta @f$
134  * with @f$ \sigma_\eta @f$ the standard deviation of the random uniform
135  * process @f$ \eta(t) @f$.
136  */
137  static GammaType sigma(double kT, double time_step, GammaType const &gamma) {
138  // random uniform noise has variance 1/12
139  constexpr auto const temp_coeff = 2.0 * 12.0;
140  return sqrt((temp_coeff * kT / time_step) * gamma);
141  }
142  /** @name Parameters */
143  /**@{*/
144  /** Translational friction coefficient @f$ \gamma_{\text{trans}} @f$. */
145  GammaType gamma = sentinel(GammaType{});
146  /** Rotational friction coefficient @f$ \gamma_{\text{rot}} @f$. */
147  GammaType gamma_rotation = sentinel(GammaType{});
148  /**@}*/
149  /** @name Prefactors */
150  /**@{*/
151  /** Prefactor for the friction.
152  * Stores @f$ \gamma_{\text{trans}} @f$.
153  */
154  GammaType pref_friction;
155  /** Prefactor for the translational velocity noise.
156  * Stores @f$ \sqrt{2 k_B T \gamma_{\text{trans}} / dt} / \sigma_\eta @f$.
157  */
158  GammaType pref_noise;
159  /** Prefactor for the angular velocity noise.
160  * Stores @f$ \sqrt{2 k_B T \gamma_{\text{rot}} / dt} / \sigma_\eta @f$.
161  */
162  GammaType pref_noise_rotation;
163  /**@}*/
164 };

166 /** %Thermostat for Brownian dynamics.
167  * Default particle mass is assumed to be unitary in these global parameters.
168  */
169 struct BrownianThermostat : public BaseThermostat {
170 private:
171  using GammaType = Thermostat::GammaType;
172 
173 public:
174  /** Recalculate prefactors.
175  * Needs to be called every time the parameters are changed.
176  */
177  void recalc_prefactors(double kT) {
178  /** The heat velocity dispersion corresponds to the Gaussian noise only,
179  * which is only valid for the BD. Just a square root of kT, see (10.2.17)
180  * and comments in 2 paragraphs afterwards, @cite Pottier2010.
181  */
182  sigma_vel = sigma(kT);
183  /** The random walk position dispersion is defined by the second eq. (14.38)
184  * of @cite Schlick2010. Its time interval factor will be added in the
185  * Brownian Dynamics functions. Its square root is the standard deviation.
186  */
187  sigma_pos = sigma(kT, gamma);
188 #ifdef ROTATION
189  /** Note: the BD thermostat assigns the brownian viscous parameters as well.
190  * They correspond to the friction tensor Z from the eq. (14.31) of
191  * @cite Schlick2010.
192  */
193  // If gamma_rotation is not set explicitly, use the translational one.
194  if (gamma_rotation < GammaType{}) {
195  gamma_rotation = gamma;
196  }
197  sigma_vel_rotation = sigma(kT);
198  sigma_pos_rotation = sigma(kT, gamma_rotation);
199 #endif // ROTATION
200  }
201  /** Calculate the noise prefactor.
202  * Evaluates the quantity @f$ \sqrt{2 k_B T / \gamma} / \sigma_\eta @f$
203  * with @f$ \sigma_\eta @f$ the standard deviation of the random gaussian
204  * process @f$ \eta(t) @f$.
205  */
206  static GammaType sigma(double kT, GammaType const &gamma) {
207  constexpr auto const temp_coeff = 2.0;
208  return sqrt(Utils::hadamard_division(temp_coeff * kT, gamma));
209  }
210  /** Calculate the noise prefactor.
211  * Evaluates the quantity @f$ \sqrt{k_B T} / \sigma_\eta @f$
212  * with @f$ \sigma_\eta @f$ the standard deviation of the random gaussian
213  * process @f$ \eta(t) @f$.
214  */
215  static double sigma(double kT) {
216  constexpr auto const temp_coeff = 1.0;
217  return sqrt(temp_coeff * kT);
218  }
219  /** @name Parameters */
220  /**@{*/
221  /** Translational friction coefficient @f$ \gamma_{\text{trans}} @f$. */
222  GammaType gamma = sentinel(GammaType{});
223  /** Rotational friction coefficient @f$ \gamma_{\text{rot}} @f$. */
224  GammaType gamma_rotation = sentinel(GammaType{});
225  /**@}*/
226  /** @name Prefactors */
227  /**@{*/
228  /** Translational noise standard deviation.
229  * Stores @f$ \sqrt{2D_{\text{trans}}} @f$ with
230  * @f$ D_{\text{trans}} = k_B T/\gamma_{\text{trans}} @f$
231  * the translational diffusion coefficient.
232  */
233  GammaType sigma_pos = sentinel(GammaType{});
234  /** Rotational noise standard deviation.
235  * Stores @f$ \sqrt{2D_{\text{rot}}} @f$ with
236  * @f$ D_{\text{rot}} = k_B T/\gamma_{\text{rot}} @f$
237  * the rotational diffusion coefficient.
238  */
239  GammaType sigma_pos_rotation = sentinel(GammaType{});
240  /** Translational velocity noise standard deviation.
241  * Stores @f$ \sqrt{k_B T} @f$.
242  */
243  double sigma_vel = 0;
244  /** Angular velocity noise standard deviation.
245  * Stores @f$ \sqrt{k_B T} @f$.
246  */
247  double sigma_vel_rotation = 0;
248  /**@}*/
249 };

251 #ifdef NPT
252 /** %Thermostat for isotropic NPT dynamics. */
253 struct IsotropicNptThermostat : public BaseThermostat {
254 private:
255  using GammaType = Thermostat::GammaType;
256 
257 public:
258  /** Recalculate prefactors.
259  * Needs to be called every time the parameters are changed.
260  */
261  void recalc_prefactors(double kT, double piston, double time_step) {
262  assert(piston > 0.0);
263  auto const half_time_step = time_step / 2.0;
264  pref_rescale_0 = -gamma0 * half_time_step;
265  pref_noise_0 = sigma(kT, gamma0, time_step);
266  pref_rescale_V = -gammav * half_time_step / piston;
267  pref_noise_V = sigma(kT, gammav, time_step);
268  }
269  /** Calculate the noise prefactor.
270  * Evaluates the quantity @f$ \sqrt{2 k_B T \gamma dt / 2} / \sigma_\eta @f$
271  * with @f$ \sigma_\eta @f$ the standard deviation of the random uniform
272  * process @f$ \eta(t) @f$.
273  */
274  static double sigma(double kT, double gamma, double time_step) {
275  // random uniform noise has variance 1/12; the temperature
276  // coefficient of 2 is canceled out by the half time step
277  constexpr auto const temp_coeff = 12.0;
278  return sqrt(temp_coeff * kT * gamma * time_step);
279  }
280  /** @name Parameters */
281  /**@{*/
282  /** Friction coefficient of the particles @f$ \gamma^0 @f$ */
283  double gamma0;
284  /** Friction coefficient for the box @f$ \gamma^V @f$ */
285  double gammav;
286  /**@}*/
287  /** @name Prefactors */
288  /**@{*/
289  /** %Particle velocity rescaling at half the time step.
290  * Stores @f$ \gamma^{0}\cdot\frac{dt}{2} @f$.
291  */
292  double pref_rescale_0;
293  /** %Particle velocity rescaling noise standard deviation.
294  * Stores @f$ \sqrt{k_B T \gamma^{0} dt} / \sigma_\eta @f$.
295  */
296  double pref_noise_0;
297  /** Volume rescaling at half the time step.
298  * Stores @f$ \frac{\gamma^{V}}{Q}\cdot\frac{dt}{2} @f$.
299  */
300  double pref_rescale_V;
301  /** Volume rescaling noise standard deviation.
302  * Stores @f$ \sqrt{k_B T \gamma^{V} dt} / \sigma_\eta @f$.
303  */
304  double pref_noise_V;
305  /**@}*/
306 };
307 #endif
