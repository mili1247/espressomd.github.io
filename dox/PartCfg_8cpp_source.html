42  auto const this_size =
43  boost::algorithm::clamp(chunk_size, 0, ids.size() - offset);
44  auto const chunk_ids =
45  Utils::make_const_span(ids.data() + offset, this_size);
46 
47  prefetch_particle_data(chunk_ids);
48
