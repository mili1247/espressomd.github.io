167 void p3m_init() {

169  // prefactor is zero: electrostatics switched off
170  p3m.params.r_cut = 0.0;
171  p3m.params.r_cut_iL = 0.0;
172  return;
173  }
174 
175  if (p3m_sanity_checks_parameters()) {
176  return;
177  }
178 
179  p3m.params.cao3 = Utils::int_pow<3>(p3m.params.cao);

181  /* initializes the (inverse) mesh constant p3m.params.a (p3m.params.ai) and
182  * the cutoff for charge assignment p3m.params.cao_cut */
183  p3m_init_a_ai_cao_cut();
184 
185  if (p3m_sanity_checks()) {
186  return;
187  }
188 
189  double elc_layer = 0.0;

192  }
193 
194  p3m_calc_local_ca_mesh(p3m.local_mesh, p3m.params, local_geo, skin,
195  elc_layer);
196 
197  p3m.sm.resize(comm_cart, p3m.local_mesh);
198 
199  int ca_mesh_size =
200  fft_init(p3m.local_mesh.dim, p3m.local_mesh.margin, p3m.params.mesh,
201  p3m.params.mesh_off, p3m.ks_pnum, p3m.fft, node_grid, comm_cart);
202  p3m.rs_mesh.resize(ca_mesh_size);
203 
204  for (auto &e : p3m.E_mesh) {

206  }

209 
210  /* fix box length dependent constants */

212 
213  p3m_count_charged_particles();

239 void p3m_set_params(double r_cut, const int *mesh, int cao, double alpha,
240  double accuracy) {
241  if (r_cut < 0)
242  throw std::runtime_error("P3M: invalid r_cut");
243 
244  if (mesh[0] < 0 || mesh[1] < 0 || mesh[2] < 0)
245  throw std::runtime_error("P3M: invalid mesh size");
246 
247  if (cao < 1 || cao > 7)
248  throw std::runtime_error("P3M: invalid cao");
249 
250  if (cao > mesh[0] || cao > mesh[1] || cao > mesh[2])
251  throw std::runtime_error("P3M: cao larger than mesh size");
252 
253  if (alpha <= 0.0 && alpha != -1.0)
254  throw std::runtime_error("P3M: invalid alpha");
255 
256  if (accuracy <= 0.0 && accuracy != -1.0)
257  throw std::runtime_error("P3M: invalid accuracy");
258 
259  if (coulomb.method != COULOMB_P3M && coulomb.method != COULOMB_ELC_P3M &&
260  coulomb.method != COULOMB_P3M_GPU)
261  coulomb.method = COULOMB_P3M;
262 
263  p3m.params.r_cut = r_cut;
264  p3m.params.r_cut_iL = r_cut * box_geo.length_inv()[0];
265  p3m.params.mesh[2] = mesh[2];
266  p3m.params.mesh[1] = mesh[1];
267  p3m.params.mesh[0] = mesh[0];
268  p3m.params.cao = cao;
269  p3m.params.alpha = alpha;
270  p3m.params.alpha_L = alpha * box_geo.length()[0];
271  p3m.params.accuracy = accuracy;
272 
273  mpi_bcast_coulomb_params();
274 }

353  void operator()(double force_prefac, const ParticleRange &particles) const {

376  }

416 /** @details Calculate the long range electrostatics part of the pressure
417  * tensor. This is part \f$\Pi_{\textrm{dir}, \alpha, \beta}\f$ eq. (2.6)
418  * in @cite essmann95a. The part \f$\Pi_{\textrm{corr}, \alpha, \beta}\f$
419  * eq. (2.8) is not present here since M is the empty set in our simulations.
420  */
421 Utils::Vector9d p3m_calc_kspace_pressure_tensor() {
422  using namespace detail::FFT_indexing;
423 
424  Utils::Vector9d node_k_space_pressure_tensor{};
425 
426  if (p3m.sum_q2 > 0) {
427  p3m.sm.gather_grid(p3m.rs_mesh.data(), comm_cart, p3m.local_mesh.dim);
428  fft_perform_forw(p3m.rs_mesh.data(), p3m.fft, comm_cart);
429 
430  double diagonal = 0;
431  int ind = 0;
432  int j[3];
433  auto const half_alpha_inv_sq = Utils::sqr(1.0 / 2.0 / p3m.params.alpha);
434  for (j[0] = 0; j[0] < p3m.fft.plan[3].new_mesh[RX]; j[0]++) {
435  for (j[1] = 0; j[1] < p3m.fft.plan[3].new_mesh[RY]; j[1]++) {
436  for (j[2] = 0; j[2] < p3m.fft.plan[3].new_mesh[RZ]; j[2]++) {
437  auto const kx = 2.0 * Utils::pi() *
438  p3m.d_op[RX][j[KX] + p3m.fft.plan[3].start[KX]] *
439  box_geo.length_inv()[RX];
440  auto const ky = 2.0 * Utils::pi() *
441  p3m.d_op[RY][j[KY] + p3m.fft.plan[3].start[KY]] *
442  box_geo.length_inv()[RY];
443  auto const kz = 2.0 * Utils::pi() *
444  p3m.d_op[RZ][j[KZ] + p3m.fft.plan[3].start[KZ]] *
445  box_geo.length_inv()[RZ];
446  auto const sqk = Utils::sqr(kx) + Utils::sqr(ky) + Utils::sqr(kz);
447 
448  auto const node_k_space_energy =
449  (sqk == 0)
450  ? 0.0
451  : p3m.g_energy[ind] * (Utils::sqr(p3m.rs_mesh[2 * ind]) +
452  Utils::sqr(p3m.rs_mesh[2 * ind + 1]));
453  ind++;
454 
455  auto const vterm =
456  (sqk == 0) ? 0. : -2.0 * (1 / sqk + half_alpha_inv_sq);
457 
458  diagonal += node_k_space_energy;
459  auto const prefactor = node_k_space_energy * vterm;
460  node_k_space_pressure_tensor[0] += prefactor * kx * kx; /* sigma_xx */
461  node_k_space_pressure_tensor[1] += prefactor * kx * ky; /* sigma_xy */
462  node_k_space_pressure_tensor[2] += prefactor * kx * kz; /* sigma_xz */
463  node_k_space_pressure_tensor[3] += prefactor * ky * kx; /* sigma_yx */
464  node_k_space_pressure_tensor[4] += prefactor * ky * ky; /* sigma_yy */
465  node_k_space_pressure_tensor[5] += prefactor * ky * kz; /* sigma_yz */
466  node_k_space_pressure_tensor[6] += prefactor * kz * kx; /* sigma_zx */
467  node_k_space_pressure_tensor[7] += prefactor * kz * ky; /* sigma_zy */
468  node_k_space_pressure_tensor[8] += prefactor * kz * kz; /* sigma_zz */
469  }
470  }
471  }
472  node_k_space_pressure_tensor[0] += diagonal;
473  node_k_space_pressure_tensor[4] += diagonal;
474  node_k_space_pressure_tensor[8] += diagonal;
475  }
476 
477  auto const force_prefac = coulomb.prefactor / (2.0 * box_geo.volume());
478  return force_prefac * node_k_space_pressure_tensor;
479 }

481 double p3m_calc_kspace_forces(bool force_flag, bool energy_flag,
482  const ParticleRange &particles) {
483  /* Gather information for FFT grid inside the nodes domain (inner local mesh)
484  * and perform forward 3D FFT (Charge Assignment Mesh). */
485  p3m.sm.gather_grid(p3m.rs_mesh.data(), comm_cart, p3m.local_mesh.dim);
486  fft_perform_forw(p3m.rs_mesh.data(), p3m.fft, comm_cart);
487 
488  // Note: after these calls, the grids are in the order yzx and not xyz
489  // anymore!!!
490  /* The dipole moment is only needed if we don't have metallic boundaries. */
491  auto const box_dipole = (p3m.params.epsilon != P3M_EPSILON_METALLIC)
492  ? boost::make_optional(calc_dipole_moment(
493  comm_cart, particles, box_geo))
494  : boost::none;
495 
496  /* === k-space force calculation === */
497  if (force_flag) {
498  /* sqrt(-1)*k differentiation */
499  int j[3];
500  int ind = 0;
501  for (j[0] = 0; j[0] < p3m.fft.plan[3].new_mesh[0]; j[0]++) {
502  for (j[1] = 0; j[1] < p3m.fft.plan[3].new_mesh[1]; j[1]++) {
503  for (j[2] = 0; j[2] < p3m.fft.plan[3].new_mesh[2]; j[2]++) {
504  auto const rho_hat = std::complex<double>(p3m.rs_mesh[2 * ind + 0],
505  p3m.rs_mesh[2 * ind + 1]);
506  auto const phi_hat = p3m.g_force[ind] * rho_hat;
507 
508  for (int d = 0; d < 3; d++) {
509  /* direction in r-space: */
510  int d_rs = (d + p3m.ks_pnum) % 3;
511  /* directions */
512  auto const k = 2.0 * Utils::pi() *
513  p3m.d_op[d_rs][j[d] + p3m.fft.plan[3].start[d]] *
514  box_geo.length_inv()[d_rs];
515 
516  /* i*k*(Re+i*Im) = - Im*k + i*Re*k (i=sqrt(-1)) */
517  p3m.E_mesh[d_rs][2 * ind + 0] = -k * phi_hat.imag();
518  p3m.E_mesh[d_rs][2 * ind + 1] = +k * phi_hat.real();
519  }
520 
521  ind++;
522  }
523  }
524  }
525 
526  /* Back FFT force component mesh */
527  for (int d = 0; d < 3; d++) {
528  fft_perform_back(p3m.E_mesh[d].data(),
529  /* check_complex */ !p3m.params.tuning, p3m.fft,
530  comm_cart);
531  }
532 
533  {
534  std::array<double *, 3> E_fields = {
535  p3m.E_mesh[0].data(), p3m.E_mesh[1].data(), p3m.E_mesh[2].data()};
536  /* redistribute force component mesh */
537  p3m.sm.spread_grid(Utils::make_span(E_fields), comm_cart,
538  p3m.local_mesh.dim);
539  }
540 
541  auto const force_prefac = coulomb.prefactor / box_geo.volume();
542  Utils::integral_parameter<AssignForces, 1, 7>(p3m.params.cao, force_prefac,
543  particles);
544 
545  if (p3m.params.epsilon != P3M_EPSILON_METALLIC) {
546  add_dipole_correction(box_dipole.value(), particles);
547  }
548  } /* if(force_flag) */
549 
550  /* === k-space energy calculation === */
551  if (energy_flag) {
552  double node_k_space_energy = 0.;
553 
554  for (int i = 0; i < p3m.fft.plan[3].new_size; i++) {
555  // Use the energy optimized influence function for energy!
556  node_k_space_energy +=
557  p3m.g_energy[i] *
558  (Utils::sqr(p3m.rs_mesh[2 * i]) + Utils::sqr(p3m.rs_mesh[2 * i + 1]));
559  }
560  node_k_space_energy *= coulomb.prefactor / (2 * box_geo.volume());
561 
562  double k_space_energy = 0.0;
563  boost::mpi::reduce(comm_cart, node_k_space_energy, k_space_energy,
564  std::plus<>(), 0);
565  if (this_node == 0) {
566  /* self energy correction */
567  k_space_energy -= coulomb.prefactor *
568  (p3m.sum_q2 * p3m.params.alpha * Utils::sqrt_pi_i());
569  /* net charge correction */
570  k_space_energy -= coulomb.prefactor * p3m.square_sum_q * Utils::pi() /
571  (2.0 * box_geo.volume() * Utils::sqr(p3m.params.alpha));
572  /* dipole correction */
573  if (p3m.params.epsilon != P3M_EPSILON_METALLIC) {
574  k_space_energy += dipole_correction_energy(box_dipole.value());
575  }
576  }
577  return k_space_energy;
578  } /* if (energy_flag) */
579 
580  return 0.0;
581 }

599 /** Get the minimal error for this combination of parameters.
600  *
601  * The real space error is tuned such that it contributes half of the
602  * total error, and then the Fourier space error is calculated.
603  * If an optimal alpha is not found, the value 0.1 is used as fallback.
604  * @param[in] mesh @copybrief P3MParameters::mesh
605  * @param[in] cao @copybrief P3MParameters::cao
606  * @param[in] r_cut_iL @copybrief P3MParameters::r_cut_iL
607  * @param[out] _alpha_L @copybrief P3MParameters::alpha_L
608  * @param[out] _rs_err real space error
609  * @param[out] _ks_err Fourier space error
610  * @returns Error magnitude
611  */
612 static double p3m_get_accuracy(const int mesh[3], int cao, double r_cut_iL,
613  double *_alpha_L, double *_rs_err,
614  double *_ks_err) {
615  double rs_err, ks_err;
616  double alpha_L;
617 
618  /* calc maximal real space error for setting */
619  rs_err = p3m_real_space_error(coulomb.prefactor, r_cut_iL, p3m.sum_qpart,
620  p3m.sum_q2, 0);
621 
622  if (Utils::sqrt_2() * rs_err > p3m.params.accuracy) {
623  /* assume rs_err = ks_err -> rs_err = accuracy/sqrt(2.0) -> alpha_L */
624  alpha_L =
625  sqrt(log(Utils::sqrt_2() * rs_err / p3m.params.accuracy)) / r_cut_iL;
626  } else {
627  /* even alpha=0 is ok, however, we cannot choose it since it kills the
628  k-space error formula.
629  Anyways, this very likely NOT the optimal solution */
630  alpha_L = 0.1;
631  }
632 
633  *_alpha_L = alpha_L;
634  /* calculate real space and k-space error for this alpha_L */
635  rs_err = p3m_real_space_error(coulomb.prefactor, r_cut_iL, p3m.sum_qpart,
636  p3m.sum_q2, alpha_L);
637 #ifdef CUDA
638  if (coulomb.method == COULOMB_P3M_GPU)
639  ks_err =
640  p3m_k_space_error_gpu(coulomb.prefactor, mesh, cao, p3m.sum_qpart,
641  p3m.sum_q2, alpha_L, box_geo.length().data());
642  else
643 #endif
644  ks_err = p3m_k_space_error(coulomb.prefactor, mesh, cao, p3m.sum_qpart,
645  p3m.sum_q2, alpha_L);
646 
647  *_rs_err = rs_err;
648  *_ks_err = ks_err;
649  return sqrt(Utils::sqr(rs_err) + Utils::sqr(ks_err));
650 }

690 /** Get the optimal alpha and the corresponding computation time for a fixed
691  * @p mesh and @p cao.
692  *
693  * The @p _r_cut_iL is determined via a simple bisection.
694  *
695  * @param[in] mesh @copybrief P3MParameters::mesh
696  * @param[in] cao @copybrief P3MParameters::cao
697  * @param[in] r_cut_iL_min lower bound for @p _r_cut_iL
698  * @param[in] r_cut_iL_max upper bound for @p _r_cut_iL
699  * @param[out] _r_cut_iL @copybrief P3MParameters::r_cut_iL
700  * @param[out] _alpha_L @copybrief P3MParameters::alpha_L
701  * @param[out] _accuracy @copybrief P3MParameters::accuracy
702  * @param[in] timings Number of test force calculations
703  * @param[in] verbose printf output
704  *
705  * @returns The integration time in case of success, otherwise
706  * -@ref P3M_TUNE_FAIL, -@ref P3M_TUNE_ACCURACY_TOO_LARGE,
707  * -@ref P3M_TUNE_CAO_TOO_LARGE, or -@ref P3M_TUNE_ELCTEST
708  */
709 static double p3m_mc_time(const int mesh[3], int cao, double r_cut_iL_min,
710  double r_cut_iL_max, double *_r_cut_iL,
711  double *_alpha_L, double *_accuracy, int timings,
712  bool verbose) {
713  double rs_err, ks_err;
714 
715  /* initial checks. */
716  auto const k_cut =
717  std::max(box_geo.length()[0] * cao / (2.0 * mesh[0]),
718  std::max(box_geo.length()[1] * cao / (2.0 * mesh[1]),
719  box_geo.length()[2] * cao / (2.0 * mesh[2])));
720 
721  auto const min_box_l = *boost::min_element(box_geo.length());
722  auto const min_local_box_l = *boost::min_element(local_geo.length());
723 
724  if (cao >= std::min(mesh[0], std::min(mesh[1], mesh[2])) ||
725  k_cut >= (std::min(min_box_l, min_local_box_l) - skin)) {
726  if (verbose) {
727  std::printf("%-4d %-3d cao too large for this mesh\n", mesh[0], cao);
728  }
729  return -P3M_TUNE_CAO_TOO_LARGE;
730  }
731 
732  /* Either low and high boundary are equal (for fixed cut), or the low border
733  is initially 0 and therefore
734  has infinite error estimate, as required. Therefore if the high boundary
735  fails, there is no possible r_cut */
736  if ((*_accuracy = p3m_get_accuracy(mesh, cao, r_cut_iL_max, _alpha_L, &rs_err,
737  &ks_err)) > p3m.params.accuracy) {
738  /* print result */
739  if (verbose) {
740  std::printf("%-4d %-3d %.5e %.5e %.5e %.3e %.3e accuracy not achieved\n",
741  mesh[0], cao, r_cut_iL_max, *_alpha_L, *_accuracy, rs_err,
742  ks_err);
743  }
744  return -P3M_TUNE_ACCURACY_TOO_LARGE;
745  }
746 
747  double r_cut_iL;
748  for (;;) {
749  r_cut_iL = 0.5 * (r_cut_iL_min + r_cut_iL_max);
750 
751  if (r_cut_iL_max - r_cut_iL_min < P3M_RCUT_PREC)
752  break;
753 
754  /* bisection */
755  if ((p3m_get_accuracy(mesh, cao, r_cut_iL, _alpha_L, &rs_err, &ks_err) >
756  p3m.params.accuracy))
757  r_cut_iL_min = r_cut_iL;
758  else
759  r_cut_iL_max = r_cut_iL;
760  }
761 
762  /* final result is always the upper interval boundary, since only there
763  we know that the desired minimal accuracy is obtained */
764  *_r_cut_iL = r_cut_iL = r_cut_iL_max;
765 
766  /* check whether we are running P3M+ELC, and whether we leave a reasonable
767  * gap
768  * space */
769  if (coulomb.method == COULOMB_ELC_P3M &&
770  elc_params.gap_size <= 1.1 * r_cut_iL * box_geo.length()[0]) {
771  /* print result */
772  if (verbose) {
773  std::printf("%-4d %-3d %.5e %.5e %.5e %.3e %.3e conflict with ELC\n",
774  mesh[0], cao, r_cut_iL, *_alpha_L, *_accuracy, rs_err,
775  ks_err);
776  }
777  return -P3M_TUNE_ELCTEST;
778  }
779 
780  auto const int_time = p3m_mcr_time(mesh, cao, r_cut_iL, *_alpha_L, timings);
781  if (int_time == -P3M_TUNE_FAIL) {
782  if (verbose) {
783  std::printf("tuning failed, test integration not possible\n");
784  }
785  return int_time;
786  }
787 
788  *_accuracy =
789  p3m_get_accuracy(mesh, cao, r_cut_iL, _alpha_L, &rs_err, &ks_err);
790 
791  /* print result */
792  if (verbose) {
793  std::printf("%-4d %-3d %.5e %.5e %.5e %.3e %.3e %-8.2f\n", mesh[0], cao,
794  r_cut_iL, *_alpha_L, *_accuracy, rs_err, ks_err, int_time);
795  }
796  return int_time;
797 }

799 /** Get the optimal alpha and the corresponding computation time for a fixed
800  * @p mesh.
801  *
802  * @p _cao should contain an initial guess, which is then adapted by stepping
803  * up and down.
804  *
805  * @param[in] mesh @copybrief P3MParameters::mesh
806  * @param[in] cao_min lower bound for @p _cao
807  * @param[in] cao_max upper bound for @p _cao
808  * @param[in,out] _cao initial guess for the
809  * @copybrief P3MParameters::cao
810  * @param[in] r_cut_iL_min lower bound for @p _r_cut_iL
811  * @param[in] r_cut_iL_max upper bound for @p _r_cut_iL
812  * @param[out] _r_cut_iL @copybrief P3MParameters::r_cut_iL
813  * @param[out] _alpha_L @copybrief P3MParameters::alpha_L
814  * @param[out] _accuracy @copybrief P3MParameters::accuracy
815  * @param[in] timings Number of test force calculations
816  * @param[in] verbose printf output
817  *
818  * @returns The integration time in case of success, otherwise
819  * -@ref P3M_TUNE_FAIL or -@ref P3M_TUNE_CAO_TOO_LARGE
820  */
821 static double p3m_m_time(const int mesh[3], int cao_min, int cao_max, int *_cao,
822  double r_cut_iL_min, double r_cut_iL_max,
823  double *_r_cut_iL, double *_alpha_L, double *_accuracy,
824  int timings, bool verbose) {
825  double best_time = -1, tmp_time, tmp_r_cut_iL = 0.0, tmp_alpha_L = 0.0,
826  tmp_accuracy = 0.0;
827  /* in which direction improvement is possible. Initially, we don't know it
828  * yet.
829  */
830  int final_dir = 0;
831  int cao = *_cao;
832 
833  /* the initial step sets a timing mark. If there is no valid r_cut, we can
834  only try
835  to increase cao to increase the obtainable precision of the far formula.
836  */
837  do {
838  tmp_time = p3m_mc_time(mesh, cao, r_cut_iL_min, r_cut_iL_max, &tmp_r_cut_iL,
839  &tmp_alpha_L, &tmp_accuracy, timings, verbose);
840  /* bail out if the force evaluation is not working */
841  if (tmp_time == -P3M_TUNE_FAIL)
842  return tmp_time;
843  /* cao is too large for this grid, but still the accuracy cannot be
844  * achieved, give up */
845  if (tmp_time == -P3M_TUNE_CAO_TOO_LARGE) {
846  return tmp_time;
847  }
848  /* we have a valid time, start optimising from there */
849  if (tmp_time >= 0) {
850  best_time = tmp_time;
851  *_r_cut_iL = tmp_r_cut_iL;
852  *_alpha_L = tmp_alpha_L;
853  *_accuracy = tmp_accuracy;
854  *_cao = cao;
855  break;
856  }
857  /* the required accuracy could not be obtained, try higher caos */
858  cao++;
859  final_dir = 1;
860  } while (cao <= cao_max);
861  /* with this mesh, the required accuracy cannot be obtained. */
862  if (cao > cao_max)
863  return -P3M_TUNE_CAO_TOO_LARGE;
864 
865  /* at the boundaries, only the opposite direction can be used for
866  * optimisation
867  */
868  if (cao == cao_min)
869  final_dir = 1;
870  else if (cao == cao_max)
871  final_dir = -1;
872 
873  if (final_dir == 0) {
874  /* check in which direction we can optimise. Both directions are possible
875  */
876  double dir_times[3];
877  for (final_dir = -1; final_dir <= 1; final_dir += 2) {
878  dir_times[final_dir + 1] = tmp_time = p3m_mc_time(
879  mesh, cao + final_dir, r_cut_iL_min, r_cut_iL_max, &tmp_r_cut_iL,
880  &tmp_alpha_L, &tmp_accuracy, timings, verbose);
881  /* bail out on errors, as usual */
882  if (tmp_time == -P3M_TUNE_FAIL)
883  return tmp_time;
884  /* in this direction, we cannot optimise, since we get into precision
885  * trouble */
886  if (tmp_time < 0)
887  continue;
888 
889  if (tmp_time < best_time) {
890  best_time = tmp_time;
891  *_r_cut_iL = tmp_r_cut_iL;
892  *_alpha_L = tmp_alpha_L;
893  *_accuracy = tmp_accuracy;
894  *_cao = cao + final_dir;
895  }
896  }
897  /* choose the direction which was optimal, if any of the two */
898  if (dir_times[0] == best_time) {
899  final_dir = -1;
900  } else if (dir_times[2] == best_time) {
901  final_dir = 1;
902  } else {
903  /* no improvement in either direction, however if one is only marginally
904  * worse, we can still try */
905  /* down is possible and not much worse, while up is either illegal or
906  * even
907  * worse */
908  if ((dir_times[0] >= 0 && dir_times[0] < best_time + P3M_TIME_GRAN) &&
909  (dir_times[2] < 0 || dir_times[2] > dir_times[0]))
910  final_dir = -1;
911  /* same for up */
912  else if ((dir_times[2] >= 0 &&
913  dir_times[2] < best_time + P3M_TIME_GRAN) &&
914  (dir_times[0] < 0 || dir_times[0] > dir_times[2]))
915  final_dir = 1;
916  else {
917  /* really no chance for optimisation */
918  return best_time;
919  }
920  }
921  /* we already checked the initial cao and its neighbor */
922  cao += 2 * final_dir;
923  } else {
924  /* here some constraint is active, and we only checked the initial cao
925  * itself */
926  cao += final_dir;
927  }
928 
929  /* move cao into the optimisation direction until we do not gain anymore. */
930  for (; cao >= cao_min && cao <= cao_max; cao += final_dir) {
931  tmp_time = p3m_mc_time(mesh, cao, r_cut_iL_min, r_cut_iL_max, &tmp_r_cut_iL,
932  &tmp_alpha_L, &tmp_accuracy, timings, verbose);
933  /* bail out on errors, as usual */
934  if (tmp_time == -P3M_TUNE_FAIL)
935  return tmp_time;
936  /* if we cannot meet the precision anymore, give up */
937  if (tmp_time < 0)
938  break;
939 
940  if (tmp_time < best_time) {
941  best_time = tmp_time;
942  *_r_cut_iL = tmp_r_cut_iL;
943  *_alpha_L = tmp_alpha_L;
944  *_accuracy = tmp_accuracy;
945  *_cao = cao;
946  }
947  /* no hope of further optimisation */
948  else if (tmp_time > best_time + P3M_TIME_GRAN)
949  break;
950  }
951  return best_time;
952 }

954 int p3m_adaptive_tune(int timings, bool verbose) {
955  double r_cut_iL_min, r_cut_iL_max, r_cut_iL = -1, tmp_r_cut_iL = 0.0;
956  int cao_min, cao_max, cao = -1, tmp_cao;

962 
963  if (p3m.params.epsilon != P3M_EPSILON_METALLIC) {

965  (box_geo.length()[1] == box_geo.length()[2]))) {

979  runtimeErrorMsg() << "no charged particles in the system";

982 
983  /* Print Status */
984  if (verbose) {
985  std::printf("P3M tune parameters: Accuracy goal = %.5e prefactor = %.5e\n"
986  "System: box_l = %.5e # charged part = %d Sum[q_i^2] = %.5e\n",

988  p3m.sum_qpart, p3m.sum_q2);
989  }
990 
991  /* Activate tuning mode */
992  p3m.params.tuning = true;
993 
994  /* parameter ranges */
995  /* if at least the number of meshpoints in one direction is not set, we have
996  * to tune it. */
997  if (p3m.params.mesh[0] == 0 || p3m.params.mesh[1] == 0 ||
998  p3m.params.mesh[2] == 0) {
999  /* Medium-educated guess for the minimal mesh */
1000  mesh_density_min = pow(p3m.sum_qpart / box_geo.volume(), 1.0 / 3.0);
1001  mesh_density_max = 512 / pow(box_geo.volume(), 1.0 / 3.0);
1002  tune_mesh = true;
1003  /* this limits the tried meshes if the accuracy cannot
1004  be obtained with smaller meshes, but normally not all these
1005  meshes have to be tested */
1006  /* avoid using more than 1 GB of FFT arrays (per default, see config.hpp)
1007  */
1008  } else if (p3m.params.mesh[1] == -1 && p3m.params.mesh[2] == -1) {
1009  double mesh_density = mesh_density_min = mesh_density_max =
1010  p3m.params.mesh[0] * box_geo.length_inv()[0];
1011  p3m.params.mesh[1] =
1012  static_cast<int>(std::round(mesh_density * box_geo.length()[1]));
1013  p3m.params.mesh[2] =
1014  static_cast<int>(std::round(mesh_density * box_geo.length()[2]));
1015  if (p3m.params.mesh[1] % 2 == 1)
1016  p3m.params.mesh[1]++; // Make sure that the mesh is even in all directions
1017  if (p3m.params.mesh[2] % 2 == 1)
1018  p3m.params.mesh[2]++;
1019 
1020  if (verbose) {
1021  std::printf("fixed mesh %d %d %d\n", p3m.params.mesh[0],
1022  p3m.params.mesh[1], p3m.params.mesh[2]);
1023  }
1024  } else {
1025  mesh_density_min = mesh_density_max =
1026  p3m.params.mesh[0] * box_geo.length_inv()[0];
1027 
1028  if (verbose) {
1029  std::printf("fixed mesh %d %d %d\n", p3m.params.mesh[0],
1030  p3m.params.mesh[1], p3m.params.mesh[2]);
1031  }
1032  }
1033 
1034  if (p3m.params.r_cut_iL == 0.0) {
1035  auto const min_box_l = *boost::min_element(box_geo.length());
1036  auto const min_local_box_l = *boost::min_element(local_geo.length());
1037 
1038  r_cut_iL_min = 0;
1039  r_cut_iL_max = std::min(min_local_box_l, min_box_l / 2.0) - skin;
1040  r_cut_iL_min *= box_geo.length_inv()[0];
1041  r_cut_iL_max *= box_geo.length_inv()[0];
1042  } else {
1043  r_cut_iL_min = r_cut_iL_max = p3m.params.r_cut_iL;
1044 
1045  if (verbose) {
1046  std::printf("fixed r_cut_iL %f\n", p3m.params.r_cut_iL);
1047  }
1048  }
1049 
1050  if (p3m.params.cao == 0) {
1051  cao_min = 1;
1052  cao_max = 7;
1053  cao = cao_max;
1054  } else {
1055  cao_min = cao_max = cao = p3m.params.cao;
1056 
1057  if (verbose) {
1058  std::printf("fixed cao %d\n", p3m.params.cao);

1186  int n_c_part, double sum_q2, double alpha_L) {
1187  double const mesh_i[3] = {1.0 / mesh[0], 1.0 / mesh[1], 1.0 / mesh[2]};
1188  auto const alpha_L_i = 1. / alpha_L;
1189  double he_q = 0.0;
1190 
1191  for (int nx = -mesh[0] / 2; nx < mesh[0] / 2; nx++) {
1192  auto const ctan_x = p3m_analytic_cotangent_sum(nx, mesh_i[0], cao);
1193  for (int ny = -mesh[1] / 2; ny < mesh[1] / 2; ny++) {
1194  auto const ctan_y =
1195  ctan_x * p3m_analytic_cotangent_sum(ny, mesh_i[1], cao);
1196  for (int nz = -mesh[2] / 2; nz < mesh[2] / 2; nz++) {
1197  if ((nx != 0) || (ny != 0) || (nz != 0)) {
1198  auto const n2 = Utils::sqr(nx) + Utils::sqr(ny) + Utils::sqr(nz);
1199  auto const cs =
1200  p3m_analytic_cotangent_sum(nz, mesh_i[2], cao) * ctan_y;
1201  double alias1, alias2;
1202  p3m_tune_aliasing_sums(nx, ny, nz, mesh, mesh_i, cao, alpha_L_i,
1203  &alias1, &alias2);
1204 
1205  auto const d = alias1 - Utils::sqr(alias2 / cs) / n2;
1206  /* at high precision, d can become negative due to extinction;
1207  also, don't take values that have no significant digits left*/
1208  if (d > 0 && (fabs(d / alias1) > ROUND_ERROR_PREC))
1209  he_q += d;
1210  }
1211  }
1212  }
1213  }
1214  return 2.0 * prefac * sum_q2 * sqrt(he_q / (double)n_c_part) /
1215  (box_geo.length()[1] * box_geo.length()[2]);
1216 }
