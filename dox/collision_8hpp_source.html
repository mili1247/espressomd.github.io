27 /** \name bits of possible modes for collision handling.
28  * The modes can be combined by or-ing together, but not all combinations are
29  * possible.
30  */
31 /**@{*/
32 #define COLLISION_MODE_OFF 0
33 /// Just create bond between centers of colliding particles
34 #define COLLISION_MODE_BOND 2
35 /** Create a bond between the centers of the colliding particles,
36  * plus two virtual sites at the point of collision and bind them
37  * together. This prevents the particles from sliding against each
38  * other. Requires VIRTUAL_SITES_RELATIVE and @ref COLLISION_MODE_BOND
39  */
40 #define COLLISION_MODE_VS 4
41 /** Glue a particle to a specific spot on the surface of an other */
42 #define COLLISION_MODE_GLUE_TO_SURF 8

45 /**@}*/

114 /** @brief Check additional criteria for the glue_to_surface collision mode */
115 inline bool glue_to_surface_criterion(Particle const &p1, Particle const &p2) {
116  return (((p1.p.type == collision_params.part_type_to_be_glued) &&
117  (p2.p.type == collision_params.part_type_to_attach_vs_to)) ||
118  ((p2.p.type == collision_params.part_type_to_be_glued) &&
119  (p1.p.type == collision_params.part_type_to_attach_vs_to)));
120 }

122 /** @brief Detect (and queue) a collision between the given particles. */
123 inline void detect_collision(Particle const &p1, Particle const &p2,
124  const double &dist_betw_part2) {
125  if (dist_betw_part2 > collision_params.distance2)
126  return;
127 
128  // If we are in the glue to surface mode, check that the particles
129  // are of the right type
130  if (collision_params.mode & COLLISION_MODE_GLUE_TO_SURF)
131  if (!glue_to_surface_criterion(p1, p2))
132  return;
133 
134  // Ignore virtual particles
135  if ((p1.p.is_virtual) || (p2.p.is_virtual))
136  return;
137 
138  // Check, if there's already a bond between the particles
139  if (pair_bond_exists_on(p1.bonds(), p2.identity(),
140  collision_params.bond_centers))
141  return;
142 
143  if (pair_bond_exists_on(p2.bonds(), p1.identity(),
144  collision_params.bond_centers))
145  return;
146 
147  /* If we're still here, there is no previous bond between the particles,
148  we have a new collision */
149 
150  // do not create bond between ghost particles
151  if (p1.l.ghost && p2.l.ghost) {
152  return;
153  }
154  queue_collision(p1.p.identity, p2.p.identity);
155 }
