50 // Center of mass of an aggregate
51 Utils::Vector3d
52 Cluster::center_of_mass_subcluster(std::vector<int> &subcl_partcicle_ids) {
53  Utils::Vector3d com{};
54 
55  // The distances between the particles "folded", such that all distances
56  // are smaller than box_l/2 in a periodic system. The 1st particle
57  // of the cluster is arbitrarily chosen as reference.
58 
59  auto const reference_position =
60  folded_position(get_particle_data(particles[0]).r.p, box_geo);
61  double total_mass = 0.;
62  for (int pid :
63  subcl_partcicle_ids) // iterate over all particle ids within a cluster
64  {
65  auto const folded_pos =
66  folded_position(get_particle_data(pid).r.p, box_geo);
67  auto const dist_to_reference = box_geo.get_mi_vector(
68  folded_pos, reference_position); // add current particle positions
69  com += dist_to_reference * get_particle_data(pid).p.mass;
70  total_mass += get_particle_data(pid).p.mass;
71  }
72 
73  // Normalize by number of particles
74  com /= total_mass;
75 
76  // Re-add reference position
77  com += reference_position;
78 
79  // Fold into simulation box
80 
81  for (int i = 0; i < 3; i++) {
82  com[i] = fmod(com[i], box_geo.length()[i]);
83  }
84  return com;
85 }

87 double Cluster::longest_distance() {
88  double ld = 0.;
89  for (auto a = particles.begin(); a != particles.end(); a++) {
90  for (auto b = a; ++b != particles.end();) {
91  auto const dist = box_geo
92  .get_mi_vector(get_particle_data(*a).r.p,
93  get_particle_data(*b).r.p)
94  .norm();
95 
96  // Larger than previous largest distance?
97  ld = std::max(ld, dist);
98  }
99  }
100  return ld;
101 }

108 double
109 Cluster::radius_of_gyration_subcluster(std::vector<int> &subcl_particle_ids) {
110  // Center of mass
111  Utils::Vector3d com = center_of_mass_subcluster(subcl_particle_ids);
112  double sum_sq_dist = 0.;
113  for (auto const pid : subcl_particle_ids) {
114  // calculate square length of this distance
115  sum_sq_dist +=
116  box_geo.get_mi_vector(com, get_particle_data(pid).r.p).norm2();
117  }
118 
119  return sqrt(sum_sq_dist / static_cast<double>(subcl_particle_ids.size()));
120 }

124 
125  // Unsorted for unsorted vector (0..n-1)

135 std::pair<double, double> Cluster::fractal_dimension(double dr) {
136 #ifdef GSL
137  Utils::Vector3d com = center_of_mass();
138  // calculate Df using linear regression on the logarithms of the radii of
139  // gyration against the number of particles in sub-clusters. Particles are
140  // included step by step from the center of mass outwards
141 
142  // Distances of particles from the center of mass
143  std::vector<double> distances;
144 
145  for (auto const &it : particles) {
146  distances.push_back(box_geo.get_mi_vector(com, get_particle_data(it).r.p)
147  .norm()); // add distance from the current particle
148  // to the com in the distances vectors
149  }
150 
151  // Get particle indices in the cluster which yield distances sorted in
152  // ascending order from center of mass.
153  auto particle_indices = sort_indices(distances);
154 
155  // Particle ids in the current sub-cluster
156  std::vector<int> subcluster_ids;
157 
158  std::vector<double> log_pcounts; // particle count
159  std::vector<double> log_diameters; // corresponding radii of gyration
160  double last_dist = 0;
161  for (auto const idx : particle_indices) {
162  subcluster_ids.push_back(particles[idx]);
163  if (distances[idx] < last_dist + dr)
164  continue;
165 
166  last_dist = distances[idx];
167  if (subcluster_ids.size() == 1)
168  continue;
169  double current_rg = radius_of_gyration_subcluster(subcluster_ids);
170  log_pcounts.push_back(log(subcluster_ids.size()));
171  log_diameters.push_back(log(current_rg * 2.0));
172  }
173  // usage: Function: int gsl_fit_linear (const double * x, const size_t
174  // xstride, const double * y, const std::size_t ystride, std::size_t n, double
175  // * c0, double * c1, double * cov00, double * cov01, double * cov11, double *
176  // sumsq)
177  const int n = log_pcounts.size();
178  double c0, c1, cov00, cov01, cov11, sumsq;
179  gsl_fit_linear(&log_diameters.front(), 1, &log_pcounts.front(), 1, n, &c0,
180  &c1, &cov00, &cov01, &cov11, &sumsq);
181  double mean_sq_residual = sumsq / static_cast<double>(log_diameters.size());
182  return {c1, mean_sq_residual};
183 #else
184  runtimeErrorMsg() << "GSL (gnu scientific library) is required for fractal "
185  "dimension calculation.";
186  return {0, 0};
187 #endif
188 }
