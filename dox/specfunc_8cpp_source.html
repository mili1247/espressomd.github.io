214 double hzeta(double s, double q) {
215  constexpr auto max_bits = 54.0;
216  constexpr auto jmax = 12;
217  constexpr auto kmax = 10;
218 
219  if ((s > max_bits && q < 1.0) || (s > 0.5 * max_bits && q < 0.25))
220  return pow(q, -s);
221  if (s > 0.5 * max_bits && q < 1.0) {
222  double p1 = pow(q, -s);
223  double p2 = pow(q / (1.0 + q), s);
224  double p3 = pow(q / (2.0 + q), s);
225  return p1 * (1.0 + p2 + p3);
226  }
227  /** Euler-Maclaurin summation formula from @cite moshier89a p. 400, with
228  * several typo corrections.
229  */
230  auto const kmax_q = static_cast<double>(kmax) + q;
231  auto const pmax = pow(kmax_q, -s);
232  auto scp = s;
233  auto pcp = pmax / kmax_q;
234  auto ans = pmax * (kmax_q / (s - 1.0) + 0.5);
235 
236  for (int k = 0; k < kmax; k++)
237  ans += pow(static_cast<double>(k) + q, -s);
238 
239  for (int j = 0; j <= jmax; j++) {
240  auto const delta = hzeta_c[j + 1] * scp * pcp;
241  ans += delta;
242  scp *= (s + 2 * j + 1) * (s + 2 * j + 2);
243  pcp /= Utils::sqr(static_cast<double>(kmax) + q);
244  }
245 
246  return ans;
247 }

249 double K0(double x) {
250  double c, I0;
251  if (x <= 2.0) {
252  c = evaluateAsChebychevSeriesAt(bk0_cs, 0.5 * x * x - 1.0);
253  I0 = evaluateAsChebychevSeriesAt(bi0_cs, x * x / 4.5 - 1.0);
254  return (-log(x) + Utils::ln_2()) * I0 + c;
255  }
256  c = (x <= 8.0) ? evaluateAsChebychevSeriesAt(ak0_cs, (16.0 / x - 5.0) / 3.0)
257  : evaluateAsChebychevSeriesAt(ak02_cs, 16.0 / x - 1.0);
258  return exp(-x) * c / sqrt(x);
259 }

261 double K1(double x) {
262  double c, I1;
263  if (x <= 2.0) {
264  c = evaluateAsChebychevSeriesAt(bk1_cs, 0.5 * x * x - 1.0);
265  I1 = x * evaluateAsChebychevSeriesAt(bi1_cs, x * x / 4.5 - 1.0);
266  return (log(x) - Utils::ln_2()) * I1 + c / x;
267  }
268  c = (x <= 8.0) ? evaluateAsChebychevSeriesAt(ak1_cs, (16.0 / x - 5.0) / 3.0)
269  : evaluateAsChebychevSeriesAt(ak12_cs, 16.0 / x - 1.0);
270  return exp(-x) * c / sqrt(x);
271 }

277 /** necessary orders for K0/1 from 2 up to 22 for 10^-14 precision. Note that

287 double LPK0(double x) {
288  if (x >= 27.) {
289  auto const tmp = .5 * exp(-x) / sqrt(x);
290  return tmp * ak0_cs[0];
291  }
292  if (x >= 23.) {
293  auto const tmp = exp(-x) / sqrt(x), xx = (16. / 3.) / x - 5. / 3.;
294  return tmp * (xx * ak0_cs[1] + 0.5 * ak0_cs[0]);
295  }
296  if (x > 2) {
297  int j = ak01_orders[static_cast<int>(x) - 2];
298  double x2;
299  double *s0;
300  if (x <= 8) {
301  s0 = ak0_cs;
302  x2 = (2. * 16. / 3.) / x - 2. * 5. / 3.;
303  } else {
304  s0 = ak02_cs;
305  x2 = (2. * 16.) / x - 2.;
306  }
307  auto dd0 = s0[j];
308  auto d0 = x2 * dd0 + s0[j - 1];
309  for (j -= 2; j >= 1; j--) {
310  auto const tmp0 = d0;
311  d0 = x2 * d0 - dd0 + s0[j];
312  dd0 = tmp0;
313  }
314  auto const tmp = exp(-x) / sqrt(x);
315  return tmp * (0.5 * (s0[0] + x2 * d0) - dd0);
316  }
317  /* x <= 2 */
318  {
319  /* I0/1 series */
320  int j = 10;
321  auto x2 = (2. / 4.5) * x * x - 2.;
322  auto dd0 = bi0_cs[j];
323  auto d0 = x2 * dd0 + bi0_cs[j - 1];
324  for (j -= 2; j >= 1; j--) {
325  auto const tmp0 = d0;
326  d0 = x2 * d0 - dd0 + bi0_cs[j];
327  dd0 = tmp0;
328  }
329  auto const tmp = log(x) - Utils::ln_2();
330  auto const ret = -tmp * (0.5 * (bi0_cs[0] + x2 * d0) - dd0);
331 
332  /* K0/K1 correction */
333  j = 9;
334  x2 = x * x - 2.;
335  dd0 = bk0_cs[j];
336  d0 = x2 * dd0 + bk0_cs[j - 1];
337  for (j -= 2; j >= 1; j--) {
338  auto const tmp0 = d0;
339  d0 = x2 * d0 - dd0 + bk0_cs[j];
340  dd0 = tmp0;
341  }
342  return ret + (0.5 * (x2 * d0 + bk0_cs[0]) - dd0);
343  }
344 }

405 std::pair<double, double> LPK01(double x) {
406  if (x >= 27.) {
407  auto const tmp = .5 * exp(-x) / sqrt(x);
408  auto const K0 = tmp * ak0_cs[0];
409  auto const K1 = tmp * ak1_cs[0];
410  return {K0, K1};
411  }
412  if (x >= 23.) {
413  auto const tmp = exp(-x) / sqrt(x), xx = (16. / 3.) / x - 5. / 3.;
414  auto const K0 = tmp * (xx * ak0_cs[1] + 0.5 * ak0_cs[0]);
415  auto const K1 = tmp * (xx * ak1_cs[1] + 0.5 * ak1_cs[0]);
416  return {K0, K1};
417  }
418  if (x > 2) {
419  int j = ak01_orders[((int)x) - 2];
420  double x2;
421  double *s0, *s1;
422  if (x <= 8) {
423  s0 = ak0_cs;
424  s1 = ak1_cs;
425  x2 = (2. * 16. / 3.) / x - 2. * 5. / 3.;
426  } else {
427  s0 = ak02_cs;
428  s1 = ak12_cs;
429  x2 = (2. * 16.) / x - 2.;
430  }
431  auto dd0 = s0[j];
432  auto dd1 = s1[j];
433  auto d0 = x2 * dd0 + s0[j - 1];
434  auto d1 = x2 * dd1 + s1[j - 1];
435  for (j -= 2; j >= 1; j--) {
436  auto const tmp0 = d0, tmp1 = d1;
437  d0 = x2 * d0 - dd0 + s0[j];
438  d1 = x2 * d1 - dd1 + s1[j];
439  dd0 = tmp0;
440  dd1 = tmp1;
441  }
442  auto const tmp = exp(-x) / sqrt(x);
443  auto const K0 = tmp * (0.5 * (s0[0] + x2 * d0) - dd0);
444  auto const K1 = tmp * (0.5 * (s1[0] + x2 * d1) - dd1);
445  return {K0, K1};
446  }
447  /* x <= 2 */
448  {
449  /* I0/1 series */
450  int j = 10;
451  auto x2 = (2. / 4.5) * x * x - 2.;
452  auto dd0 = bi0_cs[j];
453  auto dd1 = bi1_cs[j];
454  auto d0 = x2 * dd0 + bi0_cs[j - 1];
455  auto d1 = x2 * dd1 + bi1_cs[j - 1];
456  for (j -= 2; j >= 1; j--) {
457  auto const tmp0 = d0, tmp1 = d1;
458  d0 = x2 * d0 - dd0 + bi0_cs[j];
459  d1 = x2 * d1 - dd1 + bi1_cs[j];
460  dd0 = tmp0;
461  dd1 = tmp1;
462  }
463  auto const tmp = log(x) - Utils::ln_2();
464  auto K0 = -tmp * (0.5 * (bi0_cs[0] + x2 * d0) - dd0);
465  auto K1 = x * tmp * (0.5 * (bi1_cs[0] + x2 * d1) - dd1);
466 
467  /* K0/K1 correction */
468  j = 9;
469  x2 = x * x - 2.;
470  dd0 = bk0_cs[j];
471  dd1 = bk1_cs[j];
472  d0 = x2 * dd0 + bk0_cs[j - 1];
473  d1 = x2 * dd1 + bk1_cs[j - 1];
474  for (j -= 2; j >= 1; j--) {
475  auto const tmp0 = d0, tmp1 = d1;
476  d0 = x2 * d0 - dd0 + bk0_cs[j];
477  d1 = x2 * d1 - dd1 + bk1_cs[j];
478  dd0 = tmp0;
479  dd1 = tmp1;
480  }
481  K0 += (0.5 * (x2 * d0 + bk0_cs[0]) - dd0);
482  K1 += (0.5 * (x2 * d1 + bk1_cs[0]) - dd1) / x;
483  return {K0, K1};
484  }
