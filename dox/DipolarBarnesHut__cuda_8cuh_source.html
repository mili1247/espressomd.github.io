28 struct BHData {
29  /// CUDA blocks
30  int blocks;
31  /// each node corresponds to a split of the cubic box in 3D space to equal
32  /// cubic boxes hence, 8 octant nodes per particle is a theoretical octree
33  /// limit: a maximal number of octree nodes is "nnodesd" and a number of
34  /// particles "nbodiesd" respectively.
35  int nbodies;
36  int nnodes;
37  /// particle positions on the device:
38  float *r;
39  /// particle dipole moments on the device:
40  float *u;
41  /// Not a real mass. Just a node weight coefficient.
42  float *mass;
43  /// min positions' coordinates of the Barnes-Hut box.
44  float *minp;
45  /// max positions' coordinates of the Barnes-Hut box.
46  float *maxp;
