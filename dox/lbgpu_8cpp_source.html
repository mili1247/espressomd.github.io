106 /** (Re-)initialize the fluid.
107  * See @cite dunweg07a and @cite dhumieres09a.
108  */
109 void lb_reinit_parameters_gpu() {
110  lbpar_gpu.mu = 0.0;
111 
112  if (lbpar_gpu.viscosity > 0.0 && lbpar_gpu.agrid > 0.0 &&
113  lbpar_gpu.tau > 0.0) {
114  /* Eq. (80) @cite dunweg07a. */
115  lbpar_gpu.gamma_shear = 1.f - 2.f / (6.f * lbpar_gpu.viscosity + 1.f);
116  }
117 
118  if (lbpar_gpu.bulk_viscosity > 0.0) {
119  /* Eq. (81) @cite dunweg07a. */
120  lbpar_gpu.gamma_bulk = 1.f - 2.f / (9.f * lbpar_gpu.bulk_viscosity + 1.f);
121  }
122 
123  // By default, gamma_even and gamma_odd are chosen such that the MRT becomes
124  // a TRT with ghost mode relaxation factors that minimize unphysical wall
125  // slip at bounce-back boundaries. For the relation between the gammas
126  // achieving this, consult @cite dhumieres09a.
127  // Note that the relaxation operator in ESPResSo is defined as
128  // m* = m_eq + gamma * (m - m_eq)
129  // as opposed to this reference, where
130  // m* = m + lambda * (m - m_eq)
131 
132  if (lbpar_gpu.is_TRT) {
133  lbpar_gpu.gamma_bulk = lbpar_gpu.gamma_shear;
134  lbpar_gpu.gamma_even = lbpar_gpu.gamma_shear;
135  lbpar_gpu.gamma_odd =
136  -(7.0f * lbpar_gpu.gamma_even + 1.0f) / (lbpar_gpu.gamma_even + 7.0f);
137  }
138 
139  if (lbpar_gpu.kT > 0.0) { /* fluctuating hydrodynamics ? */
140 
141  /* Eq. (51) @cite dunweg07a.*/
142  /* Note that the modes are not normalized as in the paper here! */
143  lbpar_gpu.mu = lbpar_gpu.kT * Utils::sqr(lbpar_gpu.tau) /
144  D3Q19::c_sound_sq<float> / Utils::sqr(lbpar_gpu.agrid);
145  }
146 
147  lb_set_agrid_gpu(lbpar_gpu.agrid);
148 
149 #ifdef ELECTROKINETICS
150  if (ek_initialized) {
151  lbpar_gpu.tau = static_cast<float>(get_time_step());
152  }
153 #endif
154 
155  reinit_parameters_GPU(&lbpar_gpu);
156 }

188 void lb_set_agrid_gpu(double agrid) {
189  lbpar_gpu.agrid = static_cast<float>(agrid);
190 
191  lbpar_gpu.dim[0] =
192  static_cast<unsigned int>(round(box_geo.length()[0] / agrid));
193  lbpar_gpu.dim[1] =
194  static_cast<unsigned int>(round(box_geo.length()[1] / agrid));
195  lbpar_gpu.dim[2] =
196  static_cast<unsigned int>(round(box_geo.length()[2] / agrid));
197 
198  Utils::Vector<float, 3> box_from_dim(
199  Utils::Vector<unsigned int, 3>(lbpar_gpu.dim) * agrid);
200  Utils::Vector<float, 3> box_lf(box_geo.length());
201 
202  auto const rel_difference_vec =
203  Utils::hadamard_division(box_lf - box_from_dim, box_lf);
204  auto const commensurable = std::all_of(
205  rel_difference_vec.begin(), rel_difference_vec.end(), [](auto d) {
206  return std::abs(d) < std::numeric_limits<float>::epsilon();
207  });
208  if (not commensurable) {
209  runtimeErrorMsg() << "Lattice spacing agrid= " << agrid
210  << " is incompatible with one of the box dimensions: "
211  << box_geo.length()[0] << " " << box_geo.length()[1]
212  << " " << box_geo.length()[2];
213  }
214  lbpar_gpu.number_of_nodes =
215  std::accumulate(lbpar_gpu.dim.begin(), lbpar_gpu.dim.end(), 1u,
216  std::multiplies<unsigned int>());
217 }
