35 /** Parameters of the isotropic NpT-integration scheme. */
36 struct NptIsoParameters {
37  /** mass of a virtual piston representing the shaken box */
38  double piston;
39  /** inverse of \ref piston */
40  double inv_piston;
41  /** isotropic volume. Note that we use the term volume throughout,
42  * although for a 2d or 1d system we mean Area and Length respectively
43  */
44  double volume;
45  /** desired pressure to which the algorithm strives to */
46  double p_ext;
47  /** instantaneous pressure the system currently has */
48  double p_inst;
49  /** difference between \ref p_ext and \ref p_inst */
50  double p_diff;
51  /** virial (short-range) components of \ref p_inst */
52  Utils::Vector3d p_vir;
53  /** ideal gas components of \ref p_inst, derived from the velocities */
54  Utils::Vector3d p_vel;
55  /** geometry information for the NpT integrator. Holds the vector
56  * < dir, dir, dir > where a positive value for dir indicates that
57  * box movement is allowed in that direction. To check whether a
58  * given direction is turned on, use bitwise comparison with \ref
59  * nptgeom_dir
60  */
61  int geometry;
62  /** bitwise comparison values corresponding to different directions */
63  Utils::Vector3i nptgeom_dir;
64  /** The number of dimensions in which NpT boxlength motion is coupled to
65  * particles */
66  int dimension;
67  /** Set this flag if you want all box dimensions to be identical. Needed for
68  * electrostatics and magnetostatics. If the value of \ref dimension is
69  * less than 3, then box length motion in one or more directions will
70  * be decoupled from the particle motion
71  */
72  bool cubic_box;
73  /** An index to one of the non-constant dimensions. Handy if you just want
74  * the variable box_l
75  */
76  int non_const_dim;
77 };
