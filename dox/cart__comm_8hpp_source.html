105 /**
106  * @brief Calculates the numbers of the nearest neighbors for a node.
107  *
108  * @tparam dim Dimension of the communicator
109  * @param comm Cartesian communicator
110  *
111  * @return Ranks of 2*dim neighbors
112  */
113 template <std::size_t dim>
114 Utils::Vector<int, 2 * dim>
115 cart_neighbors(const boost::mpi::communicator &comm) {
116  using std::get;
117 
118  Vector<int, 2 * dim> ret;
119 
120  for (std::size_t i = 0; i < dim; i++) {
121  ret[2 * i + 0] = get<1>(cart_shift(comm, i, -1));
122  ret[2 * i + 1] = get<1>(cart_shift(comm, i, +1));
123  }
124 
125  return ret;
126 }
