42 /** Calculate the aliasing sums for the optimal influence function.
43  *
44  * Calculates the aliasing sums in the numerator and denominator of
45  * the expression for the optimal influence function (see
46  * @cite hockney88a : 8-22, p. 275).
47  *
48  * \tparam S order (2 for energy, 3 for forces)
49  * \param params DP3M parameters
50  * \param shift shift for a given n-vector
51  * \param d_op differential operator for a given n-vector
52  * \return The result of the fraction.
53  */
54 template <std::size_t S>
55 double G_opt_dipolar(P3MParameters const &params, Utils::Vector3i const &shift,
56  Utils::Vector3i const &d_op) {
57  using Utils::int_pow;
58  using Utils::sinc;
59  constexpr double limit = 30;
60 
61  double numerator = 0.0;
62  double denominator = 0.0;
63 
64  auto const f1 = 1.0 / static_cast<double>(params.mesh[0]);
65  auto const f2 = Utils::sqr(Utils::pi() / params.alpha_L);
66 
67  for (double mx = -P3M_BRILLOUIN; mx <= P3M_BRILLOUIN; mx++) {
68  auto const nmx = shift[0] + params.mesh[0] * mx;
69  auto const sx = std::pow(sinc(f1 * nmx), 2.0 * params.cao);
70  for (double my = -P3M_BRILLOUIN; my <= P3M_BRILLOUIN; my++) {
71  auto const nmy = shift[1] + params.mesh[0] * my;
72  auto const sy = sx * std::pow(sinc(f1 * nmy), 2.0 * params.cao);
73  for (double mz = -P3M_BRILLOUIN; mz <= P3M_BRILLOUIN; mz++) {
74  auto const nmz = shift[2] + params.mesh[0] * mz;
75  auto const sz = sy * std::pow(sinc(f1 * nmz), 2.0 * params.cao);
76  auto const nm2 = Utils::sqr(nmx) + Utils::sqr(nmy) + Utils::sqr(nmz);
77  auto const exponent = f2 * nm2;
78  if (exponent < limit) {
79  auto const f3 = sz * std::exp(-exponent) / nm2;
80  auto const n_nm = d_op[0] * nmx + d_op[1] * nmy + d_op[2] * nmz;
81  numerator += f3 * int_pow<S>(n_nm);
82  }
83  denominator += sz;
84  }
85  }
86  }
87  return numerator / (int_pow<S>(static_cast<double>(d_op.norm2())) *
88  Utils::sqr(denominator));
89 }

91 /**
92  * @brief Map influence function over a grid.
93  *
94  * This evaluates the optimal influence function @ref G_opt_dipolar
95  * over a regular grid of k vectors, and returns the values as a vector.
96  *
97  * @tparam S Order of the differential operator, e.g. 2 for energy, 3 for force
98  *
99  * @param params DP3M parameters
100  * @param n_start Lower left corner of the grid
101  * @param n_end Upper right corner of the grid.
102  * @param box_l Box size
103  * @return Values of the influence function at regular grid points.
104  */
105 template <std::size_t S>
106 std::vector<double> grid_influence_function(P3MParameters const &params,
107  Utils::Vector3i const &n_start,
108  Utils::Vector3i const &n_end,
109  Utils::Vector3d const &box_l) {
110 
111  auto const size = n_end - n_start;
112 
113  /* The influence function grid */
114  auto g =
115  std::vector<double>(boost::accumulate(size, 1, std::multiplies<>()), 0.);
116 
117  /* Skip influence function calculation in tuning mode,
118  the results need not be correct for timing. */
119  if (params.tuning) {
120  return g;
121  }
122 
123  double fak1 = Utils::int_pow<3>(static_cast<double>(params.mesh[0])) * 2.0 /
124  Utils::sqr(box_l[0]);
125 
126  auto const shifts =
127  detail::calc_meshift({params.mesh[0], params.mesh[1], params.mesh[2]});
128  auto const d_ops = detail::calc_meshift(
129  {params.mesh[0], params.mesh[1], params.mesh[2]}, true);
130 
131  Utils::Vector3i n{};
132  for (n[0] = n_start[0]; n[0] < n_end[0]; n[0]++) {
133  for (n[1] = n_start[1]; n[1] < n_end[1]; n[1]++) {
134  for (n[2] = n_start[2]; n[2] < n_end[2]; n[2]++) {
135  auto const ind = Utils::get_linear_index(n - n_start, size,
136  Utils::MemoryOrder::ROW_MAJOR);
137 
138  if (((n[0] % (params.mesh[0] / 2) == 0) &&
139  (n[1] % (params.mesh[0] / 2) == 0) &&
140  (n[2] % (params.mesh[0] / 2) == 0))) {
141  g[ind] = 0.0;
142  } else {
143  auto const shift = Utils::Vector3i{shifts[0][n[0]], shifts[0][n[1]],
144  shifts[0][n[2]]};
145  auto const d_op =
146  Utils::Vector3i{d_ops[0][n[0]], d_ops[0][n[1]], d_ops[0][n[2]]};
147  auto const fak2 = G_opt_dipolar<S>(params, shift, d_op);
148  g[ind] = fak1 * fak2;
149  }
150  }
151  }
152  }
153  return g;
154 }

156 double G_opt_dipolar_self_energy(P3MParameters const &params,
157  Utils::Vector3i const &shift) {
158  using Utils::sinc;
159  double u_sum = 0.0;
160  constexpr int limit = P3M_BRILLOUIN + 5;
161 
162  auto const f1 = 1.0 / static_cast<double>(params.mesh[0]);
163 
164  for (double mx = -limit; mx <= limit; mx++) {
165  auto const nmx = shift[0] + params.mesh[0] * mx;
166  auto const sx = std::pow(sinc(f1 * nmx), 2.0 * params.cao);
167  for (double my = -limit; my <= limit; my++) {
168  auto const nmy = shift[1] + params.mesh[0] * my;
169  auto const sy = sx * std::pow(sinc(f1 * nmy), 2.0 * params.cao);
170  for (double mz = -limit; mz <= limit; mz++) {
171  auto const nmz = shift[2] + params.mesh[0] * mz;
172  auto const sz = sy * std::pow(sinc(f1 * nmz), 2.0 * params.cao);
173  u_sum += sz;
174  }
175  }
176  }
177  return u_sum;
178 }

180 /**
181  * @brief Calculate self-energy of the influence function.
182  *
183  * @param params DP3M parameters
184  * @param n_start Lower left corner of the grid
185  * @param n_end Upper right corner of the grid.
186  * @param g Energies on the grid.
187  * @return Total self-energy.
188  */
189 double grid_influence_function_self_energy(P3MParameters const &params,
190  Utils::Vector3i const &n_start,
191  Utils::Vector3i const &n_end,
192  std::vector<double> const &g) {
193  auto const size = n_end - n_start;
194 
195  auto const shifts =
196  detail::calc_meshift({params.mesh[0], params.mesh[1], params.mesh[2]});
197  auto const d_ops = detail::calc_meshift(
198  {params.mesh[0], params.mesh[1], params.mesh[2]}, true);
199 
200  double energy = 0.0;
201  Utils::Vector3i n{};
202  for (n[0] = n_start[0]; n[0] < n_end[0]; n[0]++) {
203  for (n[1] = n_start[1]; n[1] < n_end[1]; n[1]++) {
204  for (n[2] = n_start[2]; n[2] < n_end[2]; n[2]++) {
205  if (((n[0] % (params.mesh[0] / 2) == 0) &&
206  (n[1] % (params.mesh[0] / 2) == 0) &&
207  (n[2] % (params.mesh[0] / 2) == 0))) {
208  energy += 0.0;
209  } else {
210  auto const ind = Utils::get_linear_index(
211  n - n_start, size, Utils::MemoryOrder::ROW_MAJOR);
212  auto const shift = Utils::Vector3i{shifts[0][n[0]], shifts[0][n[1]],
213  shifts[0][n[2]]};
214  auto const d_op =
215  Utils::Vector3i{d_ops[0][n[0]], d_ops[0][n[1]], d_ops[0][n[2]]};
216  auto const U2 = G_opt_dipolar_self_energy(params, shift);
217  energy += g[ind] * U2 * d_op.norm2();
218  }
219  }
220  }
221  }
222  return energy;
223 }
