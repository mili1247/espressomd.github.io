34 /** Type codes for the type of dipolar interaction.
35  * Enumeration of implemented methods for the magnetostatic interaction.
36  */
37 enum DipolarInteraction {
38  /** Dipolar interaction switched off. */
39  DIPOLAR_NONE = 0,
40  /** Dipolar method is P3M. */
41  DIPOLAR_P3M,
42  /** Dipolar method is P3M plus DLC. */
43  DIPOLAR_MDLC_P3M,
44  /** Dipolar method is all with all and no replicas. */
45  DIPOLAR_ALL_WITH_ALL_AND_NO_REPLICA,
46  /** Dipolar method is magnetic dipolar direct summation. */
47  DIPOLAR_DS,
48  /** Dipolar method is direct summation plus DLC. */
49  DIPOLAR_MDLC_DS,
50  /** Dipolar method is direct summation on GPU. */
51  DIPOLAR_DS_GPU,
52 #ifdef DIPOLAR_BARNES_HUT
53  /** Dipolar method is direct summation on GPU by Barnes-Hut algorithm. */
54  DIPOLAR_BH_GPU,
55 #endif
56  /** Dipolar method is ScaFaCoS. */
57  DIPOLAR_SCAFACOS
58 };
