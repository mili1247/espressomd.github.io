80 void on_program_start() {
81 #ifdef CUDA
82  if (this_node == 0) {
83  try {
84  cuda_init();
85  } catch (cuda_runtime_error const &err) {
86  // pass
87  }
88  }
89 #endif
90 
91  init_node_grid();
92 
93  /* initially go for domain decomposition */
94  cells_re_init(CELL_STRUCTURE_DOMDEC);
95 
96  if (this_node == 0) {

102 void on_integration_start(double time_step) {
103  /********************************************/
104  /* sanity checks */
105  /********************************************/
106 
107  integrator_sanity_checks();
108 #ifdef NPT
109  integrator_npt_sanity_checks();
110 #endif
111  interactions_sanity_checks();
112  lb_lbfluid_sanity_checks(time_step);
113 
114  /********************************************/
115  /* end sanity checks */
116  /********************************************/
117 
118  lb_lbfluid_on_integration_start();
119 
120 #ifdef CUDA
121  MPI_Bcast(gpu_get_global_particle_vars_pointer_host(),
122  sizeof(CUDA_global_part_vars), MPI_BYTE, 0, comm_cart);
123 #endif
124 
125  /* Prepare the thermostat */
126  if (reinit_thermo) {
127  thermo_init(time_step);
128  reinit_thermo = false;
129  recalc_forces = true;
130  }
131 
132 #ifdef NPT
133  npt_ensemble_init(box_geo);
134 #endif
135 
136  partCfg().invalidate();
137  invalidate_fetch_cache();
138 
139 #ifdef ADDITIONAL_CHECKS
140  if (!Utils::Mpi::all_compare(comm_cart, cell_structure.use_verlet_list)) {
141  runtimeErrorMsg() << "Nodes disagree about use of verlet lists.";
142  }
143 #ifndef OPENMPI_BUG_MPI_ALLOC_MEM
144 #ifdef ELECTROSTATICS
145  if (!Utils::Mpi::all_compare(comm_cart, coulomb.method))
146  runtimeErrorMsg() << "Nodes disagree about Coulomb long range method";
147 #endif
148 #ifdef DIPOLES
149  if (!Utils::Mpi::all_compare(comm_cart, dipole.method))
150  runtimeErrorMsg() << "Nodes disagree about dipolar long range method";
151 #endif
152 #endif
153 #endif /* ADDITIONAL_CHECKS */
154 
155  on_observable_calc();
156 }

158 void on_observable_calc() {
159  /* Prepare particle structure: Communication step: number of ghosts and ghost
160  * information */
161  cells_update_ghosts(global_ghost_flags());
162  update_dependent_particles();
163 #ifdef ELECTROSTATICS
164  if (reinit_electrostatics) {
165  Coulomb::on_observable_calc();
166  reinit_electrostatics = false;
167  }
168 #endif /* ELECTROSTATICS */
169 
170 #ifdef DIPOLES
171  if (reinit_magnetostatics) {
172  Dipole::on_observable_calc();
173  reinit_magnetostatics = false;
174  }
175 #endif /* DIPOLES */
176 
177 #ifdef ELECTROKINETICS
178  if (ek_initialized) {
179  ek_integrate_electrostatics();
180  }
181 #endif /* ELECTROKINETICS */
182 
183  clear_particle_node();
184 }

326 /**
327  * @brief Returns the ghost flags required for running pair
328  * kernels for the global state, e.g. the force calculation.
329  * @return Required data parts;
330  */
331 unsigned global_ghost_flags() {
332  /* Position and Properties are always requested. */
333  unsigned data_parts = Cells::DATA_PART_POSITION | Cells::DATA_PART_PROPERTIES;
334 
335  if (lattice_switch == ActiveLB::CPU)
336  data_parts |= Cells::DATA_PART_MOMENTUM;
337 
338  if (thermo_switch & THERMO_DPD)
339  data_parts |= Cells::DATA_PART_MOMENTUM;
340 
341  if (n_thermalized_bonds) {
342  data_parts |= Cells::DATA_PART_MOMENTUM;
343  data_parts |= Cells::DATA_PART_BONDS;
344  }
345 
346 #ifdef COLLISION_DETECTION
347  if (collision_params.mode) {
348  data_parts |= Cells::DATA_PART_BONDS;
349  }
350 #endif
351 
352  return data_parts;
353 }
