21 /** \file
22  * main header-file for MDLC (magnetic dipolar layer correction).
23  *
24  * Developer: Joan J. Cerda.
25  *
26  * Purpose: get the corrections for dipolar 3D algorithms
27  * when applied to a slab geometry and dipolar
28  * particles. DLC & co
29  *
30  * Article: @cite brodka04a
31  *
32  * We also include a tuning function that returns the
33  * cut-off necessary to attend a certain accuracy.
34  *
35  * Restrictions: the slab must be such that the z is the short
36  * direction. Otherwise we get trash.
37  *
38  * Limitations: at this moment it is restricted to work with 1 cpu
39  */

49 /** parameters for the MDLC method */
50 struct DLC_struct {
51  /** maximal pairwise error of the potential and force */
52  double maxPWerror;
53 
54  /** Cutoff of the exponential sum. Since in all other MMM methods this is
55  * the far formula, we call it here the same, although in the ELC context
56  * it does not make much sense.
57  */
58  double far_cut;
59 
60  /** Size of the empty gap. Note that MDLC relies on the user to make sure
61  * that this condition is fulfilled.
62  */
63  double gap_size;
64 
65  /** Flag whether #far_cut was set by the user, or calculated by ESPResSo.
66  * In the latter case, the cutoff will be adapted if important parameters,
67  * such as the box dimensions, change.
68  */
69  bool far_calculated;
70

72  double h;
