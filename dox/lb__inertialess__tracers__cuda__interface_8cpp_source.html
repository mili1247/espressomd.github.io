44 static void pack_particles(ParticleRange const &particles,
45  std::vector<IBM_CUDA_ParticleDataInput> &buffer) {
46 
47  int i = 0;
48  for (auto const &part : particles) {
49  auto const pos = folded_position(part.r.p, box_geo);
50 
51  buffer[i].pos[0] = static_cast<float>(pos[0]);
52  buffer[i].pos[1] = static_cast<float>(pos[1]);
53  buffer[i].pos[2] = static_cast<float>(pos[2]);
54 
55  buffer[i].f[0] = static_cast<float>(part.f.f[0]);
56  buffer[i].f[1] = static_cast<float>(part.f.f[1]);
57  buffer[i].f[2] = static_cast<float>(part.f.f[2]);
58 
59  buffer[i].is_virtual = part.p.is_virtual;
60 
61  i++;
62  }
63 }

65 /** Gather particle positions on the master node in order to communicate them
66  * to GPU. We transfer all particles (real and virtual), but actually we would
67  * only need the virtual ones. Room for improvement...
68  * Analogous to @ref cuda_mpi_get_particles.
69  */
70 void IBM_cuda_mpi_get_particles(ParticleRange const &particles) {
71  auto const n_part = particles.size();
72 
73  if (this_node > 0) {
74  static std::vector<IBM_CUDA_ParticleDataInput> buffer;
75  buffer.resize(n_part);
76  /* pack local parts into buffer */
77  pack_particles(particles, buffer);
78 
79  Utils::Mpi::gather_buffer(buffer, comm_cart);
80  } else {
81  /* Pack own particles */
82  pack_particles(particles, IBM_ParticleDataInput_host);
83 
84  Utils::Mpi::gather_buffer(IBM_ParticleDataInput_host, comm_cart);
85  }
86 }

88 static void set_velocities(ParticleRange const &particles,
89  std::vector<IBM_CUDA_ParticleDataOutput> &buffer) {
90  int i = 0;
91  for (auto &part : particles) {
92  if (part.p.is_virtual) {
93  for (int j = 0; j < 3; j++)
94  part.m.v[j] = static_cast<double>(buffer[i].v[j]);
95  }
96  i++;
97  }
98 }
