32 std::pair<double, double> WidomInsertion::measure_excess_chemical_potential(
33  SingleReaction &current_reaction) {
34 
35  if (!all_reactant_particles_exist(current_reaction))
36  throw std::runtime_error("Trying to remove some non-existing particles "
37  "from the system via the inverse Widom scheme.");
38 
39  const double E_pot_old = calculate_current_potential_energy_of_system();
40 
41  // make reaction attempt
42  std::vector<int> p_ids_created_particles;
43  std::vector<StoredParticleProperty> hidden_particles_properties;
44  std::vector<StoredParticleProperty> changed_particles_properties;
45 
46  // save p_id, charge and type of the reactant particle, only thing we
47  // need to hide the particle and recover it
48  const int number_of_saved_properties = 3;
49 
50  std::tie(changed_particles_properties, p_ids_created_particles,
51  hidden_particles_properties) =
52  make_reaction_attempt(current_reaction);
53 
54  const double E_pot_new = calculate_current_potential_energy_of_system();
55  // reverse reaction attempt
56  // reverse reaction
57  // 1) delete created product particles
58  for (int p_ids_created_particle : p_ids_created_particles) {
59  delete_particle(p_ids_created_particle);
60  }
61  // 2) restore previously hidden reactant particles
62  restore_properties(hidden_particles_properties, number_of_saved_properties);
63  // 3) restore previously changed reactant particles
64  restore_properties(changed_particles_properties, number_of_saved_properties);
65  std::vector<double> exponential = {exp(-1.0 / kT * (E_pot_new - E_pot_old))};
66  current_reaction.accumulator_potential_energy_difference_exponential(
67  exponential);
68 
69  // calculate mean excess chemical potential and standard error of the mean
70  auto const &accumulator =
71  current_reaction.accumulator_potential_energy_difference_exponential;
72  return {-kT * log(accumulator.mean()[0]),
73  std::abs(-kT / accumulator.mean()[0] * accumulator.std_error()[0])};
74 }
