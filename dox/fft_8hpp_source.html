23 /** \file
24  *
25  * Routines, row decomposition, data structures and communication for the
26  * 3D-FFT.
27  *
28  * The 3D-FFT is split into 3 ond dimensional FFTs. The data is
29  * distributed in such a way, that for the actual direction of the
30  * FFT each node has a certain number of rows for which it performs a
31  * 1D-FFT. After performing the FFT on that direction the data is
32  * redistributed.
33  *
34  * For simplicity at the moment I have implemented a full complex to
35  * complex FFT (even though a real to complex FFT would be
36  * sufficient)
37  *
38  * \todo Combine the forward and backward structures.
39  * \todo The packing routines could be moved to utils.hpp when they are needed
40  * elsewhere.
41  *
42  * For more information about FFT usage, see \ref fft.cpp "fft.cpp".
43  */

58 /** Aligned allocator for fft data. */
59 template <class T> struct fft_allocator {
60  typedef T value_type;
61  fft_allocator() noexcept = default; // default ctor not required
62  template <class U> explicit fft_allocator(const fft_allocator<U> &) {}
63  template <class U> bool operator==(const fft_allocator<U> &) const {
64  return true;
65  }
66  template <class U> bool operator!=(const fft_allocator<U> &) const {
67  return false;
68  }
69 
70  T *allocate(const std::size_t n) const {
71  if (n == 0) {
72  return nullptr;
73  }
74  if (n > static_cast<std::size_t>(-1) / sizeof(T)) {
75  throw std::bad_array_new_length();
76  }
77  void *const pv = fftw_malloc(n * sizeof(T));
78  if (!pv) {
79  throw std::bad_alloc();
80  }
81  return static_cast<T *>(pv);
82  }
83  void deallocate(T *const p, std::size_t) const noexcept { fftw_free(p); }
84 };

211 /** Pack a block (<tt>size[3]</tt> starting at <tt>start[3]</tt>) of an input
212  * 3d-grid with dimension <tt>dim[3]</tt> into an output 3d-block with
213  * dimension <tt>size[3]</tt>.
214  *
215  * The block with dimensions <tt>(size[0], size[1], size[2])</tt> is stored
216  * in 'row-major-order' or 'C-order', that means the first index is
217  * changing slowest when running through the linear array. The
218  * element (@c i0 (slow), @c i1 (mid), @c i2 (fast)) has the linear index
219  * <tt>li = i2 + size[2] * (i1 + (size[1] * i0))</tt>.
220  *
221  * \param[in] in pointer to input 3d-grid.
222  * \param[out] out pointer to output 3d-grid (block).
223  * \param[in] start start index of the block in the in-grid.
224  * \param[in] size size of the block (=dimension of the out-grid).
225  * \param[in] dim size of the in-grid.
226  * \param[in] element size of a grid element (e.g. 1 for Real, 2 for Complex).
227  */
228 void fft_pack_block(double const *in, double *out, int const start[3],
229  int const size[3], int const dim[3], int element);
