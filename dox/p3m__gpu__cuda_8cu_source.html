120 template <int cao>
121 __device__ void static Aliasing_sums_ik(const P3MGpuData p, int NX, int NY,
122  int NZ, REAL_TYPE *Zaehler,
123  REAL_TYPE *Nenner) {
124  REAL_TYPE S1, S2, S3;
125  REAL_TYPE zwi;
126  int MX, MY, MZ;
127  REAL_TYPE NMX, NMY, NMZ;
128  REAL_TYPE NM2;
129  REAL_TYPE TE;
130  REAL_TYPE Leni[3];
131  REAL_TYPE Meshi[3];
132  for (int i = 0; i < 3; ++i) {
133  Leni[i] = 1.0f / p.box[i];
134  Meshi[i] = 1.0f / static_cast<REAL_TYPE>(p.mesh[i]);
135  }
136 
137  Zaehler[0] = Zaehler[1] = Zaehler[2] = *Nenner = 0.0;
138 
139  for (MX = -P3M_BRILLOUIN; MX <= P3M_BRILLOUIN; MX++) {
140  NMX = static_cast<REAL_TYPE>(((NX > p.mesh[0] / 2) ? NX - p.mesh[0] : NX) +
141  p.mesh[0] * MX);
142  S1 = int_pow<2 * cao>(Utils::sinc(Meshi[0] * NMX));
143  for (MY = -P3M_BRILLOUIN; MY <= P3M_BRILLOUIN; MY++) {
144  NMY = static_cast<REAL_TYPE>(
145  ((NY > p.mesh[1] / 2) ? NY - p.mesh[1] : NY) + p.mesh[1] * MY);
146  S2 = S1 * int_pow<2 * cao>(Utils::sinc(Meshi[1] * NMY));
147  for (MZ = -P3M_BRILLOUIN; MZ <= P3M_BRILLOUIN; MZ++) {
148  NMZ = static_cast<REAL_TYPE>(
149  ((NZ > p.mesh[2] / 2) ? NZ - p.mesh[2] : NZ) + p.mesh[2] * MZ);
150  S3 = S2 * int_pow<2 * cao>(Utils::sinc(Meshi[2] * NMZ));
151 
152  NM2 = sqr(NMX * Leni[0]) + sqr(NMY * Leni[1]) + sqr(NMZ * Leni[2]);
153  *Nenner += S3;
154 
155  TE = exp(-sqr(Utils::pi<float>() / (p.alpha)) * NM2);
156  zwi = S3 * TE / NM2;
157  Zaehler[0] += NMX * zwi * Leni[0];
158  Zaehler[1] += NMY * zwi * Leni[1];
159  Zaehler[2] += NMZ * zwi * Leni[2];
160  }
161  }
162  }
163 }

165 /* Calculate influence function */
166 template <int cao>
167 __global__ void calculate_influence_function_device(const P3MGpuData p) {
168 
169  const auto NX = static_cast<int>(blockDim.x * blockIdx.x + threadIdx.x);
170  const auto NY = static_cast<int>(blockDim.y * blockIdx.y + threadIdx.y);
171  const auto NZ = static_cast<int>(blockDim.z * blockIdx.z + threadIdx.z);
172  REAL_TYPE Dnx, Dny, Dnz;
173  REAL_TYPE Zaehler[3] = {0.0, 0.0, 0.0}, Nenner = 0.0;
174  REAL_TYPE zwi;
175  int ind = 0;
176  REAL_TYPE Leni[3];
177  for (int i = 0; i < 3; ++i)
178  Leni[i] = 1.0f / p.box[i];
179 
180  if ((NX >= p.mesh[0]) || (NY >= p.mesh[1]) || (NZ >= (p.mesh[2] / 2 + 1)))
181  return;
182 
183  ind = NX * p.mesh[1] * (p.mesh[2] / 2 + 1) + NY * (p.mesh[2] / 2 + 1) + NZ;
184 
185  if (((NX == 0) && (NY == 0) && (NZ == 0)) ||
186  ((NX % (p.mesh[0] / 2) == 0) && (NY % (p.mesh[1] / 2) == 0) &&
187  (NZ % (p.mesh[2] / 2) == 0)))
188  p.G_hat[ind] = 0.0;
189  else {
190  Aliasing_sums_ik<cao>(p, NX, NY, NZ, Zaehler, &Nenner);
191 
192  Dnx = static_cast<REAL_TYPE>((NX > p.mesh[0] / 2) ? NX - p.mesh[0] : NX);
193  Dny = static_cast<REAL_TYPE>((NY > p.mesh[1] / 2) ? NY - p.mesh[1] : NY);
194  Dnz = static_cast<REAL_TYPE>((NZ > p.mesh[2] / 2) ? NZ - p.mesh[2] : NZ);
195 
196  zwi = Dnx * Zaehler[0] * Leni[0] + Dny * Zaehler[1] * Leni[1] +
197  Dnz * Zaehler[2] * Leni[2];
198  zwi /= ((sqr(Dnx * Leni[0]) + sqr(Dny * Leni[1]) + sqr(Dnz * Leni[2])) *
199  sqr(Nenner));
200  p.G_hat[ind] = 2.0f * zwi / Utils::pi<float>();
201  }
202 }

205 __device__ double atomicAdd(double *address, double val) {
206  unsigned long long int *address_as_ull = (unsigned long long int *)address;
207  unsigned long long int old = *address_as_ull, assumed;
208  do {
209  assumed = old;
210  old = atomicCAS(address_as_ull, assumed,
211  __double_as_longlong(val + __longlong_as_double(assumed)));
212  } while (assumed != old);
213  return __longlong_as_double(old);
214 }

218 __device__ inline int linear_index_r(P3MGpuData const &p, int i, int j, int k) {
219  return p.mesh[1] * p.mesh_z_padded * i + p.mesh_z_padded * j + k;

222 __device__ inline int linear_index_k(P3MGpuData const &p, int i, int j, int k) {
223  return p.mesh[1] * (p.mesh[2] / 2 + 1) * i + (p.mesh[2] / 2 + 1) * j + k;

227 __global__ void apply_diff_op(const P3MGpuData p) {
228  const int linear_index = linear_index_k(p, static_cast<int>(blockIdx.x),
229  static_cast<int>(blockIdx.y),
230  static_cast<int>(threadIdx.x));
231 
232  auto const nx = static_cast<int>(
233  (blockIdx.x > p.mesh[0] / 2) ? blockIdx.x - p.mesh[0] : blockIdx.x);
234  auto const ny = static_cast<int>(
235  (blockIdx.y > p.mesh[1] / 2) ? blockIdx.y - p.mesh[1] : blockIdx.y);
236  auto const nz = static_cast<int>(threadIdx.x);
237 
238  const FFT_TYPE_COMPLEX meshw = p.charge_mesh[linear_index];
239  FFT_TYPE_COMPLEX buf;
240  buf.x = -2.0f * Utils::pi<float>() * meshw.y;
241  buf.y = 2.0f * Utils::pi<float>() * meshw.x;
242 
243  p.force_mesh_x[linear_index].x =
244  static_cast<decltype(FFT_TYPE_COMPLEX::x)>(nx) * buf.x / p.box[0];
245  p.force_mesh_x[linear_index].y =
246  static_cast<decltype(FFT_TYPE_COMPLEX::x)>(nx) * buf.y / p.box[0];
247 
248  p.force_mesh_y[linear_index].x =
249  static_cast<decltype(FFT_TYPE_COMPLEX::x)>(ny) * buf.x / p.box[1];
250  p.force_mesh_y[linear_index].y =
251  static_cast<decltype(FFT_TYPE_COMPLEX::x)>(ny) * buf.y / p.box[1];
252 
253  p.force_mesh_z[linear_index].x =
254  static_cast<decltype(FFT_TYPE_COMPLEX::x)>(nz) * buf.x / p.box[2];
255  p.force_mesh_z[linear_index].y =
256  static_cast<decltype(FFT_TYPE_COMPLEX::x)>(nz) * buf.y / p.box[2];
257 }

259 __device__ inline int wrap_index(const int ind, const int mesh) {
260  if (ind < 0)
261  return ind + mesh;
262  if (ind >= mesh)
263  return ind - mesh;
264  return ind;

276 template <int cao, bool shared>
277 __global__ void assign_charge_kernel(const CUDA_particle_data *const pdata,
278  const P3MGpuData par,
279  const int parts_per_block) {
280  const int part_in_block = threadIdx.x / cao;
281  const int cao_id_x = threadIdx.x - part_in_block * cao;
282  /* id of the particle */
283  int id =
284  parts_per_block * (blockIdx.x * gridDim.y + blockIdx.y) + part_in_block;
285  if (id >= par.n_part)
286  return;
287  /* position relative to the closest gird point */
288  REAL_TYPE m_pos[3];
289  /* index of the nearest mesh point */
290  int nmp_x, nmp_y, nmp_z;
291 
292  const CUDA_particle_data p = pdata[id];
293  auto *charge_mesh = (REAL_TYPE *)par.charge_mesh;
294 
295  m_pos[0] = p.p[0] * par.hi[0] - par.pos_shift;
296  m_pos[1] = p.p[1] * par.hi[1] - par.pos_shift;
297  m_pos[2] = p.p[2] * par.hi[2] - par.pos_shift;
298 
299  nmp_x = static_cast<int>(floorf(m_pos[0] + 0.5f));
300  nmp_y = static_cast<int>(floorf(m_pos[1] + 0.5f));
301  nmp_z = static_cast<int>(floorf(m_pos[2] + 0.5f));
302 
303  m_pos[0] -= static_cast<REAL_TYPE>(nmp_x);
304  m_pos[1] -= static_cast<REAL_TYPE>(nmp_y);
305  m_pos[2] -= static_cast<REAL_TYPE>(nmp_z);
306 
307  nmp_x = wrap_index(nmp_x + cao_id_x, par.mesh[0]);
308  nmp_y = wrap_index(nmp_y + static_cast<int>(threadIdx.y), par.mesh[1]);
309  nmp_z = wrap_index(nmp_z + static_cast<int>(threadIdx.z), par.mesh[2]);
310 
311  const int ind = linear_index_r(par, nmp_x, nmp_y, nmp_z);
312 
313  extern __shared__ float weights[];
314 
315  if (shared) {
316  if ((threadIdx.y < 3) && (threadIdx.z == 0)) {
317  weights[3 * cao * part_in_block + 3 * cao_id_x + threadIdx.y] =
318  Utils::bspline<cao>(cao_id_x, m_pos[threadIdx.y]);
319  }
320 
321  __syncthreads();
322 
323  atomicAdd(&(charge_mesh[ind]),
324  weights[3 * cao * part_in_block + 3 * cao_id_x + 0] *
325  weights[3 * cao * part_in_block + 3 * threadIdx.y + 1] *
326  weights[3 * cao * part_in_block + 3 * threadIdx.z + 2] * p.q);
327 
328  } else {
329  atomicAdd(&(charge_mesh[ind]),
330  Utils::bspline<cao>(cao_id_x, m_pos[0]) *
331  Utils::bspline<cao>(threadIdx.y, m_pos[1]) *
332  Utils::bspline<cao>(threadIdx.z, m_pos[2]) * p.q);
333  }
334 }

336 void assign_charges(const CUDA_particle_data *const pdata, const P3MGpuData p) {
337  dim3 grid, block;
338  grid.z = 1;
339  const int cao3 = p.cao * p.cao * p.cao;
340  const int cao = p.cao;
341  int parts_per_block = 1, n_blocks = 1;
342 
343  while ((parts_per_block + 1) * cao3 <= 1024) {
344  parts_per_block++;
345  }
346  if ((p.n_part % parts_per_block) == 0)
347  n_blocks = std::max<int>(1, p.n_part / parts_per_block);
348  else
349  n_blocks = p.n_part / parts_per_block + 1;
350 
351  grid.x = n_blocks;
352  grid.y = 1;
353  while (grid.x > 65536) {
354  grid.y++;
355  if ((n_blocks % grid.y) == 0)
356  grid.x = std::max<int>(1, n_blocks / grid.y);
357  else
358  grid.x = n_blocks / grid.y + 1;
359  }
360 
361  block.x = parts_per_block * cao;
362  block.y = cao;
363  block.z = cao;
364 
365  switch (cao) {
366  case 1:
367  (assign_charge_kernel<1, false>)<<<dim3(grid), dim3(block), 0, nullptr>>>(
368  pdata, p, parts_per_block);
369  break;
370  case 2:
371  (assign_charge_kernel<2, false>)<<<dim3(grid), dim3(block), 0, nullptr>>>(
372  pdata, p, parts_per_block);
373  break;
374  case 3:
375  (assign_charge_kernel<
376  3, true>)<<<dim3(grid), dim3(block),
377  3 * parts_per_block * cao * sizeof(REAL_TYPE), nullptr>>>(
378  pdata, p, parts_per_block);
379  break;
380  case 4:
381  (assign_charge_kernel<
382  4, true>)<<<dim3(grid), dim3(block),
383  3 * parts_per_block * cao * sizeof(REAL_TYPE), nullptr>>>(
384  pdata, p, parts_per_block);
385  break;
386  case 5:
387  (assign_charge_kernel<
388  5, true>)<<<dim3(grid), dim3(block),
389  3 * parts_per_block * cao * sizeof(REAL_TYPE), nullptr>>>(
390  pdata, p, parts_per_block);
391  break;
392  case 6:
393  (assign_charge_kernel<
394  6, true>)<<<dim3(grid), dim3(block),
395  3 * parts_per_block * cao * sizeof(REAL_TYPE), nullptr>>>(
396  pdata, p, parts_per_block);
397  break;
398  case 7:
399  (assign_charge_kernel<
400  7, true>)<<<dim3(grid), dim3(block),
401  3 * parts_per_block * cao * sizeof(REAL_TYPE), nullptr>>>(
402  pdata, p, parts_per_block);
403  break;
404  default:
405  break;
406  }
407  cuda_check_errors_exit(block, grid, "assign_charge", __FILE__, __LINE__);
408 }

410 template <int cao, bool shared>
411 __global__ void assign_forces_kernel(const CUDA_particle_data *const pdata,
412  const P3MGpuData par,
413  float *lb_particle_force_gpu,
414  REAL_TYPE prefactor, int parts_per_block) {
415  const int part_in_block = threadIdx.x / cao;
416  const int cao_id_x = threadIdx.x - part_in_block * cao;
417  /* id of the particle */
418  int id =
419  parts_per_block * (blockIdx.x * gridDim.y + blockIdx.y) + part_in_block;
420  if (id >= par.n_part)
421  return;
422  /* position relative to the closest grid point */
423  REAL_TYPE m_pos[3];
424  /* index of the nearest mesh point */
425  int nmp_x, nmp_y, nmp_z;
426 
427  const CUDA_particle_data p = pdata[id];
428 
429  m_pos[0] = p.p[0] * par.hi[0] - par.pos_shift;
430  m_pos[1] = p.p[1] * par.hi[1] - par.pos_shift;
431  m_pos[2] = p.p[2] * par.hi[2] - par.pos_shift;
432 
433  nmp_x = static_cast<int>(floorf(m_pos[0] + 0.5f));
434  nmp_y = static_cast<int>(floorf(m_pos[1] + 0.5f));
435  nmp_z = static_cast<int>(floorf(m_pos[2] + 0.5f));
436 
437  m_pos[0] -= static_cast<REAL_TYPE>(nmp_x);
438  m_pos[1] -= static_cast<REAL_TYPE>(nmp_y);
439  m_pos[2] -= static_cast<REAL_TYPE>(nmp_z);
440 
441  nmp_x = wrap_index(nmp_x + cao_id_x, par.mesh[0]);
442  nmp_y = wrap_index(nmp_y + static_cast<int>(threadIdx.y), par.mesh[1]);
443  nmp_z = wrap_index(nmp_z + static_cast<int>(threadIdx.z), par.mesh[2]);
444 
445  REAL_TYPE c;
446  const int index = linear_index_r(par, nmp_x, nmp_y, nmp_z);
447 
448  extern __shared__ float weights[];
449 
450  if (shared) {
451  if ((threadIdx.y < 3) && (threadIdx.z == 0)) {
452  weights[3 * cao * part_in_block + 3 * cao_id_x + threadIdx.y] =
453  Utils::bspline<cao>(cao_id_x, m_pos[threadIdx.y]);
454  }
455 
456  __syncthreads();
457 
458  c = -prefactor * weights[3 * cao * part_in_block + 3 * cao_id_x + 0] *
459  weights[3 * cao * part_in_block + 3 * threadIdx.y + 1] *
460  weights[3 * cao * part_in_block + 3 * threadIdx.z + 2] * p.q;
461  } else {
462  c = -prefactor * Utils::bspline<cao>(cao_id_x, m_pos[0]) *
463  Utils::bspline<cao>(threadIdx.y, m_pos[1]) *
464  Utils::bspline<cao>(threadIdx.z, m_pos[2]) * p.q;
465  }
466 
467  const REAL_TYPE *force_mesh_x = (REAL_TYPE *)par.force_mesh_x;
468  const REAL_TYPE *force_mesh_y = (REAL_TYPE *)par.force_mesh_y;
469  const REAL_TYPE *force_mesh_z = (REAL_TYPE *)par.force_mesh_z;
470 
471  atomicAdd(&(lb_particle_force_gpu[3 * id + 0]), c * force_mesh_x[index]);
472  atomicAdd(&(lb_particle_force_gpu[3 * id + 1]), c * force_mesh_y[index]);
473  atomicAdd(&(lb_particle_force_gpu[3 * id + 2]), c * force_mesh_z[index]);
474 }

476 void assign_forces(const CUDA_particle_data *const pdata, const P3MGpuData p,
477  float *lb_particle_force_gpu, REAL_TYPE prefactor) {
478  dim3 grid, block;
479  grid.z = 1;
480 
481  const int cao = p.cao;
482  const int cao3 = cao * cao * cao;
483  int parts_per_block = 1, n_blocks = 1;
484 
485  while ((parts_per_block + 1) * cao3 <= 1024) {
486  parts_per_block++;
487  }
488 
489  if ((p3m_gpu_data.n_part % parts_per_block) == 0)
490  n_blocks = std::max<int>(1, p3m_gpu_data.n_part / parts_per_block);
491  else
492  n_blocks = p3m_gpu_data.n_part / parts_per_block + 1;
493 
494  grid.x = n_blocks;
495  grid.y = 1;
496  while (grid.x > 65536) {
497  grid.y++;
498  if ((n_blocks % grid.y) == 0)
499  grid.x = std::max<int>(1, n_blocks / grid.y);
500  else
501  grid.x = n_blocks / grid.y + 1;
502  }
503 
504  block.x = parts_per_block * cao;
505  block.y = cao;
506  block.z = cao;
507 
508  /* Switch for assignment templates, the shared version only is faster for cao
509  * > 2 */
510  switch (cao) {
511  case 1:
512  (assign_forces_kernel<1, false>)<<<dim3(grid), dim3(block), 0, nullptr>>>(
513  pdata, p, lb_particle_force_gpu, prefactor, parts_per_block);
514  break;
515  case 2:
516  (assign_forces_kernel<2, false>)<<<dim3(grid), dim3(block), 0, nullptr>>>(
517  pdata, p, lb_particle_force_gpu, prefactor, parts_per_block);
518  break;
519  case 3:
520  (assign_forces_kernel<

555 /* Init the internal data structures of the P3M GPU.
556  * Mainly allocation on the device and influence function calculation.
557  * Be advised: this needs mesh^3*5*sizeof(REAL_TYPE) of device memory.
558  * We use real to complex FFTs, so the size of the reciprocal mesh
559  * is (cuFFT convention) Nx x Ny x [ Nz /2 + 1 ].
560  */
561 void p3m_gpu_init(int cao, const int mesh[3], double alpha) {
562  if (mesh[0] == -1 && mesh[1] == -1 && mesh[2] == -1)
563  throw std::runtime_error("P3M: invalid mesh size");
564 
565  espressoSystemInterface.requestParticleStructGpu();
566 
567  bool reinit_if = false, mesh_changed = false;
568  p3m_gpu_data.n_part = gpu_get_particle_pointer().size();
569 
570  if ((p3m_gpu_data_initialized == 0) || (p3m_gpu_data.alpha != alpha)) {
571  p3m_gpu_data.alpha = static_cast<REAL_TYPE>(alpha);
572  reinit_if = true;
573  }
574 
575  if ((p3m_gpu_data_initialized == 0) || (p3m_gpu_data.cao != cao)) {
576  p3m_gpu_data.cao = cao;
577  // NOLINTNEXTLINE(bugprone-integer-division)
578  p3m_gpu_data.pos_shift = static_cast<REAL_TYPE>((p3m_gpu_data.cao - 1) / 2);
579  reinit_if = true;
580  }
581 
582  if ((p3m_gpu_data_initialized == 0) || (p3m_gpu_data.mesh[0] != mesh[0]) ||
583  (p3m_gpu_data.mesh[1] != mesh[1]) || (p3m_gpu_data.mesh[2] != mesh[2])) {
584  std::copy(mesh, mesh + 3, p3m_gpu_data.mesh);
585  mesh_changed = true;
586  reinit_if = true;
587  }
588 
589  auto const box_l = box_geo.length();
590 
591  if ((p3m_gpu_data_initialized == 0) || (p3m_gpu_data.box[0] != box_l[0]) ||
592  (p3m_gpu_data.box[1] != box_l[1]) || (p3m_gpu_data.box[2] != box_l[2])) {
593  std::copy(box_l.begin(), box_l.end(), p3m_gpu_data.box);
594  reinit_if = true;
595  }
596 
597  p3m_gpu_data.mesh_z_padded = (mesh[2] / 2 + 1) * 2;
598  p3m_gpu_data.mesh_size = mesh[0] * mesh[1] * p3m_gpu_data.mesh_z_padded;
599 
600  for (int i = 0; i < 3; i++) {
601  p3m_gpu_data.hi[i] =
602  static_cast<REAL_TYPE>(p3m_gpu_data.mesh[i]) / p3m_gpu_data.box[i];
603  }
604 
605  if ((p3m_gpu_data_initialized == 1) && mesh_changed) {
606  cuda_safe_mem(cudaFree(p3m_gpu_data.charge_mesh));
607  p3m_gpu_data.charge_mesh = nullptr;
608  cuda_safe_mem(cudaFree(p3m_gpu_data.force_mesh_x));
609  p3m_gpu_data.force_mesh_x = nullptr;
610  cuda_safe_mem(cudaFree(p3m_gpu_data.force_mesh_y));
611  p3m_gpu_data.force_mesh_y = nullptr;
612  cuda_safe_mem(cudaFree(p3m_gpu_data.force_mesh_z));
613  p3m_gpu_data.force_mesh_z = nullptr;
614  cuda_safe_mem(cudaFree(p3m_gpu_data.G_hat));
615  p3m_gpu_data.G_hat = nullptr;
616 
617  cufftDestroy(p3m_gpu_fft_plans.forw_plan);
618  cufftDestroy(p3m_gpu_fft_plans.back_plan);
619 
620  p3m_gpu_data_initialized = 0;
621  }
622 
623  if ((p3m_gpu_data_initialized == 0) && (p3m_gpu_data.mesh_size > 0)) {
624  /* Size of the complex mesh Nx * Ny * ( Nz / 2 + 1 ) */
625  const int cmesh_size = p3m_gpu_data.mesh[0] * p3m_gpu_data.mesh[1] *
626  (p3m_gpu_data.mesh[2] / 2 + 1);
627 
628  cuda_safe_mem(cudaMalloc((void **)&(p3m_gpu_data.charge_mesh),
629  cmesh_size * sizeof(FFT_TYPE_COMPLEX)));
630  cuda_safe_mem(cudaMalloc((void **)&(p3m_gpu_data.force_mesh_x),
631  cmesh_size * sizeof(FFT_TYPE_COMPLEX)));
632  cuda_safe_mem(cudaMalloc((void **)&(p3m_gpu_data.force_mesh_y),
633  cmesh_size * sizeof(FFT_TYPE_COMPLEX)));
634  cuda_safe_mem(cudaMalloc((void **)&(p3m_gpu_data.force_mesh_z),
635  cmesh_size * sizeof(FFT_TYPE_COMPLEX)));
636  cuda_safe_mem(cudaMalloc((void **)&(p3m_gpu_data.G_hat),
637  cmesh_size * sizeof(REAL_TYPE)));
638 
639  if (cufftPlan3d(&(p3m_gpu_fft_plans.forw_plan), mesh[0], mesh[1], mesh[2],
640  FFT_PLAN_FORW_FLAG) != CUFFT_SUCCESS ||
641  cufftPlan3d(&(p3m_gpu_fft_plans.back_plan), mesh[0], mesh[1], mesh[2],
642  FFT_PLAN_BACK_FLAG) != CUFFT_SUCCESS) {
643  throw std::runtime_error("Unable to create fft plan");
644  }
645  }
646 
647  if ((reinit_if || (p3m_gpu_data_initialized == 0)) &&
648  (p3m_gpu_data.mesh_size > 0)) {
649  dim3 grid(1, 1, 1);
650  dim3 block(1, 1, 1);
651  block.x = 512 / mesh[0] + 1;
652  block.y = mesh[1];
653  block.z = 1;
654  grid.x = mesh[0] / block.x + 1;
655  grid.z = mesh[2] / 2 + 1;
656 
657  switch (p3m_gpu_data.cao) {
658  case 1:
659  KERNELCALL(calculate_influence_function_device<1>, grid, block,
660  p3m_gpu_data);
661  break;
662  case 2:
663  KERNELCALL(calculate_influence_function_device<2>, grid, block,
664  p3m_gpu_data);
665  break;
666  case 3:
667  KERNELCALL(calculate_influence_function_device<3>, grid, block,
668  p3m_gpu_data);
669  break;
670  case 4:
671  KERNELCALL(calculate_influence_function_device<4>, grid, block,
672  p3m_gpu_data);
673  break;
674  case 5:
675  KERNELCALL(calculate_influence_function_device<5>, grid, block,
676  p3m_gpu_data);
677  break;
678  case 6:
679  KERNELCALL(calculate_influence_function_device<6>, grid, block,
680  p3m_gpu_data);
681  break;
682  case 7:
683  KERNELCALL(calculate_influence_function_device<7>, grid, block,
684  p3m_gpu_data);
685  break;
686  }
687  }
688  if (p3m_gpu_data.mesh_size > 0)
689  p3m_gpu_data_initialized = 1;
690 }

692 /**
693  * \brief The long range part of the P3M algorithm.
694  */
695 void p3m_gpu_add_farfield_force() {
696  auto device_particles = gpu_get_particle_pointer();
697  CUDA_particle_data *lb_particle_gpu = device_particles.data();
698  p3m_gpu_data.n_part = device_particles.size();
699 
700  float *lb_particle_force_gpu = gpu_get_particle_force_pointer();
701 
702  if (p3m_gpu_data.n_part == 0)
703  return;
704 
705  dim3 gridConv(p3m_gpu_data.mesh[0], p3m_gpu_data.mesh[1], 1);
706  dim3 threadsConv(p3m_gpu_data.mesh[2] / 2 + 1, 1, 1);
707 
708  auto prefactor =
709  REAL_TYPE(coulomb.prefactor / (p3m_gpu_data.box[0] * p3m_gpu_data.box[1] *
710  p3m_gpu_data.box[2] * 2.0));
711 
712  cuda_safe_mem(cudaMemset(p3m_gpu_data.charge_mesh, 0,
713  p3m_gpu_data.mesh_size * sizeof(REAL_TYPE)));
714 
715  /* Interpolate the charges to the mesh */
716  assign_charges(lb_particle_gpu, p3m_gpu_data);
717 
718  /* Do forward FFT of the charge mesh */
719  if (FFT_FORW_FFT(p3m_gpu_fft_plans.forw_plan,
720  (REAL_TYPE *)p3m_gpu_data.charge_mesh,
721  p3m_gpu_data.charge_mesh) != CUFFT_SUCCESS) {
722  fprintf(stderr, "CUFFT error: Forward FFT failed\n");
723  return;
724  }
725 
726  /* Do convolution */
727  KERNELCALL(apply_influence_function, gridConv, threadsConv, p3m_gpu_data);
728 
729  /* Take derivative */
730  KERNELCALL(apply_diff_op, gridConv, threadsConv, p3m_gpu_data);
731 
732  /* Transform the components of the electric field back */
733  FFT_BACK_FFT(p3m_gpu_fft_plans.back_plan, p3m_gpu_data.force_mesh_x,
734  (REAL_TYPE *)p3m_gpu_data.force_mesh_x);
735  FFT_BACK_FFT(p3m_gpu_fft_plans.back_plan, p3m_gpu_data.force_mesh_y,
736  (REAL_TYPE *)p3m_gpu_data.force_mesh_y);
737  FFT_BACK_FFT(p3m_gpu_fft_plans.back_plan, p3m_gpu_data.force_mesh_z,
738  (REAL_TYPE *)p3m_gpu_data.force_mesh_z);
739 
740  /* Assign the forces from the mesh back to the particles */
741  assign_forces(lb_particle_gpu, p3m_gpu_data, lb_particle_force_gpu,
742  prefactor);
743 }
