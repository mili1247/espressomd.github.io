44 /** @brief Calculate the derivatives of the quaternion and angular
45  * acceleration for a given particle.
46  * See @cite sonnenschein85a. Please note that ESPResSo uses scalar-first
47  * notation for quaternions, while @cite sonnenschein85a uses scalar-last
48  * notation.
49  * @param[in] p %Particle
50  * @param[out] Qd First derivative of the particle quaternion
51  * @param[out] Qdd Second derivative of the particle quaternion
52  * @param[out] S Function of @p Qd and @p Qdd, used to evaluate the
53  * Lagrange parameter lambda
54  * @param[out] Wd Angular acceleration of the particle
55  */
56 static void define_Qdd(Particle const &p, Utils::Quaternion<double> &Qd,
57  Utils::Quaternion<double> &Qdd, Utils::Vector3d &S,
58  Utils::Vector3d &Wd) {
59  /* calculate the first derivative of the quaternion */
60  /* Eq. (4) @cite sonnenschein85a */
61  Qd[0] = 0.5 * (-p.r.quat[1] * p.m.omega[0] - p.r.quat[2] * p.m.omega[1] -
62  p.r.quat[3] * p.m.omega[2]);
63 
64  Qd[1] = 0.5 * (p.r.quat[0] * p.m.omega[0] - p.r.quat[3] * p.m.omega[1] +
65  p.r.quat[2] * p.m.omega[2]);
66 
67  Qd[2] = 0.5 * (p.r.quat[3] * p.m.omega[0] + p.r.quat[0] * p.m.omega[1] -
68  p.r.quat[1] * p.m.omega[2]);
69 
70  Qd[3] = 0.5 * (-p.r.quat[2] * p.m.omega[0] + p.r.quat[1] * p.m.omega[1] +
71  p.r.quat[0] * p.m.omega[2]);
72 
73  /* Calculate the angular acceleration. */
74  /* Eq. (5) @cite sonnenschein85a */
75  if (p.p.rotation & ROTATION_X)
76  Wd[0] = (p.f.torque[0] + p.m.omega[1] * p.m.omega[2] *
77  (p.p.rinertia[1] - p.p.rinertia[2])) /
78  p.p.rinertia[0];
79  if (p.p.rotation & ROTATION_Y)
80  Wd[1] = (p.f.torque[1] + p.m.omega[2] * p.m.omega[0] *
81  (p.p.rinertia[2] - p.p.rinertia[0])) /
82  p.p.rinertia[1];
83  if (p.p.rotation & ROTATION_Z)
84  Wd[2] = (p.f.torque[2] + p.m.omega[0] * p.m.omega[1] *
85  (p.p.rinertia[0] - p.p.rinertia[1])) /
86  p.p.rinertia[2];
87 
88  auto const S1 = Qd.norm2();
89 
90  /* Calculate the second derivative of the quaternion. */
91  /* Eq. (8) @cite sonnenschein85a */
92  Qdd[0] =
93  0.5 * (-p.r.quat[1] * Wd[0] - p.r.quat[2] * Wd[1] - p.r.quat[3] * Wd[2]) -
94  p.r.quat[0] * S1;
95 
96  Qdd[1] =
97  0.5 * (p.r.quat[0] * Wd[0] - p.r.quat[3] * Wd[1] + p.r.quat[2] * Wd[2]) -
98  p.r.quat[1] * S1;
99 
100  Qdd[2] =
101  0.5 * (p.r.quat[3] * Wd[0] + p.r.quat[0] * Wd[1] - p.r.quat[1] * Wd[2]) -
102  p.r.quat[2] * S1;
103 
104  Qdd[3] =
105  0.5 * (-p.r.quat[2] * Wd[0] + p.r.quat[1] * Wd[1] + p.r.quat[0] * Wd[2]) -
106  p.r.quat[3] * S1;
107 
108  S[0] = S1;
109  S[1] = Utils::dot(Qd, Qdd);
110  S[2] = Qdd.norm2();
111 }

113 /**
114  * See @cite omelyan98a. Please note that ESPResSo uses scalar-first
115  * notation for quaternions, while @cite omelyan98a uses scalar-last
116  * notation.
117  *
118  * For very high angular velocities (e.g. if the product of @p time_step
119  * with the largest component of @ref ParticleMomentum::omega "p.m.omega"
120  * is superior to ~2.0), the calculation might fail.
121  *
122  * \todo implement for fixed_coord_flag
123  */
124 void propagate_omega_quat_particle(Particle &p, double time_step) {
125 
126  // If rotation for the particle is disabled entirely, return early.
127  if (p.p.rotation == ROTATION_FIXED)
128  return;
129 
130  Utils::Quaternion<double> Qd{}, Qdd{};
131  Utils::Vector3d S{}, Wd{};
132 
133  // Clear rotational velocity for blocked rotation axes.
134  p.m.omega = Utils::mask(p.p.rotation, p.m.omega);
135 
136  define_Qdd(p, Qd, Qdd, S, Wd);
137 
138  auto const time_step_squared = time_step * time_step;
139  auto const time_step_half = 0.5 * time_step;
140 
141  /* Eq. (12) @cite omelyan98a. */
142  auto const square =
143  1 - time_step_squared *
144  (S[0] +
145  time_step * (S[1] + time_step_half / 2. * (S[2] - S[0] * S[0])));
146  assert(square >= 0.);
147  auto const lambda = 1 - S[0] * 0.5 * time_step_squared - sqrt(square);
148 
149  p.m.omega += time_step_half * Wd;
150  p.r.quat += time_step * (Qd + time_step_half * Qdd) - lambda * p.r.quat;
151 
152  /* and rescale quaternion, so it is exactly of unit length */
153  auto const scale = p.r.quat.norm();
154  if (scale == 0) {
155  p.r.quat = Utils::Quaternion<double>::identity();
156  } else {
157  p.r.quat /= scale;
158  }
159 }

161 void convert_torques_propagate_omega(const ParticleRange &particles,
162  double time_step) {
163  for (auto &p : particles) {
164  // Skip particle if rotation is turned off entirely for it.
165  if (p.p.rotation == ROTATION_FIXED)
166  continue;
167 
168  convert_torque_to_body_frame_apply_fix(p);
169 
170  // Propagation of angular velocities
171  p.m.omega += hadamard_division(0.5 * time_step * p.f.torque, p.p.rinertia);
172 
173  // zeroth estimate of omega
174  Utils::Vector3d omega_0 = p.m.omega;
175 
176  /* if the tensor of inertia is isotropic, the following refinement is not
177  needed.
178  Otherwise repeat this loop 2-3 times depending on the required accuracy
179  */
180 
181  const double rinertia_diff_01 = p.p.rinertia[0] - p.p.rinertia[1];
182  const double rinertia_diff_12 = p.p.rinertia[1] - p.p.rinertia[2];
183  const double rinertia_diff_20 = p.p.rinertia[2] - p.p.rinertia[0];
184  for (int times = 0; times <= 5; times++) {
185  Utils::Vector3d Wd;
186 
187  Wd[0] = p.m.omega[1] * p.m.omega[2] * rinertia_diff_12 / p.p.rinertia[0];
188  Wd[1] = p.m.omega[2] * p.m.omega[0] * rinertia_diff_20 / p.p.rinertia[1];
189  Wd[2] = p.m.omega[0] * p.m.omega[1] * rinertia_diff_01 / p.p.rinertia[2];
190 
191  p.m.omega = omega_0 + (0.5 * time_step) * Wd;
192  }
193  }
194 }
