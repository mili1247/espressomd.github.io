45 /** Returns pointer to the cell which corresponds to the position if the
46  * position is in the nodes spatial domain otherwise a nullptr pointer.
47  */
48 Cell *DomainDecomposition::position_to_cell(const Utils::Vector3d &pos) {
49  Utils::Vector3i cpos;
50 
51  for (int i = 0; i < 3; i++) {
52  cpos[i] = static_cast<int>(std::floor(pos[i] * inv_cell_size[i])) + 1 -
53  cell_offset[i];
54 
55  /* particles outside our box. Still take them if
56  nonperiodic boundary. We also accept the particle if we are at
57  the box boundary, and the particle is within the box. In this case
58  the particle belongs here and could otherwise potentially be dismissed
59  due to rounding errors. */
60  if (cpos[i] < 1) {
61  if ((!m_box.periodic(i) or (pos[i] >= m_box.length()[i])) &&
62  m_local_box.boundary()[2 * i])
63  cpos[i] = 1;
64  else
65  return nullptr;
66  } else if (cpos[i] > cell_grid[i]) {
67  if ((!m_box.periodic(i) or (pos[i] < m_box.length()[i])) &&
68  m_local_box.boundary()[2 * i + 1])
69  cpos[i] = cell_grid[i];
70  else
71  return nullptr;
72  }
73  }
74 
75  auto const ind = get_linear_index(cpos, ghost_cell_grid);
76  return &(cells.at(ind));
77 }

96 void DomainDecomposition::move_left_or_right(ParticleList &src,
97  ParticleList &left,
98  ParticleList &right,
99  int dir) const {
100  for (auto it = src.begin(); it != src.end();) {
101  if ((m_box.get_mi_coord(it->r.p[dir], m_local_box.my_left()[dir], dir) <
102  0.0) and
103  (m_box.periodic(dir) || (m_local_box.boundary()[2 * dir] == 0))) {
104  left.insert(std::move(*it));
105  it = src.erase(it);
106  } else if ((m_box.get_mi_coord(it->r.p[dir], m_local_box.my_right()[dir],
107  dir) >= 0.0) and
108  (m_box.periodic(dir) ||
109  (m_local_box.boundary()[2 * dir + 1] == 0))) {
110  right.insert(std::move(*it));
111  it = src.erase(it);
112  } else {
113  ++it;
114  }
115  }

118 void DomainDecomposition::exchange_neighbors(
119  ParticleList &pl, std::vector<ParticleChange> &modified_cells) {
120  auto const node_neighbors = Utils::Mpi::cart_neighbors<3>(m_comm);
121  static ParticleList send_buf_l, send_buf_r, recv_buf_l, recv_buf_r;
122 
123  for (int dir = 0; dir < 3; dir++) {
124  /* Single node direction, no action needed. */
125  if (Utils::Mpi::cart_get<3>(m_comm).dims[dir] == 1) {
126  continue;
127  /* In this (common) case left and right neighbors are
128  the same, and we need only one communication */
129  }
130  if (Utils::Mpi::cart_get<3>(m_comm).dims[dir] == 2) {
131  move_left_or_right(pl, send_buf_l, send_buf_l, dir);
132 
133  Utils::Mpi::sendrecv(m_comm, node_neighbors[2 * dir], 0, send_buf_l,
134  node_neighbors[2 * dir], 0, recv_buf_l);
135 
136  send_buf_l.clear();
137  } else {
138  using boost::mpi::request;
139  using Utils::Mpi::isendrecv;
140 
141  move_left_or_right(pl, send_buf_l, send_buf_r, dir);
142 
143  auto req_l = isendrecv(m_comm, node_neighbors[2 * dir], 0, send_buf_l,
144  node_neighbors[2 * dir], 0, recv_buf_l);
145  auto req_r = isendrecv(m_comm, node_neighbors[2 * dir + 1], 0, send_buf_r,
146  node_neighbors[2 * dir + 1], 0, recv_buf_r);
147 
148  std::array<request, 4> reqs{{req_l[0], req_l[1], req_r[0], req_r[1]}};
149  boost::mpi::wait_all(reqs.begin(), reqs.end());
150 
151  send_buf_l.clear();
152  send_buf_r.clear();
153  }
154 
155  move_if_local(recv_buf_l, pl, modified_cells);
156  move_if_local(recv_buf_r, pl, modified_cells);
157  }
158 }

171 void DomainDecomposition::resort(bool global,
172  std::vector<ParticleChange> &diff) {
173  ParticleList displaced_parts;
174 
175  for (auto &c : local_cells()) {
176  for (auto it = c->particles().begin(); it != c->particles().end();) {
177  fold_and_reset(*it, m_box);
178 
179  auto target_cell = particle_to_cell(*it);
180 
181  /* Particle is in place */
182  if (target_cell == c) {
183  std::advance(it, 1);
184  continue;
185  }
186 
187  auto p = std::move(*it);
188  it = c->particles().erase(it);
189  diff.emplace_back(ModifiedList{c->particles()});
190 
191  /* Particle is not local */
192  if (target_cell == nullptr) {
193  diff.emplace_back(RemovedParticle{p.identity()});
194  displaced_parts.insert(std::move(p));
195  }
196  /* Particle belongs on this node but is in the wrong cell. */
197  else if (target_cell != c) {
198  target_cell->particles().insert(std::move(p));
199  diff.emplace_back(ModifiedList{target_cell->particles()});
200  }
201  }
202  }
203 
204  if (global) {
205  auto const grid = Utils::Mpi::cart_get<3>(m_comm).dims;
206  /* Worst case we need grid - 1 rounds per direction.
207  * This correctly implies that if there is only one node,
208  * no action should be taken. */
209  int rounds_left = grid[0] + grid[1] + grid[2] - 3;
210  for (; rounds_left > 0; rounds_left--) {
211  exchange_neighbors(displaced_parts, diff);
212 
213  auto left_over = boost::mpi::all_reduce(m_comm, displaced_parts.size(),
214  std::plus<std::size_t>());
215 
216  if (left_over == 0) {
217  break;
218  }
219  }
220  } else {
221  exchange_neighbors(displaced_parts, diff);
222  }
223 
224  if (not displaced_parts.empty()) {
225  auto sort_cell = local_cells()[0];
226 
227  for (auto &part : displaced_parts) {
228  runtimeErrorMsg() << "Particle " << part.identity()
229  << " moved more than"
230  " one local box length in one timestep.";
231  sort_cell->particles().insert(std::move(part));
232 
233  diff.emplace_back(ModifiedList{sort_cell->particles()});
234  }
235  }
236 }

238 void DomainDecomposition::mark_cells() {
239  int cnt_c = 0;
240 
241  m_local_cells.clear();
242  m_ghost_cells.clear();
243 
244  for (int o = 0; o < ghost_cell_grid[2]; o++)
245  for (int n = 0; n < ghost_cell_grid[1]; n++)
246  for (int m = 0; m < ghost_cell_grid[0]; m++) {
247  if ((m > 0 && m < ghost_cell_grid[0] - 1 && n > 0 &&
248  n < ghost_cell_grid[1] - 1 && o > 0 && o < ghost_cell_grid[2] - 1))
249  m_local_cells.push_back(&cells.at(cnt_c++));
250  else
251  m_ghost_cells.push_back(&cells.at(cnt_c++));

274 int DomainDecomposition::calc_processor_min_num_cells() const {
275  /* the minimal number of cells can be lower if there are at least two nodes
276  serving a direction,
277  since this also ensures that the cell size is at most half the box
278  length. However, if there is only one processor for a direction, there
279  have to be at least two cells for this direction. */
280  return boost::accumulate(Utils::Mpi::cart_get<3>(m_comm).dims, 1,
281  [](int n_cells, int grid) {
282  return (grid == 1) ? 2 * n_cells : n_cells;
283  });

286 void DomainDecomposition::create_cell_grid(double range) {
287  auto const cart_info = Utils::Mpi::cart_get<3>(m_comm);
288 
289  int n_local_cells;
290  double cell_range[3];
291 
292  /* initialize */
293  cell_range[0] = cell_range[1] = cell_range[2] = range;
294 
295  /* Min num cells can not be smaller than calc_processor_min_num_cells. */
296  int min_num_cells = calc_processor_min_num_cells();
297 
298  if (range <= 0.) {
299  /* this is the non-interacting case */
300  auto const cells_per_dir =
301  static_cast<int>(std::ceil(std::pow(min_num_cells, 1. / 3.)));
302 
303  cell_grid[0] = cells_per_dir;
304  cell_grid[1] = cells_per_dir;
305  cell_grid[2] = cells_per_dir;
306 
307  n_local_cells = cell_grid[0] * cell_grid[1] * cell_grid[2];
308  } else {
309  /* Calculate initial cell grid */
310  double volume = m_local_box.length()[0];
311  for (int i = 1; i < 3; i++)
312  volume *= m_local_box.length()[i];
313  double scale = pow(DomainDecomposition::max_num_cells / volume, 1. / 3.);
314  for (int i = 0; i < 3; i++) {
315  /* this is at least 1 */
316  cell_grid[i] = (int)ceil(m_local_box.length()[i] * scale);
317  cell_range[i] = m_local_box.length()[i] / cell_grid[i];
318 
319  if (cell_range[i] < range) {
320  /* ok, too many cells for this direction, set to minimum */
321  cell_grid[i] = (int)floor(m_local_box.length()[i] / range);
322  if (cell_grid[i] < 1) {
323  runtimeErrorMsg() << "interaction range " << range << " in direction "
324  << i << " is larger than the local box size "
325  << m_local_box.length()[i];
326  cell_grid[i] = 1;
327  }
328  cell_range[i] = m_local_box.length()[i] / cell_grid[i];
329  }
330  }
331 
332  /* It may be necessary to asymmetrically assign the scaling to the
333  coordinates, which the above approach will not do.
334  For a symmetric box, it gives a symmetric result. Here we correct that.
335  */
336  for (;;) {
337  n_local_cells = cell_grid[0] * cell_grid[1] * cell_grid[2];
338 
339  /* done */
340  if (n_local_cells <= DomainDecomposition::max_num_cells)
341  break;
342 
343  /* find coordinate with the smallest cell range */
344  int min_ind = 0;
345  double min_size = cell_range[0];
346 
347  for (int i = 1; i < 3; i++) {
348  if (cell_grid[i] > 1 && cell_range[i] < min_size) {
349  min_ind = i;
350  min_size = cell_range[i];
351  }
352  }
353 
354  cell_grid[min_ind]--;
355  cell_range[min_ind] = m_local_box.length()[min_ind] / cell_grid[min_ind];
356  }
357 
358  /* sanity check */
359  if (n_local_cells < min_num_cells) {
360  runtimeErrorMsg()
361  << "number of cells " << n_local_cells << " is smaller than minimum "
362  << min_num_cells
363  << " (interaction range too large or min_num_cells too large)";
364  }
365  }
366 
367  /* quit program if unsuccessful */
368  if (n_local_cells > DomainDecomposition::max_num_cells) {
369  runtimeErrorMsg() << "no suitable cell grid found ";
370  }
371 
372  auto const node_pos = cart_info.coords;
373 
374  /* now set all dependent variables */
375  int new_cells = 1;
376  for (int i = 0; i < 3; i++) {
377  ghost_cell_grid[i] = cell_grid[i] + 2;
378  new_cells *= ghost_cell_grid[i];
379  cell_size[i] = m_local_box.length()[i] / (double)cell_grid[i];
380  inv_cell_size[i] = 1.0 / cell_size[i];
381  cell_offset[i] = node_pos[i] * cell_grid[i];
382  }
383 
384  /* allocate cell array and cell pointer arrays */
385  cells.clear();
386  cells.resize(new_cells);
387  m_local_cells.resize(n_local_cells);
388  m_ghost_cells.resize(new_cells - n_local_cells);
389 }

391 void DomainDecomposition::init_cell_interactions() {
392  /* loop all local cells */
393  for (int o = 1; o < cell_grid[2] + 1; o++)
394  for (int n = 1; n < cell_grid[1] + 1; n++)
395  for (int m = 1; m < cell_grid[0] + 1; m++) {
396 
397  auto const ind1 = get_linear_index(m, n, o, ghost_cell_grid);
398 
399  std::vector<Cell *> red_neighbors;
400  std::vector<Cell *> black_neighbors;
401 
402  /* loop all neighbor cells */
403  int lower_index[3] = {m - 1, n - 1, o - 1};
404  int upper_index[3] = {m + 1, n + 1, o + 1};
405 
406  for (int p = lower_index[2]; p <= upper_index[2]; p++)
407  for (int q = lower_index[1]; q <= upper_index[1]; q++)
408  for (int r = lower_index[0]; r <= upper_index[0]; r++) {
409  auto const ind2 = get_linear_index(r, q, p, ghost_cell_grid);
410  if (ind2 > ind1) {
411  red_neighbors.push_back(&cells.at(ind2));
412  } else {
413  black_neighbors.push_back(&cells.at(ind2));
414  }
415  }
416  cells.at(ind1).m_neighbors =
417  Neighbors<Cell *>(red_neighbors, black_neighbors);
418  }

422 /** Revert the order of a communicator: After calling this the
423  * communicator is working in reverted order with exchanged
424  * communication types GHOST_SEND <-> GHOST_RECV.
425  */
426 void revert_comm_order(GhostCommunicator &comm) {
427  /* revert order */
428  boost::reverse(comm.communications);
429 
430  /* exchange SEND/RECV */
431  for (auto &c : comm.communications) {
432  if (c.type == GHOST_SEND)
433  c.type = GHOST_RECV;
434  else if (c.type == GHOST_RECV)
435  c.type = GHOST_SEND;
436  else if (c.type == GHOST_LOCL) {
437  boost::reverse(c.part_lists);
438  }
439  }
440 }

445 void assign_prefetches(GhostCommunicator &comm) {
446  for (auto it = comm.communications.begin(); it != comm.communications.end();

456 /* Calc the ghost shift vector for dim dir in direction lr */
457 Utils::Vector3d shift(BoxGeometry const &box, LocalBox<double> const &local_box,
458  int dir, int lr) {
459  Utils::Vector3d ret{};
460 
461  /* Shift is non-zero only in periodic directions, if we are at the box
462  * boundary */
463  ret[dir] = box.periodic(dir) * local_box.boundary()[2 * dir + lr] *
464  box.length()[dir];
465 
466  return ret;

470 GhostCommunicator DomainDecomposition::prepare_comm() {
471  int dir, lr, i, cnt, n_comm_cells[3];
472  Utils::Vector3i lc{}, hc{}, done{};
473 
474  auto const comm_info = Utils::Mpi::cart_get<3>(m_comm);
475  auto const node_neighbors = Utils::Mpi::cart_neighbors<3>(m_comm);
476 
477  /* calculate number of communications */
478  std::size_t num = 0;
479  for (dir = 0; dir < 3; dir++) {
480  for (lr = 0; lr < 2; lr++) {
481  /* No communication for border of non periodic direction */
482  if (comm_info.dims[dir] == 1)
483  num++;
484  else
485  num += 2;
486  }
487  }
488 
489  /* prepare communicator */
490  auto ghost_comm = GhostCommunicator{m_comm, num};
491 
492  /* number of cells to communicate in a direction */
493  n_comm_cells[0] = cell_grid[1] * cell_grid[2];
494  n_comm_cells[1] = cell_grid[2] * ghost_cell_grid[0];
495  n_comm_cells[2] = ghost_cell_grid[0] * ghost_cell_grid[1];
496 
497  cnt = 0;
498  /* direction loop: x, y, z */
499  for (dir = 0; dir < 3; dir++) {
500  lc[(dir + 1) % 3] = 1 - done[(dir + 1) % 3];
501  lc[(dir + 2) % 3] = 1 - done[(dir + 2) % 3];
502  hc[(dir + 1) % 3] = cell_grid[(dir + 1) % 3] + done[(dir + 1) % 3];
503  hc[(dir + 2) % 3] = cell_grid[(dir + 2) % 3] + done[(dir + 2) % 3];
504  /* lr loop: left right */
505  /* here we could in principle build in a one sided ghost
506  communication, simply by taking the lr loop only over one
507  value */
508  for (lr = 0; lr < 2; lr++) {
509  if (comm_info.dims[dir] == 1) {
510  /* just copy cells on a single node */
511  ghost_comm.communications[cnt].type = GHOST_LOCL;
512  ghost_comm.communications[cnt].node = m_comm.rank();
513 
514  /* Buffer has to contain Send and Recv cells -> factor 2 */
515  ghost_comm.communications[cnt].part_lists.resize(2 * n_comm_cells[dir]);
516  /* prepare folding of ghost positions */
517  ghost_comm.communications[cnt].shift =
518  shift(m_box, m_local_box, dir, lr);
519 
520  /* fill send ghost_comm cells */
521  lc[dir] = hc[dir] = 1 + lr * (cell_grid[dir] - 1);
522 
523  fill_comm_cell_lists(ghost_comm.communications[cnt].part_lists.data(),
524  lc, hc);
525 
526  /* fill recv ghost_comm cells */
527  lc[dir] = hc[dir] = 0 + (1 - lr) * (cell_grid[dir] + 1);
528 
529  /* place receive cells after send cells */
530  fill_comm_cell_lists(
531  &ghost_comm.communications[cnt].part_lists[n_comm_cells[dir]], lc,
532  hc);
533 
534  cnt++;
535  } else {
536  /* i: send/recv loop */
537  for (i = 0; i < 2; i++) {
538  if ((comm_info.coords[dir] + i) % 2 == 0) {
539  ghost_comm.communications[cnt].type = GHOST_SEND;
540  ghost_comm.communications[cnt].node = node_neighbors[2 * dir + lr];
541  ghost_comm.communications[cnt].part_lists.resize(n_comm_cells[dir]);
542  /* prepare folding of ghost positions */
543  ghost_comm.communications[cnt].shift =
544  shift(m_box, m_local_box, dir, lr);
545 
546  lc[dir] = hc[dir] = 1 + lr * (cell_grid[dir] - 1);
547 
548  fill_comm_cell_lists(
549  ghost_comm.communications[cnt].part_lists.data(), lc, hc);
550  cnt++;
551  }
552  if ((comm_info.coords[dir] + (1 - i)) % 2 == 0) {
553  ghost_comm.communications[cnt].type = GHOST_RECV;
554  ghost_comm.communications[cnt].node =
555  node_neighbors[2 * dir + (1 - lr)];
556  ghost_comm.communications[cnt].part_lists.resize(n_comm_cells[dir]);
557 
558  lc[dir] = hc[dir] = (1 - lr) * (cell_grid[dir] + 1);
559 
560  fill_comm_cell_lists(
561  ghost_comm.communications[cnt].part_lists.data(), lc, hc);
562  cnt++;
563  }
564  }
565  }
566  done[dir] = 1;
567  }
568  }
569 
570  return ghost_comm;
571 }

573 DomainDecomposition::DomainDecomposition(boost::mpi::communicator comm,
574  double range,
575  const BoxGeometry &box_geo,
576  const LocalBox<double> &local_geo)
577  : m_comm(std::move(comm)), m_box(box_geo), m_local_box(local_geo) {
578  /* set up new domain decomposition cell structure */
579  create_cell_grid(range);
580 
581  /* setup cell neighbors */
582  init_cell_interactions();
583 
584  /* mark local and ghost cells */
585  mark_cells();
586 
587  /* create communicators */
588  m_exchange_ghosts_comm = prepare_comm();
589  m_collect_ghost_force_comm = prepare_comm();
590 
591  /* collect forces has to be done in reverted order! */
592  revert_comm_order(m_collect_ghost_force_comm);
593 
594  assign_prefetches(m_exchange_ghosts_comm);
595  assign_prefetches(m_collect_ghost_force_comm);
